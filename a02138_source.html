29bool canDefaultPoolAllocate(unsigned int setCount,
30 const DescriptorPool::SetBindingInfo& allocInfo) {
31 if (setCount > MaxSets) { return false; }
32
33 auto available = PoolSizes;
34
35 for (std::uint32_t i = 0; i < allocInfo.bindingCount; ++i) {
36 const auto& binding = allocInfo.bindings[i];
37 const unsigned int bindingCount = binding.descriptorCount * setCount;
38 if (bindingCount > available[binding.descriptorType]) { return false; }
39 available[binding.descriptorType] -= bindingCount;
40 }
41 return true;
42}

82DescriptorPool::AllocationHandle DescriptorPool::allocate(VkDescriptorSetLayout layout,
83 VkDescriptorSet* sets,
84 std::size_t setCount, bool dedicated) {
85 std::unique_lock lock(mutex);
86
87 // find layout info
88 const auto lit = layoutMap.find(layout);
89 if (lit == layoutMap.end()) {
90 throw std::runtime_error(
91 "Failed to find descriptor set layout. Create the layout with the pool");
92 }
93 const SetBindingInfo& allocInfo = lit->second;
94
95 // test if requires dedicated pool
96 if (!dedicated) {
97 if (!canDefaultPoolAllocate(setCount, allocInfo)) {
98 BL_LOG_WARN << "Descriptor set allocation requires dedicated pool due to size";
99 dedicated = true;
100 }
101 }
102
103 // create allocation record
104 AllocationHandle handle = AllocationHandle(
105 this, allocations.emplace(allocations.begin(), dedicated, layout, setCount, sets));
106
107 // create dedicated pool if required
108 if (dedicated) {
109 handle.it->pool = pools.emplace(pools.end(), vulkanState, allocInfo, setCount);
110 handle.it->pool->allocate(allocInfo, layout, sets, setCount);
111 return handle;
112 }
113
114 // see if existing pool can allocate
115 for (auto it = pools.begin(); it != pools.end(); ++it) {
116 if (it->canAllocate(allocInfo, setCount)) {
117 handle.it->pool = it;
118 goto newPoolNotNeeded;
119 }
120 }
121
122 // create new pool to allocate from
123 handle.it->pool = pools.emplace(pools.end(), vulkanState);
124
125newPoolNotNeeded:
126 handle.it->pool->allocate(allocInfo, layout, sets, setCount);
127 return handle;
128}

130void DescriptorPool::release(AllocationHandle& handle, const VkDescriptorSet* sets) {
131 if (!handle.pool) { return; }
132 if (handle.pool != this) {
133 BL_LOG_ERROR << "Attempted to release descriptor set from wrong pool";
134 return;
135 }
136
137 std::unique_lock lock(mutex);
138
139 // find layout info
140 const auto lit = layoutMap.find(handle.it->layout);
141 if (lit == layoutMap.end()) {
142 throw std::runtime_error(
143 "Failed to find descriptor set layout. Create the layout with the pool");
144 }
145 const SetBindingInfo& allocInfo = lit->second;
146
147 // free sets
148 if (handle.it->dedicated) { pools.erase(handle.it->pool); }
149 else {
150 handle.it->pool->release(
151 allocInfo, sets != nullptr ? sets : handle.it->sets, handle.it->setCount);
152 }
153 allocations.erase(handle.it);
154
155 handle.pool = nullptr;
156}

211bool DescriptorPool::Subpool::canAllocate(const SetBindingInfo& allocInfo,
212 std::size_t setCount) const {
213 if (freeSets < setCount) { return false; }
214
215 auto availLocal = available;
216
217 for (std::size_t j = 0; j < allocInfo.bindingCount; ++j) {
218 const auto& binding = allocInfo.bindings[j];
219 const unsigned int bindingCount = binding.descriptorCount * setCount;
220 if (availLocal[binding.descriptorType] < bindingCount) { return false; }
221 availLocal[binding.descriptorType] -= bindingCount;
222 }
223
224 return true;
225}

227void DescriptorPool::Subpool::allocate(const SetBindingInfo& allocInfo,
228 VkDescriptorSetLayout layout, VkDescriptorSet* sets,
229 std::size_t setCount) {
230 // update metadata
231 freeSets -= setCount;
232 for (std::size_t j = 0; j < allocInfo.bindingCount; ++j) {
233 const auto& binding = allocInfo.bindings[j];
234 available[binding.descriptorType] -= binding.descriptorCount * setCount;
235 }
236
237 // duplicate layout for vulkan api
238 std::array<VkDescriptorSetLayout, 128> layoutsArray;
239 std::vector<VkDescriptorSetLayout> layoutsVector;
240 VkDescriptorSetLayout* pLayouts;
241 if (setCount <= layoutsArray.size()) {
242 for (unsigned int i = 0; i < setCount; ++i) { layoutsArray[i] = layout; }
243 pLayouts = layoutsArray.data();
244 }
245 else {
246 layoutsVector.resize(setCount, layout);
247 pLayouts = layoutsVector.data();
248 }
249
250 // allocate sets
251 VkDescriptorSetAllocateInfo setAllocInfo{};
252 setAllocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
253 setAllocInfo.descriptorPool = pool;
254 setAllocInfo.descriptorSetCount = setCount;
255 setAllocInfo.pSetLayouts = pLayouts;
256 if (vkAllocateDescriptorSets(vulkanState.getDevice(), &setAllocInfo, sets) != VK_SUCCESS) {
257 throw std::runtime_error("Failed to allocate descriptor sets");
258 }
259}
