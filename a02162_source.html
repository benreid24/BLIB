23RenderPassParameters& RenderPassParameters::useSubpassOutputsAsInputs(std::uint32_t index,
24 bool depth) {
25 if (index == 0 || index >= subpasses.size()) {
26 throw std::runtime_error("Invalid subpass index");
27 }
28
29 SubPass& prevPass = subpasses[index - 1];
30 const auto& inputs = prevPass.colorAttachments;
31 if (inputs.empty()) {
32 throw std::runtime_error("Subpass " + std::to_string(index - 1) +
33 " has no color attachments to use as inputs");
34 }
35
36 const bool useDepth = depth && prevPass.depthAttachment.has_value();
37 SubPass& subpass = subpasses[index];
38 subpass.inputAttachments.resize(inputs.size() + (useDepth ? 1 : 0));
39 for (std::size_t i = 0; i < inputs.size(); ++i) {
40 subpass.inputAttachments[i].attachment = inputs[i].attachment;
41 subpass.inputAttachments[i].layout = inputs[i].layout;
42 }
43 if (useDepth) {
44 subpass.inputAttachments.back().attachment = prevPass.depthAttachment.value().attachment;
45 subpass.inputAttachments.back().layout = prevPass.depthAttachment.value().layout;
46 }
47
48 return *this;
49}

82RenderPassParameters&& RenderPassParameters::build() {
83 if (subpasses.empty()) {
84 throw std::runtime_error("RenderPass must have at least one subpass");
85 }
86 if (attachments.empty()) {
87 throw std::runtime_error("RenderPass must have at least one attachment");
88 }
89
90 isCleared = true;
91 for (const auto& attachment : attachments) {
92 if (attachment.loadOp != VK_ATTACHMENT_LOAD_OP_CLEAR) {
93 isCleared = false;
94 break;
95 }
96 }
97
98 // regardless of msaa state we can populate the resolve attachments and append them to the
99 // attachment list
100 if (msaaBehavior & MSAABehavior::ResolveAttachments) {
101 if (subpasses.size() > 1) {
102 throw std::runtime_error("Resolve attachments can only be used with a single subpass");
103 }
104 auto& pass = subpasses[0];
105 pass.resolveAttachments = pass.colorAttachments;
106 for (auto& resolve : pass.resolveAttachments) {
107 resolve.attachment += pass.colorAttachments.size();
108 }
109 unsigned int i = 0;
110 for (const auto& color : pass.colorAttachments) {
111 semanticFormats.emplace_back(semanticFormats[i++]);
112 const auto& src = attachments[color.attachment];
113 auto& resolved = attachments.emplace_back(src);
114 resolved.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
115 resolved.loadOp = VK_ATTACHMENT_LOAD_OP_DONT_CARE;
116 resolved.samples = VK_SAMPLE_COUNT_1_BIT;
117 }
118
119 // move the depth attachment to the end of the list if it exists
120 if (pass.depthAttachment.has_value()) {
121 const auto semantic = semanticFormats[pass.depthAttachment.value().attachment];
122 const auto depth = attachments[pass.depthAttachment.value().attachment];
123 attachments.erase(pass.depthAttachment.value().attachment);
124 attachments.emplace_back(depth);
125 semanticFormats.erase(pass.depthAttachment.value().attachment);
126 semanticFormats.emplace_back(semantic);
127 pass.depthAttachment.value().attachment = attachments.size() - 1;
128 }
129 }
130
131 return std::move(*this);
132}
