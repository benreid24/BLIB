25scene::SceneObject* BatchedScene::doAdd(ecs::Entity entity, rcom::DrawableBase& obj,
26 UpdateSpeed updateFreq) {
27 const auto alloc = objects.allocate(updateFreq, entity);
28 const std::uint32_t sceneId = alloc.newObject->sceneKey.sceneId;
29 auto& transCache = updateFreq == UpdateSpeed::Static ? staticTransCache : dynamicTransCache;
30
31 if (alloc.addressesChanged) {
32 transCache.resize(alloc.newCapacity);
33 handleAddressChange(updateFreq, alloc.originalBaseAddress);
34 }
35
36 transCache[sceneId] = obj.getContainsTransparency();
37 auto& batch = obj.getContainsTransparency() ? transparentObjects : opaqueObjects;
38 mat::MaterialPipeline* pipeline = obj.getCurrentPipeline();
39
40 // find or create layout batch
41 LayoutBatch* layoutBatch = nullptr;
42 for (LayoutBatch& lb : batch.batches) {
43 if (lb.layout.rawLayout() == pipeline->getLayout().rawLayout()) {
44 layoutBatch = &lb;
45 break;
46 }
47 }
48 if (!layoutBatch) {
49 batch.batches.emplace_back(descriptorSets, pipeline->getLayout());
50 layoutBatch = &batch.batches.back();
51 }
52
53 // init descriptor sets, return on fail
54 for (std::uint8_t i = 0; i < layoutBatch->descriptorCount; ++i) {
55 if (!layoutBatch->descriptors[i]->allocateObject(entity, alloc.newObject->sceneKey)) {
56 for (std::int8_t j = i - 1; j >= 0; --j) {
57 layoutBatch->descriptors[j]->releaseObject(entity, alloc.newObject->sceneKey);
58 }
59 return nullptr;
60 }
61 }
62
63 // find or create pipeline batch
64 auto& pipelineBatches = updateFreq == UpdateSpeed::Static ? layoutBatch->staticBatches :
65 layoutBatch->dynamicBatches;
66 PipelineBatch* pipelineBatch = nullptr;
67 for (PipelineBatch& pb : pipelineBatches) {
68 if (&pb.pipeline == pipeline) {
69 pipelineBatch = &pb;
70 break;
71 }
72 }
73 if (!pipelineBatch) {
74 pipelineBatches.emplace_back(*pipeline);
75 pipelineBatch = &pipelineBatches.back();
76 }
77
78 // add to pipeline batch
79 pipelineBatch->objects.emplace_back(alloc.newObject);
80
81 // add scene link
82 engine.ecs().emplaceComponent<com::BatchSceneLink>(entity, alloc.newObject->sceneKey);
83
84 return alloc.newObject;

107void BatchedScene::releaseObject(SceneObject* object, mat::MaterialPipeline* pipeline) {
108 // lookup object
109 const ecs::Entity entity = objects.getObjectEntity(object->sceneKey);
110 auto& transCache =
111 object->sceneKey.updateFreq == UpdateSpeed::Static ? staticTransCache : dynamicTransCache;
112 auto& batch = transCache[object->sceneKey.sceneId] ? transparentObjects : opaqueObjects;
113
114 // remove from batch
115 for (LayoutBatch& p : batch.batches) {
116 if (p.layout.rawLayout() == pipeline->getLayout().rawLayout()) {
117 auto& pipelineBatches = object->sceneKey.updateFreq == UpdateSpeed::Static ?
118 p.staticBatches :
119 p.dynamicBatches;
120 for (PipelineBatch& pb : pipelineBatches) {
121 if (&pb.pipeline == pipeline) {
122 for (auto it = pb.objects.begin(); it != pb.objects.end(); ++it) {
123 if (*it == object) {
124 pb.objects.erase(it);
125 goto removeDescriptors;
126 }
127 }
128 break;
129 }
130 }
131 removeDescriptors:
132 for (std::uint8_t i = 0; i < p.descriptorCount; ++i) {
133 p.descriptors[i]->releaseObject(entity, object->sceneKey);
134 }
135 break;
136 }
137 }
138
139 // cleanup scene link
140 engine.ecs().removeComponent<com::BatchSceneLink>(entity);

143void BatchedScene::doBatchChange(const BatchChange& change, mat::MaterialPipeline* oldPipeline) {
144 const bool isStatic = change.changed->sceneKey.updateFreq == UpdateSpeed::Static;
145 auto& transCache = isStatic ? staticTransCache : dynamicTransCache;
146 const bool wasTrans = transCache[change.changed->sceneKey.sceneId];
147 transCache[change.changed->sceneKey.sceneId] = change.newTrans;
148
149 // determine what changed
150 const bool transChanged = wasTrans != change.newTrans;
151 const bool pipelineChanged = oldPipeline != change.newPipeline;
152 bool layoutChanged = false;
153
154 // determine if layout changed
155 if (pipelineChanged) {
156 layoutChanged =
157 change.newPipeline->getLayout().rawLayout() != oldPipeline->getLayout().rawLayout();
158 }
159
160 // re-bucket if trans, layout, or pipeline changed
161 if (transChanged || layoutChanged || pipelineChanged) {
162 // remove from old bucket
163 LayoutBatch* oldLayoutBatch = nullptr;
164 auto& oldObjectBatch = wasTrans ? transparentObjects : opaqueObjects;
165 for (LayoutBatch& lb : oldObjectBatch.batches) {
166 if (lb.layout.rawLayout() == oldPipeline->getLayout().rawLayout()) {
167 oldLayoutBatch = &lb;
168 auto& pipelineBatches = isStatic ? lb.staticBatches : lb.dynamicBatches;
169 for (PipelineBatch& pb : pipelineBatches) {
170 if (&pb.pipeline == oldPipeline) {
171 for (auto it = pb.objects.begin(); it != pb.objects.end(); ++it) {
172 if (*it == change.changed) {
173 pb.objects.erase(it);
174 break;
175 }
176 }
177 break;
178 }
179 }
180 break;
181 }
182 }
183
184 auto& batch = change.newTrans ? transparentObjects : opaqueObjects;
185
186 // find or create new layout batch
187 LayoutBatch* layoutBatch = nullptr;
188 for (LayoutBatch& lb : batch.batches) {
189 if (lb.layout.rawLayout() == change.newPipeline->getLayout().rawLayout()) {
190 layoutBatch = &lb;
191 break;
192 }
193 }
194 if (!layoutBatch) {
195 layoutBatch =
196 &batch.batches.emplace_back(descriptorSets, change.newPipeline->getLayout());
197 }
198
199 // find or create pipeline batch
200 auto& pipelineBatches = isStatic ? layoutBatch->staticBatches : layoutBatch->dynamicBatches;
201 PipelineBatch* pipelineBatch = nullptr;
202 for (PipelineBatch& pb : pipelineBatches) {
203 if (&pb.pipeline == change.newPipeline) {
204 pipelineBatch = &pb;
205 break;
206 }
207 }
208 if (!pipelineBatch) { pipelineBatch = &pipelineBatches.emplace_back(*change.newPipeline); }
209
210 // add to pipeline batch
211 pipelineBatch->objects.emplace_back(change.changed);
212
213 // update descriptors if layout changed
214 if (layoutChanged) {
215 SceneObject& object = *change.changed;
216 const ecs::Entity entity = objects.getObjectEntity(object.sceneKey);
217 change.newPipeline->getLayout().updateDescriptorSets(descriptorSets,
218 oldLayoutBatch->descriptors.data(),
219 oldLayoutBatch->descriptorCount,
220 entity,
221 object.sceneKey.sceneId,
222 object.sceneKey.updateFreq);
223 }

227void BatchedScene::renderScene(scene::SceneRenderContext& ctx) {
228 for (ObjectBatch* batch : std::array<ObjectBatch*, 2>{&opaqueObjects, &transparentObjects}) {
229 for (LayoutBatch& lb : batch->batches) {
230 const VkPipelineLayout layout = lb.layout.rawLayout();
231 UpdateSpeed speed = UpdateSpeed::Dynamic;
232
233 for (auto* pipelineBatches : std::array<std::vector<PipelineBatch>*, 2>{
234 &lb.dynamicBatches, &lb.staticBatches}) {
235 // bind layout descriptors
236 ctx.bindDescriptors(layout, speed, lb.descriptors.data(), lb.descriptorCount);
237
238 // render each pipeline
239 if (!lb.bindless) {
240 for (PipelineBatch& pb : *pipelineBatches) {
241 ctx.bindPipeline(pb.pipeline);
242 for (SceneObject* obj : pb.objects) {
243 if (obj->hidden) { continue; }
244 for (std::uint8_t i = lb.perObjStart; i < lb.descriptorCount; ++i) {
245 lb.descriptors[i]->bindForObject(ctx, layout, i, obj->sceneKey);
246 }
247 ctx.renderObject(*obj);
248 }
249 }
250 }
251 else {
252 for (PipelineBatch& pb : *pipelineBatches) {
253 ctx.bindPipeline(pb.pipeline);
254 for (SceneObject* obj : pb.objects) {
255 if (obj->hidden) { continue; }
256 ctx.renderObject(*obj);
257 }
258 }
259 }
260
261 speed = UpdateSpeed::Static;
262 }
263 }
