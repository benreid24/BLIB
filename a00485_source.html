116 if (range.start + range.size == totalSize) { backRange = &range; }
117 if (range.size >= size) {
118 if (!allocFrom || allocFrom->size > range.size) { allocFrom = &range; }
119 }
120 }

123 result.range.size = size;
124 result.newPoolSize = totalSize;

126 // alloc from range and update range
127 if (allocFrom) {

129 result.range.start = allocFrom->start;

132 allocFrom->size -= size;
133 }
134 else {

136 freeRanges.erase(freeRanges.begin() + i);

138 }
139 // expand pool if no free range large enough
140 else {

144 const T expandedBy = newSize - totalSize;

162template<typename T>
163void RangeAllocatorUnbounded<T>::release(Range range) {
164 // find adjacent range(s)
165 Range* frontCon = nullptr;
166 Range* backCon = nullptr;
167 for (Range& fr : freeRanges) {
168 if (fr.start + fr.size == range.start) { frontCon = &fr; }
169 else if (range.start + range.size == fr.start) { backCon = &fr; }
170 }
171
172 // expand or combine ranges if found
173 if (frontCon && backCon) {
174 frontCon->size += backCon->size + range.size;
175 const std::size_t bi = backCon - freeRanges.data();
176 freeRanges.erase(freeRanges.begin() + bi);
177 }
178 else if (frontCon) { frontCon->size += range.size; }
179 else if (backCon) {
180 backCon->start -= range.size;
181 backCon->size += range.size;
182 }
183 else { freeRanges.emplace_back(range); }
