228 void AnimationData::computeFrameSize(Frame& frame) {
229  sf::FloatRect bounds(1000000.f, 1000000.f, -1000000.f, -1000000.f);
230  for (auto& shard : frame.shards) {
231  const sf::FloatRect source(shard.source);
232  const sf::Vector2f center(source.width * 0.5f, source.height * 0.5f);
233  sf::Transform transform;
234  transform.rotate(shard.rotation, center);
235  transform.scale(shard.scale, center);
236  transform.translate(shard.offset);
237 
238  const sf::FloatRect shardBounds = transform.transformRect(source);
239  bounds.left = std::min(bounds.left, shardBounds.left);
240  bounds.top = std::min(bounds.top, shardBounds.top);
241  bounds.width = std::max(bounds.width, shardBounds.left + shardBounds.width);
242  bounds.height = std::max(bounds.height, shardBounds.top + shardBounds.height);
243  }
244 
245  frame.size = {bounds.width - bounds.left, bounds.height - bounds.top};
246 }

275  BL_LOG_ERROR

277  return false;
278  }
279  }
280

282  if (frame.shards.size() != frames.front().shards.size()) {
283  BL_LOG_ERROR << "Animation combine failed: State animation has different shard "

287  }
288 
289  if (ss) {

291  if (!isValidSlideshow(anim)) {
292  BL_LOG_ERROR << "Animation combine failed: State animation was not Slideshow";
293  return false;

296  }
297  }

301  stateOffsets[stateIndex] = offset;

304  for (std::size_t i = offset; i < frames.size() - 1; ++i) { frames[i].nextFrame = i + 1; }
