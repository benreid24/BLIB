91bool Buffer::ensureSize(VkDeviceSize newSize, bool skipCopy) {
92 if (newSize > size) {
93 doDefer();
94
95 const VkDeviceSize oldSize = size;
96 const VkBuffer oldBuffer = buffer;
97 size = newSize;
98 if (!doCreate()) { throw std::runtime_error("Failed to resize buffer"); }
99
100 if (!skipCopy) {
101 auto commandBuffer = vulkanState->sharedCommandPool.createBuffer();
102
103 // barrier to ensure writes to old buffer are complete
104 VkBufferMemoryBarrier barrier{};
105 barrier.sType = VK_STRUCTURE_TYPE_BUFFER_MEMORY_BARRIER;
106 barrier.buffer = oldBuffer;
107 barrier.offset = 0;
108 barrier.size = oldSize;
109 barrier.srcAccessMask = VK_ACCESS_MEMORY_WRITE_BIT;
110 barrier.dstAccessMask = VK_ACCESS_MEMORY_READ_BIT;
111 vkCmdPipelineBarrier(commandBuffer,
112 VK_PIPELINE_STAGE_TRANSFER_BIT,
113 VK_PIPELINE_STAGE_TRANSFER_BIT,
114 VK_DEPENDENCY_BY_REGION_BIT,
115 0,
116 nullptr,
117 1,
118 &barrier,
119 0,
120 nullptr);
121
122 // copy old buffer into new
123 VkBufferCopy copyCmd{};
124 copyCmd.dstOffset = 0;
125 copyCmd.size = oldSize;
126 copyCmd.srcOffset = 0;
127 vkCmdCopyBuffer(commandBuffer, oldBuffer, buffer, 1, &copyCmd);
128
129 // insert pipeline barrier
130 VkBufferMemoryBarrier bufBarrier{};
131 bufBarrier.sType = VK_STRUCTURE_TYPE_BUFFER_MEMORY_BARRIER;
132 bufBarrier.srcAccessMask = VK_ACCESS_MEMORY_WRITE_BIT;
133 bufBarrier.dstAccessMask = VK_ACCESS_MEMORY_READ_BIT;
134 bufBarrier.buffer = buffer;
135 bufBarrier.offset = 0;
136 bufBarrier.size = oldSize;
137 vkCmdPipelineBarrier(commandBuffer,
138 VK_PIPELINE_STAGE_TRANSFER_BIT,
139 VK_PIPELINE_STAGE_VERTEX_INPUT_BIT,
140 VK_DEPENDENCY_BY_REGION_BIT,
141 0,
142 nullptr,
143 1,
144 &bufBarrier,
145 0,
146 nullptr);

148 commandBuffer.submit();
149
150 return true;
151 }
