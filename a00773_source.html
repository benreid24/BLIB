14 const Value* getNestedValue(const Group& group, std::string path) {
15  const Group* nestedGroup = &group;
16 
17  while (!path.empty()) {
18  const auto n = path.find_first_of('/');
19  if (n != std::string::npos) {
20  const std::string subname = path.substr(0, n);
21  const Value* val = nestedGroup->getField(subname);
22  if (!val || !val->getAsGroup()) return nullptr;
23  nestedGroup = val->getAsGroup();
24  path.erase(0, n + 1);
25  }
26  else
27  return nestedGroup->getField(path);
28  }
29 
30  return nullptr;
31 }
