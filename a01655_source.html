144bool Engine::loop() {
145 BL_LOG_INFO << "Starting engine with state: " << states.top()->name();
146
147 sf::Clock loopTimer;
148 sf::Clock updateOuterTimer;
149 const float minFrameLength =
150 engineSettings.maximumFramerate() > 0 ? 1.f / engineSettings.maximumFramerate() : 0.f;
151 float lag = 0.f;
152
153 sf::Clock updateMeasureTimer;
154 float updateTimestep = engineSettings.updateTimestep();
155 float averageUpdateTime = engineSettings.updateTimestep();
156
157 float lastWarnTime = -6.f; // always log first warning
158 bool followupLog = false;
159 auto fallBehindWarning = [&lastWarnTime, &updateMeasureTimer, &followupLog](float behind) {
160 if (updateMeasureTimer.getElapsedTime().asSeconds() - lastWarnTime >= 5.f) {
161 lastWarnTime = updateMeasureTimer.getElapsedTime().asSeconds();
162 followupLog = true;
163 BL_LOG_WARN << "Can't catch up, running " << behind << " seconds behind";
164 }
165 };
166 auto skipUpdatesInfo = [&followupLog](int frameCount) {
167 if (followupLog) {
168 followupLog = false;
169 BL_LOG_INFO << "Skipping " << frameCount << " updates";
170 }
171 };
172
173 sf::Clock fpsTimer;
174 float frameCount = 0.f;
175
176 workers.start();
177 backgroundWorkers.start(2);
178 states.top()->activate(*this);
179 bl::event::Dispatcher::dispatch<event::Startup>({states.top()});
180
181 while (true) {
182 // Clear flags from last loop
183 engineFlags.clear();
184
185 if (renderWindow.isOpen()) {
186 sf::Event event;
187 while (renderWindow.pollEvent(event)) {
188 bl::event::Dispatcher::dispatch<sf::Event>(event);
189
190 switch (event.type) {
191 case sf::Event::Closed:
192 bl::event::Dispatcher::dispatch<event::Shutdown>(
193 {event::Shutdown::WindowClosed});
194 return true;
195
196 case sf::Event::LostFocus:
197 bl::event::Dispatcher::dispatch<event::Paused>({});
198 if (!awaitFocus()) {
199 bl::event::Dispatcher::dispatch<event::Shutdown>(
200 {event::Shutdown::WindowClosed});
201 return true;
202 }
203 bl::event::Dispatcher::dispatch<event::Resumed>({});
204 updateOuterTimer.restart();
205 loopTimer.restart();
206 break;
207
208 case sf::Event::Resized:
209 handleResize(event.size, true);
210 break;
211
212 default:
213 break;
214 }
215 }
216 }
217
218 ecsSystems.notifyFrameStart();
219
220 // Update and render
221 lag += updateOuterTimer.getElapsedTime().asSeconds() * timeScale;
222 updateOuterTimer.restart();
223 const float startingLag = lag;
224 float totalDt = 0.f;
225 updateMeasureTimer.restart();
226
227 // update until caught up
228 while (lag >= updateTimestep) {
229 const float updateStart = updateMeasureTimer.getElapsedTime().asSeconds();
230
231 // core update game logic
232 totalDt += updateTimestep;
233 input.update();
234 states.top()->update(*this, updateTimestep, updateTimestep / timeScale);
235 ecsSystems.update(FrameStage::FrameStart,
236 FrameStage::MARKER_OncePerFrame,
237 states.top()->systemsMask(),
238 updateTimestep,
239 updateTimestep / timeScale,
240 lag,
241 lag / timeScale);
242 bl::event::Dispatcher::syncListeners();
243
244 // check if we should end early
245 if (engineFlags.stateChangeReady()) {
246 lag = 0.f;
247 break;
248 }
249
250 // handle timing
251 lag -= updateTimestep;
252 averageUpdateTime =
253 0.8f * averageUpdateTime +
254 0.2f * (updateMeasureTimer.getElapsedTime().asSeconds() - updateStart);
255 if (updateMeasureTimer.getElapsedTime().asSeconds() > startingLag * 1.1f) {
256 fallBehindWarning(updateMeasureTimer.getElapsedTime().asSeconds() - startingLag);
257 if (engineSettings.allowVariableTimestep()) {
258 const float newTs = updateTimestep * 1.05f;
259 BL_LOG_INFO << "Adjusting update timestep from " << updateTimestep << "s to "
260 << newTs << "s";
261 updateTimestep = newTs;
262 averageUpdateTime = updateTimestep;
263 }
264 else {
265 skipUpdatesInfo(std::ceil(lag / updateTimestep));
266 lag = 0.f;
267 }
268 }
269 }
270
271 // update timing
272 if (averageUpdateTime < updateTimestep * 0.9f &&
273 updateTimestep > engineSettings.updateTimestep()) {
274 const float newTs = std::max(engineSettings.updateTimestep(), updateTimestep * 0.95f);
275 BL_LOG_INFO << "Performance improved, adjusting timestep from " << updateTimestep
276 << "s to " << newTs << "s";
277 updateTimestep = newTs;
278 }
279
280 // wait for render thread to finish current frame & block it
281 std::unique_lock renderLock(renderingMutex);
282
283 // Free world slots
284 for (auto& world : worlds) {
285 if (world && world.refCount() == 1) {
286 bl::event::Dispatcher::dispatch<event::WorldDestroyed>({*world});
287 world.release();
288 }
289 }
290
291 // Process flags
292 bool stateChanged = false;
293 while (engineFlags.stateChangeReady()) {
294 stateChanged = true;
295
296 if (engineFlags.active(Flags::Terminate)) {
297 bl::event::Dispatcher::dispatch<event::Shutdown>({event::Shutdown::Terminated});
298 return true;
299 }
300 else if (engineFlags.active(Flags::PopState)) {
301 BL_LOG_INFO << "Popping state: " << states.top()->name();
302 State::Ptr prev = states.top();
303 prev->deactivate(*this);
304 states.pop();
305 if (states.empty()) { // exit if no states left
306 BL_LOG_INFO << "Final state popped, exiting";
307 bl::event::Dispatcher::dispatch<event::Shutdown>(
308 {event::Shutdown::FinalStatePopped});
309 return true;
310 }
311 BL_LOG_INFO << "New engine state (popped): " << states.top()->name();
312 postStateChange(prev);
313 }
314 else if (engineFlags.active(Flags::_priv_ReplaceState)) {
315 BL_LOG_INFO << "New engine state (replaced): " << newState->name();
316 State::Ptr prev = states.top();
317 prev->deactivate(*this);
318 states.pop();
319 states.push(newState);
320 postStateChange(prev);
321 }
322 else if (engineFlags.active(Flags::_priv_PushState)) {
323 BL_LOG_INFO << "New engine state (pushed): " << newState->name();
324 State::Ptr& prev = states.top();
325 prev->deactivate(*this);
326 states.push(newState);
327 postStateChange(prev);
328 }
329
330 updateOuterTimer.restart();
331 loopTimer.restart();
332 }
333
334 if (!stateChanged) {
335 // signal render next frame
336 if (renderWindow.isOpen()) {
337 // sync descriptors
338 ecsSystems.update(FrameStage::MARKER_OncePerFrame,
339 FrameStage::COUNT,
340 states.top()->systemsMask(),
341 totalDt,
342 totalDt / timeScale,
343 lag,
344 lag / timeScale);
345
346 // flush scene object changes
347 renderingSystem.syncSceneObjects();
348
349 // Flush ECS deletion queues
350 entityRegistry.flushDeletions();
351
352 // signal rendering thread to start
353 renderLock.unlock();
354 renderingCv.notify_one();
355 }
356
357 // Adhere to FPS cap
358 if (minFrameLength > 0) {
359 const float st = minFrameLength - loopTimer.getElapsedTime().asSeconds();
360 if (st > 0) sf::sleep(sf::seconds(st));
361 loopTimer.restart();
362 }
363
364 frameCount += 1.f;
365 if (renderWindow.isOpen() && fpsTimer.getElapsedTime().asSeconds() >= 1.f &&
366 engineSettings.logFps()) {
367 const float fps = frameCount / fpsTimer.getElapsedTime().asSeconds();
368 renderWindow.getSfWindow().setTitle(engineSettings.windowParameters().title() +
369 " (" + std::to_string(int(std::roundf(fps))) +
370 " fps)");
371 frameCount = 0.f;
372 fpsTimer.restart();
373 }
374 }
375 else {
376 // Flush ECS deletion queues
377 entityRegistry.flushDeletions();
378 }
379 }
380
381 return false; // shouldn't be able to get here
382}

437void Engine::handleResize(const sf::Event::SizeEvent& resize, bool ss) {
438 const sf::Vector2f modeSize(sf::Vector2u(engineSettings.windowParameters().videoMode().width,
439 engineSettings.windowParameters().videoMode().height));
440 const sf::Vector2f& ogSize = engineSettings.windowParameters().initialViewSize().x > 0.f ?
441 engineSettings.windowParameters().initialViewSize() :
442 modeSize;
443
444 const float newWidth = static_cast<float>(resize.width);
445 const float newHeight = static_cast<float>(resize.height);
446
447 sf::FloatRect viewport(0.f, 0.f, 1.f, 1.f);
448 if (engineSettings.windowParameters().letterBox()) {
449 const float xScale = newWidth / ogSize.x;
450 const float yScale = newHeight / ogSize.y;
451
452 if (xScale >= yScale) { // constrained by height, bars on sides
453 windowScale = yScale;
454 viewport.width = ogSize.x * yScale / newWidth;
455 viewport.left = (1.f - viewport.width) * 0.5f;
456 }
457 else { // constrained by width, bars on top and bottom
458 windowScale = xScale;
459 viewport.height = ogSize.y * xScale / newHeight;
460 viewport.top = (1.f - viewport.height) * 0.5f;
461 }
462 }
463
464 if (renderingSystem.vulkanState().device) {
465 renderingSystem.processResize(sf::Rect<std::uint32_t>(newWidth * viewport.left,
466 newHeight * viewport.top,
467 newWidth * viewport.width,
468 newHeight * viewport.height));
469 if (engineSettings.windowParameters().syncOverlaySize() &&
470 !engineSettings.windowParameters().letterBox()) {
471 cam::OverlayCamera::setOverlayCoordinateSpace(newWidth, newHeight);
472 }
473 }
474
475 if (ss) {
476 Settings::WindowParameters params = engineSettings.windowParameters();
477 params.withVideoMode(
478 sf::VideoMode(resize.width, resize.height, params.videoMode().bitsPerPixel));
479 engineSettings.withWindowParameters(params);
480 params.syncToConfig();
481 }
482
483 bl::event::Dispatcher::dispatch<event::WindowResized>({renderWindow});
484}
