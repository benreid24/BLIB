139 Value str(SymbolTable& t, const std::vector<Value>& args) {
140  if (args.size() != 1) throw Error("str() takes 1 argument");
141 
142  switch (args[0].getType()) {
143  case Value::TBool:
144  return Value(args[0].getAsBool() ? "true" : "false");
145  case Value::TNumeric: {
146  const int v = args[0].getAsNum();
147  if (v == args[0].getAsNum()) return Value(std::to_string(v));
148  return Value(std::to_string(args[0].getAsNum()));
149  }
150  case Value::TString:
151  return Value(args[0].getAsString());
152  case Value::TArray: {
153  std::string result = "[";
154  if (!args[0].getAsArray().empty())
155  result += str(t, {*(args[0].getAsArray()[0])}).getAsString();
156  for (unsigned int i = 1; i < args[0].getAsArray().size(); ++i) {
157  result += ", " + str(t, {*(args[0].getAsArray()[i])}).getAsString();
158  }
159  result += "]";
160  return Value(result);
161  }
162  case Value::TRef:
163  return str(t, {args[0].deref()});
164 
165  case Value::TFunction:
166  return Value("<function>");
167 
168  case Value::TVoid:
169  return Value("<void>");
170 
171  default:
172  throw Error("Internal error: Invalid Value type");
173  }
