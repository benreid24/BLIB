54void Scene::syncObjects() {
55 std::unique_lock lock(objectMutex);
56 std::unique_lock queueLock(queueMutex);
57 isClearingQueues = true;
58
59 // do batch changes
60 for (const auto& change : queuedBatchChanges) {
61 auto& objectPipelines = change.changed->sceneKey.updateFreq == UpdateSpeed::Static ?
62 staticPipelines :
63 dynamicPipelines;
64 doBatchChange(change, objectPipelines[change.changed->sceneKey.sceneId]);
65 objectPipelines[change.changed->sceneKey.sceneId] = change.newPipeline;
66 }
67 queuedBatchChanges.clear();
68
69 // remove queued objects
70 for (auto* obj : queuedRemovals) { removeQueuedObject(obj); }
71 queuedRemovals.clear();
72
73 // add queued objects
74 ecs::Transaction<ecs::tx::EntityRead> tx(engine.ecs());
75 for (auto& add : queuedAdds) {
76 if (engine.ecs().entityExists(add.entity, tx)) { addQueuedObject(add); }
77 }
78 queuedAdds.clear();
79
80 isClearingQueues = false;
81}

105void Scene::removeQueuedObject(scene::SceneObject* obj) {
106 std::unique_lock lock(queueMutex);
107
108 auto& objectPipelines =
109 obj->sceneKey.updateFreq == UpdateSpeed::Static ? staticPipelines : dynamicPipelines;
110 const auto cachedKey = obj->sceneKey.sceneId;
111 mat::MaterialPipeline* pipeline =
112 cachedKey < objectPipelines.size() ? objectPipelines[cachedKey] : nullptr;
113 if (pipeline != nullptr) {
114 doObjectRemoval(obj, pipeline);
115 objectPipelines[cachedKey] = nullptr;

119void Scene::addQueuedObject(ObjectAdd& add) {
120 rcom::DrawableBase& object = *add.object;
121 scene::SceneObject* sobj = doAdd(add.entity, object, add.updateFreq);
122 if (sobj) {
123 sobj->entity = add.entity;
124 object.sceneRef.object = sobj;
125 object.sceneRef.scene = this;
126 sobj->component = &object;
127
128 auto& objectPipelines =
129 sobj->sceneKey.updateFreq == UpdateSpeed::Static ? staticPipelines : dynamicPipelines;
130 if (sobj->sceneKey.sceneId >= objectPipelines.size()) {
131 objectPipelines.resize(sobj->sceneKey.sceneId + 1, nullptr);
132 }
133 objectPipelines[sobj->sceneKey.sceneId] = object.getCurrentPipeline();
134 }
135 else { BL_LOG_ERROR << "Failed to add " << add.entity << " to scene " << this; }
