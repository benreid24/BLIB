49void RenderPass::doCreate() {
50 const bool msaaEnabled =
51 (createParams.msaaBehavior & RenderPassParameters::MSAABehavior::UseSettings) &&
52 renderer.getSettings().getAntiAliasing() != Settings::AntiAliasing::None;
53 const bool resolveAttachments =
54 (createParams.msaaBehavior & RenderPassParameters::MSAABehavior::ResolveAttachments) &&
55 msaaEnabled;
56
57 ctr::StaticVector<VkSubpassDescription, RenderPassParameters::MaxSubpassCount> subpasses;
58 for (const auto& sp : createParams.subpasses) {
59 subpasses.emplace_back(VkSubpassDescription{});
60 subpasses.back().pipelineBindPoint = VK_PIPELINE_BIND_POINT_GRAPHICS;
61 subpasses.back().colorAttachmentCount = sp.colorAttachments.size();
62 subpasses.back().pColorAttachments = sp.colorAttachments.data();
63 subpasses.back().inputAttachmentCount = sp.inputAttachments.size();
64 subpasses.back().pInputAttachments = sp.inputAttachments.data();
65 subpasses.back().preserveAttachmentCount = sp.preserveAttachments.size();
66 subpasses.back().pPreserveAttachments = sp.preserveAttachments.data();
67 if (sp.depthAttachment.has_value()) {
68 subpasses.back().pDepthStencilAttachment = &sp.depthAttachment.value();
69 }
70 if (msaaEnabled && resolveAttachments) {
71 subpasses.back().pResolveAttachments = sp.resolveAttachments.data();
72 }
73 }
74
75 // set sample count on color attachments based on MSAA behavior
76 if (createParams.msaaBehavior & RenderPassParameters::MSAABehavior::UseSettings) {
77 for (const auto& color : createParams.subpasses[0].colorAttachments) {
78 auto& attachment = createParams.attachments[color.attachment];
79 attachment.samples = renderer.getSettings().getMSAASampleCount();
80 }
81 }
82
83 // create local attachment list so we can reorder it
84 auto semanticFormats = createParams.semanticFormats;
85 auto attachments = createParams.attachments;
86
87 // attachments is always full length, need to remove resolve attachments if msaa is disabled
88 if ((createParams.msaaBehavior & RenderPassParameters::MSAABehavior::ResolveAttachments) &&
89 !msaaEnabled) {
90 for (unsigned int i = 0; i < createParams.subpasses[0].resolveAttachments.size(); ++i) {
91 attachments.erase(createParams.subpasses[0].colorAttachments.size());
92 semanticFormats.erase(createParams.subpasses[0].colorAttachments.size());
93 }
94 }
95
96 // update depth index if it exists
97 if (createParams.subpasses[0].depthAttachment.has_value()) {
98 auto& di = createParams.subpasses[0].depthAttachment.value().attachment;
99 di = msaaEnabled ? attachments.size() - 1 :
100 createParams.subpasses[0].colorAttachments.size();
101 attachments[di].samples =
102 msaaEnabled ? renderer.getSettings().getMSAASampleCount() : VK_SAMPLE_COUNT_1_BIT;
103 }
104
105 // update formats based on semantic formats
106 for (unsigned int i = 0; i < attachments.size(); ++i) {
107 if (createParams.semanticFormats[i] != SemanticTextureFormat::NonSematic) {
108 attachments[i].format =
109 renderer.getTextureFormatManager().getFormat(semanticFormats[i]);
110 }
111 }
112
113 VkRenderPassCreateInfo renderPassInfo{};
114 renderPassInfo.sType = VK_STRUCTURE_TYPE_RENDER_PASS_CREATE_INFO;
115 renderPassInfo.attachmentCount = attachments.size();
116 renderPassInfo.pAttachments = attachments.data();
117 renderPassInfo.subpassCount = subpasses.size();
118 renderPassInfo.pSubpasses = subpasses.data();
119 renderPassInfo.dependencyCount = createParams.dependencies.size();
120 renderPassInfo.pDependencies = createParams.dependencies.data();
121
122 if (vkCreateRenderPass(
123 renderer.vulkanState().getDevice(), &renderPassInfo, nullptr, &renderPass) !=
124 VK_SUCCESS) {
125 throw std::runtime_error("Failed to create render pass");
126 }
127
128 if (createParams.debugName) {
129 renderer.vulkanState().setVulkanObjectDebugName(
130 renderPass, VK_OBJECT_TYPE_RENDER_PASS, createParams.debugName);
131 }
132}
