156 glm::vec2 RenderTarget::transformToWorldSpace(const glm::vec2& sp) const {
157  const glm::vec2 ndc((sp.x - viewport.x) / viewport.width * 2.f - 1.f,
158  (sp.y - viewport.y) / viewport.height * 2.f - 1.f);
159  glm::mat4 tform(1.f);
160  if (hasScene() && scenes.back().camera) {
161  tform = scenes.back().camera->getProjectionMatrix(viewport) *
162  scenes.back().camera->getViewMatrix();
163  }
164  tform = glm::inverse(tform);
165  const glm::vec4 result = tform * glm::vec4(ndc, 0.f, 1.f);
166  return {result.x, result.y};
