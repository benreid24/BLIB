147  maxLineHeight = it.getText().cachedLineHeight;

152  it.getText().cachedLineHeight = maxLineHeight;
153  glm::vec2 advance =

155  prevChar = it.getChar();
156

158  if (lineOnNextSpace) {
159  it.makeNewline();

161  wordStartPos = glm::vec2{0.f, advance.y + maxLineHeight};
162  }
