48// FreeType callbacks that operate on a sf::InputStream
49unsigned long read(FT_Stream rec, unsigned long offset, unsigned char* buffer,
50 unsigned long count) {
51 std::uint64_t convertedOffset = static_cast<std::uint64_t>(offset);
52 sf::InputStream* stream = static_cast<sf::InputStream*>(rec->descriptor.pointer);
53 if (stream->seek(convertedOffset) == convertedOffset) {
54 if (count > 0) {
55 auto read =
56 stream->read(reinterpret_cast<char*>(buffer), static_cast<std::uint64_t>(count));
57 return read.value_or(0);
58 }
59 else { return 0; }
60 }
61 else
62 return count > 0 ? 0 : 1; // error code is 0 if we're reading, or nonzero if we're seeking
63}
64void close(FT_Stream) {}

66// Helper to interpret memory as a specific type
67template<typename T, typename U>
68inline T reinterpret(const U& input) {
69 T output;
70 std::memcpy(&output, &input, sizeof(U));
71 return output;
72}

74// Combine outline thickness, boldness and font glyph index into a single 64-bit key
75std::uint64_t combine(std::uint8_t size, float outlineThickness, bool bold, std::uint32_t index) {
76 return (static_cast<std::uint64_t>(reinterpret<std::uint32_t>(outlineThickness)) << 32) |
77 (static_cast<std::uint64_t>(bold) << 31) | (static_cast<std::uint64_t>(size) << 32) |
78 index;
79}

146bool VulkanFont::loadFromFile(const std::string& filename) {
147 needsUpload = true;
148
149#ifndef SFML_SYSTEM_ANDROID
150
151 // Cleanup the previous resources
152 cleanup();
153 m_refCount = new int(1);
154
155 // Initialize FreeType
156 // Note: we initialize FreeType for every font instance in order to avoid having a single
157 // global manager that would create a lot of issues regarding creation and destruction order.
158 FT_Library library;
159 if (FT_Init_FreeType(&library) != 0) {
160 err() << "Failed to load font \"" << filename << "\" (failed to initialize FreeType)"
161 << std::endl;
162 return false;
163 }
164 m_library = library;
165
166 // Load the new font face from the specified file
167 FT_Face face;
168 if (FT_New_Face(static_cast<FT_Library>(m_library), filename.c_str(), 0, &face) != 0) {
169 err() << "Failed to load font \"" << filename << "\" (failed to create the font face)"
170 << std::endl;
171 return false;
172 }
173
174 // Load the stroker that will be used to outline the font
175 FT_Stroker stroker;
176 if (FT_Stroker_New(static_cast<FT_Library>(m_library), &stroker) != 0) {
177 err() << "Failed to load font \"" << filename << "\" (failed to create the stroker)"
178 << std::endl;
179 FT_Done_Face(face);
180 return false;
181 }
182
183 // Select the unicode character map
184 if (FT_Select_Charmap(face, FT_ENCODING_UNICODE) != 0) {
185 err() << "Failed to load font \"" << filename
186 << "\" (failed to set the Unicode character set)" << std::endl;
187 FT_Stroker_Done(stroker);
188 FT_Done_Face(face);
189 return false;
190 }
191
192 // Store the loaded font in our ugly void* :)
193 m_stroker = stroker;
194 m_face = face;
195
196 // Store the font information
197 m_info.family = face->family_name ? face->family_name : std::string();
198
199 return true;
200
201#else
202
203 if (m_stream) delete static_cast<priv::ResourceStream*>(m_stream);
204
205 m_stream = new priv::ResourceStream(filename);
206 return loadFromStream(*static_cast<priv::ResourceStream*>(m_stream));
207
208#endif
209}

272bool VulkanFont::loadFromStream(InputStream& stream) {
273 needsUpload = true;
274
275 // Cleanup the previous resources
276 cleanup();
277 m_refCount = new int(1);
278
279 // Initialize FreeType
280 // Note: we initialize FreeType for every font instance in order to avoid having a single
281 // global manager that would create a lot of issues regarding creation and destruction order.
282 FT_Library library;
283 if (FT_Init_FreeType(&library) != 0) {
284 err() << "Failed to load font from stream (failed to initialize FreeType)" << std::endl;
285 return false;
286 }
287 m_library = library;
288
289 // Make sure that the stream's reading position is at the beginning
290 stream.seek(0);
291
292 // Prepare a wrapper for our stream, that we'll pass to FreeType callbacks
293 FT_StreamRec* rec = new FT_StreamRec;
294 std::memset(rec, 0, sizeof(*rec));
295 rec->base = NULL;
296 rec->size = static_cast<unsigned long>(stream.getSize().value_or(0));
297 rec->pos = 0;
298 rec->descriptor.pointer = &stream;
299 rec->read = &read;
300 rec->close = &close;
301
302 // Setup the FreeType callbacks that will read our stream
303 FT_Open_Args args;
304 args.flags = FT_OPEN_STREAM;
305 args.stream = rec;
306 args.driver = 0;
307
308 // Load the new font face from the specified stream
309 FT_Face face;
310 if (FT_Open_Face(static_cast<FT_Library>(m_library), &args, 0, &face) != 0) {
311 err() << "Failed to load font from stream (failed to create the font face)" << std::endl;
312 delete rec;
313 return false;
314 }
315
316 // Load the stroker that will be used to outline the font
317 FT_Stroker stroker;
318 if (FT_Stroker_New(static_cast<FT_Library>(m_library), &stroker) != 0) {
319 err() << "Failed to load font from stream (failed to create the stroker)" << std::endl;
320 FT_Done_Face(face);
321 delete rec;
322 return false;
323 }
324
325 // Select the Unicode character map
326 if (FT_Select_Charmap(face, FT_ENCODING_UNICODE) != 0) {
327 err() << "Failed to load font from stream (failed to set the Unicode character set)"
328 << std::endl;
329 FT_Done_Face(face);
330 FT_Stroker_Done(stroker);
331 delete rec;
332 return false;
333 }
334
335 // Store the loaded font in our ugly void* :)
336 m_stroker = stroker;
337 m_face = face;
338 m_streamRec = rec;
339
340 // Store the font information
341 m_info.family = face->family_name ? face->family_name : std::string();
342
343 return true;
344}

350const Glyph& VulkanFont::getGlyph(std::uint32_t codePoint, unsigned int characterSize, bool bold,
351 float outlineThickness) const {
352 // Build the key by combining the glyph index (based on code point), bold flag, and outline
353 // thickness
354 std::uint64_t key = combine(characterSize,
355 outlineThickness,
356 bold,
357 FT_Get_Char_Index(static_cast<FT_Face>(m_face), codePoint));
358
359 // Search the glyph into the cache
360 GlyphTable::const_iterator it = glyphs.find(key);
361 if (it != glyphs.end()) {
362 // Found: just return it
363 return it->second;
364 }
365 else {
366 // Not found: we have to load it
367 const_cast<sf::VulkanFont*>(this)->needsUpload = true;
368 Glyph glyph = loadGlyph(codePoint, characterSize, bold, outlineThickness);
369 return glyphs.insert(std::make_pair(key, glyph)).first->second;
370 }
371}

379float VulkanFont::getKerning(std::uint32_t first, std::uint32_t second, unsigned int characterSize,
380 bool bold) const {
381 // Special case where first or second is 0 (null character)
382 if (first == 0 || second == 0) return 0.f;
383
384 FT_Face face = static_cast<FT_Face>(m_face);
385
386 if (face && setCurrentSize(characterSize)) {
387 // Convert the characters to indices
388 FT_UInt index1 = FT_Get_Char_Index(face, first);
389 FT_UInt index2 = FT_Get_Char_Index(face, second);
390
391 // Retrieve position compensation deltas generated by FT_LOAD_FORCE_AUTOHINT flag
392 float firstRsbDelta = static_cast<float>(getGlyph(first, characterSize, bold).rsbDelta);
393 float secondLsbDelta = static_cast<float>(getGlyph(second, characterSize, bold).lsbDelta);
394
395 // Get the kerning vector if present
396 FT_Vector kerning;
397 kerning.x = kerning.y = 0;
398 if (FT_HAS_KERNING(face))
399 FT_Get_Kerning(face, index1, index2, FT_KERNING_UNFITTED, &kerning);
400
401 // X advance is already in pixels for bitmap fonts
402 if (!FT_IS_SCALABLE(face)) return static_cast<float>(kerning.x);
403
404 // Combine kerning with compensation deltas and return the X advance
405 // Flooring is required as we use FT_KERNING_UNFITTED flag which is not quantized in 64
406 // based grid
407 return std::floor((secondLsbDelta - firstRsbDelta + static_cast<float>(kerning.x) + 32) /
408 static_cast<float>(1 << 6));
409 }
410 else {
411 // Invalid font
412 return 0.f;
413 }

484void VulkanFont::cleanup() {
485 // Check if we must destroy the FreeType pointers
486 if (m_refCount) {
487 // Decrease the reference counter
488 (*m_refCount)--;
489
490 // Free the resources only if we are the last owner
491 if (*m_refCount == 0) {
492 // Delete the reference counter
493 delete m_refCount;
494
495 // Destroy the stroker
496 if (m_stroker) FT_Stroker_Done(static_cast<FT_Stroker>(m_stroker));
497
498 // Destroy the font face
499 if (m_face) FT_Done_Face(static_cast<FT_Face>(m_face));
500
501 // Destroy the stream rec instance, if any (must be done after FT_Done_Face!)
502 if (m_streamRec) delete static_cast<FT_StreamRec*>(m_streamRec);
503
504 // Close the library
505 if (m_library) FT_Done_FreeType(static_cast<FT_Library>(m_library));
506 }
507 }
508
509 // Reset members
510 m_library = NULL;
511 m_face = NULL;
512 m_stroker = NULL;
513 m_streamRec = NULL;
514 m_refCount = NULL;
515 glyphs.clear();
516 rows.clear();
517 nextRow = 3;
518 needsUpload = true;
519 vulkanTexture.release();
520 std::vector<std::uint8_t>().swap(m_pixelBuffer);
521 buffer.clear();
522}

525Glyph VulkanFont::loadGlyph(std::uint32_t codePoint, unsigned int characterSize, bool bold,
526 float outlineThickness) const {
527 // The glyph to return
528 Glyph glyph;
529
530 // First, transform our ugly void* to a FT_Face
531 FT_Face face = static_cast<FT_Face>(m_face);
532 if (!face) return glyph;
533
534 // Set the character size
535 if (!setCurrentSize(characterSize)) return glyph;
536
537 // Load the glyph corresponding to the code point
538 FT_Int32 flags = FT_LOAD_TARGET_NORMAL | FT_LOAD_FORCE_AUTOHINT;
539 if (outlineThickness != 0) flags |= FT_LOAD_NO_BITMAP;
540 if (FT_Load_Char(face, codePoint, flags) != 0) return glyph;
541
542 // Retrieve the glyph
543 FT_Glyph glyphDesc;
544 if (FT_Get_Glyph(face->glyph, &glyphDesc) != 0) return glyph;
545
546 // Apply bold and outline (there is no fallback for outline) if necessary -- first technique
547 // using outline (highest quality)
548 FT_Pos weight = 1 << 6;
549 bool outline = (glyphDesc->format == FT_GLYPH_FORMAT_OUTLINE);
550 if (outline) {
551 if (bold) {
552 FT_OutlineGlyph outlineGlyph = reinterpret_cast<FT_OutlineGlyph>(glyphDesc);
553 FT_Outline_Embolden(&outlineGlyph->outline, weight);
554 }
555
556 if (outlineThickness != 0) {
557 FT_Stroker stroker = static_cast<FT_Stroker>(m_stroker);
558
559 FT_Stroker_Set(stroker,
560 static_cast<FT_Fixed>(outlineThickness * static_cast<float>(1 << 6)),
561 FT_STROKER_LINECAP_ROUND,
562 FT_STROKER_LINEJOIN_ROUND,
563 0);
564 FT_Glyph_Stroke(&glyphDesc, stroker, true);
565 }
566 }
567
568 // Convert the glyph to a bitmap (i.e. rasterize it)
569 // Warning! After this line, do not read any data from glyphDesc directly, use
570 // bitmapGlyph.root to access the FT_Glyph data.
571 FT_Glyph_To_Bitmap(&glyphDesc, FT_RENDER_MODE_NORMAL, 0, 1);
572 FT_BitmapGlyph bitmapGlyph = reinterpret_cast<FT_BitmapGlyph>(glyphDesc);
573 FT_Bitmap& bitmap = bitmapGlyph->bitmap;
574
575 // Apply bold if necessary -- fallback technique using bitmap (lower quality)
576 if (!outline) {
577 if (bold) FT_Bitmap_Embolden(static_cast<FT_Library>(m_library), &bitmap, weight, weight);
578
579 if (outlineThickness != 0)
580 err() << "Failed to outline glyph (no fallback available)" << std::endl;
581 }
582
583 // Compute the glyph's advance offset
584 glyph.advance = static_cast<float>(bitmapGlyph->root.advance.x >> 16);
585 if (bold) glyph.advance += static_cast<float>(weight) / static_cast<float>(1 << 6);
586
587 glyph.lsbDelta = static_cast<int>(face->glyph->lsb_delta);
588 glyph.rsbDelta = static_cast<int>(face->glyph->rsb_delta);
589
590 unsigned int width = bitmap.width;
591 unsigned int height = bitmap.rows;
592
593 if ((width > 0) && (height > 0)) {
594 // Leave a small padding around characters, so that filtering doesn't
595 // pollute them with pixels from neighbors
596 const unsigned int padding = 2;
597
598 width += 2 * padding;
599 height += 2 * padding;
600
601 // Find a good position for the new glyph into the texture
602 glyph.textureRect = findGlyphRect(width, height);
603
604 // Make sure the texture data is positioned in the center
605 // of the allocated texture rectangle
606 glyph.textureRect.position.x += static_cast<int>(padding);
607 glyph.textureRect.position.y += static_cast<int>(padding);
608 glyph.textureRect.size.x -= static_cast<int>(2 * padding);
609 glyph.textureRect.size.y -= static_cast<int>(2 * padding);
610
611 // Compute the glyph's bounding box
612 glyph.bounds.position.x = static_cast<float>(bitmapGlyph->left);
613 glyph.bounds.position.y = static_cast<float>(-bitmapGlyph->top);
614 glyph.bounds.size.x = static_cast<float>(bitmap.width);
615 glyph.bounds.size.y = static_cast<float>(bitmap.rows);
616
617 // Resize the pixel buffer to the new size and fill it with transparent white pixels
618 m_pixelBuffer.resize(width * height * 4);
619
620 std::uint8_t* current = &m_pixelBuffer[0];
621 std::uint8_t* end = current + width * height * 4;
622
623 while (current != end) {
624 (*current++) = 255;
625 (*current++) = 255;
626 (*current++) = 255;
627 (*current++) = 0;
628 }
629
630 // Extract the glyph's pixels from the bitmap
631 const std::uint8_t* pixels = bitmap.buffer;
632 if (bitmap.pixel_mode == FT_PIXEL_MODE_MONO) {
633 // Pixels are 1 bit monochrome values
634 for (unsigned int y = padding; y < height - padding; ++y) {
635 for (unsigned int x = padding; x < width - padding; ++x) {
636 // The color channels remain white, just fill the alpha channel
637 std::size_t index = x + y * width;
638 m_pixelBuffer[index * 4 + 3] =
639 ((pixels[(x - padding) / 8]) & (1 << (7 - ((x - padding) % 8)))) ? 255 : 0;
640 }
641 pixels += bitmap.pitch;
642 }
643 }
644 else {
645 // Pixels are 8 bits gray levels
646 for (unsigned int y = padding; y < height - padding; ++y) {
647 for (unsigned int x = padding; x < width - padding; ++x) {
648 // The color channels remain white, just fill the alpha channel
649 std::size_t index = x + y * width;
650 m_pixelBuffer[index * 4 + 3] = pixels[x - padding];
651 }
652 pixels += bitmap.pitch;
653 }
654 }
655
656 // Write the pixels to the texture
657 unsigned int x = static_cast<unsigned int>(glyph.textureRect.position.x) - padding;
658 unsigned int y = static_cast<unsigned int>(glyph.textureRect.position.y) - padding;
659 unsigned int w = static_cast<unsigned int>(glyph.textureRect.size.x) + 2 * padding;
660 unsigned int h = static_cast<unsigned int>(glyph.textureRect.size.y) + 2 * padding;
661 for (unsigned int xi = 0; xi < w; ++xi) {
662 for (unsigned int yi = 0; yi < h; ++yi) {
663 std::size_t index = xi + yi * width;
664 texture.setPixel({x + xi, y + yi},
665 sf::Color{255, 255, 255, m_pixelBuffer[index * 4 + 3]});
666 }
667 }
668 }
669
670 // Delete the FT glyph
671 FT_Done_Glyph(glyphDesc);
672
673 // Done :)
674 return glyph;
675}

678IntRect VulkanFont::findGlyphRect(unsigned int width, unsigned int height) const {
679 // Find the line that fits well the glyph
680 Row* row = NULL;
681 float bestRatio = 0;
682 for (std::vector<Row>::iterator it = rows.begin(); it != rows.end() && !row; ++it) {
683 float ratio = static_cast<float>(height) / static_cast<float>(it->height);
684
685 // Ignore rows that are either too small or too high
686 if ((ratio < 0.7f) || (ratio > 1.f)) continue;
687
688 // Check if there's enough horizontal space left in the row
689 if (width > texture.getSize().x - it->width) continue;
690
691 // Make sure that this new row is the best found so far
692 if (ratio < bestRatio) continue;
693
694 // The current row passed all the tests: we can select it
695 row = &*it;
696 bestRatio = ratio;
697 }
698
699 // If we didn't find a matching row, create a new one (10% taller than the glyph)
700 if (!row) {
701 unsigned int rowHeight = height + height / 10;
702 while ((nextRow + rowHeight >= texture.getSize().y) || (width >= texture.getSize().x)) {
703 // Not enough space: resize the texture if possible
704 unsigned int textureWidth = texture.getSize().x;
705 unsigned int textureHeight = texture.getSize().y;
706
707 // Make the texture 2 times bigger
708 Image newTexture;
709 newTexture.resize({textureWidth * 2, textureHeight * 2}, sf::Color(255, 255, 255, 0));
710 newTexture.copy(texture, {0, 0});
711 texture = std::move(newTexture);
712 }
713
714 // We can now create the new row
715 rows.emplace_back(nextRow, rowHeight);
716 nextRow += rowHeight;
717 row = &rows.back();
718 }
719
720 // Find the glyph's rectangle on the selected row
721 IntRect rect(Rect<unsigned int>({row->width, row->top}, {width, height}));
722
723 // Update the row informations
724 row->width += width;
725
726 return rect;
727}

730bool VulkanFont::setCurrentSize(unsigned int characterSize) const {
731 // FT_Set_Pixel_Sizes is an expensive function, so we must call it
732 // only when necessary to avoid killing performances
733
734 FT_Face face = static_cast<FT_Face>(m_face);
735 FT_UShort currentSize = face->size->metrics.x_ppem;
736
737 if (currentSize != characterSize) {
738 FT_Error result = FT_Set_Pixel_Sizes(face, 0, characterSize);
739
740 if (result == FT_Err_Invalid_Pixel_Size) {
741 // In the case of bitmap fonts, resizing can
742 // fail if the requested size is not available
743 if (!FT_IS_SCALABLE(face)) {
744 err() << "Failed to set bitmap font size to " << characterSize << std::endl;
745 err() << "Available sizes are: ";
746 for (int i = 0; i < face->num_fixed_sizes; ++i) {
747 const long size = (face->available_sizes[i].y_ppem + 32) >> 6;
748 err() << size << " ";
749 }
750 err() << std::endl;
751 }
752 else { err() << "Failed to set font size to " << characterSize << std::endl; }
753 }
754
755 return result == FT_Err_Ok;
756 }
757
758 return true;
759}
