129 void TextEntryComponent::positionItems() {
130  constexpr float Padding = 2.f;
131 
132  TextEntry& owner = getOwnerAs<TextEntry>();
133  const sf::FloatRect& acq = owner.getAcquisition();
134  const sf::FloatRect bounds = text.getLocalBounds();
135  const glm::vec2 cpos = text.findCharacterPosition(0, owner.getCursorPosition());
136 
137  glm::vec2 textOffset(0.f, 0.f);
138  if ((bounds.width + bounds.left) > acq.width || (bounds.top + bounds.height) > acq.height) {
139  const glm::vec2 gcpos = cpos + text.getTransform().getLocalPosition();
140  const sf::Glyph& g = text.getFont().getGlyph(
141  text.getSection().getString()[owner.getCursorPosition()],
142  text.getSection().getCharacterSize(),
143  (text.getSection().getStyle() & sf::Text::Style::Bold) == sf::Text::Style::Bold,
144  text.getSection().getOutlineThickness());
145 
146  const float right = acq.width - Padding * 2.f;
147  if (gcpos.x < 0.f) { textOffset.x = -cpos.x; }
148  else if (cpos.x > right) {
149  textOffset.x = right - cpos.x - g.bounds.width - g.advance - CaratWidth;
150  }
151 
152  const float lineHeight =
153  text.getFont().getLineSpacing(text.getSection().getCharacterSize());
154  const float y = static_cast<float>(owner.getCurrentLine()) * lineHeight;
155  const float by = y + lineHeight;
156  const float bottom = acq.height - Padding * 2.f;
157  if (y < 0.f) { textOffset.y = -y; }
158  else if (by > bottom) { textOffset.y = bottom - by; }
159  }
160 
161  text.getTransform().setPosition(glm::vec2{Padding, Padding} + textOffset);
162  carat.getTransform().setPosition(cpos);
163 }
