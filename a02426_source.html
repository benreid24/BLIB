31void GlobalDescriptors::init() {
32 auto& vulkanState = renderer.vulkanState();
33
34 VkDescriptorSetLayoutBinding setBindings[6] = {texturePool.getTextureLayoutBinding(),
35 materialPool.getLayoutBinding(),
36 texturePool.getCubemapLayoutBinding(),
37 {},
38 {},
39 {}};
40
41 VkDescriptorSetLayoutBinding& frameDataBinding = setBindings[3];
42 frameDataBinding.descriptorCount = 1;
43 frameDataBinding.binding = 3;
44 frameDataBinding.stageFlags = VK_SHADER_STAGE_ALL;
45 frameDataBinding.descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
46 frameDataBinding.pImmutableSamplers = 0;
47
48 VkDescriptorSetLayoutBinding& settingsBinding = setBindings[4];
49 settingsBinding.descriptorCount = 1;
50 settingsBinding.binding = 4;
51 settingsBinding.stageFlags = VK_SHADER_STAGE_ALL;
52 settingsBinding.descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
53 settingsBinding.pImmutableSamplers = 0;
54
55 VkDescriptorSetLayoutBinding& dynamicSettingsBinding = setBindings[5];
56 dynamicSettingsBinding.descriptorCount = 1;
57 dynamicSettingsBinding.binding = 5;
58 dynamicSettingsBinding.stageFlags = VK_SHADER_STAGE_ALL;
59 dynamicSettingsBinding.descriptorType = VK_DESCRIPTOR_TYPE_STORAGE_BUFFER;
60 dynamicSettingsBinding.pImmutableSamplers = 0;
61
62 // create descriptor layout
63 constexpr std::size_t NBindings = std::size(setBindings);
64 VkDescriptorSetLayoutCreateInfo descriptorCreateInfo{};
65 descriptorCreateInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_LAYOUT_CREATE_INFO;
66 descriptorCreateInfo.bindingCount = NBindings;
67 descriptorCreateInfo.pBindings = setBindings;
68 if (VK_SUCCESS !=
69 vkCreateDescriptorSetLayout(
70 vulkanState.getDevice(), &descriptorCreateInfo, nullptr, &descriptorSetLayout)) {
71 throw std::runtime_error("Failed to create texture pool descriptor set layout");
72 }
73
74 // create descriptor pool
75 ctr::StaticVector<VkDescriptorPoolSize, 10> poolSizes;
76 for (std::size_t i = 0; i < NBindings; ++i) {
77 VkDescriptorPoolSize* poolSize = nullptr;
78 for (VkDescriptorPoolSize& ps : poolSizes) {
79 if (ps.type == setBindings[i].descriptorType) {
80 poolSize = &ps;
81 break;
82 }
83 }
84 if (!poolSize) { poolSize = &poolSizes.emplace_back(VkDescriptorPoolSize{}); }
85 poolSize->type = setBindings[i].descriptorType;
86 poolSize->descriptorCount +=
87 setBindings[i].descriptorCount * cfg::Limits::MaxConcurrentFrames * 2;
88 }
89
90 VkDescriptorPoolCreateInfo poolCreate{};
91 poolCreate.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_POOL_CREATE_INFO;
92 poolCreate.maxSets = 2 * cfg::Limits::MaxConcurrentFrames;
93 poolCreate.poolSizeCount = poolSizes.size();
94 poolCreate.pPoolSizes = poolSizes.data();
95 if (VK_SUCCESS !=
96 vkCreateDescriptorPool(vulkanState.getDevice(), &poolCreate, nullptr, &descriptorPool)) {
97 throw std::runtime_error("Failed to create texture descriptor pool");
98 }
99
100 // allocate descriptor set
101 VkDescriptorSet allocatedSets[2 * cfg::Limits::MaxConcurrentFrames];
102 std::array<VkDescriptorSetLayout, 2 * cfg::Limits::MaxConcurrentFrames> setLayouts;
103 setLayouts.fill(descriptorSetLayout);
104
105 VkDescriptorSetAllocateInfo setAlloc{};
106 setAlloc.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
107 setAlloc.descriptorPool = descriptorPool;
108 setAlloc.descriptorSetCount = setLayouts.size();
109 setAlloc.pSetLayouts = setLayouts.data();
110 if (VK_SUCCESS != vkAllocateDescriptorSets(vulkanState.getDevice(), &setAlloc, allocatedSets)) {
111 throw std::runtime_error("Failed to allocate texture descriptor set");
112 }
113
114 // write descriptor set with texture and material pools
115 unsigned int i = 0;
116 descriptorSets.init(vulkanState, [&i, &allocatedSets](auto& set) { set = allocatedSets[i++]; });
117 rtDescriptorSets.init(vulkanState,
118 [&i, &allocatedSets](auto& set) { set = allocatedSets[i++]; });
119
120 texturePool.init(descriptorSets, rtDescriptorSets);
121 materialPool.init(descriptorSets, rtDescriptorSets);
122
123 // write descriptor set with settings uniform
124 auto& pool = renderer.getGlobalShaderResources();
125 settingsResource = pool.getShaderResourceWithKey(sri::SettingsUniformResourceKey);
126 frameTimeResource = pool.getShaderResourceWithKey(sri::FrameDataUniformResourceKey);
127 dynamicSettingsResource = pool.getShaderResourceWithKey(sri::DynamicSettingsUniformResourceKey);
128 ds::SetWriteHelper writer;
129 const auto writeGlobalsSet = [this, &writer](VkDescriptorSet set, std::uint32_t i, bool forRt) {
130 // frame data
131 auto& frameDataBufferInfo = writer.getNewBufferInfo();
132 frameDataBufferInfo.buffer = frameTimeResource->getBuffer().getRawBuffer(i);
133 frameDataBufferInfo.offset = 0;
134 frameDataBufferInfo.range = frameTimeResource->getBuffer().getAlignedElementSize();
135
136 auto& frameDataWrite = writer.getNewSetWrite(set);
137 frameDataWrite.descriptorCount = 1;
138 frameDataWrite.dstBinding = 3;
139 frameDataWrite.dstArrayElement = 0;
140 frameDataWrite.pBufferInfo = &frameDataBufferInfo;
141 frameDataWrite.descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
142
143 // settings
144 auto& settingsBufferInfo = writer.getNewBufferInfo();
145 settingsBufferInfo.buffer = settingsResource->getBuffer().getCurrentFrameRawBuffer();
146 settingsBufferInfo.offset =
147 forRt ? settingsResource->getBuffer().getAlignedElementSize() : 0;
148 settingsBufferInfo.range = settingsResource->getBuffer().getAlignedElementSize();
149
150 auto& settingsWrite = writer.getNewSetWrite(set);
151 settingsWrite.descriptorCount = 1;
152 settingsWrite.dstBinding = 4;
153 settingsWrite.dstArrayElement = 0;
154 settingsWrite.pBufferInfo = &settingsBufferInfo;
155 settingsWrite.descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
156
157 // dynamic settings
158 auto& dynamicSettingsBufferInfo = writer.getNewBufferInfo();
159 dynamicSettingsBufferInfo.buffer =
160 dynamicSettingsResource->getBuffer().getCurrentFrameBuffer().getBuffer();
161 dynamicSettingsBufferInfo.offset = 0;
162 dynamicSettingsBufferInfo.range =
163 dynamicSettingsResource->getBuffer().getTotalAlignedSize();
164
165 auto& dynamicSettingsWrite = writer.getNewSetWrite(set);
166 dynamicSettingsWrite.descriptorCount = 1;
167 dynamicSettingsWrite.dstBinding = 5;
168 dynamicSettingsWrite.dstArrayElement = 0;
169 dynamicSettingsWrite.pBufferInfo = &dynamicSettingsBufferInfo;
170 dynamicSettingsWrite.descriptorType = VK_DESCRIPTOR_TYPE_STORAGE_BUFFER;
171 };
172
173 settingsResource->getBuffer().setCopyFullRange(true);
174 dynamicSettingsResource->getBuffer().setCopyFullRange(true);
175
176 frameTimeResource->getBuffer().transferEveryFrame();
177 frameTimeResource->getBuffer().setCopyFullRange(true);
178
179 i = 0;
180 descriptorSets.visit(
181 [&writeGlobalsSet, &i](VkDescriptorSet set) { writeGlobalsSet(set, i++, false); });
182 i = 0;
183 rtDescriptorSets.visit(
184 [&writeGlobalsSet, &i](VkDescriptorSet set) { writeGlobalsSet(set, i++, true); });
185 writer.performWrite(vulkanState.getDevice());
186 updateSettings(renderer.getSettings());
187}

197void GlobalDescriptors::onFrameStart() {
198 texturePool.onFrameStart(
199 descriptorWriter, descriptorSets.current(), rtDescriptorSets.current());
200 materialPool.onFrameStart();
201 descriptorWriter.performWrite(renderer.vulkanState().getDevice());
202 accumulatedTimings.realFrameDt = frameTimer.restart().asSeconds();
203 accumulatedTimings.frameDt = accumulatedTimings.realFrameDt * engine.getTimeScale();
204 frameTimer.restart();
205 frameTimeResource->getBuffer().writeDirect(&accumulatedTimings, 1, 0);
206 accumulatedTimings = {};

217void GlobalDescriptors::updateSettings(const Settings& settings) {
218 // reset current hdr exposure level
219 dynamicSettingsResource->getBuffer()[0].currentHdrExposure = settings.getExposureFactor();
220 dynamicSettingsResource->getBuffer().queueTransfer();
221
222 // populate global settings
223 settingsResource->getBuffer()[0].gamma = settings.getGamma();
224 settingsResource->getBuffer()[0].hdrSettings = settings.getAutoHDRSettings();
225
226 // handle render texture specific settings
227 settingsResource->getBuffer()[1] = settingsResource->getBuffer()[0];
228 settingsResource->getBuffer()[1].gamma = 1.f; // cancel out gamma correction
229 settingsResource->getBuffer().queueTransfer();
230}
