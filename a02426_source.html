13void generateErrorPattern(sf::Image& img, unsigned int left, unsigned int top, unsigned int width,
14 unsigned int height, unsigned int numBoxes) {
15 const unsigned int ErrorBoxWidth = width / numBoxes;
16 const unsigned int ErrorBoxHeight = height / numBoxes;
17 for (unsigned int x = left; x < left + width; ++x) {
18 for (unsigned int y = top; y < top + height; ++y) {
19 const unsigned int xi = x / ErrorBoxWidth;
20 const unsigned int yi = y / ErrorBoxHeight;
21 if ((xi % 2) == (yi % 2)) { img.setPixel({x, y}, sf::Color(230, 66, 245)); }
22 else { img.setPixel({x, y}, sf::Color(255, 254, 196)); }
23 }
24 }
25}

68void TexturePool::init(vk::PerFrame<VkDescriptorSet>& descriptorSets,
69 vk::PerFrame<VkDescriptorSet>& rtDescriptorSets) {
70 // create error texture
71 constexpr unsigned int ErrorSize = 128;
72 constexpr unsigned int ErrorBoxCount = 16;
73 errorPattern.resize({ErrorSize, ErrorSize});
74 errorPatternCube.resize({ErrorSize, ErrorSize * 6});
75 generateErrorPattern(errorPattern, 0, 0, ErrorSize, ErrorSize, ErrorBoxCount);
76 for (std::size_t i = 0; i < 6; ++i) {
77 generateErrorPattern(
78 errorPatternCube, 0, i * ErrorSize, ErrorSize, ErrorSize, ErrorBoxCount);
79 }
80 errorTexture.altImg = &errorPattern;
81 errorCubemap.altImg = &errorPatternCube;
82 errorTexture.createFromContentsAndQueue(
83 vk::Texture::Type::Texture2D,
84 {.format = vk::CommonTextureFormats::SRGBA32Bit,
85 .sampler = vk::SamplerOptions::Type::FilteredRepeated});
86 errorCubemap.createFromContentsAndQueue(
87 vk::Texture::Type::Cubemap,
88 {.format = vk::CommonTextureFormats::SRGBA32Bit,
89 .sampler = vk::SamplerOptions::Type::FilteredRepeated});
90 renderer.getTransferEngine().executeTransfers();
91
92 // init all textures to error pattern
93 for (vk::Texture& txtr : textures) { txtr.currentView = errorTexture.currentView; }
94 for (vk::Texture& txtr : cubemaps) { txtr.currentView = errorCubemap.getView(); }
95
96 // fill descriptor set
97 VkDescriptorImageInfo errorInfo{};
98 errorInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
99 errorInfo.imageView = errorTexture.getView();
100 errorInfo.sampler = renderer.samplerCache().getSampler({.type = errorTexture.getSampler()});
101 std::vector<VkDescriptorImageInfo> imageInfos(cfg::Limits::MaxTextureCount, errorInfo);
102
103 VkDescriptorImageInfo errorCubeInfo{};
104 errorCubeInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
105 errorCubeInfo.imageView = errorCubemap.getView();
106 errorCubeInfo.sampler = errorInfo.sampler;
107 std::vector<VkDescriptorImageInfo> imageCubeInfos(cfg::Limits::MaxCubemapCount, errorCubeInfo);
108
109 std::array<VkWriteDescriptorSet, 4 * cfg::Limits::MaxConcurrentFrames> setWrites{};
110 unsigned int i = 0;
111 const auto visitor = [this, &i, &setWrites](auto& set,
112 std::uint32_t bindIndex,
113 const std::vector<VkDescriptorImageInfo>& infos) {
114 setWrites[i].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
115 setWrites[i].descriptorCount = infos.size();
116 setWrites[i].dstBinding = bindIndex;
117 setWrites[i].dstArrayElement = 0;
118 setWrites[i].dstSet = set;
119 setWrites[i].pImageInfo = infos.data();
120 setWrites[i].descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
121 ++i;
122 };
123 descriptorSets.visit(
124 [&visitor, &imageInfos](auto& set) { visitor(set, TextureArrayBindIndex, imageInfos); });
125 rtDescriptorSets.visit(
126 [&visitor, &imageInfos](auto& set) { visitor(set, TextureArrayBindIndex, imageInfos); });
127 descriptorSets.visit([&visitor, &imageCubeInfos](auto& set) {
128 visitor(set, CubemapArrayBindIndex, imageCubeInfos);
129 });
130 rtDescriptorSets.visit([&visitor, &imageCubeInfos](auto& set) {
131 visitor(set, CubemapArrayBindIndex, imageCubeInfos);
132 });
133 vkUpdateDescriptorSets(vulkanState.getDevice(), setWrites.size(), setWrites.data(), 0, nullptr);
134}

180void TexturePool::doRelease(vk::Texture* texture) {
181 if (texture->getType() != vk::Texture::Type::Cubemap) {
182 std::uint32_t i = texture - textures.data();
183 refCounts[i] = 0;
184
185 if (i < reverseAssetMap.size()) {
186 freeSlots.release(i);
187 if (reverseAssetMap[i] != util::UUID()) {
188 assetMap.erase(reverseAssetMap[i]);
189 reverseAssetMap[i] = util::UUID();
190 }
191 if (reverseImageMap[i]) {
192 imageMap.erase(reverseImageMap[i]);
193 reverseImageMap[i] = nullptr;
194 }
195 }
196 else { freeRtSlots.release(i - reverseAssetMap.size()); }
197 }
198 else {
199 std::uint32_t i = texture - cubemaps.data();
200
201 cubemapRefCounts[i] = 0;
202 cubemapFreeSlots.release(i);
203 cubemapAssetMap.erase(reverseCubemapAssetMap[i]);
204 reverseCubemapAssetMap[i] = util::UUID();
205 }
206 resetTexture(texture);
207}

286TextureRef TexturePool::getOrLoadTexture(as::TypedRef<asi::ImagePayload> imageAsset,
287 const vk::TextureOptions& options) {
288 if (!imageAsset) {
289 BL_LOG_ERROR << "Attempted to get or load texture with null image asset";
290 return TextureRef{*this, errorTexture, ErrorTextureId};
291 }
292
293 std::unique_lock lock(mutex);
294
295 auto it = assetMap.find(imageAsset.getUUID());
296 if (it != assetMap.end()) {
297 auto& t = textures[it->second];
298 cancelRelease(&t);
299 return TextureRef{*this, t, it->second};
300 }
301
302 if (imageAsset.getState() != as::State::Loaded) {
303 BL_LOG_INFO << "Requested texture asset " << imageAsset.getUUID()
304 << " is not loaded, attempting to load";
305 if (!imageAsset.getAsset().load()) {
306 BL_LOG_ERROR << "Failed to load texture asset " << imageAsset.getUUID();
307
308 return TextureRef{*this, errorTexture, ErrorTextureId};
309 }
310 }
311
312 TextureRef txtr = allocateTexture();
313 prepareTextureUpdate(txtr.get(), imageAsset);
314 it = assetMap.try_emplace(imageAsset.getUUID(), txtr.id()).first;
315 reverseAssetMap[txtr.id()] = it->first;
316 txtr->createFromContentsAndQueue(vk::Texture::Type::Texture2D, options);
317 updateTexture(txtr.get());
318
319 return txtr;
320}

363TextureRef TexturePool::getOrCreateCubemap(as::TypedRef<asi::CubemapPayload> cubemap,
364 vk::SamplerOptions::Type sampler) {
365 // check if we already have it
366 const auto it = cubemapAssetMap.find(cubemap.getUUID());
367 if (it != cubemapAssetMap.end()) {
368 auto& t = cubemaps[it->second];
369 cancelRelease(&t);
370 return TextureRef{*this, t, it->second};
371 }
372
373 // validate faces
374 std::array<as::TypedRef<asi::TexturePayload>, 6> faceRefs = {cubemap->right.getRef(),
375 cubemap->left.getRef(),
376 cubemap->top.getRef(),
377 cubemap->bottom.getRef(),
378 cubemap->front.getRef(),
379 cubemap->back.getRef()};
380 std::array<as::TypedRef<asi::TexturePayload>*, 6> faces = {
381 &faceRefs[0], &faceRefs[1], &faceRefs[2], &faceRefs[3], &faceRefs[4], &faceRefs[5]};
382 asi::TexturePayload::ColorSpace colorSpace = asi::TexturePayload::ColorSpace::sRGB;
383 glm::u32vec2 faceSize(0, 0);
384 for (as::TypedRef<asi::TexturePayload>*& face : faces) {
385 if (!*face) {
386 BL_LOG_ERROR << "Cubemap face ref is not valid";
387 face = nullptr;
388 continue;
389 }
390
391 if (face->getState() != as::State::Loaded) {
392 BL_LOG_INFO << "Cubemap face asset " << face->getAsset().getUUID()
393 << " is not loaded, attempting to load";
394 if (!face->getAsset().load()) {
395 BL_LOG_ERROR << "Failed to load cubemap face asset " << face->getUUID();
396 face = nullptr;
397 continue;
398 }
399 }
400
401 if (!(*face)->image.isValid()) {
402 BL_LOG_ERROR << "Cubemap face " << face->getAsset().getUUID() << " has invalid image";
403 face = nullptr;
404 continue;
405 }
406
407 const sf::Image& img = (*face)->image.get().get();
408 if (img.getSize().x > 0 && img.getSize().y > 0) {
409 if (faceSize.x == 0) {
410 faceSize.x = img.getSize().x;
411 faceSize.y = img.getSize().y;
412 colorSpace = (*face)->colorSpace;
413 }
414 else {
415 if (faceSize.x != img.getSize().x || faceSize.y != img.getSize().y) {
416 BL_LOG_ERROR << "Cubemap faces must all be the same size";
417 face = nullptr;
418 }
419 if (colorSpace != (*face)->colorSpace) {
420 BL_LOG_WARN
421 << "Cubemap faces must all have the same color space. Offending cubemap: "
422 << cubemap.getUUID();
423 }
424 }
425 }
426 else {
427 BL_LOG_ERROR << "Cubemap face " << face->getAsset().getUUID() << " has invalid size";
428 face = nullptr;
429 }
430 }
431 if (faceSize.x == 0) {
432 BL_LOG_ERROR << "No valid cubemap faces provided";
433 faceSize = {64, 64};
434 }
435
436 std::unique_lock lock(mutex);
437
438 TextureRef cm = allocateCubemap();
439
440 // stitch images into new image
441 cm->altImg = &cm->localImage;
442 cm->localImage.resize({faceSize.x, faceSize.y * 6}, sf::Color::Transparent);
443 for (std::size_t i = 0; i < 6; ++i) {
444 if (faces[i]) {
445 cm->localImage.copy((*faces[i])->image.get().get(), sf::Vector2u(0, i * faceSize.y));
446 }
447 else {
448 generateErrorPattern(cm->localImage, 0, i * faceSize.y, faceSize.x, faceSize.y, 16);
449 BL_LOG_WARN << "Cubemap face " << i << " is invalid, using error pattern";
450 }
451 }
452
453 cm->createFromContentsAndQueue(
454 vk::Texture::Type::Cubemap,
455 {.format = mapColorSpaceToFormat(colorSpace), .sampler = sampler});
456 updateTexture(cm.get());
457
458 return cm;
459}

461void TexturePool::onFrameStart(ds::SetWriteHelper& setWriter, VkDescriptorSet currentSet,
462 VkDescriptorSet currentRtSet) {
463 if (!queuedUpdates.current().empty()) {
464 // prepare descriptor updates before waiting
465 setWriter.hintWriteCount(queuedUpdates.current().size() * 2);
466 setWriter.hintImageInfoCount(queuedUpdates.current().size() * 2);
467
468 for (unsigned int j = 0; j < queuedUpdates.current().size(); ++j) {
469 vk::Texture* texture = queuedUpdates.current()[j];
470 const bool isCubemap = texture->getType() == vk::Texture::Type::Cubemap;
471 const std::uint32_t bindIndex =
472 isCubemap ? CubemapArrayBindIndex : TextureArrayBindIndex;
473 const vk::Texture* base = isCubemap ? cubemaps.data() : textures.data();
474 const std::uint32_t i = texture - base;
475 const bool isRT = texture->getType() == vk::Texture::Type::RenderTexture;
476 const VkImageView view = texture->currentView;
477
478 auto& regularInfo = setWriter.getNewImageInfo();
479 regularInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
480 regularInfo.imageView = view;
481 regularInfo.sampler =
482 renderer.samplerCache().getSampler({.type = texture->getSampler()});
483
484 auto& regularWrite = setWriter.getNewSetWrite(currentSet);
485 regularWrite.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
486 regularWrite.descriptorCount = 1;
487 regularWrite.dstBinding = bindIndex;
488 regularWrite.dstArrayElement = i;
489 regularWrite.dstSet = currentSet;
490 regularWrite.pImageInfo = &regularInfo;
491 regularWrite.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
492
493 // write error texture to rt set if is rt itself
494 auto& rtInfo = setWriter.getNewImageInfo();
495 rtInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
496 rtInfo.imageView = isRT ? errorTexture.getView() : view;
497 rtInfo.sampler =
498 isRT ? renderer.samplerCache().getSampler({.type = errorTexture.getSampler()}) :
499 regularInfo.sampler;
500
501 auto& rtWrite = setWriter.getNewSetWrite(currentRtSet);
502 rtWrite.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
503 rtWrite.descriptorCount = 1;
504 rtWrite.dstBinding = bindIndex;
505 rtWrite.dstArrayElement = i;
506 rtWrite.pImageInfo = &rtInfo;
507 rtWrite.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
508 }
509
510 queuedUpdates.current().clear();
511 }
