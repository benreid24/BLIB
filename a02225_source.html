181PipelineParameters&& PipelineParameters::build() {
182 if (shaders.empty()) { throw std::runtime_error("Pipeline creation must have shaders"); }
183 if (vertexAttributes.empty()) {
184 withVertexFormat<3>(prim::Vertex::bindingDescription(),
185 prim::Vertex::attributeDescriptions());
186 }
187 if (colorAttachmentBlendStates.empty()) {
188 colorAttachmentBlendStates.emplace_back();
189 auto& ca = colorAttachmentBlendStates.back();
190 ca.colorWriteMask = VK_COLOR_COMPONENT_R_BIT | VK_COLOR_COMPONENT_G_BIT |
191 VK_COLOR_COMPONENT_B_BIT | VK_COLOR_COMPONENT_A_BIT;
192 ca.blendEnable = VK_TRUE;
193 ca.srcColorBlendFactor = VK_BLEND_FACTOR_SRC_ALPHA;
194 ca.dstColorBlendFactor = VK_BLEND_FACTOR_ONE_MINUS_SRC_ALPHA;
195 ca.colorBlendOp = VK_BLEND_OP_ADD;
196 ca.srcAlphaBlendFactor = VK_BLEND_FACTOR_ONE;
197 ca.dstAlphaBlendFactor = VK_BLEND_FACTOR_ONE;
198 ca.alphaBlendOp = VK_BLEND_OP_MAX;
199 }
200 colorBlending.pAttachments = colorAttachmentBlendStates.data();
201 colorBlending.attachmentCount = colorAttachmentBlendStates.size();
202
203 return std::move(*this);
204}
