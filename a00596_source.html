141  break;

143  default:

158

161  input.update();
162  states.top()->update(*this, updateTimestep);
163  bl::event::Dispatcher::syncListeners();

165  // check if we should end early

167  newState) {

169  break;
170  }
171 
172  // handle timing
173  lag -= updateTimestep;
174  averageUpdateTime =
175  0.8f * averageUpdateTime +
176  0.2f * (updateMeasureTimer.getElapsedTime().asSeconds() - updateStart);
177  if (updateMeasureTimer.getElapsedTime().asSeconds() > startingLag * 1.1f) {
178  fallBehindWarning(updateMeasureTimer.getElapsedTime().asSeconds() - startingLag);
179  if (engineSettings.allowVariableTimestep()) {
180  const float newTs = updateTimestep * 1.05f;
181  BL_LOG_INFO << "Adjusting update timestep from " << updateTimestep << "s to "
182  << newTs << "s";
183  updateTimestep = newTs;
184  averageUpdateTime = updateTimestep;
185  }
186  else {
187  skipUpdatesInfo(std::ceil(lag / updateTimestep));
188  lag = 0.f;
189  }
190  }
191  }
192 
193  // update timing
194  if (averageUpdateTime < updateTimestep * 0.9f &&
195  updateTimestep > engineSettings.updateTimestep()) {
196  const float newTs = std::max(engineSettings.updateTimestep(), updateTimestep * 0.95f);
197  BL_LOG_INFO << "Performance improved, adjusting timestep from " << updateTimestep
198  << "s to " << newTs << "s";
199  updateTimestep = newTs;
200  }
201 
202  // do render
203  if (!engineFlags.active(Flags::PopState) && !engineFlags.active(Flags::Terminate) &&
204  !newState) {
205  if (renderWindow) { renderingSystem.cameras().configureView(*renderWindow); }
206  states.top()->render(*this, lag);
207  }
208 
209  // Process flags
210  if (engineFlags.active(Flags::Terminate)) {
211  bl::event::Dispatcher::dispatch<event::Shutdown>({event::Shutdown::Terminated});
212  if (renderWindow) renderWindow->close();
213  return true;
214  }
215  else if (engineFlags.active(Flags::PopState)) {
216  BL_LOG_INFO << "Popping state: " << states.top()->name();
217  auto prev = states.top();
218  prev->deactivate(*this);
219  states.pop();
220  if (states.empty() && !newState) { // exit if no states left
221  BL_LOG_INFO << "Final state popped, exiting";
222  bl::event::Dispatcher::dispatch<event::Shutdown>(
223  {event::Shutdown::FinalStatePopped});
224  if (renderWindow) renderWindow->close();
225  return true;
226  }
227  else if (!newState) { // plain pop state
228  BL_LOG_INFO << "New engine state (popped): " << states.top()->name();
229  states.top()->activate(*this);
230  bl::event::Dispatcher::dispatch<event::StateChange>({states.top(), prev});
231  }
232  else { // replace state
233  BL_LOG_INFO << "New engine state (replaced): " << newState->name();
234  states.push(newState);
235  states.top()->activate(*this);
236  bl::event::Dispatcher::dispatch<event::StateChange>({states.top(), prev});
237  newState = nullptr;
238  }
239  updateOuterTimer.restart();
240  loopTimer.restart();
241  }
242 
243  // Handle state push
244  if (newState) {
245  BL_LOG_INFO << "New engine state (pushed): " << newState->name();
246  auto prev = states.top();
247  prev->deactivate(*this);
248  states.push(newState);
249  states.top()->activate(*this);
250  bl::event::Dispatcher::dispatch<event::StateChange>({states.top(), prev});
251  newState = nullptr;
252  updateOuterTimer.restart();
253  loopTimer.restart();
254  }
255 
256  // Adhere to FPS cap
257  if (minFrameLength > 0) {
258  const float st = minFrameLength - loopTimer.getElapsedTime().asSeconds();
259  if (st > 0) sf::sleep(sf::seconds(st));
260  loopTimer.restart();
261  }
262 
263  frameCount += 1.f;
264  if (fpsTimer.getElapsedTime().asSeconds() >= 1.f && engineSettings.logFps()) {
265  BL_LOG_INFO << "Running at " << frameCount / fpsTimer.getElapsedTime().asSeconds()
266  << " fps";
267  frameCount = 0.f;
268  fpsTimer.restart();
269  }
270  }
271 
272  return false; // shouldn't be able to get here

337 void Engine::handleResize(const sf::Event::SizeEvent& resize, bool ss) {
338  const sf::Vector2f modeSize(sf::Vector2u(engineSettings.windowParameters().videoMode().width,
339  engineSettings.windowParameters().videoMode().height));
340  const sf::Vector2f& ogSize = engineSettings.windowParameters().initialViewSize().x > 0.f ?
341  engineSettings.windowParameters().initialViewSize() :
342  modeSize;
343 
344  const float newWidth = static_cast<float>(resize.width);
345  const float newHeight = static_cast<float>(resize.height);
346 
347  const float xScale = newWidth / ogSize.x;
348  const float yScale = newHeight / ogSize.y;
349 
350  // it's ok to change view size here, cameras reset it every frame anyways
351  sf::View view(sf::FloatRect(0.f, 0.f, ogSize.x, ogSize.y));
352  sf::FloatRect viewPort(0.f, 0.f, 1.f, 1.f);
353 
354  if (xScale >= yScale) { // constrained by height, bars on sides
355  viewPort.width = ogSize.x * yScale / newWidth;
356  viewPort.left = (1.f - viewPort.width) * 0.5f;
357  }
358  else { // constrained by width, bars on top and bottom
359  viewPort.height = ogSize.y * xScale / newHeight;
360  viewPort.top = (1.f - viewPort.height) * 0.5f;
361  }
362 
363  view.setViewport(viewPort);
364  renderWindow->setView(view);
365 
366  if (ss) {
367  Settings::WindowParameters params = engineSettings.windowParameters();
368  params.withVideoMode(
369  sf::VideoMode(resize.width, resize.height, params.videoMode().bitsPerPixel));
370  engineSettings.withWindowParameters(params);
371  params.syncToConfig();
372  bl::event::Dispatcher::dispatch<event::WindowResized>({*renderWindow});
373  }
