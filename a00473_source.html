328template<typename C, typename U>
329struct DefaultHolder<C, U, true> {
330 std::unique_ptr<U> defVal;
331 void (*copyFn)(U&, const U&);
332
333 DefaultHolder()
334 : copyFn(nullptr) {}
335
336 template<typename... TArgs>
337 void make(TArgs&&... args) {
338 defVal = std::make_unique<U>(std::forward<TArgs>(args)...);
339 copyFn = [](U& dst, const U& src) { dst = src; };
340 }
341
342 void assign(U& result) const {
343 if (defVal && copyFn) { copyFn(result, *defVal); }
344 }
345
346 U* get() { return defVal.get(); }
347};
