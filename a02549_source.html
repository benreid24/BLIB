125void ScrollArea::onAcquisition() {
126 refreshSize();
127
128 bool showH = alwaysShowH && !neverShowH;
129 if (!neverShowH) { showH = showH || availableSize.x < totalSize.x; }
130 bool showV = alwaysShowV && !neverShowV;
131 if (!neverShowV) { showV = showV || availableSize.y < totalSize.y; }
132
133 if (showV) {
134 availableSize.x = getAcquisition().size.x - BarSize;
135 if (!neverShowH) showH = showH || availableSize.x < totalSize.x;
136 }
137 if (showH) {
138 availableSize.y = getAcquisition().size.y - BarSize;
139 if (!!neverShowV) showV = showV || availableSize.y < totalSize.y;
140 }
141 // one more time in case V is now visible
142 if (showV) {
143 availableSize.x = getAcquisition().size.x - BarSize;
144 if (!neverShowH) showH = showH || availableSize.x < totalSize.x;
145 }
146
147 if (showH) {
148 const sf::Vector2f barSize(getAcquisition().size.x - BarSize, BarSize);
149 const sf::Vector2f barPos(getAcquisition().position.x,
150 getAcquisition().position.y + getAcquisition().size.y -
151 barSize.y);
152 horScrollbar->setVisible(true);
153 horScrollbar->setSliderSize(computeButtonSize(totalSize.x, availableSize.x));
154 horScrollbar->setSliderIncrement(computeIncrement(totalSize.x, availableSize.x));
155 Packer::manuallyPackElement(horScrollbar, sf::FloatRect(barPos, barSize), true);
156 }
157 else
158 horScrollbar->setVisible(false);
159
160 if (showV) {
161 const sf::Vector2f barSize(BarSize, getAcquisition().size.y - BarSize);
162 const sf::Vector2f barPos(getAcquisition().position.x + getAcquisition().size.x - barSize.x,
163 getAcquisition().position.y);
164 vertScrollbar->setVisible(true);
165 vertScrollbar->setSliderSize(computeButtonSize(totalSize.y, availableSize.y));
166 vertScrollbar->setSliderIncrement(computeIncrement(totalSize.y, availableSize.y));
167 Packer::manuallyPackElement(vertScrollbar, sf::FloatRect(barPos, barSize), true);
168 }
169 else
170 vertScrollbar->setVisible(false);
171
172 Packer::manuallyPackElement(contentWrapper, sf::FloatRect(getPosition(), availableSize), true);
173 const sf::Vector2f contentArea(std::max(totalSize.x, availableSize.x),
174 std::max(totalSize.y, availableSize.y));
175 Packer::manuallyPackElement(content, sf::FloatRect(getPosition(), contentArea), true);
176}

178void ScrollArea::scrolled() {
179 offset = {0, 0};
180 if (horScrollbar->visible()) {
181 const float freeSpace = totalSize.x - availableSize.x;
182 offset.x = -freeSpace * horScrollbar->getValue();
183 }
184 if (vertScrollbar->visible()) {
185 const float freeSpace = totalSize.y - availableSize.y;
186 offset.y = -freeSpace * vertScrollbar->getValue();
187 }
188
189 updateContentPos();
190
191 const Event mockMove(Event::MouseMoved, boxMousePos);
192 content->processEvent(mockMove);
193}

195bool ScrollArea::handleScroll(const Event& scroll) {
196 if (getAcquisition().contains(scroll.mousePosition())) {
197 if (content->handleScroll(scroll)) return true;
198
199 if (totalSize.x > availableSize.x || totalSize.y > availableSize.y) {
200 if (sf::Keyboard::isKeyPressed(sf::Keyboard::Key::LControl) ||
201 sf::Keyboard::isKeyPressed(sf::Keyboard::Key::RControl)) {
202 horScrollbar->incrementValue(-scroll.scrollDelta());
203 }
204 else { vertScrollbar->incrementValue(-scroll.scrollDelta()); }
205 }
206 return true;
207 }
208 return false;
209}
