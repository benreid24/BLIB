176 for (auto& step : timeline) {

179 if (!asset->outputtedBy) {

181 asset->firstAvailableStep = 0;
182 }
183 }
184 for (GraphAsset* asset : task->assets.optionalInputs) {
185 if (asset && !asset->outputtedBy) {

193 const auto isInput = [](GraphAsset* asset, Task* task) -> bool {
194 for (GraphAsset* i : task->assets.requiredInputs) {
195 if (i == asset) { return true; }
196 }
197 for (GraphAsset* i : task->assets.optionalInputs) {
198 if (i == asset) { return true; }
199 }
200 return false;
201 };

203 // traverse graph up from leafs and populate firstAvailableStep for each asset
204 while (!toVisit.empty()) {

206 toVisit.pop();
207

209 node.second->assets.output->firstAvailableStep, node.first->firstAvailableStep + 1);
210
211 for (auto& task : tasks) {
212 if (isInput(node.second->assets.output, task.get())) {
213 toVisit.emplace(node.second->assets.output, task.get());
214 }
215 }
216 }

218 // for each task now determine soonest step it can run and move it

224 }
225 for (GraphAsset* asset : task->assets.requiredInputs) {

228 }

233 unsigned int i = 0;
234 for (auto& step : timeline) {

236 Task* task = *it;

238 if (soonest != i) {
