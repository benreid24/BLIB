123 AudioSystem::Handle AudioSystem::getOrLoadSound(const std::string& path) {
124  std::unique_lock lock(Runner::get().soundMutex);
125  auto& r = Runner::get();
126 
127  // Create or find handle
128  Handle handle = AudioSystem::InvalidHandle;
129  auto it = r.soundHandles.find(path);
130  if (it != r.soundHandles.end()) { handle = it->second; }
131  else {
132  handle = makeHandle();
133  r.soundHandles.emplace(path, handle);
134  r.soundSources.emplace(handle, path);

136 
137  // Ensure sound loaded

139  if (sit == r.sounds.end()) {

431 void Runner::run() {
432  while (!shouldStop) {
433  {
434  std::unique_lock plock(pauseMutex);
435  if (paused) { pauseCond.wait(plock); }
436 
437  // sound cleanup and fadeouts
438  {
439  std::unique_lock slock(soundMutex);
440 
441  for (auto i = soundHandles.begin(); i != soundHandles.end();) {
442  auto j = i++;
443  auto it = sounds.find(j->second);
444  if (it != sounds.end()) {
445  if (timer.getElapsedTime().asSeconds() - it->second.lastInteractTime >=
446  unloadTimeout + it->second.buffer->getDuration().asSeconds() &&
447  it->second.sound.getStatus() != sf::Sound::Playing) {
448  sounds.erase(it);
449  soundHandles.erase(j);
450  }
451  }
452  }
453 
454  for (auto i = fadingSounds.begin(); i != fadingSounds.end();) {
455  auto j = i++;
456  j->update();
457  }
458  soundTimer.restart();
459  }
460 
461  // playlist crossfade update and song update
462  {
463  std::unique_lock plock(playlistMutex);
464 
465  if (!playlistStack.empty() && playlistStack.front()->isPlaying()) {
466  playlistStack.front()->update();
467  }
468 
469  fadeIn.update();
470  fadeOut.update();
471  fadeTimer.restart();
472  }
473  }
474 
475  sf::sleep(sf::milliseconds(30));
476  }
477 
478  BL_LOG_INFO << "AudioSystem thread terminated";
479 }
