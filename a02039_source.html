35bool AnimationData::isValidSlideshow(const AnimationData& data) {
36 if (data.frames.empty()) return true;
37
38 for (const auto& frame : data.frames) {
39 // TODO - remove ugly hack when animation editor rewritten
40 removeDuplicateShard(frame);
41
42 if (frame.shards.size() != 1) {
43 BL_LOG_DEBUG << "Slideshow check fail: Frame has more than 1 shard";
44 return false;
45 }
46 if (frame.shards.front().rotation != 0.f) {
47 BL_LOG_DEBUG << "Slideshow check fail: Frame has non-zero rotation";
48 return false;
49 }
50 }
51
52 const sf::Vector2f pos = data.frames.front().shards.front().offset;
53 const sf::Vector2f scale = data.frames.front().shards.front().scale;
54 for (const auto& frame : data.frames) {
55 if (!compareVectors(scale, frame.shards.front().scale)) {
56 BL_LOG_DEBUG << "Slideshow check fail: Frame scales inconsistent";
57 return false;
58 }
59 if (!compareVectors(pos, frame.shards.front().offset)) {
60 BL_LOG_DEBUG << "Slideshow check fail: Frame positions inconsistent";
61 return false;
62 }
63 }
64
65 return true;
66}

118bool AnimationData::doLoad(serial::binary::InputStream& input, const std::string& ogPath,
119 bool forBundle) {
120 frames.clear();
121 stateOffsets.clear();
122 slideshow = false;
123
124 stateOffsets.reserve(16);
125 stateOffsets.emplace_back(0);
126
127 const std::string path = util::FileUtil::getPath(ogPath);
128 const std::string& spritesheetDir =
129 engine::Configuration::get<std::string>("blib.animation.spritesheet_path");
130
131 std::string sheet;
132 if (!input.read(sheet)) return false;
133 spritesheetSource = sheet;
134 if (!resource::FileSystem::resourceExists(sheet)) {
135 const std::string localSheet = util::FileUtil::joinPath(path, sheet);
136 const std::string dirSheet = util::FileUtil::joinPath(spritesheetDir, sheet);
137
138 if (resource::FileSystem::resourceExists(localSheet)) { sheet = localSheet; }
139 else if (resource::FileSystem::resourceExists(dirSheet)) { sheet = dirSheet; }
140 else { return false; }
141
142 if (forBundle) { spritesheetSource = sheet; }
143 }
144 actualSpritesheetPath = sheet;
145
146 if (!input.read(loop)) return false;
147
148 std::uint16_t nFrames = 0;
149 if (!input.read(nFrames)) return false;
150 frames.reserve(nFrames);
151 for (unsigned int i = 0; i < nFrames; ++i) {
152 Frame& frame = frames.emplace_back();
153
154 std::uint32_t length;
155 if (!input.read(length)) return false;
156 frame.length = static_cast<float>(length) / 1000.0f;
157
158 std::uint16_t nShards = 0;
159 if (!input.read(nShards)) return false;
160 frame.shards.reserve(nShards);
161
162 for (unsigned int j = 0; j < nShards; ++j) {
163 frame.shards.emplace_back();
164 Frame::Shard& shard = frame.shards.back();
165
166 uint32_t u32;
167 int32_t s32;
168 if (!input.read(u32)) return false;
169 shard.source.left = u32;
170 if (!input.read(u32)) return false;
171 shard.source.top = u32;
172 if (!input.read(u32)) return false;
173 shard.source.width = u32;
174 if (!input.read(u32)) return false;
175 shard.source.height = u32;
176 if (!input.read(u32)) return false;
177 shard.scale.x = static_cast<float>(u32) / 100.0f;
178 if (!input.read(u32)) return false;
179 shard.scale.y = static_cast<float>(u32) / 100.0f;
180 if (!input.read(s32)) return false;
181 shard.offset.x = s32;
182 if (!input.read(s32)) return false;
183 shard.offset.y = s32;
184 if (!input.read(u32)) return false;
185 shard.rotation = u32;
186 if (!input.read(shard.alpha)) return false;
187 frame.shards.push_back(shard);
188 }
189 }
190
191 // bool value not present in legacy files
192 std::uint8_t u8;
193 if (!input.read(u8)) { centerShards = false; }
194 else { centerShards = u8 == 1; }
195
196 populateDerivedState();
197
198 return true;

201void AnimationData::populateDerivedState() {
202 totalLength = 0.f;
203 std::uint32_t shardIndex = 0;
204
205 const sf::Vector2u sheetSize =
206 resource::ResourceManager<sf::Image>::load(resolvedSpritesheet())->getSize();
207 const sf::Vector2f size(sheetSize);
208
209 for (unsigned int i = 0; i < frames.size(); ++i) {
210 auto& frame = frames[i];
211 frame.shardIndex = shardIndex;
212 frame.nextFrame = i + 1;
213 shardIndex += frame.shards.size();
214 computeFrameSize(frame);
215 totalLength += frame.length;
216
217 for (auto& shard : frame.shards) {
218 const sf::FloatRect source(shard.source);
219 shard.normalizedSource.left = source.left / size.x;
220 shard.normalizedSource.top = source.top / size.y;
221 shard.normalizedSource.width = source.width / size.x;
222 shard.normalizedSource.height = source.height / size.y;
223 }
224 }
225 if (!frames.empty()) { frames.back().nextFrame = 0; }
226
227 slideshow = isValidSlideshow(*this);
228}

236void AnimationData::computeFrameSize(Frame& frame) {
237 sf::FloatRect bounds(1000000.f, 1000000.f, -1000000.f, -1000000.f);
238 for (auto& shard : frame.shards) {
239 const sf::FloatRect source(shard.source);
240 const sf::Vector2f center(source.width * 0.5f, source.height * 0.5f);
241 sf::Transform transform;
242 transform.rotate(shard.rotation, center);
243 transform.scale(shard.scale, center);
244 transform.translate(shard.offset);
245
246 const sf::FloatRect shardBounds = transform.transformRect(source);
247 bounds.left = std::min(bounds.left, shardBounds.left);
248 bounds.top = std::min(bounds.top, shardBounds.top);
249 bounds.width = std::max(bounds.width, shardBounds.left + shardBounds.width);
250 bounds.height = std::max(bounds.height, shardBounds.top + shardBounds.height);
251 }
252
253 frame.size = {bounds.width - bounds.left, bounds.height - bounds.top};
254}

268bool AnimationData::addState(std::size_t stateIndex, const AnimationData& anim, bool ss) {
269 // validate compatible
270 if (!frames.empty()) {
271 if (anim.frames.empty()) {
272 BL_LOG_ERROR << "Animation combine failed: State animation is empty";
273 return false;
274 }
275
276 if (spritesheetSource != anim.spritesheetSource) {
277 BL_LOG_ERROR << "Animation combine failed: Spritesheets are different";
278 return false;
279 }
280
281 for (auto& frame : frames) {
282 if (frame.shards.size() != frames.front().shards.size()) {
283 BL_LOG_ERROR
284 << "Animation combine failed: Base animation has inconsistent shard counts";
285 return false;
286 }
287 }
288
289 for (const auto& frame : anim.frames) {
290 if (frame.shards.size() != frames.front().shards.size()) {
291 BL_LOG_ERROR << "Animation combine failed: State animation has different shard "
292 "counts than base";
293 return false;
294 }
295 }
296
297 if (ss) {
298 if (isValidSlideshow(*this)) {
299 if (!isValidSlideshow(anim)) {
300 BL_LOG_ERROR << "Animation combine failed: State animation was not Slideshow";
301 return false;
302 }
303 }
304 }
305 }
306
307 // copy in frames and update next frame pointers
308 const std::size_t offset = frames.size();
309 stateOffsets[stateIndex] = offset;
310 frames.reserve(frames.size() + anim.frames.size());
311 std::copy(anim.frames.begin(), anim.frames.end(), std::back_inserter(frames));
312 for (std::size_t i = offset; i < frames.size() - 1; ++i) { frames[i].nextFrame = i + 1; }
313 frames.back().nextFrame = offset;
314
315 return true;
316}
