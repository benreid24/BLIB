50void Image::create(Renderer& r, const ImageOptions& options) {
51 if (renderer && compareOptions(createOptions, options)) { return; }
52
53 deferDestroy();
54 renderer = &r;
55 createOptions = options;
56 if (createOptions.viewAspect == VK_IMAGE_ASPECT_FLAG_BITS_MAX_ENUM) {
57 createOptions.viewAspect = createOptions.aspect;
58 }
59
60 VmaAllocationCreateInfo allocInfo{};
61 allocInfo.requiredFlags = createOptions.memoryLocation;
62 allocInfo.usage = VMA_MEMORY_USAGE_AUTO;
63 allocInfo.flags = createOptions.allocFlags;
64
65 VkImageCreateInfo createInfo{};
66 createInfo.sType = VK_STRUCTURE_TYPE_IMAGE_CREATE_INFO;
67 createInfo.imageType = VK_IMAGE_TYPE_2D;
68 createInfo.extent.width = createOptions.extent.width;
69 createInfo.extent.height = createOptions.extent.height;
70 createInfo.extent.depth = 1;
71 createInfo.mipLevels = createOptions.mipLevels;
72 createInfo.arrayLayers = getLayerCount();
73 createInfo.format = createOptions.format;
74 createInfo.tiling = VK_IMAGE_TILING_OPTIMAL;
75 createInfo.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
76 createInfo.usage = createOptions.usage | VK_IMAGE_USAGE_TRANSFER_DST_BIT;
77 createInfo.sharingMode = VK_SHARING_MODE_EXCLUSIVE;
78 createInfo.samples = createOptions.samples;
79 createInfo.flags = getCreateFlags(createOptions.type) | createOptions.extraCreateFlags;
80 if (vmaCreateImage(renderer->vulkanState().getVmaAllocator(),

99void Image::resize(const glm::u32vec2& newSize, bool copyContents) {
100 if (!renderer) {
101 BL_LOG_WARN << "Cannot resize image before calling create()";
102 return;
103 }
104
105 // cache data needed to delete original image
106 VkImage oldImage = imageHandle;
107
108 // transition original image to transfer source layout
109 if (copyContents) {
110 auto cb = renderer->getSharedCommandPool().createBuffer();
111 renderer->vulkanState().transitionImageLayout(cb,
112 imageHandle,
113 VK_IMAGE_LAYOUT_UNDEFINED,
114 VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
115 getLayerCount(),
116 createOptions.aspect);
117 currentLayout = VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL;
118 cb.submit();
119 }
120
121 // create new image
122 const glm::u32vec2 oldSize(createOptions.extent.width, createOptions.extent.height);
123 ImageOptions newOptions = createOptions;
124 newOptions.extent.width = newSize.x;
125 newOptions.extent.height = newSize.y;
126 create(*renderer, newOptions);
127 currentLayout = VK_IMAGE_LAYOUT_UNDEFINED;
128
129 // copy from old to new
130 if (copyContents && oldImage != imageHandle) {
131 auto cb = renderer->getSharedCommandPool().createBuffer();
132
133 renderer->vulkanState().transitionImageLayout(cb,
134 imageHandle,
135 currentLayout,
136 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
137 getLayerCount(),
138 createOptions.aspect);
139 currentLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
140
141 if (createOptions.mipLevels > 1) {
142 BL_LOG_WARN << "Copying mip image contents on resize is not "
143 "supported. Only the first level will be copied";
144 }
145
146 VkImageCopy copyInfo{};
147 copyInfo.extent.width = std::min(oldSize.x, newSize.x);
148 copyInfo.extent.height = std::min(oldSize.y, oldSize.y);
149 copyInfo.extent.depth = 1;
150 copyInfo.srcSubresource.aspectMask = createOptions.aspect;
151 copyInfo.srcSubresource.mipLevel = 0;
152 copyInfo.srcSubresource.baseArrayLayer = 0;
153 copyInfo.srcSubresource.layerCount = getLayerCount();
154 copyInfo.dstSubresource = copyInfo.srcSubresource;
155
156 vkCmdCopyImage(cb,
157 oldImage,
158 VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
159 imageHandle,
160 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
161 1,
162 &copyInfo);
163
164 cb.submit();
165 }

283void Image::recordBarrier(VkCommandBuffer commandBuffer, VkPipelineStageFlags srcStages,
284 VkAccessFlags srcAccess, VkPipelineStageFlags dstStages,
285 VkAccessFlags dstAccess, VkImageLayout newLayout) {
286 newLayout = (newLayout == VK_IMAGE_LAYOUT_UNDEFINED) ? currentLayout : newLayout;
287
288 VkImageMemoryBarrier barrier{};
289 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
290 barrier.oldLayout = currentLayout;
291 barrier.newLayout = newLayout;
292 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
293 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
294 barrier.image = imageHandle;
295 barrier.subresourceRange.aspectMask = getAspect();
296 barrier.subresourceRange.baseMipLevel = 0;
297 barrier.subresourceRange.levelCount = createOptions.mipLevels;
298 barrier.subresourceRange.baseArrayLayer = 0;
299 barrier.subresourceRange.layerCount = getLayerCount();
300 barrier.srcAccessMask = srcAccess;
301 barrier.dstAccessMask = dstAccess;
302 vkCmdPipelineBarrier(

323void Image::genMipMapsOnGpu(VkCommandBuffer commandBuffer, VkImageLayout finalLayout) {
324 if (!canGenMipMapsOnGpu()) {
325 BL_LOG_WARN << "Cannot generate mipmaps on GPU for image with format: "
326 << createOptions.format;
327 return;
328 }
329 if (createOptions.type != ImageOptions::Type::Image2D) {
330 BL_LOG_ERROR << "Cannot generate mipmaps for image type: " << createOptions.type;
331 return;
332 }
333
334 // transition first level to transfer source
335 renderer->vulkanState().transitionImageLayout(commandBuffer,
336 imageHandle,
337 currentLayout,
338 VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
339 1,
340 createOptions.aspect,
341 0,
342 1);
343
344 // transfer remaining levels to transfer dest
345 renderer->vulkanState().transitionImageLayout(commandBuffer,
346 imageHandle,
347 VK_IMAGE_LAYOUT_UNDEFINED,
348 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
349 1,
350 createOptions.aspect,
351 1,
352 createOptions.mipLevels - 1);
353
354 VkImageMemoryBarrier barrier{};
355 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
356 barrier.image = imageHandle;
357 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
358 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
359 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
360 barrier.subresourceRange.baseArrayLayer = 0;
361 barrier.subresourceRange.layerCount = 1;
362 barrier.subresourceRange.levelCount = 1;
363
364 std::int32_t mipWidth = createOptions.extent.width;
365 std::int32_t mipHeight = createOptions.extent.height;
366
367 for (std::uint32_t i = 1; i < createOptions.mipLevels; ++i) {
368 // transition prior level to transfer source
369 barrier.subresourceRange.baseMipLevel = i - 1;
370 if (i > 1) { // we already transitioned the first level
371 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
372 barrier.newLayout = VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL;
373 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
374 barrier.dstAccessMask = VK_ACCESS_TRANSFER_READ_BIT;
375
376 vkCmdPipelineBarrier(commandBuffer,
377 VK_PIPELINE_STAGE_TRANSFER_BIT,
378 VK_PIPELINE_STAGE_TRANSFER_BIT,
379 0,
380 0,
381 nullptr,
382 0,
383 nullptr,
384 1,
385 &barrier);
386 }
387
388 // copy prior level to current level and downsample
389 VkImageBlit blit{};
390 blit.srcOffsets[0] = {0, 0, 0};
391 blit.srcOffsets[1] = {mipWidth, mipHeight, 1};
392 blit.srcSubresource.aspectMask = createOptions.aspect;
393 blit.srcSubresource.mipLevel = i - 1;
394 blit.srcSubresource.baseArrayLayer = 0;
395 blit.srcSubresource.layerCount = 1;
396 blit.dstOffsets[0] = {0, 0, 0};
397 blit.dstOffsets[1] = {
398 mipWidth > 1 ? mipWidth / 2 : 1, mipHeight > 1 ? mipHeight / 2 : 1, 1};
399 blit.dstSubresource.aspectMask = createOptions.aspect;
400 blit.dstSubresource.mipLevel = i;
401 blit.dstSubresource.baseArrayLayer = 0;
402 blit.dstSubresource.layerCount = 1;
403
404 vkCmdBlitImage(commandBuffer,
405 imageHandle,
406 VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
407 imageHandle,
408 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
409 1,
410 &blit,
411 VK_FILTER_LINEAR);
412
413 // transition source level to final layout
414 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL;
415 barrier.newLayout = finalLayout;
416 barrier.srcAccessMask = VK_ACCESS_TRANSFER_READ_BIT;
417 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
418
419 vkCmdPipelineBarrier(commandBuffer,
420 VK_PIPELINE_STAGE_TRANSFER_BIT,
421 VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT,
422 0,
423 0,
424 nullptr,
425 0,
426 nullptr,
427 1,
428 &barrier);
429
430 if (mipWidth > 1) { mipWidth /= 2; }
431 if (mipHeight > 1) { mipHeight /= 2; }
432 }
433
434 // transition last level to final layout
435 barrier.subresourceRange.baseMipLevel = createOptions.mipLevels - 1;
436 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
437 barrier.newLayout = finalLayout;
438 barrier.srcAccessMask = VK_ACCESS_TRANSFER_READ_BIT;
439 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
440
441 vkCmdPipelineBarrier(commandBuffer,
442 VK_PIPELINE_STAGE_TRANSFER_BIT,
443 VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT,
444 0,
445 0,
446 nullptr,
447 0,
448 nullptr,
449 1,
450 &barrier);
451
452 currentLayout = finalLayout;
453}

455sf::Image Image::genMipMapsOnCpu(const sf::Image& image) {
456 const std::uint32_t mipLevels =
457 computeMipLevelsForGeneration(image.getSize().x, image.getSize().y);
458
459 std::uint32_t sourceX = 0;
460 std::uint32_t sourceY = 0;
461 std::uint32_t mipWidth = image.getSize().x / 2;
462 std::uint32_t mipHeight = image.getSize().y / 2;
463 std::uint32_t mipX = image.getSize().x;
464 std::uint32_t mipY = 0;
465
466 std::uint32_t totalMipHeight = 0;
467 for (std::uint32_t i = 1; i < mipLevels; ++i) {
468 totalMipHeight += mipHeight;
469 if (mipHeight > 1) { mipHeight /= 2; }
470 }
471 mipHeight = image.getSize().y / 2;
472
473 sf::Image result;
474 result.create(image.getSize().x + mipWidth,
475 std::max(image.getSize().y, totalMipHeight),
476 sf::Color::Transparent);
477 result.copy(image, 0, 0, sf::IntRect(0, 0, image.getSize().x, image.getSize().y));
478
479 for (std::uint32_t i = 1; i < mipLevels; ++i) {
480 for (std::uint32_t x = 0; x < mipWidth; ++x) {
481 for (std::uint32_t y = 0; y < mipHeight; ++y) {
482 std::uint32_t rSum = 0;
483 std::uint32_t gSum = 0;
484 std::uint32_t bSum = 0;
485 std::uint32_t aSum = 0;
486 std::uint32_t read = 0;
487
488 const auto addColor = [&rSum, &gSum, &bSum, &aSum, &read, &result](
489 std::uint32_t rx, std::uint32_t ry) {
490 if (rx < result.getSize().x && ry < result.getSize().y) {
491 ++read;
492 const sf::Color c = result.getPixel(rx, ry);
493 rSum += static_cast<std::uint32_t>(c.r);
494 gSum += static_cast<std::uint32_t>(c.g);
495 bSum += static_cast<std::uint32_t>(c.b);
496 aSum += static_cast<std::uint32_t>(c.a);
497 }
498 };
499
500 addColor(sourceX + x * 2, sourceY + y * 2);
501 addColor(sourceX + x * 2 + 1, sourceY + y * 2);
502 addColor(sourceX + x * 2, sourceY + y * 2 + 1);
503 addColor(sourceX + x * 2 + 1, sourceY + y * 2 + 1);
504
505 const sf::Color avg(rSum / read, gSum / read, bSum / read, aSum / read);
506 result.setPixel(mipX + x, mipY + y, avg);
507 }
508 }
509 sourceX = mipX;
510 sourceY = mipY;
511 mipY += mipHeight;
512 if (mipWidth > 1) { mipWidth /= 2; }
513 if (mipHeight > 1) { mipHeight /= 2; }
514 }
515
516 return result;
517}
