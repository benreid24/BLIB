222sf::Vector2u Grid<T>::getIndexAtPosition(const sf::Vector2f& pos) {
223 sf::Vector2f v2f(pos - origin);
224 v2f.x = std::max(v2f.x, 0.f);

241void Grid<T>::forAllInRegion(const sf::FloatRect& region, const TCb& cb) {
242 static_assert(std::is_invocable<TCb, T>::value,
243 "Visitor signature is void(T val) or bool(T val");
244
245 const sf::Vector2u sp = getIndexAtPosition({region.left, region.top});
246 const sf::Vector2u ep =
247 getIndexAtPosition({region.left + region.width, region.top + region.height});
248 for (unsigned int x = sp.x; x <= ep.x; ++x) {
249 for (unsigned int y = sp.y; y <= ep.y; ++y) {
250 Cell& cell = cells(x, y);
251 for (T val : cell) {
252 if constexpr (std::is_same_v<std::invoke_result_t<TCb, T>, bool>) {
253 if (cb(val)) return;
254 }
255 else { cb(val); }

257 }

264 static_assert(std::is_invocable<TCb, T>::value,
265 "Visitor signature is void(T val) or bool(T val");
266
267 const sf::Vector2u i = getIndexAtPosition(pos);
268 const unsigned int sx = i.x > 0 ? i.x - 1 : 0;
269 const unsigned int ex = std::min(i.x + 1, cells.getWidth() - 1);
270 const unsigned int sy = i.y > 0 ? i.y - 1 : 0;
271 const unsigned int ey = std::min(i.y + 1, cells.getHeight() - 1);
272
273 for (unsigned int x = sx; x <= ex; ++x) {
274 for (unsigned int y = sy; y <= ey; ++y) {
275 Cell& cell = cells(x, y);
