62bool SerializableObjectBase::deserializeBinary(stream::InputStream& stream, void* obj) const {
63 binary::detail::InputStreamWrapper wrapper(stream);
64
65 char foundFields[MaxFieldCount];

70 };

78 std::uint16_t id = 0;

81 logReadError("field id");
82 return false;

86 return false;

92 << ". Message type: " << debugName;
93 return false;
94 }

96 BL_LOG_DEBUG << "Failed to deserialize '" << debugName << "' field failed: " << id;
97 return false;
98 }

100 }
101 else {

103 logReadError("skip field");

106 }
107 }

110 for (const auto& field : fieldsBinary) {
111 if (foundFields[field.first] != 0) continue;
112 if (!field.second->optional()) {
113 BL_LOG_DEBUG << "Failed to deserialize '" << debugName << "', missing required field "
114 << field.first;
115 return false;
116 }
117 field.second->makeDefault(obj);
118 }
119 return true;

168bool SerializableObjectBase::deserializeJsonStream(stream::InputStream& stream, void* obj) {
169 using FIter = decltype(fieldsJson)::iterator;
170 std::pair<bool, FIter> fields[MaxFieldCount];
171 unsigned int fieldCount = 0;
172 for (auto it = fieldsJson.begin(); it != fieldsJson.end(); ++it) {
173 fields[fieldCount].first = false;
174 fields[fieldCount].second = it;
175 ++fieldCount;
176 }
177
178 const auto markVisited = [&fields, fieldCount](FIter it) {
179 for (unsigned int i = 0; i < fieldCount; ++i) {
180 if (fields[i].second == it) {
181 fields[i].first = true;
182 break;
183 }
184 }
185 };
186
187 const auto logParseError = [this](const std::string& err) {
188 BL_LOG_DEBUG << "Failed to deserialize '" << debugName << "', parse error: " << err;
189 };
190
191 util::StreamUtil::skipWhitespace(stream);
192 if (stream.peek() != '{') {
193 logParseError(std::string("Expected '{' but got ") + static_cast<char>(stream.peek()));
194 return false;
195 }
196 stream.get();
197
198 while (stream.isValid()) {
199 util::StreamUtil::skipWhitespace(stream);
200 if (stream.peek() != '"') {
201 logParseError(std::string("Expected '\"' but got ") + static_cast<char>(stream.peek()));
202 return false;
203 }
204 stream.get();
205
206 std::string name;
207 util::StreamUtil::getline(stream, name, '"');
208 if (!stream.isValid()) {
209 logParseError("End of stream while reading field name");
210 return false;
211 }
212 if (!util::StreamUtil::skipUntil(stream, ':')) {
213 logParseError("End of stream while reading to ':'");
214 return false;
215 }
216 stream.get();
217
218 const auto it = fieldsJson.find(name);
219 if (it != fieldsJson.end()) {
220 if (!it->second->deserializeJsonStream(stream, obj)) {
221 BL_LOG_DEBUG << "Failed to deserialize '" << debugName << "', field failed '"
222 << it->first << "'";
223 return false;
224 }
225 markVisited(it);
226 }
227 else {
228 json::Loader loader(stream);
229 json::Value trash(false);
230 if (!loader.loadValue(trash)) {
231 logParseError("Failed to skip past unknown json data");
232 return false;
233 }
234 }
235
236 util::StreamUtil::skipWhitespace(stream);
237 const char c = stream.get();
238 if (c == '}') {
239 // check fields not found for required
240 for (unsigned int i = 0; i < fieldCount; ++i) {
241 if (!fields[i].first) {
242 const auto& field = fields[i].second;
243 if (!field->second->optional()) {
244 BL_LOG_DEBUG << "Failed to deserialize '" << debugName
245 << "', missing required field '" << field->first << "'";
246 return false;
247 }
248 field->second->makeDefault(obj);
249 }
250 }
251 return true;
252 }
253 else if (c == ',') { continue; }
254 logParseError(std::string("Expected ',' but got ") + c);
255 return false;
256 }
257 logParseError("Unexpected end of stream");
