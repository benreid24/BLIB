11bool circleIntersectsRect(const glm::vec2& circleCenter, float radius, const glm::vec2& rectCenter,
12 const glm::vec2& size) {
13 const glm::vec2 halfSize = size * 0.5f;
14 const glm::vec2 centerDist = glm::abs(circleCenter - rectCenter);
15
16 if (centerDist.x > halfSize.x + radius || centerDist.y > halfSize.y + radius) { return false; }
17 if (centerDist.x <= halfSize.x || centerDist.x <= halfSize.y) { return true; }
18
19 const glm::vec2 cornerDist = centerDist - halfSize;
20 return (cornerDist.x * cornerDist.x + cornerDist.y * cornerDist.y) <= radius * radius;
21}

91bool Hitbox2D::intersectsRect(const glm::vec2& corner, const glm::vec2& otherSize) const {
92 switch (type) {

94 return circleIntersectsRect(
95 transform->getGlobalPosition(), radius, corner + otherSize * 0.5f, otherSize);
96 }
97 case Rectangle: {
98 const glm::vec2 myCorner = transform->getGlobalPosition() - transform->getOrigin();
99 const glm::vec2 myOtherSide = myCorner + size;
100 const glm::vec2 theirOtherSide = corner + otherSize;
101 const float interLeft = std::max(myCorner.x, corner.x);
102 const float interTop = std::max(myCorner.y, corner.y);
103 const float interRight = std::min(myOtherSide.x, theirOtherSide.x);
104 const float interBottom = std::min(myOtherSide.y, theirOtherSide.y);
105 return interLeft < interRight && interTop < interBottom;
106 }
107 case Point: {
108 const glm::vec2 lp = (transform->getGlobalPosition() - transform->getOrigin()) - corner;
