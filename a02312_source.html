86Overlay* RenderTarget::createSceneOverlay() {
87 if (scenes.empty()) {
88 BL_LOG_ERROR << "Tried to create Overlay with no current scene";
89 throw std::runtime_error("Tried to create Overlay with no current scene");
90 }
91
92 scenes.back().overlayRef = renderer.scenePool().allocateScene<Overlay>();
93 scenes.back().overlay = static_cast<Overlay*>(scenes.back().overlayRef.get());
94 scenes.back().overlayIndex = scenes.back().overlay->registerObserver(this);
95 scenes.back().overlayDescriptorCache.emplace(
96 ds::InitContext{engine,
97 engine.renderer(),
98 *this,
99 *scenes.back().overlay,
100 renderer.getGlobalShaderResources(),
101 getShaderResources(),
102 scenes.back().overlay->getShaderResources()});
103 graphAssets.replaceAsset<rgi::SceneAsset>(scenes.back().overlay, rg::AssetTags::OverlayInput);
104 scenes.back().graph.removeTasks<rgi::RenderOverlayTask>();
105 scenes.back().graph.putTask<rgi::RenderOverlayTask>(&scenes.back().overlayIndex);
106
107 return scenes.back().overlay;
108}

167void RenderTarget::updateDescriptorsAndQueueTransfers() {
168 if (hasScene()) {
169 auto& currentScene = scenes.back();
170
171 if (!currentScene.camera) {
172 if (!dynamic_cast<Overlay*>(currentScene.scene.get())) {
173 BL_LOG_WARN
174 << "Scene being rendered before having a camera set. Creating default camera";
175 }
176 currentScene.camera = currentScene.scene->createDefaultCamera();
177 currentScene.scene->setDefaultNearAndFarPlanes(*currentScene.camera);
178 }
179
180 if (currentScene.graph.needsRepopulation()) {
181 currentScene.graph.populate(*currentScene.scene);
182 if (currentScene.overlay) {
183 currentScene.graph.putTask<rgi::RenderOverlayTask>(&currentScene.overlayIndex);
184 }
185 }
186
187 shaderResources.performTransfers();
188 currentScene.descriptorCache.updateDescriptors();
189 if (currentScene.overlay) {
190 currentScene.overlayDescriptorCache.value().updateDescriptors();
191 }
192 }
193}

216glm::vec2 RenderTarget::transformToWorldSpace(const glm::vec2& sp) const {
217 const glm::vec2 ndc((sp.x - viewport.x) / viewport.width * 2.f - 1.f,
218 (sp.y - viewport.y) / viewport.height * 2.f - 1.f);
219 glm::mat4 tform(1.f);
220 if (hasScene() && scenes.back().camera) {
221 tform = scenes.back().camera->getProjectionMatrix(viewport) *
222 scenes.back().camera->getViewMatrix();
223 }
224 tform = glm::inverse(tform);
225 const glm::vec4 result = tform * glm::vec4(ndc, 0.f, 1.f);
226 return {result.x, result.y};
227}

234glm::vec2 RenderTarget::transformToOverlaySpace(const glm::vec2& sp) const {
235 const glm::vec2 ndc((sp.x - viewport.x) / viewport.width * 2.f - 1.f,
236 (sp.y - viewport.y) / viewport.height * 2.f - 1.f);
237 cam::OverlayCamera& cam = const_cast<cam::OverlayCamera&>(overlayCamera);
238 glm::mat4 tform = cam.getProjectionMatrix(viewport) * cam.getViewMatrix();
