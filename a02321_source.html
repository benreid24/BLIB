43void Overlay::renderScene(scene::SceneRenderContext& ctx) {
44 std::unique_lock lock(objectMutex);
45 ecs::Transaction<ecs::tx::EntityUnlocked, ecs::tx::ComponentRead<ovy::OverlayObject>>
46 transaction(engine.ecs());
47
48 std::copy(roots.begin(), roots.end(), std::inserter(renderStack, renderStack.begin()));
49
50 if (needRefreshAll ||
51 static_cast<std::uint32_t>(ctx.parentViewport().width) != cachedTargetSize.x ||
52 static_cast<std::uint32_t>(ctx.parentViewport().height) != cachedTargetSize.y) {
53 cachedParentViewport = ctx.parentViewport();
54 cachedTargetSize.x = static_cast<std::uint32_t>(ctx.parentViewport().width);
55 cachedTargetSize.y = static_cast<std::uint32_t>(ctx.parentViewport().height);
56 needRefreshAll = false;
57 refreshAll();
58 }
59
60 VkPipelineLayout currentPipelineLayout = nullptr;
61 VkPipeline currentPipeline = nullptr;
62 UpdateSpeed currentSpeed{};
63 while (!renderStack.empty()) {
64 ovy::OverlayObject& obj = *renderStack.back();
65 renderStack.pop_back();
66
67 if (obj.hidden) { continue; }
68
69 if (!obj.entity.flagSet(ecs::Flags::Dummy)) {
70 vkCmdSetScissor(ctx.getCommandBuffer(), 0, 1, &obj.cachedScissor);
71
72 const vk::PipelineLayout& layout = obj.pipeline->pipelineLayout();
73 const VkPipelineLayout vkl = layout.rawLayout();
74 const VkPipeline vkp = obj.pipeline->rawPipeline(ctx.currentRenderPass());
75 if (vkl != currentPipelineLayout || currentSpeed != obj.sceneKey.updateFreq) {
76 currentSpeed = obj.sceneKey.updateFreq;
77 currentPipelineLayout = vkl;
78 currentPipeline = vkp;
79 ctx.bindPipeline(*obj.pipeline);
80 ctx.bindDescriptors(obj.pipeline->pipelineLayout().rawLayout(),
81 obj.sceneKey.updateFreq,
82 obj.descriptors.data(),
83 obj.descriptorCount);
84 }
85 else if (currentPipeline != vkp) {
86 currentPipeline = vkp;
87 ctx.bindPipeline(*obj.pipeline);
88 }
89 for (std::uint8_t i = obj.perObjStart; i < obj.descriptorCount; ++i) {
90 obj.descriptors[i]->bindForObject(
91 ctx, obj.pipeline->pipelineLayout().rawLayout(), i, obj.sceneKey);
92 }
93 ctx.renderObject(obj);
94 }
95
96 std::copy(obj.getChildren().rbegin(),
97 obj.getChildren().rend(),
98 std::inserter(renderStack, renderStack.end()));

131void Overlay::doObjectRemoval(scene::SceneObject* object, std::uint32_t) {
132 ovy::OverlayObject* obj = static_cast<ovy::OverlayObject*>(object);
133 obj->overlayViewport = nullptr;
134
135 auto childCopy = obj->getChildren();
136 for (ovy::OverlayObject* child : childCopy) {
137 if (engine.ecs().getEntityParentDestructionBehavior(child->entity) !=
138 ecs::ParentDestructionBehavior::OrphanedByParent &&
139 child->sceneKey.sceneId != scene::Key::InvalidSceneId) {
140 removeObject(child);
141 }
142 }
143
144 const auto it = std::find(roots.begin(), roots.end(), obj);
145 if (it != roots.end()) {
146 roots.erase(it);
147 sortRoots();
148 }
149
150 for (unsigned int i = 0; i < obj->descriptorCount; ++i) {
151 obj->descriptors[i]->releaseObject(obj->entity, object->sceneKey);
152 }
153 objects.release(obj->sceneKey);
154 engine.ecs().removeComponent<ovy::OverlayObject>(obj->entity);
155 bl::event::Dispatcher::dispatch<rc::event::SceneObjectRemoved>({this, obj->entity});
156}

221void Overlay::sortRoots() {
222 std::sort(
223 roots.begin(), roots.end(), [this](ovy::OverlayObject* left, ovy::OverlayObject* right) {
224 com::Transform2D* lpos = engine.ecs().getComponent<com::Transform2D>(left->entity);
225 com::Transform2D* rpos = engine.ecs().getComponent<com::Transform2D>(right->entity);
226 if (!lpos || !rpos) { return left < right; }
227 return lpos->getGlobalDepth() < rpos->getGlobalDepth();
