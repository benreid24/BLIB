118template<typename T>
119typename SceneObjectStorage<T>::AllocateResult SceneObjectStorage<T>::allocate(
120 UpdateSpeed updateFreq, ecs::Entity entity) {
121 AllocateResult result{};
122
123 Bucket& bucket = updateFreq == UpdateSpeed::Static ? staticBucket : dynamicBucket;
124 result.originalBaseAddress = bucket.objects.data();
125 if (!bucket.freeIds.empty()) {
126 const std::uint32_t id = bucket.freeIds.front();
127 bucket.freeIds.pop();
128
129 result.newObject = &bucket.objects[id];
130 result.newObject->sceneKey.sceneId = id;
131 bucket.entityMap[id] = entity;
132 }
133 else {
134 const std::uint32_t id = bucket.objects.size();
135 result.addressesChanged = bucket.objects.size() == bucket.objects.capacity();
136 result.newObject = &bucket.objects.emplace_back();
137 bucket.entityMap.resize(bucket.objects.capacity(), ecs::InvalidEntity);
138 bucket.entityMap[id] = entity;
139 result.newObject->sceneKey.sceneId = id;
140 }
141
142 result.newObject->sceneKey.updateFreq = updateFreq;
143 result.newCapacity = bucket.objects.capacity();
144
145 return result;
146}

172 for (unsigned int i = 0; i < bucket->objects.size(); ++i) {
