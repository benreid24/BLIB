329 bool Registry::addComponent(Entity entity, const T& component) {
330  ComponentPool::Iterator it;
331 
332  {
333  const Component::IdType& id = Component::getId<T>();
334  std::unique_lock lock(entityMutex);
335 
336  // entity does not exist
337  if (entities.find(entity) == entities.end()) return false;
338 
339  // ensure pool exists
340  auto poolIter = componentPools.find(id);
341  if (poolIter == componentPools.end()) { poolIter = componentPools.try_emplace(id).first; }
342 
343  // ensure component link exists
344  auto linkIter = componentEntities.find(id);
345  if (linkIter == componentEntities.end()) {
346  linkIter = componentEntities.try_emplace(id).first;
347  }
348 
349  // ensure entity does not already have component
350  auto indexIter = entityComponentIterators.find(entity);
351  if (indexIter == entityComponentIterators.end()) {
352  indexIter = entityComponentIterators.try_emplace(entity).first;
353  }
354  if (indexIter->second.find(id) != indexIter->second.end()) return false;
355 
356  // add and track component
357  it = poolIter->second.add(component);
358  indexIter->second.emplace(id, it);
359  linkIter->second.insert(entity);
360 
361  invalidateViews(id);
362  }
363  if (dispatcher) dispatcher->dispatch<event::ComponentAdded<T>>({entity, it->get<T>()});
364  return true;
365 }
