103 bool Element::processEvent(const Event& event) {
104  if (!visible()) return false;
105 
106  if (propagateEvent(event)) return true;
107 
108  const bool eventOnMe = cachedArea.contains(event.mousePosition());
109 
110  if (event.type() == Event::Scrolled) {
111  if (eventOnMe) { return handleScroll(event); }
112  return false;
113  }
114 
115  switch (event.type()) {
116  case Event::TextEntered:
117  case Event::KeyPressed:
118  case Event::KeyReleased:
119  return processAction(event); // TODO - is this right?
120 
121  case Event::LeftMousePressed:
122  case Event::RightMousePressed:
123  if (eventOnMe) {
124  if (!active()) return true;
125 
126  if (takeFocus()) {
127  if (event.type() == Event::LeftMousePressed) {
128  dragStart = event.mousePosition();
129  isLeftPressed = true;
130  return processAction(event);
131  }
132  else {
133  isRightPressed = true;
134  return true;
135  }
136  }
137  return true; // click blocked
138  }
139  return false;
140 
141  case Event::LeftMouseReleased:
142  case Event::RightMouseReleased:
143  if (!active()) {
144  isLeftPressed = isRightPressed = false;
145  return eventOnMe;
146  }
147  processAction(event);
148  if (event.type() == Event::LeftMouseReleased) {
149  if (isLeftPressed) {
150  isLeftPressed = false;
151  if (eventOnMe) {
152  processAction(Event(Event::LeftClicked, event.mousePosition()));
153  return true;
154  }
155  }
156  }
157  else if (event.type() == Event::RightMouseReleased) {
158  if (isRightPressed) {
159  isRightPressed = false;
160  if (eventOnMe) {
161  processAction(Event(Event::RightClicked, event.mousePosition()));
162  return true;
163  }
164  }
165  }
166  return false;
167 
168  case Event::MouseMoved:
169  if (!active()) { return eventOnMe; }
170  if (eventOnMe) { fireSignal(event); }
171  if (isLeftPressed) {
172  isMouseOver = eventOnMe;
173  const bool r = processAction(Event(Event::Dragged, dragStart, event.mousePosition()));
174  dragStart = event.mousePosition();
175  return r;
176  }
177  else if (eventOnMe) {
178  if (!isMouseOver) {
179  isMouseOver = true;
180  processAction(Event(Event::MouseEntered, event.mousePosition()));
181  }
182  hoverTime = 0.f;
183  return true;
184  }
185  else if (isMouseOver) {
186  isMouseOver = false;
187  processAction(Event(Event::MouseLeft, event.mousePosition()));
188  }
189  return false;
190 
191  case Event::MouseOutsideWindow:
192  isMouseOver = false;
193  isLeftPressed = false;
194  isRightPressed = false;
195  return false;
196 
197  default:
198  return false;
199  }
