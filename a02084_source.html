28vk::Pipeline& PipelineCache::createPipline(std::uint32_t id, vk::PipelineParameters&& params) {
29 const auto insertResult =
30 cache.try_emplace(id, renderer, std::forward<vk::PipelineParameters>(params));
31 if (!insertResult.second) { BL_LOG_WARN << "Pipeline with id " << id << " already exists"; }
32 return insertResult.first->second;

35vk::Pipeline& PipelineCache::getPipeline(std::uint32_t id) {
36 auto it = cache.find(id);
37 if (it == cache.end()) {
38 BL_LOG_CRITICAL << "Failed to find pipeline with id: " << id;
39 throw std::runtime_error("Failed to find pipeline");
40 }
41 return it->second;
42}
