299 void Runner::run() {
300  while (running) {
301  {
302  std::unique_lock plock(pauseMutex);
303  if (paused) { pauseCond.wait(plock); }
304 
305  // sound cleanup and fadeouts
306  {
307  std::unique_lock slock(soundMutex);
308 
309  for (auto i = soundHandles.begin(); i != soundHandles.end();) {
310  auto j = i++;
311  auto it = sounds.find(j->second);
312  if (it != sounds.end()) {
313  if (timer.getElapsedTime().asSeconds() - it->second.lastInteractTime >=
314  unloadTimeout + it->second.buffer.getDuration().asSeconds()) {
315  sounds.erase(it);
316  }
317  }
318  soundHandles.erase(j);
319  }
320
