120AudioSystem::Handle AudioSystem::getOrLoadSound(const std::string& path) {
121 std::unique_lock lock(Runner::get().soundMutex);
122 auto& r = Runner::get();
123
124 // Create or find handle
125 Handle handle = AudioSystem::InvalidHandle;
126 auto it = r.soundHandles.find(path);
127 if (it != r.soundHandles.end()) { handle = it->second; }
128 else {
129 handle = makeHandle();
130 r.soundHandles.emplace(path, handle);
131 r.soundSources.emplace(handle, path);
132 }
133
134 // Ensure sound loaded
135 auto sit = r.sounds.find(handle);
136 if (sit == r.sounds.end()) {
137 sit = r.sounds.try_emplace(handle).first;
138 sit->second.buffer = resource::ResourceManager<sf::SoundBuffer>::load(path);
139 if (sit->second.buffer->getSampleCount() == 0) {
140 r.sounds.erase(sit);
141 BL_LOG_ERROR << "Failed to load sound: " << path;
142 return InvalidHandle;
143 }
144 sit->second.sound.setBuffer(*sit->second.buffer);
145 }
146
147 return handle;
148}

150bool AudioSystem::playSound(Handle sound, float fadeIn, bool loop) {
151 std::shared_lock slock(Runner::get().soundMutex);
152 auto& r = Runner::get();
153 if (r.paused) return false;
154
155 const auto it = r.validateAndLoadSound(sound);
156 if (it == r.sounds.end()) return false;
157
158 // play the sound
159 it->second.sound.setLoop(loop);
160 it->second.lastInteractTime = r.timer.getElapsedTime().asSeconds();
161 if (it->second.sound.getStatus() == sf::Sound::Playing) return true;
162
163 if (fadeIn > 0.f) {
164 it->second.sound.setVolume(0.f);
165 r.fadingSounds.emplace_back(it->second.sound, 100.f / fadeIn);
166 r.fadingSounds.back().me = --r.fadingSounds.end();
167 }
168 else { it->second.sound.setVolume(100.f); }
169
170 it->second.sound.play();
171 return true;
172}

427void Runner::run() {
428 while (!shouldStop) {
429 {
430 std::unique_lock plock(pauseMutex);
431 if (paused) { pauseCond.wait(plock); }
432
433 // sound cleanup and fadeouts
434 {
435 std::unique_lock slock(soundMutex);
436
437 for (auto i = soundHandles.begin(); i != soundHandles.end();) {
438 auto j = i++;
439 auto it = sounds.find(j->second);
440 if (it != sounds.end()) {
441 if (timer.getElapsedTime().asSeconds() - it->second.lastInteractTime >=
442 unloadTimeout + it->second.buffer->getDuration().asSeconds() &&
443 it->second.sound.getStatus() != sf::Sound::Playing) {
444 sounds.erase(it);
445 soundHandles.erase(j);
446 }
447 }
448 }
449
450 for (auto i = fadingSounds.begin(); i != fadingSounds.end();) {
451 auto j = i++;
452 j->update();
453 }
454 soundTimer.restart();
455 }
456
457 // playlist crossfade update and song update
458 {
459 std::unique_lock plock(playlistMutex);
460
461 if (!playlistStack.empty() && playlistStack.front()->isPlaying()) {
462 playlistStack.front()->update();
463 }
464
465 fadeIn.update();
466 fadeOut.update();
467 fadeTimer.restart();
468 }
469 }
470
471 sf::sleep(sf::milliseconds(30));
472 }
473
474 BL_LOG_INFO << "AudioSystem thread terminated";
475}

504void initiateCrossfade(Playlist* in, Playlist* out, float inTime, float outTime) {
505 if (Runner::get().fadeIn.playlist) {
506 Runner::get().fadeIn.playlist->pause();
507 Runner::get().fadeIn.playlist = nullptr;
508 }
509 if (in) {
510 if (inTime > 0.f) {
511 Runner::get().fadeIn.playlist = in;
512 Runner::get().fadeIn.fvel = 100.f / inTime;
513 in->setVolume(0.f);
514 }
515 else { in->setVolume(100.f); }
516 if (!Runner::get().paused) in->play();
517 }
518
519 if (Runner::get().fadeOut.playlist) {
520 Runner::get().fadeOut.playlist->pause();
521 Runner::get().fadeOut.playlist = nullptr;
522 }
523 if (out) {
524 if (outTime > 0.f) {
525 Runner::get().fadeOut.playlist = out;
526 Runner::get().fadeOut.fvel = -out->getVolume() / outTime;
527 }
528 else { out->pause(); }
529 }
