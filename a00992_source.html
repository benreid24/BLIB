174 void setParent(Entity child, Entity parent) {
175 T* childCom = get(child);
176 T* parentCom = get(parent);
177
178 if (childCom && parentCom) { childCom->parent = parentCom; }
179 else {
180 // only child is checked before this is called so no log for bad parent
181 if (!childCom) { BL_LOG_ERROR << "Invalid child entity: " << child; }
182 }
183 }
184
185 void addChild(Entity child, Entity parent) {
186 T* parentCom = get(parent);
187 if (!parentCom) { return; }
188 T* childCom = get(child);
189 if (childCom) {
190 auto& c = parentCom->children;
191 const auto it = std::find(c.begin(), c.end(), childCom);
192 if (it == c.end()) { parentCom->children.emplace_back(childCom); }
193 }
194 else { BL_LOG_ERROR << "Invalid child entity " << child << " for parent " << parent; }
195 }
196
197 virtual void onParentRemove(Entity parent, Entity orphan) override {
198 constexpr bool needsLock = IsParentAware || IsChildAware;
199 if constexpr (needsLock) { poolLock.lockWrite(); }
200
201 if constexpr (std::is_base_of_v<trait::ParentAware<T>, T>) { removeParent(orphan); }
202 if constexpr (std::is_base_of_v<trait::ChildAware<T>, T>) { removeChild(parent, orphan); }
203
204 if constexpr (needsLock) { poolLock.unlockWrite(); }
205 }
206
207 void removeParent(Entity orphan) {
208 T* com = get(orphan);
209 if (com) { com->parent = nullptr; }
210 else { BL_LOG_WARN << "Invalid orphan entity: " << orphan; }
211 }
212
213 void removeChild(Entity parent, Entity orphan) {
214 T* pcom = get(parent);
215 if (!pcom) {
216 // only child is checked before this is called so no log for bad parent
217 return;
218 }
219 T* com = get(orphan);
220 if (com) {
221 auto& c = pcom->children;
222 const auto it = std::find(c.begin(), c.end(), com);
223 if (it != c.end()) { c.erase(it); }
224 }
225 else { BL_LOG_ERROR << "Invalid orphan entity: " << orphan; }
226 }
