165 Light3D<T> createNew(Scene3DLighting* owner, TArgs&&... args) {
166 if (!idAllocator.available()) {
167 BL_LOG_ERROR << "Exceeded max light count";
168 return {owner, lights, util::Random::get<std::size_t>(0, lights.size() - 1)};
169 }
170 const std::size_t i = idAllocator.allocate();
171 new (&lights[i]) T(std::forward<TArgs>(args)...);
172 active.emplace_back(i);

174 }

178 if (i < lights.size()) {
179 idAllocator.release(i);
180 for (auto it = active.begin(); it != active.end(); ++it) {
181 if (*it == i) {

185 }
186 }
187 }
