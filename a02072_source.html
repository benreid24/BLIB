53MaterialPipelineSettings&& MaterialPipelineSettings::build() {
54 const auto validatePipeline = [this](const PipelineInfo& info) {
55 if (info.pipelineParams != nullptr) {
56 if (info.id != cfg::PipelineIds::None) {
57 BL_LOG_CRITICAL
58 << "Material pipeline cannot have both a pipeline id and parameters";
59 return false;
60 }
61 if (info.overrideBehavior != PhasePipelineOverride::NoOverride) {
62 BL_LOG_CRITICAL << "Material pipeline cannot have both an override and parameters";
63 return false;
64 }
65
66 info.pipelineParams->build();
67 }
68
69 if (info.id != cfg::PipelineIds::None) {
70 if (info.overrideBehavior != PhasePipelineOverride::NoOverride) {
71 BL_LOG_CRITICAL << "Material pipeline cannot have both an override and id";
72 return false;
73 }
74 if (info.pipelineParams != nullptr) {
75 BL_LOG_CRITICAL
76 << "Material pipeline cannot have both a pipeline id and parameters";
77 return false;
78 }
79 }
80
81 if (info.overrideBehavior != PhasePipelineOverride::NoOverride) {
82 if (info.pipelineParams != nullptr) {
83 BL_LOG_CRITICAL << "Material pipeline cannot have both an override behavior and "
84 "pipeline parameters";
85 return false;
86 }
87 if (info.id != cfg::PipelineIds::None) {
88 BL_LOG_CRITICAL
89 << "Material pipeline cannot have both an override behavior and pipeline id";
90 return false;
91 }
92 }
93
94 return true;
95 };
96
97 if (mainPipeline.pipelineParams == nullptr && mainPipeline.id == cfg::PipelineIds::None &&
98 mainPipeline.overrideBehavior == PhasePipelineOverride::NoOverride) {
99 BL_LOG_CRITICAL << "Material pipeline is missing a valid pipeline";
100 throw std::runtime_error("Invalid material pipeline settings");
101 }
102
103 if (!validatePipeline(mainPipeline)) {
104 throw std::runtime_error("Invalid material pipeline settings");
105 }
106
107 for (unsigned int i = 0; i < 8; ++i) {
108 if (!validatePipeline(renderPhaseOverrides[i])) {
109 throw std::runtime_error("Invalid material pipeline settings for phase: " +
110 std::to_string(0x1 << i));
111 }
112 }
113
114 return std::move(*this);
115}
