18vk::RenderPass& RenderPassCache::createRenderPass(std::uint32_t id,
19 vk::RenderPassParameters&& sceneParams) {
20 const auto insertResult =
21 cache.try_emplace(id, id, renderer, std::forward<vk::RenderPassParameters>(sceneParams));
22 if (!insertResult.second) {
23 BL_LOG_WARN << "Duplicate creation of render pass with id: " << id;
24 }
25 return insertResult.first->second;
26}

28vk::RenderPass& RenderPassCache::getRenderPass(std::uint32_t id) {
29 auto it = cache.find(id);
30 if (it == cache.end()) {
31 BL_LOG_CRITICAL << "Failed to find render pass with id: " << id;
32 throw std::runtime_error("Failed to find gfx pass");
33 }
34 return it->second;
35}

37vk::RenderPass* RenderPassCache::getRenderPassMaybe(std::uint32_t id) {
38 auto it = cache.find(id);
39 if (it == cache.end()) { return nullptr; }
40 return &it->second;
41}
