23 std::uint32_t skinnedMaterial, std::uint32_t unskinnedMaterial) {
24 if (model->getNodes().size() == 0) { return false; }
25
26 ecs = &world.engine().ecs();

36 skeleton->bones.resize(model->getBones().numBones(), {});

67void ModelSkeletal::processNode(engine::World& world, Tx& tx,
68 std::uint32_t skinnedMaterialPipelineId,
69 std::uint32_t nonSkinnedMaterialPipelineId, com::Skeleton& skeleton,
70 const resource::Ref<mdl::Model>& model, const mdl::Node& node,
71 ecs::Entity parentEntity) {
72 ecs::Entity nodeEntity = world.createEntity(tx);
73 world.engine().ecs().setEntityParent(nodeEntity, parentEntity, tx);
74
75 // Create transform on node entity and initialize to bind pose
76 com::Transform3D* transform =
77 world.engine().ecs().emplaceComponentWithTx<com::Transform3D>(nodeEntity, tx);
78 transform->setTransform(node.getTransform());
79
80 // create the bone component and setup links if this node is a bone
81 if (node.getBoneIndex().has_value()) {
82 com::Bone* bone = world.engine().ecs().emplaceComponentWithTx<com::Bone>(nodeEntity, tx);
83 bone->nodeBindPoseLocal = node.getTransform();
84 bone->transform = transform;
85 bone->boneIndex = node.getBoneIndex().value();
86 bone->boneOffset = model->getBones().getBone(bone->boneIndex).transform;
87 skeleton.bones[bone->boneIndex].transform = transform;
88 skeleton.bones[bone->boneIndex].bone = bone;
89 }
90
91 // create entities per mesh
92 for (const std::uint32_t meshIndex : node.getMeshes()) {
93 const mdl::Mesh& mesh = model->getMeshes().getMesh(meshIndex);
94 ecs::Entity meshEntity = ecs::InvalidEntity;
95
96 // try to put one mesh on root entity
97 if (!createdMeshOnSelf && mesh.getIsSkinned()) {
98 meshEntity = entity();
99 createdMeshOnSelf = true;
100 }
101 else {
102 meshEntity = world.createEntity(tx);
103
104 // mesh transforms are always identity
105 world.engine().ecs().emplaceComponentWithTx<com::Transform3D>(meshEntity, tx);
106 }
107
108 // mesh material
109 const std::uint32_t materialId =
110 mesh.getIsSkinned() ? skinnedMaterialPipelineId : nonSkinnedMaterialPipelineId;
111 auto mat = world.engine().renderer().materialPool().getOrCreateFromModelMaterial(
112 model->getMaterials().getMaterial(mesh.getMaterialIndex()));
113
114 // skinned meshes are direct children of the root node. bone transforms provide the
115 // accumulated node transforms
116 if (mesh.getIsSkinned()) {
117 if (meshEntity != entity()) {
118 world.engine().ecs().setEntityParent(meshEntity, entity(), tx);
119
120 auto* meshComponent =
121 world.engine().ecs().emplaceComponentWithTx<com::SkinnedMesh>(meshEntity, tx);
122 meshComponent->create(world.engine().renderer(), mesh);
123
124 children.emplace_back(meshEntity, meshComponent);
125 }
126 else { // add mesh to root
127 component().create(world.engine().renderer(), mesh);
128 auto* matInstance =
129 world.engine().ecs().emplaceComponentWithTx<com::MaterialInstance>(
130 meshEntity, tx, world.engine().renderer(), component(), materialId, mat);
131 component().init(matInstance);
132 }
133 world.engine().ecs().emplaceComponentWithTx<com::SkeletonIndexLink>(meshEntity, tx);
134 }
135 else {
136 world.engine().ecs().setEntityParent(meshEntity, nodeEntity, tx);
137
138 auto* meshComponent =
139 world.engine().ecs().emplaceComponentWithTx<com::BasicMesh>(meshEntity, tx);
140 meshComponent->create(world.engine().renderer(), mesh);
141
142 children.emplace_back(meshEntity, meshComponent);
143 }
144
145 // create material instance on the mesh (root gets it from Drawable::createWithMaterial
146 if (meshEntity != entity()) {
147 std::visit(
148 [&world, &tx, &mat, meshEntity, materialId](auto& meshComponent) {
149 auto* matInstance =
150 world.engine().ecs().emplaceComponentWithTx<com::MaterialInstance>(
151 meshEntity,
152 tx,
153 world.engine().renderer(),
154 *meshComponent,
155 materialId,
156 mat);
157 meshComponent->init(matInstance);
158 },
159 children.back().mesh);
160 }
161 }
162
163 // create entities recursively for child nodes
164 for (std::uint32_t ci : node.getChildren()) {
165 const mdl::Node& childNode = model->getNodes().getNode(ci);
166 processNode(world,
167 tx,
168 skinnedMaterialPipelineId,
169 nonSkinnedMaterialPipelineId,
170 skeleton,
171 model,
172 childNode,
173 nodeEntity);
174 }
