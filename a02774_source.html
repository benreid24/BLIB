110Ref Repository::createAssetShared(std::string_view type, const std::string& name,
111 const CreateContext::CreateData& createData, bool sync) {
112 if (!getDriver(type)) {
113 BL_LOG_ERROR << "Attempted to create asset with type " << type
114 << " but no driver is registered for that type";
115 return Ref();
116 }
117
118 std::unique_lock assetLock(assetMutex);
119 util::UUID uuid = util::UUID::generate();
120 auto it = assets.try_emplace(uuid, *this).first;
121 it->second.getMetadata().setDisplayName(name);
122 it->second.uuid = uuid;
123 it->second.type = type;
124 if (!it->second.create(createData)) {
125 BL_LOG_ERROR << "Failed to create new asset '" << name << "' of type " << type;
126 util::FileUtil::deleteDirectory(EditorPaths::getAssetPath(assetDirectory, it->second));
127 assets.erase(it);
128 return Ref();
129 }
130
131 if (!createData.getPath().empty()) {
132 std::filesystem::path source = createData.getPath();
133 if (util::FileUtil::isSubpath(createData.getPath())) {
134 source = std::filesystem::proximate(createData.getPath());
135 }
136 else {
137 BL_LOG_WARN << "Source file path '" << createData.getPath()
138 << "' is not a subpath of the current working directory. Storing absolute "
139 "path in asset metadata, which may cause issues if the asset is moved";
140 }
141 it->second.getMetadata().setSourceFileInfo(source.generic_string());
142 }
143
144 if (mode == Mode::Editor && sync) {
145 it->second.markReadyForAutoSync();
146
147 if (!it->second.saveEditor()) {
148 BL_LOG_WARN << "Failed to save newly created asset '" << name << "' of type " << type
149 << " to disk";
150 }
151 if (!writeManifestLocked()) {
152 BL_LOG_WARN << "Failed to save asset manifest after creating new asset '" << name
153 << "' of type " << type;
154 }
155 }
156
157 return Ref(this, &it->second);
158}

160Ref Repository::getOrCreateAsset(std::string_view key, std::string_view type,
161 const std::string& name,
162 const CreateContext::CreateData& createData) {
163 std::string keyStr(key);
164
165 std::unique_lock lock(assetMutex);

167 if (it != keyToAsset.end()) {
168 auto asset = getAsset(it->second, State::Loaded);
169 if (asset) { return asset; }

172 }
173

197Ref Repository::getAssetFromSourcePath(std::string_view type, const std::string& path,
198 State desiredState) {
199 std::unique_lock lock(sourceLinkMutex);
200
201 auto& linksByType = sourceLinks[std::string(type)];
202 auto it = linksByType.find(path);
203 if (it == linksByType.end()) {
204 Ref ref = createAssetShared(type, path, CreateContext::CreateData(path), false);
205 if (!ref) { return ref; }
206 linksByType.try_emplace(path, ref->getUUID(), path, type);
207 ref->getMetadata().setPath(makeStaticPath(type));
208 ref->markReadyForAutoSync();
209 if (mode == Mode::Editor) {
210 if (!writeManifestLocked()) {
211 BL_LOG_WARN
212 << "Failed to save asset manifest after creating static asset with path "
213 << path << " and type " << type;
214 }
215 if (!ref.getAsset().saveEditor()) {
216 BL_LOG_WARN << "Failed to save newly created static asset '" << path << "' of type "
217 << type << " to disk";
218 }
219 }
220 return ref;
221 }
222 else {
223 if (it->second.type != type) {
224 BL_LOG_ERROR << "Attempted to get asset with path " << path << " and type " << type
225 << " but an asset with that path exists with type " << it->second.type;
226 return Ref();
227 }
228 return getAsset(it->second.uuid, desiredState);

353bool Repository::loadRepository() {
354 std::unique_lock lock(assetMutex);
355 std::unique_lock unloadLock(unloadQueueMutex);
356
357 assets.clear();
358 unloadQueue.clear();
359
360 if (mode == Mode::Editor) {
361 const std::string manifestPath = util::FileUtil::joinPath(assetDirectory, "manifest.json");
362 stream::InputStream manifestFile(manifestPath);
363 if (!manifestFile.isValid()) {
364 BL_LOG_ERROR << "Failed to open asset manifest at " << manifestPath;
365 return false;
366 }
367
368 EditorRuntimeData data{
369 .assets = &assets,
370 .sourceLinks = &sourceLinks,
371 .keyToAsset = &keyToAsset,
372 };
373 if (!serial::json::Serializer<EditorRuntimeData>::deserializeStream(manifestFile, data)) {
374 BL_LOG_ERROR << "Failed to deserialize asset manifest at " << manifestPath;
375 return false;
376 }
377
378 // load asset metadata from filesystem to find missing or moved assets
379 std::unordered_map<std::string, std::string> foundAssetPaths;
380 const std::vector<std::string> metadataFiles = util::FileUtil::listDirectory(
381 assetDirectory, std::string(EditorPaths::MetadataExtension), true);
382 for (const auto& path : metadataFiles) {
383 const std::string uuid = util::FileUtil::getBaseName(path);
384 foundAssetPaths[uuid] = path;
385 }
386
387 // find and relink moved assets
388 for (auto& pair : assets) {
389 const std::string assetPath =
390 EditorPaths::getAssetMetadataFilePath(assetDirectory, pair.second);
391 if (!util::FileUtil::exists(assetPath)) {
392 BL_LOG_WARN << "Asset with UUID " << pair.first.toString()
393 << " is missing its metadata file at expected path " << assetPath;
394
395 // see if our missing asset was moved or deleted
396 const auto it = foundAssetPaths.find(pair.first.toString());
397 if (it != foundAssetPaths.end()) {
398 BL_LOG_INFO << "Found metadata file for asset with UUID "
399 << pair.first.toString() << " at " << it->second
400 << ". Updating asset path and display name";
401 const std::string assetFolder = util::FileUtil::getFolder(it->second);
402 auto parsed = EditorPaths::parseAssetFolderName(assetFolder);
403 const std::string displayName = parsed ? parsed->second : "RecoveredAsset";
404 std::string folder = it->second;
405 // remove folder + filename + assetDirectory
406 const std::size_t rmStart = assetDirectory.size() + 1;
407 const std::size_t rmSize = rmStart + assetFolder.size() + 1 +
408 EditorPaths::MetadataExtension.size() +
409 util::UUID::StringLength;
410 folder = folder.substr(rmStart, folder.size() - rmSize);
411
412 if (!folder.empty()) {
413 if (folder.front() == '/' || folder.front() == '\\') { folder.erase(0, 1); }
414 }
415 if (!folder.empty()) {
416 if (folder.back() == '/' || folder.back() == '\\') { folder.pop_back(); }
417 }
418
419 pair.second.getMetadata().setDisplayName(displayName);
420 pair.second.getMetadata().setPath(folder);
421 }
422 else {
423 BL_LOG_ERROR << "Asset with UUID " << pair.first.toString()
424 << " is missing its metadata file and could not be found in "
425 "asset directory";
426 pair.second.state = State::Missing;
427 }
428 }
429 }
430
431 // search discovered assets for any unregistered and add them to manifest
432 for (const auto& pair : foundAssetPaths) {
433 const util::UUID uuid(pair.first);
434 if (assets.find(uuid) == assets.end()) {
435 BL_LOG_INFO << "Found unregistered asset with UUID " << pair.first << " at "
436 << pair.second << ". Adding to repository manifest";
437
438 // deserialize metadata file
439 stream::InputStream assetFile(pair.second);
440 if (!assetFile.isValid()) {
441 BL_LOG_ERROR << "Failed to open asset file at " << pair.second;
442 continue;
443 }
444
445 // load asset info
446 auto it = assets.try_emplace(uuid, *this).first;
447 if (!serial::json::Serializer<Asset>::deserializeStream(assetFile, it->second)) {
448 BL_LOG_ERROR << "Failed to deserialize asset metadata at " << pair.second;
449 assets.erase(it);
450 continue;
451 }
452
453 // validate
454 if (it->second.getUUID() != uuid) {
455 BL_LOG_ERROR << "Deserialized asset metadata at " << pair.second
456 << " has mismatched UUID " << it->second.getUUID().toString()
457 << " (expected " << uuid.toString() << ")";
458 assets.erase(it);
459 continue;
460 }
461 }
462 }
463 }
464 else {
465 const std::string manifestPath =
466 util::FileUtil::joinPath(assetDirectory, std::string(bdl::RuntimePaths::ManifestPath));
467 stream::InputStream manifestFile(manifestPath);
468 if (!manifestFile.isValid()) {
469 BL_LOG_ERROR << "Failed to open asset manifest at " << manifestPath;
470 return false;
471 }
472
473 GameRuntimeData data{.assets = &assets,
474 .keyToAsset = &keyToAsset,
475 .bundleManifest = &bundleRuntime.manifest};
476 if (!serial::binary::Serializer<GameRuntimeData>::deserialize(manifestFile, data)) {
477 BL_LOG_ERROR << "Failed to read asset manifest file";
478 return false;
479 }
480
481 streamCache.useBundles(&bundleRuntime);
482 }
483
484 // auto load assets that are marked as such
485 for (auto& pair : assets) {
486 const bool hasAllDeps = pair.second.initAfterDeserialize(*this);
487 if (pair.second.getMetadata().getIsAutoLoaded()) {
488 if (hasAllDeps) {
489 if (!pair.second.load()) {
490 BL_LOG_ERROR << "Failed to auto load auto-load asset with UUID "
491 << pair.first.toString();
492 }
493 }
494 else {
495 BL_LOG_ERROR << "Skipping auto-load of asset " << pair.first.toString()
496 << " due to missing dependencies";
497 }
498 }
499 }
500
501 // recheck source files on load in editor mode
502 if (mode == Mode::Editor) {
503 for (auto& pair : assets) {
504 Asset& asset = pair.second;
505 if (asset.getState() < State::Unloaded) { continue; }
506 const bool wasLoaded = asset.getState() == State::Loaded;
507 if (!asset.reloadFromSource()) {
508 BL_LOG_WARN << "Failed to reload asset with UUID " << pair.first.toString()
509 << " from source";
510 }
511 if (!wasLoaded) { asset.unload(); }
512 }
513 }
514 else { bundleRuntime.performPostMountAutoload(); } // autoload from bundles of autoload assets
515
516 return true;
517}

525bool Repository::exportRepository(const std::string& path) {
526 std::unique_lock assetLock(assetMutex);
527 std::unique_lock driverLock(driverMutex);
528
529 // create bundle directory
530 if (!util::FileUtil::createDirectory(path)) {
531 BL_LOG_ERROR << "Failed to create bundle directory at " << path;
532 return false;
533 }
534
535 // build set of root assets
536 std::unordered_set<util::UUID> roots;
537 for (const auto& asset : assets) {
538 for (const auto& dep : asset.second.dependencies) { roots.erase(dep.uuid); }
539 roots.emplace(asset.second.getUUID());
540 }
541
542 // second pass to eliminate middle of the tree assets
543 for (auto possibleRoot : roots) {
544 const auto it = assets.find(possibleRoot);
545 for (const auto& dep : it->second.dependencies) { roots.erase(dep.uuid); }
546 }
547
548 constexpr std::size_t BundleSoftSize = 10485760; // 10M
549 bdl::Manifest manifest;
550 std::unordered_set<util::UUID> bundled;
551 std::queue<util::UUID> toVisit;
552 for (auto root : roots) { toVisit.emplace(root); }
553
554 const auto addToBundle = [this, &manifest](bdl::BundleData& bundle, Asset& asset) -> bool {
555 WriteContext context(*this, asset, bundle);
556 detail::DriverBase* driver = getDriverLocked(asset.getType());
557 if (!driver) {
558 BL_LOG_ERROR << "Failed to find driver for asset with type " << asset.getType()
559 << " while adding to bundle";
560 return false;
561 }
562 if (!driver->write(context)) { return false; }
563 manifest.assetToBundle[asset.getUUID()] = bundle.uuid;
564 if (driver->getBundleConfig().onMount == bdl::AssetBundleConfig::OnMount::AutoLoad) {
565 bundle.autoLoadAssets.emplace_back(asset.getUUID());
566 }
567 return true;
568 };
569
570 // add root trees to export based on affinity
571 bdl::BundleData currentBundle;
572 while (!toVisit.empty()) {
573 util::UUID uuid = toVisit.front();
574 toVisit.pop();
575
576 Asset& asset = assets[uuid];
577 detail::DriverBase* driver = getDriverLocked(asset.getType());
578 if (!driver) {
579 BL_LOG_ERROR << "Failed to find driver for asset with type " << asset.getType()
580 << " while adding to bundle";
581 continue;
582 }
583 if (driver->getBundleConfig().affinity != bdl::AssetBundleConfig::Affinity::Parent) {
584 continue;
585 }
586
587 bundled.emplace(uuid);
588
589 // check if we should flush if next asset is a root
590 if (roots.find(asset.getUUID()) != roots.end()) {
591 if (currentBundle.data.size() >= BundleSoftSize) {
592 if (!currentBundle.flush(path)) {
593 BL_LOG_ERROR << "Failed to flush bundle";
594 return false;
595 }
596 currentBundle.reset();
597 }
598 }
599
600 if (!addToBundle(currentBundle, asset)) {
601 BL_LOG_ERROR << "Failed to bundle asset " << asset.getUUID();
602 }
603
604 for (auto dep : asset.dependencies) {
605 if (bundled.find(dep.uuid) != bundled.end()) { continue; }
606 toVisit.emplace(dep.uuid);
607 }
608 }
609
610 // final flush
611 if (!currentBundle.data.empty()) {
612 if (!currentBundle.flush(path)) {
613 BL_LOG_ERROR << "Failed to flush bundle";
614 return false;
615 }
616 currentBundle.reset();
617 }
618
619 // bundle remaining assets by type
620 std::unordered_map<std::string_view, bdl::BundleData> bundlesByType;
621 for (auto& asset : assets) {
622 if (bundled.find(asset.second.getUUID()) != bundled.end()) { continue; }
623 bundled.emplace(asset.second.getUUID());
624
625 detail::DriverBase* driver = getDriverLocked(asset.second.getType());
626 if (!driver) {
627 BL_LOG_ERROR << "Failed to find driver for asset with type " << asset.second.getType()
628 << " while adding to bundle";
629 continue;
630 }
631 bdl::BundleData& bundleData = bundlesByType[driver->getSupportedType()];
632 if (!addToBundle(bundleData, asset.second)) {
633 BL_LOG_ERROR << "Failed to bundle asset " << asset.second.getUUID();
634 }
635
636 // flush if necessary
637 if (bundleData.data.size() >= BundleSoftSize) {
638 if (!bundleData.flush(path)) {
639 BL_LOG_ERROR << "Failed to flush bundle";
640 return false;
641 }
642 bundleData.reset();
643 }
644 }
645
646 // final flush bundles
647 for (auto& bundle : bundlesByType) {
648 if (!bundle.second.data.empty()) {
649 if (!bundle.second.flush(path)) {
650 BL_LOG_ERROR << "Failed to flush bundle";
651 return false;
652 }
653 bundle.second.reset();
654 }
655 }
656
657 // write manifest & metadata
658 GameRuntimeData manifestData{
659 .assets = &assets, .keyToAsset = &keyToAsset, .bundleManifest = &manifest};
660 stream::OutputStream output(
661 util::FileUtil::joinPath(path, std::string(bdl::RuntimePaths::ManifestPath)));
662 if (!output.isValid()) {
663 BL_LOG_ERROR << "Failed to open manifest file for writing";
664 return false;
665 }
666 return serial::binary::Serializer<GameRuntimeData>::serialize(output, manifestData);
667}

669void Repository::releaseUnused() {
670 std::unique_lock lock(assetMutex);
671 std::unique_lock unloadLock(unloadQueueMutex);
672
673 for (const util::UUID& uuid : unloadQueue) {
674 auto it = assets.find(uuid);
675 if (it == assets.end()) {
676 BL_LOG_ERROR << "Attempted to unload asset with UUID " << uuid.toString()
677 << " but it does not exist";
678 continue;
679 }
