87bool ComboBox::propagateEvent(const Event& event) {
88 const bool contained = labelRegion.contains(event.mousePosition());
89 if (contained && opened) {
90 bool updated = false;
91
92 // determine which option the event was on
93 const auto findEventOption = [this, &event]() -> int {
94 const sf::Vector2f translated(event.mousePosition().x,
95 event.mousePosition().y + scroll);
96 sf::FloatRect test(labelRegion.left, labelRegion.top, labelSize.x, labelSize.y);
97
98 for (unsigned int i = 0; i < options.size(); ++i) {
99 if (test.contains(translated)) { return i; }
100 test.top += labelSize.y;
101 }
102 return -1;
103 };
104
105 // handle event
106 switch (event.type()) {
107 case Event::Scrolled:
108 scrolled(event);
109 moused = findEventOption();
110 updated = true;
111 break;
112
113 case Event::LeftMousePressed:
114 setSelectedOption(findEventOption());
115 updated = false;
116 break;
117
118 case Event::MouseMoved: {
119 const int priorMoused = moused;
120 moused = findEventOption();
121 updated = moused != priorMoused;
122 } break;
123 }
124
125 if (updated && getComponent()) { getComponent()->onElementUpdated(); }
126 return true;
127 }
128 if (opened && !contained && event.type() == Event::LeftMousePressed) {
129 opened = false;
130 if (getComponent()) { getComponent()->onElementUpdated(); }
131 return true; // eat it
132 }
133
134 return false;
135}
