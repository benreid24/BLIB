9std::string PrimitiveValue::typeToString(Type t) {
10 switch (t) {
11 case TVoid:
12 return "Void";
13 case TBool:
14 return "Bool";
15 case TInteger:
16 return "Integer";
17 case TFloat:
18 return "Float";
19 case TString:
20 return "String";
21 case TArray:
22 return "Array";
23 case TFunction:
24 return "Function";
25 case TRef:
26 return "Reference";
27 default: {
28 std::string ret = "(";
29 for (unsigned int i = 0; i < Type::_TYPE_COUNT; ++i) {
30 const Type v = static_cast<Type>(0x1 << i);
31 if ((t & v) != 0) { ret += typeToString(v) + ", "; }
32 }
33 ret.pop_back();
34 ret.back() = ')';
35 return ret;
36 }
37 }

115bool PrimitiveValue::getAsBool() const {
116 switch (type) {
117 case TVoid:
118 return false;
119 case TBool:
120 return *std::get_if<bool>(&value);
121 case TInteger:

123 case TFloat:

125 case TString:
126 return !getAsString().empty();
127 case TArray:
128 return !getAsArray().empty();
129 case TRef:
130 return getAsRef().deref().value().getAsBool();
131 default:
