95 int scorePhysicalDevice(VkPhysicalDevice device, const VkPhysicalDeviceProperties& deviceProperties,
96  const VkPhysicalDeviceFeatures& deviceFeatures,
97  const QueueFamilyLocator& queueFamilies, VkSurfaceKHR surface) {
