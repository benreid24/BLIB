80bool deviceHasRequiredExtensions(VkPhysicalDevice device, const char* name) {
81 std::uint32_t extensionCount;
82 vkCheck(vkEnumerateDeviceExtensionProperties(device, nullptr, &extensionCount, nullptr));
83
84 std::vector<VkExtensionProperties> availableExtensions(extensionCount);
85 vkCheck(vkEnumerateDeviceExtensionProperties(
86 device, nullptr, &extensionCount, availableExtensions.data()));
87
88 for (const char* ext : RequiredDeviceExtensions) {
89 if (std::find_if(availableExtensions.begin(),
90 availableExtensions.end(),
91 [ext](const VkExtensionProperties& c) {
92 return strcmp(c.extensionName, ext) == 0;
93 }) == availableExtensions.end()) {
94 BL_LOG_WARN << name << " is missing required extension " << ext;
95 return false;
96 }
97 }
98
99 return true;
100}

102int scoreDeviceForOptionalExtensions(VkPhysicalDevice device, const char* name) {
103 std::uint32_t extensionCount;

105
106 std::vector<VkExtensionProperties> availableExtensions(extensionCount);
107 vkCheck(vkEnumerateDeviceExtensionProperties(

109
110 int score = 0;
111 for (const char* ext : OptionalDeviceExtensions) {
112 if (std::find_if(availableExtensions.begin(),

115 return strcmp(c.extensionName, ext) == 0;

118 }
119 else { score += 500; }
120 }
121

138int scorePhysicalDevice(VkPhysicalDevice device, const VkPhysicalDeviceProperties& deviceProperties,
139 const VkPhysicalDeviceFeatures& deviceFeatures,
140 const QueueFamilyLocator& queueFamilies, VkSurfaceKHR surface) {
141 // requirements
142 if (!deviceFeatures.geometryShader) { return -1; }
143 if (!deviceFeatures.samplerAnisotropy) { return -1; }
144 if (!queueFamilies.complete()) { return -1; }
145 if (!deviceHasRequiredExtensions(device, deviceProperties.deviceName)) { return -1; }
146 if (!deviceSupportsAtomicAdd(device)) { return -1; }
147 SwapChainSupportDetails swapChainDetails(device, surface);
148 if (swapChainDetails.formats.empty() || swapChainDetails.presentModes.empty()) { return -1; }
149
150 // preferred optional features
151 int score = 0;
152 score += deviceProperties.limits.maxImageDimension2D;
153 if (deviceProperties.deviceType == VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU) { score += 50000; }
154 score += scoreDeviceForOptionalExtensions(device, deviceProperties.deviceName);
155
156 return score;
157}

203void VulkanLayer::createInstance() {
204 // app info
205 const std::string appName =
206 engine::Configuration::getOrDefault<std::string>("blib.app.name", "BLIB Application");
207 VkApplicationInfo appInfo{};
208 appInfo.sType = VK_STRUCTURE_TYPE_APPLICATION_INFO;
209 appInfo.pApplicationName = appName.c_str();
210 appInfo.applicationVersion = VK_MAKE_VERSION(1, 0, 0);
211 appInfo.pEngineName = "BLIB";
212 appInfo.engineVersion = VK_MAKE_VERSION(3, 0, 0);
213 appInfo.apiVersion = VK_API_VERSION_1_3;
214
215 // instance data to populate
216 VkInstanceCreateInfo createInfo{};
217 createInfo.sType = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO;
218 createInfo.pApplicationInfo = &appInfo;
219
220// validation layers
221#ifdef BLIB_DEBUG
222 std::vector<VkLayerProperties> validLayers;
223 std::vector<const char*> validationLayers;
224 std::uint32_t layerCount = 0;
225 if (vkEnumerateInstanceLayerProperties(&layerCount, 0) != VK_SUCCESS) {
226 throw std::runtime_error("Failed to enumerate validation layers");
227 }
228 validLayers.resize(layerCount);
229 if (vkEnumerateInstanceLayerProperties(&layerCount, validLayers.data()) != VK_SUCCESS) {
230 throw std::runtime_error("Failed to retrieve validation layers");
231 }
232 validationLayers.reserve(RequestedValidationLayers.size());
233 for (const VkLayerProperties& layer : validLayers) {
234 if (RequestedValidationLayers.find(layer.layerName) != RequestedValidationLayers.end()) {
235 validationLayers.emplace_back(layer.layerName);
236 }
237 }
238 createInfo.enabledLayerCount = validationLayers.size();
239 createInfo.ppEnabledLayerNames = validationLayers.data();
240#else
241 createInfo.enabledLayerCount = 0;
242#endif
243
244 // query valid extensions
245 std::uint32_t extensionCount = 0;
246 vkCheck(vkEnumerateInstanceExtensionProperties(nullptr, &extensionCount, nullptr));
247 std::vector<VkExtensionProperties> extensions(extensionCount);
248 vkCheck(vkEnumerateInstanceExtensionProperties(nullptr, &extensionCount, extensions.data()));
249 std::unordered_set<std::string> extensionsSet;
250 for (const VkExtensionProperties& ext : extensions) {
251 extensionsSet.emplace(ext.extensionName);
252 }
253
254 // required extensions
255 std::vector<const char*> requiredExtentions = RenderWindow::getRequiredInstanceExtensions();
256#ifdef BLIB_DEBUG
257 requiredExtentions.emplace_back(VK_EXT_DEBUG_UTILS_EXTENSION_NAME);
258 requiredExtentions.emplace_back(VK_EXT_DEBUG_REPORT_EXTENSION_NAME);
259#endif
260
261 bool missing = false;
262 for (const char* ext : requiredExtentions) {
263 if (extensionsSet.find(ext) == extensionsSet.end()) {
264 BL_LOG_ERROR << "Missing required extension: " << ext;
265 missing = true;
266 }
267 }
268 if (missing) { throw std::runtime_error("Missing required extension"); }
269
270 createInfo.enabledExtensionCount = requiredExtentions.size();
271 createInfo.ppEnabledExtensionNames = requiredExtentions.data();
272
273#ifdef BLIB_MACOS
274 createInfo.flags |= VK_INSTANCE_CREATE_ENUMERATE_PORTABILITY_BIT_KHR;
275#endif
276
277 // actual instance creation
278 const VkResult result = vkCreateInstance(&createInfo, nullptr, &instance);
279 if (result != VK_SUCCESS) {
280 BL_LOG_ERROR << "Got instance creation error code: " << result;
281 throw std::runtime_error("Failed to create Vulkan instance");
282 }
283}

319void VulkanLayer::pickPhysicalDevice() {
320 std::uint32_t deviceCount = 0;
321 vkCheck(vkEnumeratePhysicalDevices(instance, &deviceCount, nullptr));
322 if (deviceCount == 0) { throw std::runtime_error("Failed to find GPUs with Vulkan support"); }
323 std::vector<VkPhysicalDevice> devices(deviceCount);
324 vkCheck(vkEnumeratePhysicalDevices(instance, &deviceCount, devices.data()));
325
326 int pscore = -1;
327 for (const auto& device : devices) {
328 VkPhysicalDeviceProperties deviceProperties;
329 VkPhysicalDeviceFeatures deviceFeatures;
330 vkGetPhysicalDeviceProperties(device, &deviceProperties);
331 vkGetPhysicalDeviceFeatures(device, &deviceFeatures);
332
333 QueueFamilyLocator queueFamilies;
334 queueFamilies.populate(device, surface);
335
336 const int s =
337 scorePhysicalDevice(device, deviceProperties, deviceFeatures, queueFamilies, surface);
338 if (s >= pscore) {
339 physicalDevice = device;
340 pscore = s;
341 }
342 }
343
344 if (pscore < 0 || physicalDevice == VK_NULL_HANDLE) {
345 throw std::runtime_error("Failed to find a suitable GPU");
346 }
347
348 vkGetPhysicalDeviceFeatures(physicalDevice, &physicalDeviceFeatures);
349 vkGetPhysicalDeviceProperties(physicalDevice, &physicalDeviceProperties);
350
351 std::uint32_t extensionCount = 0;
352 vkCheck(
353 vkEnumerateDeviceExtensionProperties(physicalDevice, nullptr, &extensionCount, nullptr));
354 availableExtensions.resize(extensionCount);
355 vkCheck(vkEnumerateDeviceExtensionProperties(
356 physicalDevice, nullptr, &extensionCount, availableExtensions.data()));
357}

359void VulkanLayer::createLogicalDevice() {
360 QueueFamilyLocator indices;
361 indices.populate(physicalDevice, surface);
362
363 // queue(s) info
364 std::vector<VkDeviceQueueCreateInfo> queueCreateInfos;
365 std::set<std::uint32_t> uniqueQueueFamilies = {indices.graphicsFamily.value(),
366 indices.presentFamily.value()};
367 const float queuePriorities[] = {1.f};
368 queueCreateInfos.reserve(uniqueQueueFamilies.size());
369 for (std::uint32_t queueIndex : uniqueQueueFamilies) {
370 queueCreateInfos.emplace_back();
371 VkDeviceQueueCreateInfo& queueCreateInfo = queueCreateInfos.back();
372 queueCreateInfo.sType = VK_STRUCTURE_TYPE_DEVICE_QUEUE_CREATE_INFO;
373 queueCreateInfo.queueFamilyIndex = queueIndex;
374 queueCreateInfo.queueCount = 1;
375 queueCreateInfo.pQueuePriorities = queuePriorities;
376 }
377
378 // required features
379 VkPhysicalDeviceFeatures deviceFeatures{};
380 deviceFeatures.geometryShader = VK_TRUE; // TODO - disable point shadows instead of requiring?
381 deviceFeatures.samplerAnisotropy = VK_TRUE;
382 deviceFeatures.wideLines =
383 physicalDeviceProperties.limits.lineWidthRange[1] > 1.f ? VK_TRUE : VK_FALSE;
384 deviceFeatures.sampleRateShading = physicalDeviceFeatures.sampleRateShading;
385
386 // extension features
387 VkPhysicalDeviceShaderAtomicFloatFeaturesEXT atomicFloatFeatures{};
388 atomicFloatFeatures.sType = VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_SHADER_ATOMIC_FLOAT_FEATURES_EXT;
389 atomicFloatFeatures.shaderBufferFloat32AtomicAdd = VK_TRUE;
390
391 VkDeviceCreateInfo createInfo{};
392 createInfo.sType = VK_STRUCTURE_TYPE_DEVICE_CREATE_INFO;
393 createInfo.pQueueCreateInfos = queueCreateInfos.data();
394 createInfo.queueCreateInfoCount = queueCreateInfos.size();
395 createInfo.pEnabledFeatures = &deviceFeatures;
396 createInfo.enabledExtensionCount = RequiredDeviceExtensions.size();
397 createInfo.ppEnabledExtensionNames = RequiredDeviceExtensions.data();
398 createInfo.enabledLayerCount = 0; // never used
399 createInfo.pNext = &atomicFloatFeatures;
400
401 if (vkCreateDevice(physicalDevice, &createInfo, nullptr, &device) != VK_SUCCESS) {
402 throw std::runtime_error("Failed to create logical device");
403 }
404
405 vkGetDeviceQueue(device, indices.graphicsFamily.value(), 0, &graphicsQueue);
406 vkGetDeviceQueue(device, indices.presentFamily.value(), 0, &presentQueue);
407}

472std::uint32_t VulkanLayer::findMemoryType(uint32_t typeFilter, VkMemoryPropertyFlags properties) {
473 VkPhysicalDeviceMemoryProperties memProperties;
474 vkGetPhysicalDeviceMemoryProperties(physicalDevice, &memProperties);
475 for (std::uint32_t i = 0; i < memProperties.memoryTypeCount; i++) {
476 if ((typeFilter & (1 << i)) &&
477 (memProperties.memoryTypes[i].propertyFlags & properties) == properties) {
478 return i;
479 }
480 }
481
482 throw std::runtime_error("failed to find suitable memory type!");
483}

485VkFormat VulkanLayer::findSupportedFormat(const std::initializer_list<VkFormat>& candidates,
486 VkImageTiling tiling, VkFormatFeatureFlags features) {
487 for (VkFormat format : candidates) {
488 VkFormatProperties props;
489 vkGetPhysicalDeviceFormatProperties(physicalDevice, format, &props);
490
491 if (tiling == VK_IMAGE_TILING_LINEAR &&
492 (props.linearTilingFeatures & features) == features) {
493 return format;
494 }
495 else if (tiling == VK_IMAGE_TILING_OPTIMAL &&
496 (props.optimalTilingFeatures & features) == features) {
497 return format;
498 }
499 }
500
501 throw std::runtime_error("failed to find supported format!");
502}

606 barrier.image = image;

608 barrier.subresourceRange.baseMipLevel = baseMipLevel;
609 barrier.subresourceRange.levelCount = mipLevelCount;
610 barrier.subresourceRange.baseArrayLayer = 0;
611 barrier.subresourceRange.layerCount = layerCount;
612
613 VkPipelineStageFlags sourceStage;
614 VkPipelineStageFlags destinationStage;
615
616 if (oldLayout == VK_IMAGE_LAYOUT_UNDEFINED &&
617 newLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL) {
618 barrier.srcAccessMask = 0;
619 barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
620
621 sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
622 destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
623 }
624 else if (newLayout == VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL) {
625 if (oldLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL) {
626 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
627 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
628
629 sourceStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
630 destinationStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
631 }
632 else {
633 barrier.srcAccessMask = 0;
634 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
635
636 sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
637 destinationStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
638 }
639 }
640 else if (newLayout == VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL) {
641 barrier.srcAccessMask = VK_ACCESS_SHADER_READ_BIT;
642 barrier.dstAccessMask = VK_ACCESS_MEMORY_READ_BIT;
643
644 sourceStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
645 destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
646 }
647 else if (newLayout == VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL &&
648 oldLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL) {
649 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
650 barrier.dstAccessMask = VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_READ_BIT |
651 VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_WRITE_BIT;
652 sourceStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
653 destinationStage =
654 VK_PIPELINE_STAGE_EARLY_FRAGMENT_TESTS_BIT | VK_PIPELINE_STAGE_LATE_FRAGMENT_TESTS_BIT;
655 }
656 else if (newLayout == VK_IMAGE_LAYOUT_PRESENT_SRC_KHR &&
657 oldLayout == VK_IMAGE_LAYOUT_UNDEFINED) {
658 barrier.srcAccessMask = 0;
659 barrier.dstAccessMask = VK_ACCESS_MEMORY_WRITE_BIT;
660 sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
661 destinationStage = VK_PIPELINE_STAGE_BOTTOM_OF_PIPE_BIT;
662 }
663 else { throw std::invalid_argument("Unsupported layout transition"); }
664
665 vkCmdPipelineBarrier(
666 commandBuffer, sourceStage, destinationStage, 0, 0, nullptr, 0, nullptr, 1, &barrier);
667}
