330T& Engine::getPlayer(unsigned int i) {
331 static_assert(std::is_base_of_v<Player, T>, "T must derive from Player");
332
333 if constexpr (std::is_same_v<Player, T>) {
334 if (players.empty() && i == 0) {
335 BL_LOG_INFO << "No player exists, creating default";
336 addPlayer();
337 }
338 }
339
340#ifdef BLIB_DEBUG
341 T* p = dynamic_cast<T*>(players[i].get());
342 if (!p) { BL_LOG_ERROR << "Bad player cast"; }
343#else
344 T* p = static_cast<T*>(players[i].get());
345#endif
346
347 return *p;
348}

350template<typename T, typename... TArgs>
351T& Engine::addPlayer(TArgs&&... args) {
352 static_assert(std::is_base_of_v<Player, T>, "T must derive from Player");
353 static_assert(
354 std::is_constructible_v<T, Engine&, rc::Observer*, input::Actor*, TArgs...>,
355 "T must be constructible with T(Engine&, rc::Observer*, input::Actor*, TArgs...)");
356
357 auto& observer = rendererInstance->addObserver();
358 auto& actor = input.addActor();
359 T* np = new T(*this, &observer, &actor, args...);
360 auto& player = players.emplace_back(np);
361 eventEmitter.emit<event::PlayerAdded>({*player});
362 return *np;
363}

365template<typename TWorld, typename... TArgs>
366ctr::Ref<World, TWorld> Engine::createWorld(TArgs&&... args) {
367 static_assert(std::is_base_of_v<World, TWorld>, "TWorld must derive from World");
368 static_assert(std::is_constructible_v<TWorld, Engine&, TArgs...>,
369 "TWorld constructor must take Engine& as first parameter");
370
371 auto ref = worldPool.emplaceDerived<TWorld, TArgs...>(*this, std::forward<TArgs>(args)...);
372 for (unsigned int i = 0; i < worlds.size(); ++i) {
373 if (!worlds[i].isValid()) {
374 worlds[i] = ref;
375 ref->index = i;
376 eventEmitter.emit<event::WorldCreated>({*ref});
377 return ref;
378 }
379 }
380
381 throw std::runtime_error("A maximum of 8 worlds may exist concurrently");
382}
