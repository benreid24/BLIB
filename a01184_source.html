326  engineSettings.windowParameters().videoMode().height));

328  engineSettings.windowParameters().initialViewSize() :
329  modeSize;
330

332  const float newHeight = static_cast<float>(resize.height);
333 
334  sf::FloatRect viewport(0.f, 0.f, 1.f, 1.f);
335  if (engineSettings.windowParameters().letterBox()) {
336  const float xScale = newWidth / ogSize.x;
337  const float yScale = newHeight / ogSize.y;
338 
339  if (xScale >= yScale) { // constrained by height, bars on sides
340  viewport.width = ogSize.x * yScale / newWidth;
341  viewport.left = (1.f - viewport.width) * 0.5f;
342  }
343  else { // constrained by width, bars on top and bottom
344  viewport.height = ogSize.y * xScale / newHeight;
345  viewport.top = (1.f - viewport.height) * 0.5f;
346  }
347  }
348

350  renderingSystem.processResize(sf::Rect<std::uint32_t>(newWidth * viewport.left,
351  newHeight * viewport.top,
352  newWidth * viewport.width,
353  newHeight * viewport.height));
354  }
355 
356  if (ss) {
357  Settings::WindowParameters params = engineSettings.windowParameters();
358  params.withVideoMode(
359  sf::VideoMode(resize.width, resize.height, params.videoMode().bitsPerPixel));
