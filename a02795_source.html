98void Sphere::makeSphere(float radius, int detailLevel, std::vector<rc::prim::Vertex3D>& vertices,
99 std::vector<std::uint32_t>& indices) {
100 std::vector<std::uint32_t> splitIndices;
101 std::unordered_map<std::uint32_t, std::uint32_t> midpointMap;
102 vertices.clear();
103 indices.clear();
104 vertices.reserve(SourceVertices.size() * detailLevel * 3);
105 indices.reserve(SourceIndices.size() * (detailLevel * 4));
106 splitIndices.reserve(SourceIndices.size() * (detailLevel * 4));
107
108 std::transform(SourceVertices.begin(),
109 SourceVertices.end(),
110 std::back_inserter(vertices),
111 [](const glm::vec3& v) { return rc::prim::Vertex3D(v); });
112 std::copy(SourceIndices.begin(), SourceIndices.end(), std::back_inserter(indices));
113
114 // make sphere
115 for (int lod = 1; lod <= detailLevel; ++lod) {
116 for (std::uint32_t i = 0; i < indices.size(); i += 3) {
117 const std::uint32_t i0 = indices[i + 0];
118 const std::uint32_t i1 = indices[i + 1];
119 const std::uint32_t i2 = indices[i + 2];
120 const std::uint32_t m01 = getMidpoint(midpointMap, vertices, i0, i1, lod);
121 const std::uint32_t m12 = getMidpoint(midpointMap, vertices, i1, i2, lod);
122 const std::uint32_t m02 = getMidpoint(midpointMap, vertices, i0, i2, lod);
123 const std::array<std::uint32_t, 12> newIndices = {
124 i0, m01, m02, i1, m12, m01, i2, m02, m12, m02, m01, m12};
125 std::copy(newIndices.begin(), newIndices.end(), std::back_inserter(splitIndices));
126 }
127 indices.swap(splitIndices);
128 splitIndices.clear();
129 midpointMap.clear();
130 }
131
132 // normalize vertices to make sphere
133 for (auto& v : vertices) {
134 v.normal = glm::normalize(v.pos);
135 v.pos = radius * v.normal;
136 }
137}
