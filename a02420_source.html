13void generateErrorPattern(sf::Image& img, unsigned int left, unsigned int top, unsigned int width,
14 unsigned int height, unsigned int numBoxes) {
15 const unsigned int ErrorBoxWidth = width / numBoxes;
16 const unsigned int ErrorBoxHeight = height / numBoxes;
17 for (unsigned int x = left; x < left + width; ++x) {
18 for (unsigned int y = top; y < top + height; ++y) {
19 const unsigned int xi = x / ErrorBoxWidth;
20 const unsigned int yi = y / ErrorBoxHeight;
21 if ((xi % 2) == (yi % 2)) { img.setPixel(x, y, sf::Color(230, 66, 245)); }
22 else { img.setPixel(x, y, sf::Color(255, 254, 196)); }
23 }
24 }
25}

56void TexturePool::init(vk::PerFrame<VkDescriptorSet>& descriptorSets,
57 vk::PerFrame<VkDescriptorSet>& rtDescriptorSets) {
58 // create error texture
59 constexpr unsigned int ErrorSize = 128;
60 constexpr unsigned int ErrorBoxCount = 16;
61 errorPattern.create(ErrorSize, ErrorSize);
62 errorPatternCube.create(ErrorSize, ErrorSize * 6);
63 generateErrorPattern(errorPattern, 0, 0, ErrorSize, ErrorSize, ErrorBoxCount);
64 for (std::size_t i = 0; i < 6; ++i) {
65 generateErrorPattern(
66 errorPatternCube, 0, i * ErrorSize, ErrorSize, ErrorSize, ErrorBoxCount);
67 }
68 errorTexture.altImg = &errorPattern;
69 errorCubemap.altImg = &errorPatternCube;
70 errorTexture.createFromContentsAndQueue(
71 vk::Texture::Type::Texture2D,
72 {.format = vk::CommonTextureFormats::SRGBA32Bit,
73 .sampler = vk::SamplerOptions::Type::FilteredRepeated});
74 errorCubemap.createFromContentsAndQueue(
75 vk::Texture::Type::Cubemap,
76 {.format = vk::CommonTextureFormats::SRGBA32Bit,
77 .sampler = vk::SamplerOptions::Type::FilteredRepeated});
78 renderer.getTransferEngine().executeTransfers();
79
80 // init all textures to error pattern
81 for (vk::Texture& txtr : textures) { txtr.currentView = errorTexture.currentView; }
82 for (vk::Texture& txtr : cubemaps) { txtr.currentView = errorCubemap.getView(); }
83
84 // fill descriptor set
85 VkDescriptorImageInfo errorInfo{};
86 errorInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
87 errorInfo.imageView = errorTexture.getView();
88 errorInfo.sampler = renderer.samplerCache().getSampler({.type = errorTexture.getSampler()});
89 std::vector<VkDescriptorImageInfo> imageInfos(cfg::Limits::MaxTextureCount, errorInfo);
90
91 VkDescriptorImageInfo errorCubeInfo{};
92 errorCubeInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
93 errorCubeInfo.imageView = errorCubemap.getView();
94 errorCubeInfo.sampler = errorInfo.sampler;
95 std::vector<VkDescriptorImageInfo> imageCubeInfos(cfg::Limits::MaxCubemapCount, errorCubeInfo);
96
97 std::array<VkWriteDescriptorSet, 4 * cfg::Limits::MaxConcurrentFrames> setWrites{};
98 unsigned int i = 0;
99 const auto visitor = [this, &i, &setWrites](auto& set,
100 std::uint32_t bindIndex,
101 const std::vector<VkDescriptorImageInfo>& infos) {
102 setWrites[i].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
103 setWrites[i].descriptorCount = infos.size();
104 setWrites[i].dstBinding = bindIndex;
105 setWrites[i].dstArrayElement = 0;
106 setWrites[i].dstSet = set;
107 setWrites[i].pImageInfo = infos.data();
108 setWrites[i].descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
109 ++i;
110 };
111 descriptorSets.visit(
112 [&visitor, &imageInfos](auto& set) { visitor(set, TextureArrayBindIndex, imageInfos); });
113 rtDescriptorSets.visit(
114 [&visitor, &imageInfos](auto& set) { visitor(set, TextureArrayBindIndex, imageInfos); });
115 descriptorSets.visit([&visitor, &imageCubeInfos](auto& set) {
116 visitor(set, CubemapArrayBindIndex, imageCubeInfos);
117 });
118 rtDescriptorSets.visit([&visitor, &imageCubeInfos](auto& set) {
119 visitor(set, CubemapArrayBindIndex, imageCubeInfos);
120 });
121 vkUpdateDescriptorSets(vulkanState.getDevice(), setWrites.size(), setWrites.data(), 0, nullptr);
122}

168void TexturePool::doRelease(vk::Texture* texture) {
169 if (texture->getType() != vk::Texture::Type::Cubemap) {
170 std::uint32_t i = texture - textures.data();
171 refCounts[i] = 0;
172
173 if (i < reverseFileMap.size()) {
174 freeSlots.release(i);
175 if (reverseFileMap[i]) {
176 fileMap.erase(*reverseFileMap[i]);
177 reverseFileMap[i] = nullptr;
178 }
179 if (reverseImageMap[i]) {
180 imageMap.erase(reverseImageMap[i]);
181 reverseImageMap[i] = nullptr;
182 }
183 }
184 else { freeRtSlots.release(i - reverseFileMap.size()); }
185 }
186 else {
187 std::uint32_t i = texture - cubemaps.data();
188
189 cubemapRefCounts[i] = 0;
190 cubemapFreeSlots.release(i);
191 if (cubemapFileMap.find(*reverseFileMap[i]) != cubemapFileMap.end()) {
192 cubemapFileMap.erase(*cubeMapReverseFileMap[i]);
193 cubeMapReverseFileMap[i] = nullptr;
194 }
195 }
196 resetTexture(texture);
197}

336TextureRef TexturePool::createCubemap(resource::Ref<sf::Image> right, resource::Ref<sf::Image> left,
337 resource::Ref<sf::Image> top, resource::Ref<sf::Image> bottom,
338 resource::Ref<sf::Image> back, resource::Ref<sf::Image> front,
339 VkFormat format, vk::SamplerOptions::Type sampler) {
340 // validate faces
341 std::array<sf::Image*, 6> faces = {
342 right.get(), left.get(), top.get(), bottom.get(), front.get(), back.get()};
343 glm::u32vec2 faceSize(0, 0);
344 for (auto& face : faces) {
345 if (face && face->getSize().x > 0) {
346 if (faceSize.x == 0) {
347 faceSize.x = face->getSize().x;
348 faceSize.y = face->getSize().y;
349 }
350 else {
351 if (faceSize.x != face->getSize().x || faceSize.y != face->getSize().y) {
352 BL_LOG_ERROR << "Cubemap faces must all be the same size";
353 face = nullptr;
354 }
355 }
356 }
357 }
358 if (faceSize.x == 0) {
359 BL_LOG_ERROR << "No valid cubemap faces provided";
360 faceSize = {64, 64};
361 }
362
363 std::unique_lock lock(mutex);
364
365 TextureRef cm = allocateCubemap();
366
367 // stitch images into new image
368 cm->altImg = &cm->localImage;
369 cm->localImage.create(faceSize.x, faceSize.y * 6, sf::Color::Transparent);
370 for (std::size_t i = 0; i < 6; ++i) {
371 if (faces[i]) { cm->localImage.copy(*faces[i], 0, i * faceSize.y); }
372 else {
373 generateErrorPattern(cm->localImage, 0, i * faceSize.y, faceSize.x, faceSize.y, 16);
374 BL_LOG_WARN << "Cubemap face " << i << " is invalid, using error pattern";
375 }
376 }
377
378 cm->createFromContentsAndQueue(vk::Texture::Type::Cubemap,
379 {.format = format, .sampler = sampler});
380 updateTexture(cm.get());
381
382 return cm;
383}

435void TexturePool::onFrameStart(ds::SetWriteHelper& setWriter, VkDescriptorSet currentSet,
436 VkDescriptorSet currentRtSet) {
437 if (!queuedUpdates.current().empty()) {
438 // prepare descriptor updates before waiting
439 setWriter.hintWriteCount(queuedUpdates.current().size() * 2);
440 setWriter.hintImageInfoCount(queuedUpdates.current().size() * 2);
441
442 for (unsigned int j = 0; j < queuedUpdates.current().size(); ++j) {
443 vk::Texture* texture = queuedUpdates.current()[j];
444 const bool isCubemap = texture->getType() == vk::Texture::Type::Cubemap;
445 const std::uint32_t bindIndex =
446 isCubemap ? CubemapArrayBindIndex : TextureArrayBindIndex;
447 const vk::Texture* base = isCubemap ? cubemaps.data() : textures.data();
448 const std::uint32_t i = texture - base;
449 const bool isRT = texture->getType() == vk::Texture::Type::RenderTexture;
450 const VkImageView view = texture->currentView;
451
452 auto& regularInfo = setWriter.getNewImageInfo();
453 regularInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
454 regularInfo.imageView = view;
455 regularInfo.sampler =
456 renderer.samplerCache().getSampler({.type = texture->getSampler()});
457
458 auto& regularWrite = setWriter.getNewSetWrite(currentSet);
459 regularWrite.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
460 regularWrite.descriptorCount = 1;
461 regularWrite.dstBinding = bindIndex;
462 regularWrite.dstArrayElement = i;
463 regularWrite.dstSet = currentSet;
464 regularWrite.pImageInfo = &regularInfo;
465 regularWrite.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
466
467 // write error texture to rt set if is rt itself
468 auto& rtInfo = setWriter.getNewImageInfo();
469 rtInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
470 rtInfo.imageView = isRT ? errorTexture.getView() : view;
471 rtInfo.sampler =
472 isRT ? renderer.samplerCache().getSampler({.type = errorTexture.getSampler()}) :
473 regularInfo.sampler;
474
475 auto& rtWrite = setWriter.getNewSetWrite(currentRtSet);
476 rtWrite.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
477 rtWrite.descriptorCount = 1;
478 rtWrite.dstBinding = bindIndex;
479 rtWrite.dstArrayElement = i;
480 rtWrite.pImageInfo = &rtInfo;
481 rtWrite.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
482 }
483
484 queuedUpdates.current().clear();
485 }
