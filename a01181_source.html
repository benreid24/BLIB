46 bool Registry::destroyEntity(Entity start) {
47  std::lock_guard lock(entityLock);
48 
49  if (!entityExistsLocked(start)) { return false; }
50 
51  // check if we can remove due to dependencies
52  if (dependencyGraph.hasDependencies(start)) {
53  markEntityForRemoval(start);
54  return false;
55  }
56 
57  // determine list of entities to remove due to parenting and dependencies
58  std::vector<Entity> toRemove;
59  std::vector<Entity> toVisit;
60  toRemove.reserve(16);
61  toVisit.reserve(16);
62  toVisit.emplace_back(start);
63 
64  while (!toVisit.empty()) {
65  const Entity ent = toVisit.back();
66  toVisit.pop_back();
67  toRemove.emplace_back(ent);
68 
69  // add marked dependencies
70  for (Entity resource : dependencyGraph.getResources(ent)) {
71  dependencyGraph.removeDependency(resource, ent);
72  if (!dependencyGraph.hasDependencies(resource)) {
73  const std::uint32_t i = resource.getIndex();
74  if (i < markedForRemoval.size() && markedForRemoval[i]) {
75  toVisit.emplace_back(resource);
76  }
77  }
78  }
79 
80  // add children
81  for (Entity child : parentGraph.getChildren(ent)) { toVisit.emplace_back(child); }
82  }
83 
84  // remove all discovered entities
85  for (const Entity ent : toRemove) {
86  const std::uint32_t index = ent.getIndex();
87  const ComponentMask::SimpleMask mask = entityMasks[index];
88 
89  // notify external and remove from views
90  bl::event::Dispatcher::dispatch<event::EntityDestroyed>({ent});
91  for (auto& view : views) {
92  if (view->mask.passes(mask)) { view->removeEntity(ent); }
93  }
94 
95  // destroy components
96  for (ComponentPoolBase* pool : componentPools) {
97  if (ComponentMask::has(mask, pool->ComponentIndex)) { pool->remove(ent); }
98  }
99

101  entityAllocator.release(index);
102  entityMasks[index] = ComponentMask::EmptyMask;
103 
104  // remove parenting info
105  parentGraph.removeEntity(ent);
106 
107  // remove dependency info
108  if (index < markedForRemoval.size()) { markedForRemoval[index] = false; }
109  }
110 
111  return true;
112 }
