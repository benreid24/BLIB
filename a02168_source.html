43void Pipeline::createForRenderPass(std::uint32_t rpid, std::uint32_t spec) {
44 PipelineSpecialization& specialization =
45 spec > 0 && spec <= createParams.specializations.size() ?
46 createParams.specializations[spec - 1] :
47 createParams.mainSpecialization;
48 if (spec > createParams.specializations.size()) {
49 BL_LOG_ERROR << "Pipeline being used with invalid specialization: " << spec;
50 }
51
52 RenderPass& renderPass = renderer.renderPassCache().getRenderPass(rpid);
53
54 // configure color blending
55 BlendParameters& colorBlending = specialization.attachmentBlendingSpecialized ?
56 specialization.attachmentBlending :
57 createParams.colorBlending;
58 colorBlending.build();
59
60 // sample shading
61 const VkBool32 sampleShadingSetting = createParams.msaa.sampleShadingEnable;
62 const bool sampledShadingAvailable =
63 renderer.vulkanState().getPhysicalDeviceFeatures().sampleRateShading == VK_TRUE;
64 const bool wantsSampledShading = createParams.msaa.sampleShadingEnable == VK_TRUE &&
65 renderer.getSettings().getMSAASampleCountAsInt() > 1;
66 const bool useSampleShading = wantsSampledShading && sampledShadingAvailable;
67 createParams.msaa.sampleShadingEnable = useSampleShading ? VK_TRUE : VK_FALSE;
68 if (wantsSampledShading && !sampledShadingAvailable) {
69 BL_LOG_WARN << "Sample shading requested but not available on this device";
70 createParams.msaa.sampleShadingEnable = VK_FALSE;
71 }
72
73 // configure msaa
74 const RenderPassParameters::MSAABehavior msaaBehavior =
75 renderPass.getCreateParams().getMSAABehavior();
76 const bool useResolveShaders =
77 (!(msaaBehavior & RenderPassParameters::MSAABehavior::ResolveAttachments) ||
78 createParams.doesOwnResolve) &&
79 renderer.getSettings().getMSAASampleCountAsInt() > 1 && !useSampleShading;
80 if (renderPass.getCreateParams().getMSAABehavior() &
81 RenderPassParameters::MSAABehavior::UseSettings) {
82 createParams.msaa.rasterizationSamples = renderer.getSettings().getMSAASampleCount();
83 }
84 else { createParams.msaa.rasterizationSamples = VK_SAMPLE_COUNT_1_BIT; }
85
86 const auto findShaderSrc = [this, &specialization, useSampleShading, useResolveShaders](
87 const ShaderParameters& src) -> const ShaderParameters& {
88 if (useSampleShading) {
89 for (const auto& sampleShader : createParams.sampleShaders) {
90 if (sampleShader.stage == src.stage) { return sampleShader; }
91 }
92 }
93 if (useResolveShaders) {
94 for (const auto& resolveShader : createParams.resolveShaders) {
95 if (resolveShader.stage == src.stage) { return resolveShader; }
96 }
97 }
98 for (const auto& shader : specialization.shaderOverrides) {
99 if (shader.stage == src.stage) { return shader; }
100 }
101 return src;
102 };
103
104 // Load shaders
105 ctr::StaticVector<VkPipelineShaderStageCreateInfo, 5> shaderStages;
106 ctr::StaticVector<VkSpecializationInfo, 5> shaderSpecs;
107 for (const auto& shaderSrc : createParams.shaders) {
108 const auto& shader = findShaderSrc(shaderSrc);
109 auto& info = shaderStages.emplace_back(VkPipelineShaderStageCreateInfo{});
110 info.sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
111 info.stage = static_cast<VkShaderStageFlagBits>(shader.stage);
112 info.module = renderer.getShaderCache().loadShader(shader.path);
113 info.pName = shader.entrypoint.c_str();
114
115 for (auto& spec : specialization.shaderSpecializations) {
116 if ((spec.stage & shader.stage) != 0) {
117 auto& specInfo = shaderSpecs.emplace_back();
118 info.pSpecializationInfo = &specInfo;
119 specInfo.dataSize = static_cast<std::uint32_t>(spec.storage.size());
120 specInfo.pData = spec.storage.data();
121 specInfo.mapEntryCount = static_cast<std::uint32_t>(spec.entries.size());
122 specInfo.pMapEntries = spec.entries.data();
123 break;
124 }
125 }
126 }
127
128 // Configure vertices
129 VkPipelineVertexInputStateCreateInfo vertexInputInfo{};
130 vertexInputInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_VERTEX_INPUT_STATE_CREATE_INFO;
131 vertexInputInfo.vertexBindingDescriptionCount = 1;
132 vertexInputInfo.pVertexBindingDescriptions = &createParams.vertexBinding;
133 vertexInputInfo.vertexAttributeDescriptionCount = createParams.vertexAttributes.size();
134 vertexInputInfo.pVertexAttributeDescriptions = createParams.vertexAttributes.data();
135
136 // Configure primitives
137 VkPipelineInputAssemblyStateCreateInfo inputAssembly{};
138 inputAssembly.sType = VK_STRUCTURE_TYPE_PIPELINE_INPUT_ASSEMBLY_STATE_CREATE_INFO;
139 inputAssembly.topology = createParams.primitiveType;
140 inputAssembly.primitiveRestartEnable = VK_FALSE;
141
142 // Always have 1 dynamic viewport and scissor
143 VkPipelineViewportStateCreateInfo viewportState{};
144 viewportState.sType = VK_STRUCTURE_TYPE_PIPELINE_VIEWPORT_STATE_CREATE_INFO;
145 viewportState.viewportCount = 1;
146 viewportState.scissorCount = 1;
147
148 // Dynamic states
149 VkPipelineDynamicStateCreateInfo dynamicState{};
150 dynamicState.sType = VK_STRUCTURE_TYPE_PIPELINE_DYNAMIC_STATE_CREATE_INFO;
151 dynamicState.dynamicStateCount = createParams.dynamicStates.size();
152 dynamicState.pDynamicStates = createParams.dynamicStates.data();
153
154 VkGraphicsPipelineCreateInfo pipelineInfo{};
155 pipelineInfo.sType = VK_STRUCTURE_TYPE_GRAPHICS_PIPELINE_CREATE_INFO;
156 pipelineInfo.stageCount = shaderStages.size();
157 pipelineInfo.pStages = shaderStages.data();
158 pipelineInfo.pVertexInputState = &vertexInputInfo;
159 pipelineInfo.pInputAssemblyState = &inputAssembly;
160 pipelineInfo.pViewportState = &viewportState;
161 pipelineInfo.pRasterizationState = &createParams.rasterizer;
162 pipelineInfo.pMultisampleState = &createParams.msaa;
163 pipelineInfo.pDepthStencilState = specialization.depthStencilSpecialized ?
164 &specialization.depthStencil :
165 createParams.depthStencil;
166 pipelineInfo.pColorBlendState = &colorBlending.colorBlending;
167 pipelineInfo.pDynamicState = &dynamicState;
168 pipelineInfo.layout = layout->rawLayout();
169 pipelineInfo.subpass = 0; // TODO - consider dynamic subpass if needed
170 pipelineInfo.basePipelineHandle = VK_NULL_HANDLE; // Optional
171 pipelineInfo.basePipelineIndex = -1; // Optional
172 pipelineInfo.renderPass = renderPass.rawPass();
173
174 if (vkCreateGraphicsPipelines(renderer.vulkanState().getDevice(),
175 VK_NULL_HANDLE,
176 1,
177 &pipelineInfo,
178 nullptr,
179 &pipelines[spec][rpid]) != VK_SUCCESS) {
180 throw std::runtime_error("Failed to create graphics pipeline");
181 }
182
183 // reset sample shading setting
184 createParams.msaa.sampleShadingEnable = sampleShadingSetting;
185}

187void Pipeline::process(const event::SettingsChanged& event) {
188 const bool changed = createParams.handleChange(renderer, event);
189 if (changed || event.setting == event::SettingsChanged::Setting::AntiAliasing) {
190 for (std::uint32_t rpid = 0; rpid < pipelines.size(); ++rpid) {
191 RenderPass* renderPass = renderer.renderPassCache().getRenderPassMaybe(rpid);
192 if (!renderPass) { continue; }
193
194 if (changed || (renderPass->getCreateParams().getMSAABehavior() &
195 RenderPassParameters::MSAABehavior::UseSettings)) {
196 recreateForRenderPass(rpid);
197 }
198 }
199 }

206void Pipeline::recreateForRenderPass(std::uint32_t rpid) {
207 for (std::uint32_t spec = 0; spec < pipelines[rpid].size(); ++spec) {
208 if (pipelines[spec][rpid]) {
209 renderer.getCleanupManager().add(
210 [device = renderer.vulkanState().getDevice(), pipeline = pipelines[spec][rpid]]() {
211 vkDestroyPipeline(device, pipeline, nullptr);
212 });
213 createForRenderPass(rpid, spec);
214 }
