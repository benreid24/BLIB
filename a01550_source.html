58bool Bundler::createBundles() {
59 BL_LOG_INFO << "Beginning bundle creation. Saving bundles to " << config.outputDirectory();
60
61 if (!util::FileUtil::directoryExists(config.outputDirectory())) {
62 util::FileUtil::createDirectory(config.outputDirectory());
63 BL_LOG_INFO << "Created bundle directory";
64 }
65 else {
66 BL_LOG_INFO << "Clearing out old bundle files";
67 util::FileUtil::deleteDirectory(config.outputDirectory());

69 BL_LOG_INFO << "Deleted existing bundles";

72 bundle::Manifest manifest(config.outputDirectory());
73 bundle::Stats stats;
74 const auto doCreate =
75 [this, &stats, &manifest](const std::string& path,
76 const std::string& og,
77 const bundle::BundleSource& bundleSource) -> bool {
78 bundle::BundleCreator creator(path, config, bundleSource);
79 if (!creator.create(stats, manifest)) {
80 BL_LOG_ERROR << "Failed to create bundle for " << og << ", terminating";
81 return false;
82 }
83 return true;
84 };
85
86 for (const bundle::BundleSource& src : config.bundleSources()) {
87 if (src.policy == bundle::BundleSource::BundleAllFiles) {
88 if (!doCreate(src.sourcePath, src.sourcePath, src)) { return false; }
89 }
90 else if (src.policy == bundle::BundleSource::CreateBundleForEachContainedRecursive) {
91 const std::vector<std::string> all = util::FileUtil::listDirectory(src.sourcePath);
92 for (const std::string& path : all) {
93 if (config.includeFile(path)) {
94 if (src.includeFile(path)) {
95 if (!doCreate(path, src.sourcePath, src)) { return false; }
96 }
97 else { config.addExcludeFile(path); }
98 }
99 }
100 }
101 else if (src.policy == bundle::BundleSource::CreateBundleForEachContained) {
102 const std::vector<std::string> folders =
103 util::FileUtil::listDirectoryFolders(src.sourcePath);
104 for (const std::string& dir : folders) {
105 if (!doCreate(
106 bl::util::FileUtil::joinPath(src.sourcePath, dir), src.sourcePath, src)) {
107 return false;
108 }
109 }
110 const std::vector<std::string> files =
111 util::FileUtil::listDirectory(src.sourcePath, "", false);
112 for (const std::string& file : files) {
113 if (config.includeFile(file)) {
114 if (src.includeFile(file)) {
115 if (!doCreate(file, src.sourcePath, src)) { return false; }
116 }
117 else { config.addExcludeFile(file); }
118 }
119 }
120 }
121 else {
122 BL_LOG_ERROR << "Bad bundle source policy: " << src.policy << ", skipping '"
123 << src.sourcePath << "'";
124 }
125 }
126
127 if (util::FileUtil::directoryExists(config.catchAllDirectory())) {
128 BL_LOG_INFO << "Bundling remaining files in top level directory: "
129 << config.catchAllDirectory();
130 bundle::BundleSource src(config.catchAllDirectory(), bundle::BundleSource::BundleAllFiles);
131 bundle::BundleCreator finalCreator(config.catchAllDirectory(), config, src);
132 if (!finalCreator.create(stats, manifest)) {
133 BL_LOG_ERROR << "Failed to bundle remaining files";
134 return false;
135 }
136 }
137 else if (!config.catchAllDirectory().empty()) {
138 BL_LOG_WARN << "Catch-all directory " << config.catchAllDirectory() << " does not exist";
139 }
140
141 BL_LOG_INFO << "Bundled " << stats.fileCount << " files into " << stats.bundleCount
142 << " bundles with a total size of " << formatSize(stats.outputSize)
143 << ". Input files total size: " << formatSize(stats.inputSize);
144
145 if (!manifest.save()) {
146 BL_LOG_ERROR << "Failed to save manifest";
147 return false;
148 }
149 BL_LOG_INFO << "Completed bundling";
150 return true;
151}

179bool BundleCreator::create(Stats& stats, Manifest& manifest) {
180 static std::unordered_set<std::string> failed;
181
182 std::fstream tempfile(tempPath.c_str(), std::ios::binary | std::ios::out);
183
184 BundleMetadata bundleManifest;
185 unsigned int fileCount = 0;
186 while (!toBundle.empty()) {
187 const std::string path = toBundle.top();
188 toBundle.pop();
189
190 if (manifest.containsFile(path)) {
191 BL_LOG_DEBUG << "Skipping already bundled file: " << path;
192 continue;
193 }
194
195 FileHandler& handler = config.getFileHandler(path);
196 FileHandlerContext ctx(config.excludeFiles, toBundle);
197 const std::int64_t offset = tempfile.tellp();
198 if (handler.processFile(path, tempfile, ctx)) {
199 BL_LOG_DEBUG << "Added " << path << " to " << bundlePath;
200 manifest.registerFileBundle(path, bundlePath);
201
202 util::FileUtil::FileInfo info;
203 ++fileCount;
204 if (util::FileUtil::queryFileInfo(path, info)) { stats.inputSize += info.size; }
205
206 const std::int64_t len = tempfile.tellp() - offset;
207 bundleManifest.addFileInfo(path, {offset, len});
208 }
209 else {
210 if (failed.find(path) == failed.end()) {
211 failed.insert(path);
212 BL_LOG_ERROR << "Failed to process file: " << path;
213 tempfile.seekp(offset);
214 }
215 }
216 }
217
218 if (fileCount > 0) {
219 BL_LOG_INFO << "Finalizing bundle " << bundlePath;
220
221 tempfile.close();
222 tempfile.open(tempPath.c_str(), std::ios::binary | std::ios::in);
223 std::ofstream bundle(bundlePath.c_str(), std::ios::binary);
224
225 // write bundle metadata
226 if (!bundleManifest.save(bundle)) {
227 BL_LOG_ERROR << "Failed to write bundle manifest";
228 return false;
229 }
230 // copy bundled files from temp file
231 std::istreambuf_iterator<char> begin_source(tempfile);
232 std::istreambuf_iterator<char> end_source;
233 std::ostreambuf_iterator<char> begin_dest(bundle);
234 std::copy(begin_source, end_source, begin_dest);
235
236 if (!bundle.good()) {
237 BL_LOG_ERROR << "Failed to save final bundle";
238 return false;
239 }
240
241 ++stats.bundleCount;
242 stats.fileCount += fileCount;
243 stats.outputSize += bundle.tellp();
244 BL_LOG_INFO << "Completed bundle " << bundlePath << ". Bundled " << fileCount
245 << " files. Bundle size: " << formatSize(bundle.tellp());
246 }
247 else { BL_LOG_WARN << "Skipping empty bundle '" << bundlePath << "'"; }
248
249 return true;
250}
