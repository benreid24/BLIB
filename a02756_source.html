30// Combine outline thickness, boldness and font glyph index into a single 64-bit key
31std::uint64_t combine(std::uint8_t size, float outlineThickness, bool bold, std::uint32_t index) {
32 return (static_cast<std::uint64_t>(reinterpret<std::uint32_t>(outlineThickness)) << 32) |

64bool FontPayload::loadFromMemory() {
65 needsUpload = true;
66
67 // Cleanup the previous resources (excluding buffer that we are loading from)
68 cleanup(false);
69
70 // Initialize FreeType
71 // Note: we initialize FreeType for every font instance in order to avoid having a single
72 // global manager that would create a lot of issues regarding creation and destruction order.
73 FT_Library library;
74 if (FT_Init_FreeType(&library) != 0) {
75 BL_LOG_ERROR << "Failed to load font from memory (failed to initialize FreeType)";
76 return false;
77 }
78 m_library = library;
79
80 // Load the new font face from the specified file
81 FT_Face face;
82 if (FT_New_Memory_Face(static_cast<FT_Library>(m_library),
83 reinterpret_cast<const FT_Byte*>(buffer.data()),
84 static_cast<FT_Long>(buffer.size()),
85 0,
86 &face) != 0) {
87 BL_LOG_ERROR << "Failed to load font from memory (failed to create the font face)";
88 return false;
89 }
90
91 // Load the stroker that will be used to outline the font
92 FT_Stroker stroker;
93 if (FT_Stroker_New(static_cast<FT_Library>(m_library), &stroker) != 0) {
94 BL_LOG_ERROR << "Failed to load font from memory (failed to create the stroker)";
95 FT_Done_Face(face);
96 return false;
97 }
98
99 // Select the Unicode character map
100 if (FT_Select_Charmap(face, FT_ENCODING_UNICODE) != 0) {
101 BL_LOG_ERROR << "Failed to load font from memory (failed to set the Unicode character set)";
102 FT_Stroker_Done(stroker);
103 FT_Done_Face(face);
104 return false;
105 }
106
107 // Store the loaded font in our ugly void* :)
108 m_stroker = stroker;
109 m_face = face;
110
111 // Store the font information
112 m_info.family = face->family_name ? face->family_name : std::string();
113
114 return true;
115}

119const sf::Glyph& FontPayload::getGlyph(std::uint32_t codePoint, unsigned int characterSize,
120 bool bold, float outlineThickness) const {
121 // Build the key by combining the glyph index (based on code point), bold flag, and outline
122 // thickness
123 std::uint64_t key = combine(characterSize,
124 outlineThickness,
125 bold,
126 FT_Get_Char_Index(static_cast<FT_Face>(m_face), codePoint));
127
128 // Search the glyph into the cache
129 GlyphTable::const_iterator it = glyphs.find(key);
130 if (it != glyphs.end()) {
131 // Found: just return it
132 return it->second;
133 }
134 else {
135 // Not found: we have to load it
136 needsUpload = true;
137 sf ::Glyph glyph = loadGlyph(codePoint, characterSize, bold, outlineThickness);
138 return glyphs.insert(std::make_pair(key, glyph)).first->second;
139 }

146float FontPayload::getKerning(std::uint32_t first, std::uint32_t second, unsigned int characterSize,
147 bool bold) const {
148 // Special case where first or second is 0 (null character)
149 if (first == 0 || second == 0) return 0.f;
150
151 FT_Face face = static_cast<FT_Face>(m_face);
152
153 if (face && setCurrentSize(characterSize)) {
154 // Convert the characters to indices
155 FT_UInt index1 = FT_Get_Char_Index(face, first);
156 FT_UInt index2 = FT_Get_Char_Index(face, second);
157
158 // Retrieve position compensation deltas generated by FT_LOAD_FORCE_AUTOHINT flag
159 float firstRsbDelta = static_cast<float>(getGlyph(first, characterSize, bold).rsbDelta);
160 float secondLsbDelta = static_cast<float>(getGlyph(second, characterSize, bold).lsbDelta);
161
162 // Get the kerning vector if present
163 FT_Vector kerning;
164 kerning.x = kerning.y = 0;
165 if (FT_HAS_KERNING(face))
166 FT_Get_Kerning(face, index1, index2, FT_KERNING_UNFITTED, &kerning);
167
168 // X advance is already in pixels for bitmap fonts
169 if (!FT_IS_SCALABLE(face)) return static_cast<float>(kerning.x);
170
171 // Combine kerning with compensation deltas and return the X advance
172 // Flooring is required as we use FT_KERNING_UNFITTED flag which is not quantized in 64
173 // based grid
174 return std::floor((secondLsbDelta - firstRsbDelta + static_cast<float>(kerning.x) + 32) /
175 static_cast<float>(1 << 6));
176 }
177 else {
178 // Invalid font
179 return 0.f;
180 }
181}

224 if (m_library) {
225 // Destroy the stroker
226 if (m_stroker) FT_Stroker_Done(static_cast<FT_Stroker>(m_stroker));

233
234 // Close the library
235 if (m_library) FT_Done_FreeType(static_cast<FT_Library>(m_library));
236 }
237
238 // Reset members
239 m_library = NULL;
240 m_face = NULL;
241 m_stroker = NULL;
242 m_streamRec = NULL;
243 glyphs.clear();
244 rows.clear();
245 nextRow = 3;
246 needsUpload = true;
247 vulkanTexture.release();

252sf::Glyph FontPayload::loadGlyph(std::uint32_t codePoint, unsigned int characterSize, bool bold,
253 float outlineThickness) const {
254 // The glyph to return
255 sf::Glyph glyph;
256
257 // First, transform our ugly void* to a FT_Face
258 FT_Face face = static_cast<FT_Face>(m_face);
259 if (!face) return glyph;
260
261 // Set the character size
262 if (!setCurrentSize(characterSize)) return glyph;
263
264 // Load the glyph corresponding to the code point
265 FT_Int32 flags = FT_LOAD_TARGET_NORMAL | FT_LOAD_FORCE_AUTOHINT;
266 if (outlineThickness != 0) flags |= FT_LOAD_NO_BITMAP;
267 if (FT_Load_Char(face, codePoint, flags) != 0) return glyph;
268
269 // Retrieve the glyph
270 FT_Glyph glyphDesc;
271 if (FT_Get_Glyph(face->glyph, &glyphDesc) != 0) return glyph;
272
273 // Apply bold and outline (there is no fallback for outline) if necessary -- first technique
274 // using outline (highest quality)
275 FT_Pos weight = 1 << 6;
276 bool outline = (glyphDesc->format == FT_GLYPH_FORMAT_OUTLINE);
277 if (outline) {
278 if (bold) {
279 FT_OutlineGlyph outlineGlyph = reinterpret_cast<FT_OutlineGlyph>(glyphDesc);
280 FT_Outline_Embolden(&outlineGlyph->outline, weight);
281 }
282
283 if (outlineThickness != 0) {
284 FT_Stroker stroker = static_cast<FT_Stroker>(m_stroker);
285
286 FT_Stroker_Set(stroker,
287 static_cast<FT_Fixed>(outlineThickness * static_cast<float>(1 << 6)),
288 FT_STROKER_LINECAP_ROUND,
289 FT_STROKER_LINEJOIN_ROUND,
290 0);
291 FT_Glyph_Stroke(&glyphDesc, stroker, true);
292 }
293 }
294
295 // Convert the glyph to a bitmap (i.e. rasterize it)
296 // Warning! After this line, do not read any data from glyphDesc directly, use
297 // bitmapGlyph.root to access the FT_Glyph data.
298 FT_Glyph_To_Bitmap(&glyphDesc, FT_RENDER_MODE_NORMAL, 0, 1);
299 FT_BitmapGlyph bitmapGlyph = reinterpret_cast<FT_BitmapGlyph>(glyphDesc);
300 FT_Bitmap& bitmap = bitmapGlyph->bitmap;
301
302 // Apply bold if necessary -- fallback technique using bitmap (lower quality)
303 if (!outline) {
304 if (bold) FT_Bitmap_Embolden(static_cast<FT_Library>(m_library), &bitmap, weight, weight);
305
306 if (outlineThickness != 0)
307 BL_LOG_ERROR << "Failed to outline glyph (no fallback available)";
308 }
309
310 // Compute the glyph's advance offset
311 glyph.advance = static_cast<float>(bitmapGlyph->root.advance.x >> 16);
312 if (bold) glyph.advance += static_cast<float>(weight) / static_cast<float>(1 << 6);
313
314 glyph.lsbDelta = static_cast<int>(face->glyph->lsb_delta);
315 glyph.rsbDelta = static_cast<int>(face->glyph->rsb_delta);
316
317 unsigned int width = bitmap.width;
318 unsigned int height = bitmap.rows;
319
320 if ((width > 0) && (height > 0)) {
321 // Leave a small padding around characters, so that filtering doesn't
322 // pollute them with pixels from neighbors
323 const unsigned int padding = 2;
324
325 width += 2 * padding;
326 height += 2 * padding;
327
328 // Find a good position for the new glyph into the texture
329 glyph.textureRect = findGlyphRect(width, height);
330
331 // Make sure the texture data is positioned in the center
332 // of the allocated texture rectangle
333 glyph.textureRect.position.x += static_cast<int>(padding);
334 glyph.textureRect.position.y += static_cast<int>(padding);
335 glyph.textureRect.size.x -= static_cast<int>(2 * padding);
336 glyph.textureRect.size.y -= static_cast<int>(2 * padding);
337
338 // Compute the glyph's bounding box
339 glyph.bounds.position.x = static_cast<float>(bitmapGlyph->left);
340 glyph.bounds.position.y = static_cast<float>(-bitmapGlyph->top);
341 glyph.bounds.size.x = static_cast<float>(bitmap.width);
342 glyph.bounds.size.y = static_cast<float>(bitmap.rows);
343
344 // Resize the pixel buffer to the new size and fill it with transparent white pixels
345 m_pixelBuffer.resize(width * height * 4);
346
347 std::uint8_t* current = &m_pixelBuffer[0];
348 std::uint8_t* end = current + width * height * 4;
349
350 while (current != end) {
351 (*current++) = 255;
352 (*current++) = 255;
353 (*current++) = 255;
354 (*current++) = 0;
355 }
356
357 // Extract the glyph's pixels from the bitmap
358 const std::uint8_t* pixels = bitmap.buffer;
359 if (bitmap.pixel_mode == FT_PIXEL_MODE_MONO) {
360 // Pixels are 1 bit monochrome values
361 for (unsigned int y = padding; y < height - padding; ++y) {
362 for (unsigned int x = padding; x < width - padding; ++x) {
363 // The color channels remain white, just fill the alpha channel
364 std::size_t index = x + y * width;
365 m_pixelBuffer[index * 4 + 3] =
366 ((pixels[(x - padding) / 8]) & (1 << (7 - ((x - padding) % 8)))) ? 255 : 0;
367 }
368 pixels += bitmap.pitch;
369 }
370 }
371 else {
372 // Pixels are 8 bits gray levels
373 for (unsigned int y = padding; y < height - padding; ++y) {
374 for (unsigned int x = padding; x < width - padding; ++x) {
375 // The color channels remain white, just fill the alpha channel
376 std::size_t index = x + y * width;
377 m_pixelBuffer[index * 4 + 3] = pixels[x - padding];
378 }
379 pixels += bitmap.pitch;
380 }
381 }
382
383 // Write the pixels to the texture
384 unsigned int x = static_cast<unsigned int>(glyph.textureRect.position.x) - padding;
385 unsigned int y = static_cast<unsigned int>(glyph.textureRect.position.y) - padding;
386 unsigned int w = static_cast<unsigned int>(glyph.textureRect.size.x) + 2 * padding;
387 unsigned int h = static_cast<unsigned int>(glyph.textureRect.size.y) + 2 * padding;
388 for (unsigned int xi = 0; xi < w; ++xi) {
389 for (unsigned int yi = 0; yi < h; ++yi) {
390 std::size_t index = xi + yi * width;
391 texture.setPixel({x + xi, y + yi},
392 sf::Color{255, 255, 255, m_pixelBuffer[index * 4 + 3]});
393 }
394 }
395 }
396
397 // Delete the FT glyph
398 FT_Done_Glyph(glyphDesc);
399
400 // Done :)
401 return glyph;
402}

404sf::IntRect FontPayload::findGlyphRect(unsigned int width, unsigned int height) const {
405 // Find the line that fits well the glyph
406 Row* row = NULL;
407 float bestRatio = 0;
408 for (std::vector<Row>::iterator it = rows.begin(); it != rows.end() && !row; ++it) {
409 float ratio = static_cast<float>(height) / static_cast<float>(it->height);
410
411 // Ignore rows that are either too small or too high
412 if ((ratio < 0.7f) || (ratio > 1.f)) continue;
413
414 // Check if there's enough horizontal space left in the row
415 if (width > texture.getSize().x - it->width) continue;
416
417 // Make sure that this new row is the best found so far
418 if (ratio < bestRatio) continue;
419
420 // The current row passed all the tests: we can select it
421 row = &*it;
422 bestRatio = ratio;
423 }
424
425 // If we didn't find a matching row, create a new one (10% taller than the glyph)
426 if (!row) {
427 unsigned int rowHeight = height + height / 10;
428 while ((nextRow + rowHeight >= texture.getSize().y) || (width >= texture.getSize().x)) {
429 // Not enough space: resize the texture if possible
430 unsigned int textureWidth = texture.getSize().x;
431 unsigned int textureHeight = texture.getSize().y;
432
433 // Make the texture 2 times bigger
434 sf::Image newTexture;
435 newTexture.resize({textureWidth * 2, textureHeight * 2}, sf::Color(255, 255, 255, 0));
436 newTexture.copy(texture, {0, 0});
437 texture = std::move(newTexture);
438 }
439
440 // We can now create the new row
441 rows.emplace_back(nextRow, rowHeight);
442 nextRow += rowHeight;
443 row = &rows.back();
444 }
445
446 // Find the glyph's rectangle on the selected row
447 sf::IntRect rect(sf::Rect<unsigned int>({row->width, row->top}, {width, height}));
448
449 // Update the row informations
450 row->width += width;
451
452 return rect;
453}

455bool FontPayload::setCurrentSize(unsigned int characterSize) const {
456 // FT_Set_Pixel_Sizes is an expensive function, so we must call it
457 // only when necessary to avoid killing performances
458
459 FT_Face face = static_cast<FT_Face>(m_face);
460 FT_UShort currentSize = face->size->metrics.x_ppem;
461
462 if (currentSize != characterSize) {
463 FT_Error result = FT_Set_Pixel_Sizes(face, 0, characterSize);
464
465 if (result == FT_Err_Invalid_Pixel_Size) {
466 // In the case of bitmap fonts, resizing can
467 // fail if the requested size is not available
468 if (!FT_IS_SCALABLE(face)) {
469 BL_LOG_ERROR << "Failed to set bitmap font size to " << characterSize;
470 BL_LOG_ERROR << "Available sizes are: ";
471 for (int i = 0; i < face->num_fixed_sizes; ++i) {
472 const long size = (face->available_sizes[i].y_ppem + 32) >> 6;
473 BL_LOG_ERROR << size << " ";
474 }
475 BL_LOG_ERROR;
476 }
477 else { BL_LOG_ERROR << "Failed to set font size to " << characterSize; }
478 }
479
480 return result == FT_Err_Ok;
481 }
482
483 return true;
484}
