160 return false;
161 }

165 BL_LOG_ERROR

167 return false;
168 }
169 }
170
171 for (const auto& frame : stateAnim->getFrames()) {
172 if (frame.shards.size() != stateAnim->getFrames().front().shards.size()) {
173 BL_LOG_ERROR << "Animation combine failed: State animation has different shard "
174 "counts than base";
175 return false;
176 }
177 }
178
179 if (enforceSlideshow) {
180 if (animations.get(0).isSlideshow()) {
181 if (!stateAnim->isSlideshow()) {
182 BL_LOG_ERROR << "Animation combine failed: State animation was not Slideshow";
183 return false;
184 }
185 }
186 }
187
188 if (!animations.addDependency(stateAnim.getUUID())) {
189 BL_LOG_ERROR << "Failed to add state index " << stateIndex << " to animation set with "
190 << animations.getSize() << " states";
