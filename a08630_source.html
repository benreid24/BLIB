9glm::vec2 getSize(const resource::Ref<a2d::AnimationData>& animation) {
10 if (!animation || animation->frameCount() == 0) { return {1.f, 1.f}; }
11 sf::Vector2f s = animation->getFrameSize(0);
12 for (unsigned int i = 1; i < animation->frameCount(); ++i) {
13 s.x = std::max(s.x, animation->getFrame(i).size.x);
14 s.y = std::max(s.y, animation->getFrame(i).size.y);
15 }
16 return {s.x, s.y};
17}
