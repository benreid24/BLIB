672template<typename T>
673void Registry::finishComponentAdd(Entity ent, unsigned int cIndex) {
674 ComponentMask::SimpleMask& mask = entityMasks[ent.getIndex()];
675 const ComponentMask::SimpleMask ogMask = mask;
676 ComponentMask::add(mask, cIndex);
677 for (auto& view : views) {
678 if (view->mask.passes(mask)) { view->tryAddEntity(ent); }
679 else if (view->mask.passes(ogMask) && ComponentMask::has(view->mask.excluded, cIndex)) {
680 view->removeEntity(ent);
681 }
682 }
683
684 // ensure parent traits are up to date
685 auto& pool = getPool<T>();
686 for (Entity child : parentGraph.getChildren(ent)) {
687 const std::uint64_t ic = child.getIndex();
688 const ComponentMask mask{.required = entityMasks[ic]};
689 if (mask.contains(pool.ComponentIndex)) { pool.onParentSet(child, ent); }
690 }
691 const Entity parent = parentGraph.getParent(ent);
692 if (parent != InvalidEntity) {
693 const std::uint64_t ip = parent.getIndex();
694 const ComponentMask mask{.required = entityMasks[ip]};
695 if (mask.contains(pool.ComponentIndex)) { pool.onParentSet(ent, parent); }
696 }
697}

727template<typename T>
728void Registry::removeComponent(
729 Entity ent,
730 const Transaction<tx::EntityUnlocked, tx::ComponentRead<>, tx::ComponentWrite<T>>&) {
731 if (!entityExistsLocked(ent)) { return; }
732
733 auto& pool = getPool<T>();
734 const auto i = ent.getIndex();
735 if (i >= entityMasks.size()) { return; }
736 ComponentMask::SimpleMask& mask = entityMasks[i];
737 if (!ComponentMask::contains(mask, pool.ComponentIndex)) { return; }
738
739 // notify pools of parent remove
740 for (Entity child : parentGraph.getChildren(ent)) {
741 const std::uint32_t ic = child.getIndex();
742 const ComponentMask mask{.required = entityMasks[ic]};
743 if (mask.contains(pool.ComponentIndex)) { pool.onParentRemove(ent, child); }
744 }
745 const Entity parent = parentGraph.getParent(ent);
746 if (parent != InvalidEntity) {
747 const std::uint32_t ip = parent.getIndex();
748 const ComponentMask mask{.required = entityMasks[ip]};
749 if (mask.contains(pool.ComponentIndex)) { pool.onParentRemove(parent, ent); }
750 }
751
752 // do remove
753 void* com = pool.queueRemove(ent);
754 for (auto& view : views) {
755 if (ComponentMask::contains(view->mask.required, pool.ComponentIndex)) {
756 view->removeEntity(ent);
757 }
758 else if (ComponentMask::contains(view->mask.optional, pool.ComponentIndex)) {
759 view->nullEntityComponent(ent, com);
760 }
761 }
762 ComponentMask::remove(mask, pool.ComponentIndex);
763}

783template<typename TRequire, typename TOptional, typename TExclude>
784View<TRequire, TOptional, TExclude>* Registry::getOrCreateView() {
785 using TView = View<TRequire, TOptional, TExclude>;
786 std::unique_lock lock(viewMutex);
787
788 // find existing view
789 const std::type_index viewId = typeid(TView);
790 for (auto& view : views) {
791 if (view->id == viewId) return static_cast<TView*>(view.get());
792 }
793
794 // create new view
795 views.emplace_back(new TView(*this));
796 return static_cast<TView*>(views.back().get());

799template<typename T>
800ComponentPool<T>& Registry::getPool() {
801 const std::type_index tid = typeid(T);
802 auto it = poolMap.find(tid);
803 if (it == poolMap.end()) {
804#ifdef BLIB_DEBUG
805 if (poolMap.size() >= ComponentMask::MaxComponentTypeCount) {
806 BL_LOG_CRITICAL << "Maximum component types reached with component: " << tid.name()
807 << ". Specify BLIB_ECS_USE_WIDE_MASK to increase max allowed.";
808 std::exit(1);
809 }
810#endif
811 auto* np = new ComponentPool<T>(*this, poolMap.size());
812 np->emitter.connect(signalChannel);
813 it = poolMap.try_emplace(tid, np).first;
814 componentPools.emplace_back(it->second.get());
815 }
816
817 return static_cast<ComponentPool<T>&>(*it->second);
818}
