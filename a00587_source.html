335 void DefaultRenderer::renderSliderButton(sf::RenderTexture& texture, bool hor,
336  bool increasing) const {
337  texture.clear(sf::Color::Transparent);
338 
339  const float cosRatio = cos(45.f / 180.f * 3.1415f);
340  const sf::Vector2f center = static_cast<sf::Vector2f>(texture.getSize()) / 2.f;
341  const float halfWidth = static_cast<float>(texture.getSize().x) * 0.20f;
342  const float rectWidth = halfWidth / cosRatio;
343  const float rectHeight = static_cast<float>(texture.getSize().y) * 0.08f;
344 
345  sf::RenderTexture subtexture;
346  subtexture.create(texture.getSize().x, texture.getSize().y);
347  subtexture.clear(sf::Color::Transparent);
348 
349  sf::RectangleShape rect({rectWidth, rectHeight});
350  rect.setOrigin(rectWidth / 2, rectHeight / 2);
351  rect.setFillColor(sf::Color::Black);
352  rect.setRotation(45);
353  rect.setPosition(center.x + halfWidth / 2 * cosRatio, center.y * 1.1);
354  subtexture.draw(rect);
355  rect.setRotation(-45);
356  rect.setPosition(center.x - halfWidth / 2 * cosRatio, center.y * 1.1);
357  subtexture.draw(rect);
358 
359  sf::Sprite sprite(subtexture.getTexture());
360  sprite.setOrigin(center);
361  sprite.setPosition(center);
362  if (hor) {
363  if (!increasing)
364  sprite.setRotation(90); // right
365  else
366  sprite.setRotation(270); // left
367  }
368  else if (!increasing)
369  sprite.setRotation(180); // down
370 
371  texture.draw(sprite);
372  texture.display();
373 }
