211template<typename T>
212bool DynamicSSBO<T>::ensureSize(std::uint32_t desiredSize) {
213 if (desiredSize > cpuBuffer.size()) {
214 std::uint32_t newSize = cpuBuffer.size() * 2;
215 while (newSize < desiredSize) { newSize *= 2; }
216
217 dirtyThresh = newSize * 7 / 10;
218 cpuBuffer.resize(newSize);
219 gpuBuffers.visit(
220 [this](vk::Buffer& buffer) { buffer.ensureSize(cpuBuffer.alignedSize()); });
221
222 return true;
223 }
224 return false;

227template<typename T>
228void DynamicSSBO<T>::transferRange(std::uint32_t start, std::uint32_t numElements) {
229 dirtyRanges[vulkanState->currentFrameIndex()] = DirtyRange{start, numElements};
230
231 // determine if we should copy ranges or just copy entire buffer
232 std::uint32_t totalSize = 0;
233 for (auto& range : dirtyRanges) { totalSize += range.size; }
234 if (totalSize >= dirtyThresh) {
235 std::memcpy(
236 gpuBuffers.current().getMappedMemory(), cpuBuffer.data(), cpuBuffer.alignedSize());
237 }
238 else if (totalSize > 0) {
239 // consolidate ranges before copy
240 std::uint32_t start = std::numeric_limits<std::uint32_t>::max();
241 std::uint32_t end = 0;
242 for (auto& range : dirtyRanges) {
243 if (range.size == 0) continue;
244 const std::uint32_t ne = range.start + range.size;
245 start = range.start < start ? range.start : start;
246 end = ne > end ? ne : end;
247 }
248 if (end > start) {
249 const std::uint32_t size = (end - start) * cpuBuffer.elementSize();
250 void* dst = static_cast<char*>(gpuBuffers.current().getMappedMemory()) +
251 (start * cpuBuffer.elementSize());
252 std::memcpy(dst, &cpuBuffer[start], size);
253 }
254 }
