454template<typename T>
455typename ObjectPool<T>::Entry* ObjectPool<T>::doAdd() {
456 ++trackedSize;
457 if (next >= 0) {
458 Entry* e = &pool[next];
459 next = e->next();
460 return e;
461 }
462 else {
463 if (pool.size() < pool.capacity() || policy == GrowthPolicy::ExpandBuffer) {
464 pool.emplace_back();
465 return &pool.back();
466 }
467 else {
468 switch (policy) {
469 case GrowthPolicy::FailContinue:
470 BL_LOG_WARN << typeid(T).name() << " pool exceeded capacity (" << capacity()
471 << "). Increase allocation";
472 return nullptr;
473 case GrowthPolicy::FailTerminate:
474 BL_LOG_CRITICAL << typeid(T).name() << " pool exceeded capacity (" << capacity()
475 << "). Increase allocation. Terminating";
476 std::exit(1);
477 break;
478 default:
479 // unreachable
480 return nullptr;
481 }
482 }
