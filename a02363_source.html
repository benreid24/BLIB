29void InputAttachmentInstance::bind(VkCommandBuffer commandBuffer, VkPipelineBindPoint bindPoint,
30 VkPipelineLayout layout, std::uint32_t setIndex) const {
31 const vk::AttachmentSet* attachments = source.get(renderer.vulkanState().currentFrameIndex());
32 auto& cached = cachedViews.getRaw(renderer.vulkanState().currentFrameIndex());
33
34 if (!attachments) {
35 BL_LOG_WARN << "Binding input attachment with no attachments";
36 return;
37 }
38
39 bool changed = false;
40 const std::uint32_t bi = startIndex + attachments->getOutputIndex();
41 for (unsigned int i = 0; i < attachmentCount; ++i) {
42 if (cached[i] != attachments->getImageView(i + bi)) {
43 cached[i] = attachments->getImageView(i + bi);
44 changed = true;
45 break;
46 }
47 }
48
49 // update descriptor on change
50 if (changed) {
51 std::array<VkDescriptorImageInfo, 8> imageInfo{};
52 for (unsigned int i = 0; i < attachmentCount; ++i) {
53 imageInfo[i].imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
54 imageInfo[i].imageView = attachments->getImageView(i + bi);
55 imageInfo[i].sampler = sampler;
56 }
57
58 std::array<VkWriteDescriptorSet, 8> write{};
59 for (unsigned int i = 0; i < attachmentCount; ++i) {
60 write[i].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
61 write[i].dstSet = descriptorSets.getRaw(renderer.vulkanState().currentFrameIndex());
62 write[i].dstBinding = i;
63 write[i].dstArrayElement = 0;
64 write[i].descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
65 write[i].descriptorCount = 1;
66 write[i].pImageInfo = &imageInfo[i];
67 }
68
69 vkUpdateDescriptorSets(
70 renderer.vulkanState().getDevice(), attachmentCount, write.data(), 0, nullptr);
71 }
72
73 vkCmdBindDescriptorSets(
74 commandBuffer, bindPoint, layout, setIndex, 1, &descriptorSets.current(), 0, nullptr);
