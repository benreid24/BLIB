43void Overlay::renderScene(scene::SceneRenderContext& ctx) {
44 std::unique_lock lock(objectMutex);
45 ecs::Transaction<ecs::tx::EntityUnlocked, ecs::tx::ComponentRead<ovy::OverlayObject>>
46 transaction(engine.ecs());
47
48 std::copy(roots.begin(), roots.end(), std::inserter(renderStack, renderStack.begin()));
49
50 if (needRefreshAll ||
51 static_cast<std::uint32_t>(ctx.parentViewport().width) != cachedTargetSize.x ||
52 static_cast<std::uint32_t>(ctx.parentViewport().height) != cachedTargetSize.y) {
53 cachedParentViewport = ctx.parentViewport();
54 cachedTargetSize.x = static_cast<std::uint32_t>(ctx.parentViewport().width);
55 cachedTargetSize.y = static_cast<std::uint32_t>(ctx.parentViewport().height);
56 needRefreshAll = false;
57 refreshAll();
58 }
59
60 VkPipelineLayout currentPipelineLayout = nullptr;
61 VkPipeline currentPipeline = nullptr;
62 UpdateSpeed currentSpeed{};
63 while (!renderStack.empty()) {
64 ovy::OverlayObject& obj = *renderStack.back();
65 renderStack.pop_back();
66
67 if (obj.hidden) { continue; }
68
69 if (!obj.entity.flagSet(ecs::Flags::Dummy)) {
70 vkCmdSetScissor(ctx.getCommandBuffer(), 0, 1, &obj.cachedScissor);
71
72 const vk::PipelineLayout& layout = obj.pipeline->getLayout();
73 const VkPipelineLayout vkl = layout.rawLayout();
74 const VkPipeline vkp =
75 obj.pipeline->getRawPipeline(ctx.getRenderPhase(), ctx.currentRenderPass());
76 if (vkl != currentPipelineLayout || currentSpeed != obj.sceneKey.updateFreq) {
77 currentSpeed = obj.sceneKey.updateFreq;
78 currentPipelineLayout = vkl;
79 currentPipeline = vkp;
80 ctx.bindPipeline(*obj.pipeline);
81 ctx.bindDescriptors(
82 vkl, obj.sceneKey.updateFreq, obj.descriptors.data(), obj.descriptorCount);
83 }
84 else if (currentPipeline != vkp) {
85 currentPipeline = vkp;
86 ctx.bindPipeline(*obj.pipeline);
87 }
88 for (std::uint8_t i = obj.perObjStart; i < obj.descriptorCount; ++i) {
89 obj.descriptors[i]->bindForObject(ctx, vkl, i, obj.sceneKey);
90 }
91 ctx.renderObject(obj);
92 }
93
94 std::copy(obj.getChildren().rbegin(),
95 obj.getChildren().rend(),
96 std::inserter(renderStack, renderStack.end()));
97 }

129void Overlay::doObjectRemoval(scene::SceneObject* object, mat::MaterialPipeline*) {
130 ovy::OverlayObject* obj = static_cast<ovy::OverlayObject*>(object);
131 obj->overlayViewport = nullptr;
132
133 auto childCopy = obj->getChildren();
134 for (ovy::OverlayObject* child : childCopy) {
135 if (engine.ecs().getEntityParentDestructionBehavior(child->entity) !=
136 ecs::ParentDestructionBehavior::OrphanedByParent &&
137 child->sceneKey.sceneId != scene::Key::InvalidSceneId) {
138 removeObject(child);
139 }
140 }
141
142 const auto it = std::find(roots.begin(), roots.end(), obj);
143 if (it != roots.end()) {
144 roots.erase(it);
145 sortRoots();
146 }
147
148 for (unsigned int i = 0; i < obj->descriptorCount; ++i) {
149 obj->descriptors[i]->releaseObject(obj->entity, object->sceneKey);
150 }
151 objects.release(obj->sceneKey);
152 engine.ecs().removeComponent<ovy::OverlayObject>(obj->entity);
153 bl::event::Dispatcher::dispatch<rc::event::SceneObjectRemoved>({this, obj->entity});
154}
