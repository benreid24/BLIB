56void Framebuffer::beginRender(VkCommandBuffer commandBuffer, const VkRect2D& region,
57 const VkClearValue* clearColors, std::uint32_t clearColorCount,
58 bool vp, VkRenderPass rpo, bool shouldClear) {
59#ifdef BLIB_DEBUG
60 if (target == nullptr) {
61 throw std::runtime_error("Framebuffer render started without specifying target");
62 }
63#endif
64
65 bool cleared = false;
66 if (renderStartCount == 0) {
67 // begin render pass
68 VkRenderPassBeginInfo renderPassInfo{};
69 renderPassInfo.sType = VK_STRUCTURE_TYPE_RENDER_PASS_BEGIN_INFO;
70 renderPassInfo.renderPass = rpo != nullptr ? rpo : renderPass->rawPass();
71 renderPassInfo.framebuffer = framebuffer;
72 renderPassInfo.renderArea = region;
73 renderPassInfo.clearValueCount = clearColorCount;
74 renderPassInfo.pClearValues = clearColors;
75 vkCmdBeginRenderPass(commandBuffer, &renderPassInfo, VK_SUBPASS_CONTENTS_INLINE);
76 cleared = renderPass->getCreateParams().isClearedOnStart();
77 }
78 if (shouldClear && !cleared) {
79 // if we have resolve attachments we need to avoid clearing them
80 std::uint32_t colorCount = clearColorCount;
81 if (target->getOutputIndex() > 0) {
82 colorCount = std::min(colorCount, target->getOutputIndex());
83 }
84
85 // if we have a depth attachment we will handle it outside of the loop
86 const bool hasDepth =
87 target->getImageAspect(target->getAttachmentCount() - 1) & VK_IMAGE_ASPECT_DEPTH_BIT;
88 const std::uint32_t attachmentCount = hasDepth ? colorCount + 1 : colorCount;
89
90 VkClearAttachment attachments[AttachmentSet::MaxAttachments]{};
91 for (unsigned int i = 0; i < colorCount; ++i) {
92 attachments[i].aspectMask = target->getImageAspects()[i];
93 attachments[i].colorAttachment = i;
94 attachments[i].clearValue = clearColors[i];
95 }
96 if (hasDepth) {
97 attachments[colorCount].aspectMask =
98 target->getImageAspect(target->getAttachmentCount() - 1);
99 attachments[colorCount].colorAttachment = VK_ATTACHMENT_UNUSED;
100 attachments[colorCount].clearValue = clearColors[clearColorCount - 1];
101 }
102
103 VkClearRect rect;
104 rect.rect = region;
105 rect.baseArrayLayer = 0;
106 rect.layerCount = target->getLayerCount();
107
108 vkCmdClearAttachments(commandBuffer, attachmentCount, attachments, 1, &rect);
109 }
110
111 vkCmdSetScissor(commandBuffer, 0, 1, &region);
112
113 if (vp) {
114 VkViewport viewport;
115 viewport.x = static_cast<float>(region.offset.x);
116 viewport.y = static_cast<float>(region.offset.y);
117 viewport.width = static_cast<float>(region.extent.width);
118 viewport.height = static_cast<float>(region.extent.height);
119 viewport.minDepth = 0.f;
120 viewport.maxDepth = 1.f;
121
122 vkCmdSetViewport(commandBuffer, 0, 1, &viewport);
123 }
124
125 ++renderStartCount;
126}
