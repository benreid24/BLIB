143 void Renderer::renderFrame() {
144  // kick off transfers
145  textures.onFrameStart();
146  for (vk::RenderTexture* rt : renderTextures) { rt->handleDescriptorSync(); }
147  if (commonObserver.hasScene()) { commonObserver.handleDescriptorSync(); }
148  else {

150  }
151  state.transferEngine.executeTransfers();

153  // begin frame

155  VkCommandBuffer commandBuffer = nullptr;
156  state.beginFrame(currentFrame, commandBuffer);
157  framebuffers.current().recreateIfChanged(*currentFrame);
158 
159  // begin render texture rendering
160  for (vk::RenderTexture* rt : renderTextures) { rt->renderScene(commandBuffer); }
161 
162  const auto clearDepthBuffer = [this, commandBuffer]() {
163  VkClearAttachment attachment{};
164  attachment.aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT;
165  attachment.clearValue = clearColors[1];
166 
167  VkClearRect rect{};
168  rect.rect.extent.width = renderRegion.width;
169  rect.rect.extent.height = renderRegion.height;
170  rect.rect.offset.x = renderRegion.left;
171  rect.rect.offset.y = renderRegion.top;
172  rect.baseArrayLayer = 0;
173  rect.layerCount = 1;
174 
175  vkCmdClearAttachments(commandBuffer, 1, &attachment, 1, &rect);
176  };
177 
178  // record commands to render scenes
179  for (auto& o : observers) { o->renderScene(commandBuffer); }
180  if (commonObserver.hasScene()) {
181  clearDepthBuffer();
182  commonObserver.renderScene(commandBuffer);
183  }
184 
185  // perform render pass for final scene renders and overlays
186  framebuffers.current().beginRender(commandBuffer,
187  {{0, 0}, currentFrame->renderExtent()},
188  clearColors,

190  false);

194  if (commonObserver.hasScene()) {
195  clearDepthBuffer();

197  }
