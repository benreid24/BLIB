40void RenderGraph::build() {
41 needsRebuild = false;
42 for (auto& task : tasks) { task->assets.init(task->assetTags); }
43 assets.reset();

45 std::vector<Task*> missingInputs;
46 missingInputs.reserve(tasks.size());

48 const auto addTaskMissingInput = [&missingInputs](Task* task) {
49 if (std::find(missingInputs.rbegin(), missingInputs.rend(), task) == missingInputs.rend()) {
50 missingInputs.emplace_back(task);
51 }
52 };

54 const auto tryLinkConcreteInput =
55 [this, &addTaskMissingInput](Task* task, const TaskInput& tags, GraphAsset*& input) {
56 for (auto tag : tags.options) {
57 input = assets.getAssetForInput(tag, "");
58 if (input) return;
59 }
60 addTaskMissingInput(task);
61 };
62
63 // first link all inputs for external assets, mark tasks missing inputs
64 for (auto& task : tasks) {
65 for (unsigned int i = 0; i < task->assetTags.requiredInputs.size(); ++i) {
66 tryLinkConcreteInput(
67 task.get(), task->assetTags.requiredInputs[i], task->assets.requiredInputs[i]);
68 }
69 for (unsigned int i = 0; i < task->assetTags.optionalInputs.size(); ++i) {
70 tryLinkConcreteInput(
71 task.get(), task->assetTags.optionalInputs[i], task->assets.optionalInputs[i]);
72 }
73 }

75 const auto findAssetCreator =
76 [this](Task* task, const TaskInput& tags, GraphAsset*& asset) -> bool {
77 for (auto tag : tags.options) { // for all input slot options
78 for (auto& creatorTask : tasks) { // for all tasks
79 if (creatorTask.get() != task) { // that are not the requesting task
80 for (auto& output :
81 creatorTask->assetTags.outputs) { // for each task output slot
82 for (auto& option : output.options) { // for each output slot option
83 // if the slot matches and the task is allowed to create the asset
84 if (option.tag == tag &&
85 option.createMode == TaskOutput::CreatedByTask) {
86 const unsigned int i =
87 &output - creatorTask->assetTags.outputs.data();
88 auto*& outputPtr = creatorTask->assets.outputs[i];
89
90 // create the asset if it does not exist
91 if (!outputPtr) {
92 asset = assets.createAsset(tag, output.purpose);
93 asset->outputtedBy.emplace_back(creatorTask.get());
94 outputPtr = asset;
95 return true;
96 }
97
98 // allow tasks to share already created assets if permitted
99 else if (tags.shareMode != TaskInput::Exclusive &&
100 option.shareMode != TaskOutput::Exclusive &&
101 outputPtr->asset->getTag() == tag) {
102 outputPtr->outputtedBy.emplace_back(task);
103 asset = outputPtr;
104 return true;
105 }
106 }
107 }
108 }
109 }
110 }
111 }
112 return false;
113 };
114
115 // now find tasks that can provide missing inputs
116 for (Task* task : missingInputs) {
117 for (unsigned int i = 0; i < task->assetTags.requiredInputs.size(); ++i) {
118 if (task->assets.requiredInputs[i] == nullptr) {
119 if (!findAssetCreator(
120 task, task->assetTags.requiredInputs[i], task->assets.requiredInputs[i])) {
121 throw std::runtime_error("Failed to find or create required input for task");
122 }
123 }
124 }
125 for (unsigned int i = 0; i < task->assetTags.optionalInputs.size(); ++i) {
126 if (task->assets.optionalInputs[i] == nullptr) {
127 findAssetCreator(
128 task, task->assetTags.optionalInputs[i], task->assets.optionalInputs[i]);
129 }
130 }
131 }
132
133 auto& missingOutputs = missingInputs;
134 missingOutputs.clear();
135
136 const auto linkRemainingOutputs = [this, &missingOutputs](Task* task) -> bool {
137 bool allLinked = true;
138 for (unsigned int i = 0; i < task->assets.outputs.size(); ++i) {
139 if (task->assets.outputs[i]) { continue; }
140
141 const auto& params = task->assetTags.outputs[i];
142 for (auto& option : params.options) {
143 if (option.shareMode != TaskOutput::Shared) { continue; }
144
145 GraphAsset* asset = nullptr;
146
147 // if the output is created by another task, try to find it
148 if (!params.sharedWith.empty()) {
149 asset = assets.getAssetForSharedOutput(
150 option.tag, params.sharedWith, params.purpose);
151 }
152
153 // otherwise search for external assets
154 else { asset = assets.getAssetForOutput(option.tag, task, params.purpose); }
155
156 if (asset) {
157 asset->outputtedBy.emplace_back(task);
158 task->assets.outputs[i] = asset;
159 break;
160 }
161 }
162 if (!task->assets.outputs[i]) { allLinked = false; }
163 }
164 return allLinked;
165 };
166
167 // link remaining task outputs to external (or other task) assets
168 for (auto& task : tasks) {
169 if (!linkRemainingOutputs(task.get())) { missingOutputs.emplace_back(task.get()); }
170 }
171
172 // do more passes until no more outputs can be linked
173 while (!missingOutputs.empty()) {
174 bool madeProgress = false;
175 for (auto it = missingOutputs.begin(); it != missingOutputs.end();) {
176 if (linkRemainingOutputs(*it)) {
177 it = missingOutputs.erase(it);
178 madeProgress = true;
179 }
180 else { ++it; }
181 }
182 // if no progress was made, throw an error
183 if (!madeProgress) { throw std::runtime_error("Failed to link all task outputs"); }
184 }
185
186 // build timeline
187 timeline.build(tasks, assets.getFinalOutput());
188
189 // notify tasks of graph creation
190 for (auto& task : tasks) { task->onGraphInit(); }
191}
