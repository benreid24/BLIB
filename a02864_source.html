46void SingleShape2D::ensureUpdated() {

48 dirty = false;
49
50 // determine required vertex and index counts
51 const std::uint32_t vertexCount = calculateRequiredVertices();
52 const std::uint32_t indexCount = calculateRequiredIndices();
53
54 // create the index buffer
55 auto& ib = component().indexBuffer;
56 if (ib.indexCount() < indexCount || ib.vertexCount() < vertexCount) {
57 ib.create(engine().renderer(), vertexCount, indexCount);
58 }
59
