56bool doReduction(const Grammar& g, std::vector<Node::Ptr>& nonterminals,
57 std::vector<Node::Ptr>& terminals, std::stack<unsigned int>& states,
58 unsigned int& state, const Grammar::Production& prod, std::string* res) {
59 Node::Ptr nt(new Node());
60 nt->type = prod.result;
61 nt->children.resize(prod.set.size());
62
63 const int c = terminalCount(g, prod) + nonterminalCount(g, prod);
64 for (int i = 0; i < c; ++i) {
65 if (!states.empty())
66 states.pop();
67 else {
68 ErrorReporter(res).log(terminals.back())
69 << "Internal parser error, not enough states for reduction";
70 return false;
71 }
72 }
73 if (!states.empty()) state = states.top();
74
75 for (unsigned int i = 0; i < prod.set.size(); ++i) {
76 const int pi = prod.set.size() - i - 1;
77 const int si = nt->children.size() - i - 1;
78 if (g.terminal(prod.set[pi])) {
79 nt->children[si] = terminals.back();
80 terminals.pop_back();
81 }
82 else if (g.nonterminal(prod.set[pi])) {
83 nt->children[si] = nonterminals.back();
84 nonterminals.pop_back();
85 }
86 else {
87 ErrorReporter(res).log(terminals.back())
88 << "Internal parser error, grammar production uses undefined symbol";
89 return false;
90 }
91 nt->children[si]->parent = nt.get();
92 if (nt->children[si]->type != prod.set[pi]) {
93 ErrorReporter(res).log(nt->children[si])
94 << "Internal parser error, table specified reduction that does "
95 "not match the given production";
96 return false;
97 }
98 }
99
100 nt->sourceLine = nt->children.front()->sourceLine;
101 nt->sourceColumn = nt->children.front()->sourceColumn;
102 nonterminals.push_back(nt);
103 return true;
104}

143 const auto reduce = [this, &terminals, &nonterminals, &state, &stateHist, res](
144 const Grammar::Production& p) -> bool {
145 return doReduction(grammar, nonterminals, terminals, stateHist, state, p, res);
146 };

148 while (!stateHist.empty()) {

151 const Action& action = table[state].actions.at(lookahead->type);
152 if (action.type == Action::Shift) {
153 shift();
154 if (table[state].gotos.find(lookahead->type) != table[state].gotos.end()) {
155 state = table[state].gotos.at(lookahead->type);
156 stateHist.push(state);
157 }
158 else {
159 ErrorReporter(res).log(lookahead) << "Internal parser error, no goto in table";
160 return nullptr;

162 }
163 else if (action.type == Action::Reduce) {

166 << "Internal parser error, no production for reduce action";
167 return nullptr;

169 if (!reduce(action.reduction.value())) { return nullptr; }
170 if (stateHist.empty() || nonterminals.back()->type == Start) break;
171 Node::Type t = nonterminals.back()->type;
172 if (table[state].gotos.find(t) != table[state].gotos.end()) {
173 state = table[state].gotos.at(t);
174 stateHist.push(state);
175 }
176 else {

185 }
186 }

188 ErrorReporter(res).log(lookahead) << "Unexpected token '" << lookahead->data << "'";
189 return nullptr;
190 }
191 }
192
193 if (nonterminals.empty()) {
194 if (!terminals.empty())
195 ErrorReporter(res).log(terminals.back()) << "Unable to perform reduction";
196 else
197 BL_LOG_ERROR << "Unable to perform reduction";
198 return nullptr;
199 }
200 if (!terminals.empty()) {
201 ErrorReporter(res).log(terminals.back()) << "Unable to process remaining input";
202 return nullptr;
203 }
204 if (nonterminals.size() != 1) {
205 ErrorReporter(res).log(nonterminals[0]) << "Unable to reduce parse tree";
206 return nullptr;
207 }
208 if (nonterminals.back()->type != Start) {
209 ErrorReporter(res).log(nonterminals.back()) << "Invalid root symbol";
210 return nullptr;
211 }
212 if (nonterminals.back()->children[0]->type != grammar.getStart()) {
213 ErrorReporter(res).log(nonterminals.back()) << "Did not reach grammar start symbol";
214 return nullptr;
215 }
216
217 Node::Ptr result = nonterminals.back()->children[0];
218 result->parent = nullptr;
219 return result;

222bool Parser::generateTables() {
223 table.clear();
224
225 Grammar::Production s0 = {Start, Node::Sequence({grammar.getStart()})};
226 Grammar::Item i0(s0);
227 getState(grammar.closure(i0), true);
228
229 for (unsigned int i = 0; i < table.size(); ++i) {
230 Grammar::ItemSet state = table[i].state;
231 for (auto j = state.items().begin(); j != state.items().end(); ++j) {
232 const Grammar::Item& item = *j;
233 // Enter goto/shift for items not at end
234 if (!item.final()) {
235 Node::Type t = item.production.set[item.cursor];
236 Grammar::ItemSet gotoSet;
237 gotoSet.add(item.next());
238 for (const Grammar::Item& ot : table[i].state.items()) {
239 if (!ot.final() && ot.production.set[ot.cursor] == t) gotoSet.add(ot.next());
240 }
241 gotoSet = grammar.closure(gotoSet);
242 table[i].gotos[t] = getState(gotoSet, true);
243 // Shift if terminal
244 if (grammar.terminal(t)) table[i].actions[t] = Action::shift();
245 }
246 // Enter reduce for items at end
247 else {
248 Node::Sequence follow = grammar.followSet(item.production.set.back());
249 for (Node::Type t : follow) {
250 if (table[i].actions.find(t) != table[i].actions.end()) {
251 if (table[i].actions[t].type == Action::Shift) {
252 BL_LOG_ERROR << "Shift-Reduce error";
253 table.clear();
254 return false;
255 }
256 else {
257 BL_LOG_ERROR << "Reduce-Reduce error";
258 table.clear();
259 return false;
260 }
261 }
262 table[i].actions[t] = Action::reduce(item);
263 }
264 }
265 }
266 }
267
268 return true;
269}
