54std::uint32_t PipelineLayout::updateDescriptorSets(ds::DescriptorSetInstanceCache& cache,
55 ds::DescriptorSetInstance** sets,
56 std::uint32_t descriptorCount,
57 ecs::Entity entity, std::uint32_t sceneId,
58 UpdateSpeed updateSpeed) const {
59 const scene::Key key{updateSpeed, sceneId};
60
61 // capture old sets
62 std::array<ds::DescriptorSetInstance*, 4> ogSets{nullptr, nullptr, nullptr, nullptr};
63 for (std::uint32_t i = 0; i < descriptorCount; ++i) { ogSets[i] = sets[i]; }
64
65 // create new sets
66 const std::uint32_t newSetCount = initDescriptorSets(cache, sets);
67
68 // call allocate for new descriptor sets
69 for (std::uint8_t i = 0; i < newSetCount; ++i) {
70 for (std::uint8_t j = 0; j < descriptorCount; ++j) {
71 if (ogSets[j] == sets[i]) {
72 ogSets[j] = nullptr;
73 goto noAdd;
74 }
75 }
76
77 if (!sets[i]->allocateObject(entity, key)) {
78 BL_LOG_ERROR << "Unable to update entity " << entity << " (scene id: " << sceneId
79 << ") to new layout due to descriptor data missing";
80 }
81
82 noAdd:
83 continue;
84 }
85
86 // call remove for ones we no longer use
87 for (std::uint8_t i = 0; i < descriptorCount; ++i) {
88 if (ogSets[i] != nullptr) { ogSets[i]->releaseObject(entity, key); }
89 }
90
91 return newSetCount;
92}
