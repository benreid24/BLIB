88void TextureExport::performCopy() {
89 // create resources
90 commandPool = vulkanState.createCommandPool(VK_COMMAND_POOL_CREATE_TRANSIENT_BIT);
91 VkCommandBufferAllocateInfo cbAllocInfo{};
92 cbAllocInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_ALLOCATE_INFO;
93 cbAllocInfo.level = VK_COMMAND_BUFFER_LEVEL_PRIMARY;
94 cbAllocInfo.commandPool = commandPool;
95 cbAllocInfo.commandBufferCount = 1;
96 vkCheck(vkAllocateCommandBuffers(vulkanState.getDevice(), &cbAllocInfo, &commandBuffer));
97
98 VkFenceCreateInfo createFence{};
99 createFence.sType = VK_STRUCTURE_TYPE_FENCE_CREATE_INFO;
100 createFence.flags = 0;
101 vkCheck(vkCreateFence(vulkanState.getDevice(), &createFence, nullptr, &fence));
102
103 // create staging image to copy to
104 vulkanState.createImage(srcExtent.width,
105 srcExtent.height,
106 1,
107 OutputFormat,
108 VK_IMAGE_TILING_LINEAR,
109 VK_IMAGE_USAGE_TRANSFER_SRC_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT,
110 0,
111 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT |
112 VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
113 &destImage,
114 &destAlloc,
115 &destAllocInfo,
116 VMA_ALLOCATION_CREATE_HOST_ACCESS_RANDOM_BIT);
117 // Begin recording commands
118 VkCommandBufferBeginInfo begin{};
119 begin.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
120 begin.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
121 vkCheck(vkBeginCommandBuffer(commandBuffer, &begin));
122
123 // transition image to copy source layout
124 vulkanState.transitionImageLayout(
125 commandBuffer, srcImage, srcLayout, VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL);
126
127 // transition dest image to dest layout
128 vulkanState.transitionImageLayout(
129 commandBuffer, destImage, VK_IMAGE_LAYOUT_UNDEFINED, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL);
130
131 const VkFormatProperties srcFormatProps = vulkanState.getFormatProperties(srcFormat);
132 const VkFormatProperties convFormatProps = vulkanState.getFormatProperties(OutputFormat);
133
134 // blit if supported otherwise copy
135 if ((srcFormatProps.optimalTilingFeatures & VK_FORMAT_FEATURE_BLIT_SRC_BIT) != 0 &&
136 (convFormatProps.linearTilingFeatures & VK_FORMAT_FEATURE_BLIT_DST_BIT) != 0) {
137 const VkOffset3D blitSize{static_cast<std::int32_t>(srcExtent.width),
138 static_cast<std::int32_t>(srcExtent.height),
139 1};
140 VkImageBlit imageBlitRegion{};
141 imageBlitRegion.srcSubresource.aspectMask = srcAspect;
142 imageBlitRegion.srcSubresource.layerCount = 1;
143 imageBlitRegion.srcOffsets[1] = blitSize;
144 imageBlitRegion.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
145 imageBlitRegion.dstSubresource.layerCount = 1;
146 imageBlitRegion.dstOffsets[1] = blitSize;
147
148 vkCmdBlitImage(commandBuffer,
149 srcImage,
150 VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
151 destImage,
152 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
153 1,
154 &imageBlitRegion,
155 VK_FILTER_NEAREST);
156 }
157 else {
158 // check if the format requires swizzling
159 requiresManualConversion = requiresSwizzle(srcFormat);
160
161 VkImageCopy imageCopyRegion{};
162 imageCopyRegion.srcSubresource.aspectMask = srcAspect;
163 imageCopyRegion.srcSubresource.layerCount = 1;
164 imageCopyRegion.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
165 imageCopyRegion.dstSubresource.layerCount = 1;
166 imageCopyRegion.extent = srcExtent;
167
168 vkCmdCopyImage(commandBuffer,
169 srcImage,
170 VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
171 destImage,
172 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
173 1,
174 &imageCopyRegion);
175 }
176
177 // insert barrier to block until copy is complete
178 VkImageMemoryBarrier barrier{};
179 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
180 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL;
181 barrier.newLayout = srcLayout;
182 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
183 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
184 barrier.image = srcImage;
185 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
186 barrier.subresourceRange.baseMipLevel = 0;
187 barrier.subresourceRange.levelCount = 1;
188 barrier.subresourceRange.baseArrayLayer = 0;
189 barrier.subresourceRange.layerCount = 1;
190 barrier.srcAccessMask = VK_ACCESS_TRANSFER_READ_BIT;
191 barrier.dstAccessMask = VK_ACCESS_MEMORY_WRITE_BIT;
192 vkCmdPipelineBarrier(commandBuffer,
193 VK_PIPELINE_STAGE_TRANSFER_BIT,
194 VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT,
195 0,
196 0,
197 nullptr,
198 0,
199 nullptr,
200 1,
201 &barrier);
202
203 // transition dest image to general format
204 VkImageMemoryBarrier destBarrier{};
205 destBarrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
206 destBarrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
207 destBarrier.newLayout = VK_IMAGE_LAYOUT_GENERAL;
208 destBarrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
209 destBarrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
210 destBarrier.image = destImage;
211 destBarrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
212 destBarrier.subresourceRange.baseMipLevel = 0;
213 destBarrier.subresourceRange.levelCount = 1;
214 destBarrier.subresourceRange.baseArrayLayer = 0;
215 destBarrier.subresourceRange.layerCount = 1;
216 destBarrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
217 destBarrier.dstAccessMask = VK_ACCESS_MEMORY_READ_BIT;
218 vkCmdPipelineBarrier(commandBuffer,
219 VK_PIPELINE_STAGE_TRANSFER_BIT,
220 VK_PIPELINE_STAGE_TRANSFER_BIT,
221 0,
222 0,
223 nullptr,
224 0,
225 nullptr,
226 1,
227 &destBarrier);
228
229 // submit commands
230 vkCheck(vkEndCommandBuffer(commandBuffer));
231 VkSubmitInfo submit{};
232 submit.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
233 submit.pCommandBuffers = &commandBuffer;
234 submit.commandBufferCount = 1;
235 vulkanState.submitCommandBuffer(submit, fence);
236}
