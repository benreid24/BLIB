70void Swapchain::beginFrame(StandardAttachmentSet*& renderFrame, VkCommandBuffer& cb) {
71 // wait for prior frame
72 vkCheck(vkWaitForFences(
73 vulkanState.device, 1, &frameData.current().commandBufferFence, VK_TRUE, UINT64_MAX));
74 vkCheck(vkResetFences(vulkanState.device, 1, &frameData.current().commandBufferFence));
75
76 // recreate if out of date
77 if (outOfDate) { recreate(); }
78
79 // acquire next image
80 VkResult acquireResult;
81 do {
82 acquireResult = vkAcquireNextImageKHR(vulkanState.device,
83 swapchain,
84 UINT64_MAX,
85 frameData.current().imageAvailableSemaphore,
86 VK_NULL_HANDLE,
87 &currentImageIndex);
88 if (acquireResult == VK_ERROR_OUT_OF_DATE_KHR) { recreate(); }
89 } while (acquireResult == VK_ERROR_OUT_OF_DATE_KHR);
90 if (acquireResult != VK_SUCCESS && acquireResult != VK_SUBOPTIMAL_KHR) {
91 throw std::runtime_error("Failed to acquire swapchain image");
92 }
93
94 // reset prior command buffer
95 vkCheck(vkResetCommandBuffer(frameData.current().commandBuffer, 0));
96
97 // begin command buffer
98 VkCommandBufferBeginInfo beginInfo{};
99 beginInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
100 beginInfo.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
101 if (vkBeginCommandBuffer(frameData.current().commandBuffer, &beginInfo) != VK_SUCCESS) {
102 throw std::runtime_error("Failed to begin recording command buffer");
103 }
104
105 // copy output
106 renderFrame = &renderFrames[currentImageIndex];
107 cb = frameData.current().commandBuffer;
108}

110void Swapchain::completeFrame() {
111 // end command buffer
112 vkCheck(vkEndCommandBuffer(frameData.current().commandBuffer));
113
114 // submit to queue
115 VkSemaphore waitSemaphores[] = {frameData.current().imageAvailableSemaphore};
116 VkSemaphore signalSemaphores[] = {frameData.current().renderFinishedSemaphore};
117 VkPipelineStageFlags waitStages[] = {VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT};
118
119 VkSubmitInfo submitInfo{};
120 submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
121 submitInfo.waitSemaphoreCount = std::size(waitSemaphores);
122 submitInfo.pWaitSemaphores = waitSemaphores;
123 submitInfo.pWaitDstStageMask = waitStages;
124 submitInfo.signalSemaphoreCount = std::size(signalSemaphores);
125 submitInfo.pSignalSemaphores = signalSemaphores;
126 submitInfo.commandBufferCount = 1;
127 submitInfo.pCommandBuffers = &frameData.current().commandBuffer;
128 if (vkQueueSubmit(
129 vulkanState.graphicsQueue, 1, &submitInfo, frameData.current().commandBufferFence) !=
130 VK_SUCCESS) {
131 throw std::runtime_error("Failed to submit draw command buffer");
132 }
133
134 // trigger swap chain
135 VkPresentInfoKHR presentInfo{};
136 presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
137 presentInfo.waitSemaphoreCount = 1;
138 presentInfo.pWaitSemaphores = &frameData.current().renderFinishedSemaphore;
139 presentInfo.swapchainCount = 1;
140 presentInfo.pSwapchains = &swapchain;
141 presentInfo.pImageIndices = &currentImageIndex;
142 presentInfo.pResults = nullptr; // Optional
143
144 const VkResult presentResult = vkQueuePresentKHR(vulkanState.presentQueue, &presentInfo);
145 switch (presentResult) {
146 case VK_ERROR_OUT_OF_DATE_KHR:
147 case VK_SUBOPTIMAL_KHR:
148 recreate();
149 break;
150 case VK_SUCCESS:
151 break;
152 default:
153 throw std::runtime_error("Failed to present swap chain image");
154 }
155}

201void Swapchain::createSwapchain() {
202 // cache the surface that we created with
203 oldSurface = vulkanState.surface;
204
205 // get supported swap chain details
206 SwapChainSupportDetails swapChainSupport;
207 swapChainSupport.populate(vulkanState.physicalDevice, vulkanState.surface);
208 const VkSurfaceFormatKHR& surfaceFormat = swapChainSupport.swapSurfaceFormat();
209
210 // image count in swap chain
211 std::uint32_t imageCount = swapChainSupport.capabilities.minImageCount + 1;
212 if (swapChainSupport.capabilities.maxImageCount > 0 &&
213 imageCount > swapChainSupport.capabilities.maxImageCount) {
214 imageCount = swapChainSupport.capabilities.maxImageCount;
215 }
216
217 // swap chain create params
218 VkSwapchainCreateInfoKHR createInfo{};
219 createInfo.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
220 createInfo.surface = vulkanState.surface;
221 createInfo.minImageCount = imageCount;
222 createInfo.imageFormat = surfaceFormat.format;
223 createInfo.imageColorSpace = surfaceFormat.colorSpace;
224 createInfo.imageExtent = swapChainSupport.swapExtent(window.getSize());
225 createInfo.imageArrayLayers = 1;
226 createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT | VK_IMAGE_USAGE_TRANSFER_SRC_BIT;
227
228 // queue chain config
229 QueueFamilyLocator indices;
230 indices.populate(vulkanState.physicalDevice, vulkanState.surface);
231 std::uint32_t queueFamilies[] = {indices.graphicsFamily.value(), indices.presentFamily.value()};
232
233 if (indices.graphicsFamily != indices.presentFamily) {
234 createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
235 createInfo.queueFamilyIndexCount = std::size(queueFamilies);
236 createInfo.pQueueFamilyIndices = queueFamilies;
237 }
238 else {
239 createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
240 createInfo.queueFamilyIndexCount = 0; // Optional
241 createInfo.pQueueFamilyIndices = nullptr; // Optional
242 }
243
244 // more create params
245 createInfo.preTransform = swapChainSupport.capabilities.currentTransform;
246 createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR; // for other windows/applications
247 createInfo.presentMode =
248 swapChainSupport.presentationMode(!engine::Configuration::getOrDefault<bool>(
249 engine::Settings::WindowParameters::VSyncKey, true));
250 createInfo.clipped = VK_TRUE;
251 createInfo.oldSwapchain = oldSwapchain;
252
253 const VkResult result =
254 vkCreateSwapchainKHR(vulkanState.device, &createInfo, nullptr, &swapchain);
255 if (result != VK_SUCCESS) {
256 BL_LOG_ERROR << "Swap chain creation failed: " << result;
257 throw std::runtime_error("Failed to create swap chain");
258 }
259
260 // Fetch images
261 VkImage images[8];
262 vkCheck(vkGetSwapchainImagesKHR(vulkanState.device, swapchain, &imageCount, images));
263 imageFormat = surfaceFormat.format;
264
265 // create depth buffers
266 depthBuffers.resize(imageCount);
267 for (unsigned int i = 0; i < imageCount; ++i) {
268 depthBuffers[i].create(vulkanState,
269 StandardAttachmentBuffers::findDepthFormat(vulkanState),
270 VK_IMAGE_USAGE_DEPTH_STENCIL_ATTACHMENT_BIT,
271 createInfo.imageExtent);
272 }
273
274 // assign attachment sets
275 renderFrames.resize(imageCount);
276 for (unsigned int i = 0; i < imageCount; ++i) {
277 renderFrames[i].setRenderExtent(createInfo.imageExtent);
278 renderFrames[i].setAttachments(images[i],
279 vulkanState.createImageView(images[i], imageFormat),
280 depthBuffers[i].image(),
281 depthBuffers[i].view());
282 }
283}
