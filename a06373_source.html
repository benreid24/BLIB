325template<typename T>
326T& Engine::getPlayer(unsigned int i) {
327 static_assert(std::is_base_of_v<Player, T>, "T must derive from Player");
328
329 if constexpr (std::is_same_v<Player, T>) {
330 if (players.empty() && i == 0) {
331 BL_LOG_INFO << "No player exists, creating default";
332 addPlayer();
333 }
334 }
335
336#ifdef BLIB_DEBUG
337 T* p = dynamic_cast<T*>(players[i].get());
338 if (!p) { BL_LOG_ERROR << "Bad player cast"; }
339#else
340 T* p = static_cast<T*>(players[i].get());
341#endif
342
343 return *p;
344}

346template<typename T, typename... TArgs>
347T& Engine::addPlayer(TArgs&&... args) {
348 static_assert(std::is_base_of_v<Player, T>, "T must derive from Player");
349 static_assert(
350 std::is_constructible_v<T, Engine&, rc::Observer*, input::Actor*, TArgs...>,
351 "T must be constructible with T(Engine&, rc::Observer*, input::Actor*, TArgs...)");
352
353 auto& observer = renderingSystem.addObserver();
354 auto& actor = input.addActor();
355 T* np = new T(*this, &observer, &actor, args...);
356 auto& player = players.emplace_back(np);
357 bl::event::Dispatcher::dispatch<event::PlayerAdded>({*player});
358 return *np;
359}

361template<typename TWorld, typename... TArgs>
362ctr::Ref<World, TWorld> Engine::createWorld(TArgs&&... args) {
363 static_assert(std::is_base_of_v<World, TWorld>, "TWorld must derive from World");
364 static_assert(std::is_constructible_v<TWorld, Engine&, TArgs...>,
365 "TWorld constructor must take Engine& as first parameter");
366
367 auto ref = worldPool.emplaceDerived<TWorld, TArgs...>(*this, std::forward<TArgs>(args)...);
368 for (unsigned int i = 0; i < worlds.size(); ++i) {
369 if (!worlds[i].isValid()) {
370 worlds[i] = ref;
371 ref->index = i;
372 bl::event::Dispatcher::dispatch<event::WorldCreated>({*ref});
373 return ref;
374 }
375 }
376
377 throw std::runtime_error("A maximum of 8 worlds may exist concurrently");
378}
