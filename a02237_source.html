23AssetRef GraphAssetPool::getAssetForAsset(std::string_view tag, std::string_view purpose) {
24 auto it = assets.find(tag);

46GraphAsset* GraphAssetPool::getAssetForOutput(std::string_view tag, Task* task,
47 std::string_view purpose) {
48 Asset* asset = pool.getAsset(tag, task != nullptr ? this : nullptr);
49 if (asset) {
50 // dont return non-external assets for inputs
51 if (!task && !asset->isExternal()) { return nullptr; }
52
53 auto& set = assets[tag];
54 const bool createNew = !asset->isExternal() || set.empty();
55 GraphAsset* ga = createNew ? &set.emplace_back(asset) : &set.front();
56 setPurpose(*ga, purpose);
57 if (createNew) {
58 emitter.emit<event::SceneGraphAssetCreated>(
59 {.target = owner, .scene = scene, .asset = ga});
60 }
61 return ga;
62 }
63 return nullptr;

85GraphAsset* GraphAssetPool::getAssetForInput(std::string_view tag, std::string_view purpose) {
86 // search for existing asset before falling back onto pool
87 const auto it = assets.find(tag);
88 if (it != assets.end() && it->second.size() == 1) {
89 setPurpose(it->second.front(), purpose);

92
93 // fallback into pool to find external asset
