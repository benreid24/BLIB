44 const glm::vec4& color, const sf::Glyph& glyph, float italicShear) {
45 constexpr float padding = 1.0;
46
47 const float left = glyph.bounds.left - padding;

51
52 const float u1 = static_cast<float>(glyph.textureRect.left) - padding;

62 vertices[i + 1] = rc::prim::Vertex(
63 glm::vec3(position.x + right - italicShear * top, position.y + top, 0.f),
64 color,
65 glm::vec2(u2, v1));

128std::uint32_t BasicText::refreshVertices(const sf::VulkanFont& font, rc::prim::Vertex* vertices,
129 glm::vec2& cornerPos) {
130 // Clear the previous geometry
131 cachedBounds = sf::FloatRect();
132 cachedPos = cornerPos;
133
134 // No text: nothing to draw
135 if (content.isEmpty()) return 0;
136
137 // use word wrapped content
138 if (wordWrappedContent.isEmpty()) { wordWrappedContent = content; }
139
140 // Compute values related to the text style
141 const bool isBold = style & sf::Text::Bold;
142 const bool isUnderlined = style & sf::Text::Underlined;
143 const bool isStrikeThrough = style & sf::Text::StrikeThrough;
144 const float italicShear = (style & sf::Text::Italic) ? 0.209f : 0.f; // 12 degrees in radians
145 const float underlineOffset = font.getUnderlinePosition(fontSize);
146 const float underlineThickness = font.getUnderlineThickness(fontSize);
147
148 // Compute the location of the strike through dynamically
149 // We use the center point of the lowercase 'x' glyph as the reference
150 // We reuse the underline thickness as the thickness of the strike through as well
151 const sf::FloatRect xBounds = font.getGlyph(L'x', fontSize, isBold).bounds;
152 const float strikeThroughOffset = xBounds.top + xBounds.height / 2.f;
153
154 // Precompute the variables needed by the algorithm
155 const float letterSpacing = computeLetterSpacing(font);
156 const float whitespaceWidth = computeWhitespaceWidth(font);
157 const float lineSpacing = cachedLineHeight;
158 float x = cornerPos.x;
159 float y = cornerPos.y + static_cast<float>(fontSize);
160
161 // Create one quad for each character
162 float minX = static_cast<float>(fontSize);
163 float minY = static_cast<float>(fontSize);
164 float maxX = 0.f;
165 float maxY = 0.f;
166 std::uint32_t prevChar = 0;
167 std::uint32_t vi = 0;
168 for (std::size_t i = 0; i < wordWrappedContent.getSize(); ++i) {
169 const std::uint32_t curChar = wordWrappedContent[i];
170
171 // Skip the \r char to avoid weird graphical issues
172 if (curChar == L'\r') continue;
173
174 // Apply the kerning offset
175 x += font.getKerning(prevChar, curChar, fontSize, isBold);
176
177 // If we're using the underlined style and there's a new line, draw a line
178 if (isUnderlined && (curChar == L'\n' && prevChar != L'\n')) {
179 vi = addLine(vertices, vi, x, y, fillColor, underlineOffset, underlineThickness);
180
181 if (outlineThickness != 0) {
182 vi = addLine(vertices,
183 vi,
184 x,
185 y,
186 outlineColor,
187 underlineOffset,
188 underlineThickness,
189 outlineThickness);
190 }
191 }
192
193 // If we're using the strike through style and there's a new line, draw a line across all
194 // characters
195 if (isStrikeThrough && (curChar == L'\n' && prevChar != L'\n')) {
196 vi = addLine(vertices, vi, x, y, fillColor, strikeThroughOffset, underlineThickness);
197
198 if (outlineThickness != 0)
199 vi = addLine(vertices,
200 vi,
201 x,
202 y,
203 outlineColor,
204 strikeThroughOffset,
205 underlineThickness,
206 outlineThickness);
207 }
208
209 prevChar = curChar;
210
211 // Handle special characters
212 if ((curChar == L' ') || (curChar == L'\n') || (curChar == L'\t')) {
213 // Update the current bounds (min coordinates)
214 minX = std::min(minX, x);
215 minY = std::min(minY, y);
216
217 switch (curChar) {
218 case L' ':
219 x += whitespaceWidth;
220 break;
221 case L'\t':
222 x += whitespaceWidth * 4;
223 break;
224 case L'\n':
225 y += lineSpacing;
226 x = 0;
227 break;
228 }
229
230 // Update the current bounds (max coordinates)
231 maxX = std::max(maxX, x);
232 maxY = std::max(maxY, y);
233
234 // Next glyph, no need to create a quad for whitespace
235 continue;
236 }
237
238 // Apply the outline
239 if (outlineThickness != 0) {
240 const sf::Glyph& glyph = font.getGlyph(curChar, fontSize, isBold, outlineThickness);
241
242 // Add the outline glyph to the vertices
243 vi = addGlyphQuad(vertices, vi, glm::vec2(x, y), outlineColor, glyph, italicShear);
244 }
245
246 // Extract the current glyph's description
247 const sf::Glyph& glyph = font.getGlyph(curChar, fontSize, isBold);
248
249 // Add the glyph to the vertices
250 vi = addGlyphQuad(vertices, vi, glm::vec2(x, y), fillColor, glyph, italicShear);
251
252 // Update the current bounds
253 const float left = glyph.bounds.left;
254 const float top = glyph.bounds.top;
255 const float right = glyph.bounds.left + glyph.bounds.width;
256 const float bottom = glyph.bounds.top + glyph.bounds.height;
257
258 minX = std::min(minX, x + left - italicShear * bottom);
259 maxX = std::max(maxX, x + right - italicShear * top);
260 minY = std::min(minY, y + top);
261 maxY = std::max(maxY, y + bottom);
262
263 // Advance to the next character
264 x += glyph.advance + letterSpacing;
265 }
266
267 // If we're using outline, update the current bounds
268 if (outlineThickness != 0) {
269 float outline = std::abs(std::ceil(outlineThickness));
270 minX -= outline;
271 maxX += outline;
272 minY -= outline;
273 maxY += outline;
274 }
275
276 // If we're using the underlined style, add the last line
277 if (isUnderlined && (x > 0)) {
278 vi = addLine(vertices, vi, x, y, fillColor, underlineOffset, underlineThickness);
279
280 if (outlineThickness != 0)
281 vi = addLine(vertices,
282 vi,
283 x,
284 y,
285 outlineColor,
286 underlineOffset,
287 underlineThickness,
288 outlineThickness);
289 }
290
291 // If we're using the strike through style, add the last line across all characters
292 if (isStrikeThrough && (x > 0)) {
293 vi = addLine(vertices, vi, x, y, fillColor, strikeThroughOffset, underlineThickness);
294
295 if (outlineThickness != 0) {
296 vi = addLine(vertices,
297 vi,
298 x,
299 y,
300 outlineColor,
301 strikeThroughOffset,
302 underlineThickness,
303 outlineThickness);
304 }
305 }
306
307 // Update the bounding rectangle
308 cachedBounds.left = minX;
309 cachedBounds.top = minY;
310 cachedBounds.width = maxX - minX;
311 cachedBounds.height = maxY - minY;
312
313 // store next character pos
314 cornerPos.x = x;
315 cornerPos.y = y - static_cast<float>(fontSize);
316 switch (wordWrappedContent[wordWrappedContent.getSize() - 1]) {
317 case ' ':
318 case '\t':
319 case '\n':
320 break;
321 default:
322 cornerPos.x += whitespaceWidth;
323 }
324
325 return vi;
326}

365glm::vec2 BasicText::findCharacterPos(const sf::VulkanFont& font, unsigned int index) const {
366 index = index > content.getSize() ? content.getSize() : index;
367
368 const bool isBold = style & sf::Text::Bold;
369 const float letterSpacing = computeLetterSpacing(font);
370 const float whitespaceWidth = computeWhitespaceWidth(font);
371 const float lineSpacing = cachedLineHeight;
372
373 glm::vec2 position{};
374 std::uint32_t prevChar = 0;
375 for (unsigned int i = 0; i < index; ++i) {
376 const std::uint32_t curChar = content[i];
377
378 // Apply the kerning offset
379 position.x += font.getKerning(prevChar, curChar, fontSize, isBold);
380 prevChar = curChar;

383 switch (curChar) {

392 position.x = 0;

394 }
