63void BlendParameters::build() {
64 if (colorAttachmentBlendStates.empty() && colorBlendBehavior != ColorBlendBehavior::None) {
65 for (std::uint32_t i = 0; i < simpleColorBlendAttachmentCount; ++i) {
66 colorAttachmentBlendStates.emplace_back();
67 auto& ca = colorAttachmentBlendStates.back();
68 ca.colorWriteMask = VK_COLOR_COMPONENT_R_BIT | VK_COLOR_COMPONENT_G_BIT |
69 VK_COLOR_COMPONENT_B_BIT | VK_COLOR_COMPONENT_A_BIT;
70 ca.blendEnable = VK_TRUE;
71
72 switch (colorBlendBehavior) {
73 case ColorBlendBehavior::Overwrite:
74 ca.blendEnable = VK_FALSE;
75 break;
76
77 case ColorBlendBehavior::Add:
78 ca.srcColorBlendFactor = VK_BLEND_FACTOR_ONE;
79 ca.dstColorBlendFactor = VK_BLEND_FACTOR_ONE;
80 ca.colorBlendOp = VK_BLEND_OP_ADD;
81 ca.srcAlphaBlendFactor = VK_BLEND_FACTOR_ONE;
82 ca.dstAlphaBlendFactor = VK_BLEND_FACTOR_ONE;
83 ca.alphaBlendOp = VK_BLEND_OP_MAX;
84 break;
85
86 case ColorBlendBehavior::AlphaBlend:
87 default:
88 ca.srcColorBlendFactor = VK_BLEND_FACTOR_SRC_ALPHA;
89 ca.dstColorBlendFactor = VK_BLEND_FACTOR_ONE_MINUS_SRC_ALPHA;
90 ca.colorBlendOp = VK_BLEND_OP_ADD;
91 ca.srcAlphaBlendFactor = VK_BLEND_FACTOR_ONE;
92 ca.dstAlphaBlendFactor = VK_BLEND_FACTOR_ONE;
93 ca.alphaBlendOp = VK_BLEND_OP_MAX;
94 break;
95 }
96 }
97 }
98 colorBlending.pAttachments = colorAttachmentBlendStates.data();
99 colorBlending.attachmentCount = colorAttachmentBlendStates.size();
100}
