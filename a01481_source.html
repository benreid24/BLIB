40Grammar::ItemSet Grammar::closure(const Grammar::Item& item, ItemSet* r) const {
41 ItemSet tr;

80Node::Sequence Grammar::followSet(Node::Type t, Node::Sequence* rg) const {
81 if (rg) {
82 if (std::find(rg->begin(), rg->end(), t) != rg->end()) return {};
83 rg->push_back(t);
84 }
85
86 Node::Sequence result;
87 for (const Production& p : productions) {
88 for (unsigned int i = 0; i < p.set.size(); ++i) {
89 if (p.set[i] == t) {
90 // Add first set of the following symbol
91 if (i < p.set.size() - 1) {
92 Node::Sequence recurseGuard;
93 Node::Sequence pt = firstSet(p.set[i + 1], recurseGuard);
94 if (!pt.empty()) result.insert(result.begin(), pt.begin(), pt.end());
95 }
96 // Add follow set of the resulting nonterminal
97 else {
98 Node::Sequence recurseGuard = {t};
99 Node::Sequence* g = rg != nullptr ? rg : &recurseGuard;
100 Node::Sequence pt = followSet(p.result, g);
101 if (!pt.empty()) result.insert(result.begin(), pt.begin(), pt.end());
102 }
103 }
104 }
105 }
106 if (t == start) result.push_back(Node::EOI);
107
108 // dedup
109 Node::Sequence ret;
110 for (Node::Type node : result) {
111 if (std::find(ret.begin(), ret.end(), node) == ret.end()) ret.push_back(node);
112 }
113 return ret;
114}
