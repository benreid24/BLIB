498template<typename... Ts>
499struct Serializer<std::variant<Ts...>, false> {
500private:
501 struct HelperBase {
502 virtual bool deserialize(InputStream& input, std::variant<Ts...>& v) const = 0;
503 };
504
505 template<typename U>
506 struct Helper : public HelperBase {
507 virtual bool deserialize(InputStream& input, std::variant<Ts...>& v) const override {
508 v.template emplace<U>();
509 return Serializer<U>::deserialize(input, std::get<U>(v));
510 }
511 };
512
513public:
514 static bool serialize(OutputStream& output, const std::variant<Ts...>& v) {
515 const auto visitor = [&output](const auto& c) -> bool {
516 return Serializer<std::decay_t<decltype(c)>>::serialize(output, c);
517 };
518
519 if (!output.write<std::uint16_t>(v.index())) return false;
520 return std::visit(visitor, v);
521 }
522
523 static bool deserialize(InputStream& input, std::variant<Ts...>& v) {
524 static const std::tuple<Helper<Ts>...> helperTuple(Helper<Ts>{}...);
525 static const HelperBase* helpers[] = {&std::get<Helper<Ts>>(helperTuple)...};
526
527 std::uint16_t i = 0;
528 if (!input.read<std::uint16_t>(i)) return false;
529 if (i >= sizeof...(Ts)) return false;
530 return helpers[i]->deserialize(input, v);
531 }

534 static constexpr auto visitor = [](const auto& c) -> std::uint32_t {
535 return Serializer<std::decay_t<decltype(c)>>::size(c);
536 };
537
538 return std::visit(visitor, v) + sizeof(std::uint16_t);

563struct Serializer<script::Value, false> {
564 static bool serialize(OutputStream& output, const script::Value& v) {
565 const script::PrimitiveValue& value = v.value();
566 if (!Serializer<script::PrimitiveValue::Type>::serialize(output, value.getType()))
567 return false;
568 if (!supported(v.value().getType())) return true;
569
570 switch (v.value().getType()) {
571 case script::PrimitiveValue::TBool:
572 if (!Serializer<bool>::serialize(output, value.getAsBool())) return false;
573 break;
574 case script::PrimitiveValue::TInteger:
575 if (!output.write<std::int32_t>(value.getAsInt())) return false;
576 break;
577 case script::PrimitiveValue::TFloat:
578 if (!Serializer<float>::serialize(output, value.getAsFloat())) return false;
579 break;
580 case script::PrimitiveValue::TString:
581 if (!Serializer<std::string>::serialize(output, value.getAsString())) return false;
582 break;
583 case script::PrimitiveValue::TArray:
584 if (!output.write<std::uint32_t>(
585 static_cast<std::uint32_t>(value.getAsArray().size()))) {
586 return false;
587 }
588 for (const script::Value& av : value.getAsArray()) {
589 if (!serialize(output, av)) return false;
590 }
591 break;
592 default:
593 break;
594 }
595
596 if (!output.write<std::uint32_t>(v.allProperties().size())) return false;
597 for (const auto& prop : v.allProperties()) {
598 if (!Serializer<std::string>::serialize(output, prop.first)) return false;
599 if (!Serializer<script::Value>::serialize(output, prop.second.deref())) return false;
600 }
601
602 return true;
603 }
604
605 static bool deserialize(InputStream& input, script::Value& v) {
606 script::PrimitiveValue& value = v.value();
607 script::PrimitiveValue::Type type;
608 if (!Serializer<script::PrimitiveValue::Type>::deserialize(input, type)) return false;
609 if (!supported(type)) {
610 value = "<UNSUPPORTED TYPE>";
611 return true;
612 }
613
614 switch (type) {
615 case script::PrimitiveValue::TBool: {
616 bool bv;
617 if (!Serializer<bool>::deserialize(input, bv)) return false;
618 value = bv;
619 break;
620 }
621 case script::PrimitiveValue::TInteger: {
622 std::int32_t i;
623 if (!input.read<std::int32_t>(i)) return false;
624 value = i;
625 break;
626 }
627 case script::PrimitiveValue::TFloat: {
628 float f;
629 if (!Serializer<float>::deserialize(input, f)) return false;
630 value = f;
631 break;
632 }
633 case script::PrimitiveValue::TString: {
634 std::string s;
635 if (!Serializer<std::string>::deserialize(input, s)) return false;
636 value = s;
637 break;
638 }
639 case script::PrimitiveValue::TArray: {
640 std::uint32_t s;
641 if (!input.read<std::uint32_t>(s)) return false;
642 value = script::ArrayValue{};
643 value.getAsArray().resize(s);
644 for (unsigned int i = 0; i < s; ++s) {
645 if (!deserialize(input, value.getAsArray()[i])) return false;
646 }
647 break;
648 }
649 default:
650 break;
651 }
652
653 std::uint32_t s;
654 if (!input.read<std::uint32_t>(s)) return false;
655 for (unsigned int i = 0; i < s; ++i) {
656 std::string key;
657 script::Value pv;
658 if (!Serializer<std::string>::deserialize(input, key)) return false;
659 if (!Serializer<script::Value>::deserialize(input, pv)) return false;
660 v.setProperty(key, {std::move(pv)});
661 }
662
663 return true;
664 }

667 std::uint32_t s = Serializer<script::PrimitiveValue::Type>::size(v.value().getType());
668 if (!supported(v.value().getType())) return s;
669
670 switch (v.value().getType()) {
671 case script::PrimitiveValue::TBool:
672 s += Serializer<bool>::size(true);
673 break;
674 case script::PrimitiveValue::TInteger:
675 s += sizeof(std::int32_t);
676 break;
677 case script::PrimitiveValue::TFloat:

679 break;
680 case script::PrimitiveValue::TString:

682 break;
683 case script::PrimitiveValue::TArray:
684 s += sizeof(std::uint32_t);
685 for (const script::Value& av : v.value().getAsArray()) { s += size(av); }
686 break;
687 default:

691 s += sizeof(std::uint32_t);
692 for (const auto& prop : v.allProperties()) {
