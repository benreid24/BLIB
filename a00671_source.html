336  const AllocInfo alloc = *it;

339  // copy vertices to the left
340  const auto vertexStartIt = storage.vertices().begin() + alloc.vertexStart;
341  const auto nextVertexIt = vertexStartIt + alloc.vertexSize;
342  std::copy(nextVertexIt, storage.vertices().end(), vertexStartIt);
343

346  const auto nextIndexIt = indexStartIt + alloc.indexSize;
347  std::copy(nextIndexIt, storage.indices().end(), indexStartIt);
348
