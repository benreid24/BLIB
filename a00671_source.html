110 AudioSystem::Handle AudioSystem::getOrLoadSound(const std::string& path) {
111  std::unique_lock lock(Runner::get().soundMutex);
112  auto& r = Runner::get();
113 
114  // Create or find handle
115  Handle handle = AudioSystem::InvalidHandle;
116  auto it = r.soundHandles.find(path);
117  if (it != r.soundHandles.end()) { handle = it->second; }
118  else {
119  handle = makeHandle();
120  r.soundHandles.emplace(path, handle);
121  r.soundSources.emplace(handle, path);
122  }
123 
124  // Ensure sound loaded
125  auto sit = r.sounds.find(handle);
126  if (sit == r.sounds.end()) {

129  if (sit->second.buffer->getSampleCount() == 0) {
130  r.sounds.erase(sit);
131  BL_LOG_ERROR << "Failed to load sound: " << path;
132  return InvalidHandle;
133  }
134  sit->second.sound.setBuffer(*sit->second.buffer);
135  }

396 void Runner::run() {
397  while (!shouldStop) {
398  {
399  std::unique_lock plock(pauseMutex);
400  if (paused) { pauseCond.wait(plock); }
401 
402  // sound cleanup and fadeouts
403  {
404  std::unique_lock slock(soundMutex);
405 
406  for (auto i = soundHandles.begin(); i != soundHandles.end();) {
407  auto j = i++;
408  auto it = sounds.find(j->second);
409  if (it != sounds.end()) {
410  if (timer.getElapsedTime().asSeconds() - it->second.lastInteractTime >=
411  unloadTimeout + it->second.buffer->getDuration().asSeconds() &&
412  it->second.sound.getStatus() != sf::Sound::Playing) {
413  sounds.erase(it);
414  soundHandles.erase(j);
415  }
416  }
417  }
418 
419  for (auto i = fadingSounds.begin(); i != fadingSounds.end();) {
420  auto j = i++;
421  j->update();
422  }
423  soundTimer.restart();
424  }
425 
426  // playlist crossfade update and song update
427  {
428  std::unique_lock plock(playlistMutex);
429 
430  if (!playlistStack.empty() && playlistStack.front()->isPlaying()) {
431  playlistStack.front()->update();
432  }
433 
434  fadeIn.update();
435  fadeOut.update();
436  fadeTimer.restart();
437  }
438  }
439 
440  sf::sleep(sf::milliseconds(30));
441  }
442 
443  BL_LOG_INFO << "AudioSystem thread terminated";
444 }
