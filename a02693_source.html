107bool DependencyListBase::loadEntry(Entry& entry) {
108 if (entry.dependency && entry.dependency->getState() == State::Loaded) { return true; }
109
110 const State desiredState = policy == LoadPolicy::Eager ? State::Loaded : State::Unloaded;
111 entry.dependency = repo.getAsset(entry.uuid, desiredState);
112 return entry.dependency && entry.dependency->getState() >= desiredState;
113}
