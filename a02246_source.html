112void Texture::executeTransfer(VkCommandBuffer cb, tfr::TransferContext& engine) {
113 const auto transitionLayout = [this, cb, &engine]() {
114 VkImageMemoryBarrier barrier{};
115 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
116 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
117 barrier.newLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
118 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
119 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
120 barrier.image = image;
121 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
122 barrier.subresourceRange.baseMipLevel = 0;
123 barrier.subresourceRange.levelCount = 1;
124 barrier.subresourceRange.baseArrayLayer = 0;
125 barrier.subresourceRange.layerCount = 1;
126 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
127 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
128 engine.registerImageBarrier(barrier);
129 currentLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
130 };
131
132 // copy image contents if required
133 if (altImg || transferImg) {
134 const sf::Image& src = altImg ? *altImg : *transferImg;
135
136 bool fullImage = false;
137 if (source.width == 0 || source.height == 0) {
138 fullImage = true;
139 source.left = 0;
140 source.top = 0;
141 source.width = src.getSize().x;
142 source.height = src.getSize().y;
143 }
144
145 // create staging buffer
146 const VkDeviceSize stageSize = source.width * source.height * 4;
147 VkBuffer stagingBuffer;
148 void* data;
149 engine.createTemporaryStagingBuffer(stageSize, stagingBuffer, &data);
150 if (!fullImage) {
151 for (int x = source.left; x < source.left + source.width; ++x) {
152 for (int y = source.top; y < source.top + source.height; ++y) {
153 const std::ptrdiff_t offset = (x + y * source.width) * 4;
154 std::uint8_t* d = static_cast<std::uint8_t*>(data) + offset;
155 const std::uint8_t* s = src.getPixelsPtr() + offset;
156 std::memcpy(d, s, 4);
157 }
158 }
159 }
160 else { std::memcpy(data, src.getPixelsPtr(), stageSize); }
161
162 // transition to transfer dst prior to copy
163 vulkanState->transitionImageLayout(
164 cb, image, VK_IMAGE_LAYOUT_UNDEFINED, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL);
165
166 // issue copy command
167 VkBufferImageCopy copyInfo{};
168 copyInfo.bufferOffset = 0;
169 copyInfo.bufferRowLength = 0;
170 copyInfo.bufferImageHeight = 0;
171 copyInfo.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
172 copyInfo.imageSubresource.mipLevel = 0;
173 copyInfo.imageSubresource.baseArrayLayer = 0;
174 copyInfo.imageSubresource.layerCount = 1;
175 copyInfo.imageOffset.x = destPos.x;
176 copyInfo.imageOffset.y = destPos.y;
177 copyInfo.imageExtent.width = source.width;
178 copyInfo.imageExtent.height = source.height;
179 copyInfo.imageExtent.depth = 1;
180 vkCmdCopyBufferToImage(
181 cb, stagingBuffer, image, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &copyInfo);
182
183 // insert pipeline barrier
184 transitionLayout();
185
186 // cleanup
187 transferImg.release();
188 altImg = nullptr;
189 destPos.x = 0;
190 destPos.y = 0;
191 source.left = 0;
192 source.top = 0;
193 source.width = 0;
194 source.height = 0;
195 }
196 else if (currentLayout != VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL) { transitionLayout(); }
197}

217void Texture::updateTrans(const sf::Image& content) {
218 const std::uint32_t s = content.getSize().x * content.getSize().y;
219 const std::uint32_t thresh = s / 10;
220 std::uint32_t t = 0;
221
222 // jump directly from alpha to alpha
223 const std::uint8_t* end = content.getPixelsPtr() + s * 4;
224 for (const std::uint8_t* a = content.getPixelsPtr() + 3; a < end; a += 4) {
225 if (*a > 0 || *a < 255) {
226 ++t;
227 if (t >= thresh) {
228 updateTransparency(true);
229 return;
230 }
231 }
232 }
233
234 updateTransparency(false);
235}
