237void Renderer::processResize(const sf::Event::Resized& event) {
238 auto windowSettings = settings.getWindowSettings();
239 const sf::Vector2f modeSize(
240 sf::Vector2u(windowSettings.videoMode().size.x, windowSettings.videoMode().size.y));
241 const sf::Vector2f& ogSize =
242 windowSettings.initialViewSize().x > 0.f ? windowSettings.initialViewSize() : modeSize;
243
244 const float newWidth = static_cast<float>(event.size.x);
245 const float newHeight = static_cast<float>(event.size.y);
246
247 sf::FloatRect viewport({0.f, 0.f}, {1.f, 1.f});
248 if (windowSettings.letterBox()) {
249 const float xScale = newWidth / ogSize.x;
250 const float yScale = newHeight / ogSize.y;
251
252 if (xScale >= yScale) { // constrained by height, bars on sides
253 windowScale = yScale;
254 viewport.size.x = ogSize.x * yScale / newWidth;
255 viewport.position.x = (1.f - viewport.size.x) * 0.5f;
256 }
257 else { // constrained by width, bars on top and bottom
258 windowScale = xScale;
259 viewport.size.y = ogSize.y * xScale / newHeight;
260 viewport.position.y = (1.f - viewport.size.y) * 0.5f;
261 }
262 }
263
264 if (windowSettings.syncOverlaySize() && !windowSettings.letterBox()) {
265 cam::OverlayCamera::setOverlayCoordinateSpace(newWidth, newHeight);
266 }
267
268 renderRegion = sf::Rect<std::uint32_t>(
269 sf::Vector2<std::uint32_t>(static_cast<std::uint32_t>(newWidth * viewport.position.x),
270 static_cast<std::uint32_t>(newHeight * viewport.position.y)),
271 sf::Vector2<std::uint32_t>(static_cast<std::uint32_t>(newWidth * viewport.size.x),
272 static_cast<std::uint32_t>(newHeight * viewport.size.y)));
273 swapchain.invalidate();
274 assignObserverRegions();
275 commonObserver.assignRegion(window.getSfWindow().getSize(), renderRegion, 1, 0, true);
276}

300void Renderer::renderFrame() {
301 std::unique_lock lock(renderMutex);
302
303 // begin frame
304 vk::AttachmentSet* currentFrame = nullptr;
305 VkCommandBuffer commandBuffer = nullptr;
306 swapchain.beginFrame(currentFrame, commandBuffer);
307 cleanupManager.onFrameStart();
308
309 // kick off transfers
310 if (settings.dirty) {
311 globalDescriptors.updateSettings(settings);
312 settings.dirty = false;
313 }
314 globalDescriptors.onFrameStart();
315 globalShaderResources.performTransfers();
316 for (auto& rt : renderTextures) { rt->updateDescriptorsAndQueueTransfers(); }
317 if (commonObserver.hasScene()) { commonObserver.updateDescriptorsAndQueueTransfers(); }
318 else {
319 for (auto& o : observers) { o->updateDescriptorsAndQueueTransfers(); }
320 }
321 transferEngine.executeTransfers();
322
323 // reset graph asset states
324 for (auto& rt : renderTextures) { rt->resetAssets(); }
325 if (commonObserver.hasScene()) { commonObserver.resetAssets(); }
326 else {
327 for (auto& o : observers) { o->resetAssets(); }
328 }
329
330 // render offscreen textures
331 for (auto& rt : renderTextures) { rt->render(); }
332
333 // record commands to render scenes
334 for (auto& o : observers) { o->renderScene(commandBuffer); }
335 if (!virtualObservers.empty()) {
336 for (auto& o : virtualObservers) { o->renderScene(commandBuffer); }
337 }
338 if (commonObserver.hasScene()) { commonObserver.renderScene(commandBuffer); }
339
340 // complete frame
341 swapchain.completeFrame();
342 state.incrementFrame();
343
344 // submit texture exports
345 imageExporter.onFrameEnd();
346}
