267template<typename T>
268bool IndexBufferT<T>::ensureSize(std::uint32_t vertexCount, std::uint32_t indexCount) {
269 bool r = false;
270 if (vertexCount > cpuVertexBuffer.size()) {
271 cpuVertexBuffer.resize(vertexCount);
272 gpuVertexBuffer.ensureSize(vertexCount * sizeof(T));
273 vertexWriteStart = 0;
274 vertexWriteCount = vertexCount;
275 r = true;
276 }
277 if (indexCount > cpuIndexBuffer.size()) {
278 cpuIndexBuffer.resize(indexCount);
279 gpuIndexBuffer.ensureSize(indexCount * sizeof(IndexType));
280 indexWriteStart = 0;
281 indexWriteCount = indexCount;
282 r = true;
283 }
284 return r;
285}

338template<typename T>
339void IndexBufferT<T>::executeTransfer(VkCommandBuffer commandBuffer,
340 tfr::TransferContext& context) {
341 VkBuffer stagingBuf;
342 void* stagingMem;
343 VkBufferCopy copyCmd{};
344 VkBufferMemoryBarrier barrier{};
345
346 // vertex buffer
347 if (vertexWriteStart + vertexWriteCount <= cpuVertexBuffer.size() && !cpuVertexBuffer.empty()) {
348 const VkDeviceSize vertexOffset = vertexWriteStart * sizeof(T);
349 const VkDeviceSize vertexSize = vertexWriteCount * sizeof(T);
350
351 if (vertexSize > 0) {
352 context.createTemporaryStagingBuffer(vertexSize, stagingBuf, &stagingMem);
353 std::memcpy(stagingMem, &cpuVertexBuffer[vertexWriteStart], vertexSize);
354
355 copyCmd.srcOffset = 0;
356 copyCmd.dstOffset = vertexOffset;
357 copyCmd.size = vertexSize;
358 vkCmdCopyBuffer(commandBuffer, stagingBuf, gpuVertexBuffer.getBuffer(), 1, &copyCmd);
359
360 barrier.sType = VK_STRUCTURE_TYPE_BUFFER_MEMORY_BARRIER;
361 barrier.buffer = gpuVertexBuffer.getBuffer();
362 barrier.offset = vertexOffset;
363 barrier.size = vertexSize;
364 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
365 barrier.dstAccessMask = VK_ACCESS_VERTEX_ATTRIBUTE_READ_BIT;
366 context.registerBufferBarrier(barrier);
367 }
368 }
369 else { BL_LOG_WARN << "Invalid vertex buffer write, skipping"; }
370
371 // index buffer
372 if (indexWriteStart + indexWriteCount <= cpuIndexBuffer.size() && !cpuIndexBuffer.empty()) {
373 const VkDeviceSize indexOffset = indexWriteStart * sizeof(IndexType);
374 const VkDeviceSize indexSize = indexWriteCount * sizeof(IndexType);
375
376 if (indexSize > 0) {
377 context.createTemporaryStagingBuffer(indexSize, stagingBuf, &stagingMem);
378 std::memcpy(stagingMem, &cpuIndexBuffer[indexWriteStart], indexSize);
379
380 copyCmd.size = indexSize;
381 copyCmd.dstOffset = indexOffset;
382 vkCmdCopyBuffer(commandBuffer, stagingBuf, gpuIndexBuffer.getBuffer(), 1, &copyCmd);
383
384 barrier.buffer = gpuIndexBuffer.getBuffer();
385 barrier.size = indexSize;
386 context.registerBufferBarrier(barrier);
387 }
388 }
389 else { BL_LOG_WARN << "Invalid index buffer write, skipping"; }
390}
