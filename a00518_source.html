104 bool Element::handleEvent(const RawEvent& event) {
105  if (!visible()) return false;
106 
107  if (handleRawEvent(event)) return true;
108 
109  const bool eventOnMe =
110  acquisition.contains(sf::Vector2i(event.localMousePos.x, event.localMousePos.y));
111 
112  switch (event.event.type) {
113  case sf::Event::TextEntered:
114  case sf::Event::KeyPressed:
115  case sf::Event::KeyReleased:
116  case sf::Event::MouseWheelScrolled:
117  return processAction(Action::fromRaw(event));
118 
119  case sf::Event::MouseButtonPressed:
120  if (eventOnMe) {
121  if (!active()) return true;
122 
123  if (takeFocus()) {
124  if (event.event.mouseButton.button == sf::Mouse::Left) {
125  dragStart = event.localMousePos;
126  isLeftPressed = true;
127  return processAction(Action(Action::Pressed, event.localMousePos));
128  }
129  else if (event.event.mouseButton.button == sf::Mouse::Right) {
130  isRightPressed = true;
131  return true;
132  }
133  }
134  return true; // click blocked
135  }
136  return false;
137 
138  case sf::Event::MouseButtonReleased:
139  if (!active()) {
140  isLeftPressed = isRightPressed = false;
141  return eventOnMe;
142  }
143  if (event.event.mouseButton.button == sf::Mouse::Left) {
144  if (isLeftPressed) {
145  isLeftPressed = false;
146  processAction(Action(Action::Released, event.localMousePos));
147  if (eventOnMe) {
148  processAction(Action(Action::LeftClicked, event.localMousePos));
149  return true;
150  }
151  }
152  }
153  else if (event.event.mouseButton.button == sf::Mouse::Right) {
154  if (isRightPressed) {
155  isRightPressed = false;
156  if (eventOnMe) {
157  processAction(Action(Action::RightClicked, event.localMousePos));
158  return true;
159  }
160  }
161  }
162  return false;
163 
164  case sf::Event::MouseMoved:
165  if (!active()) { return eventOnMe; }
166  if (isLeftPressed) {
167  isMouseOver = eventOnMe;
168  const bool r = processAction(Action(Action::Dragged, dragStart, event.localMousePos));
169  return r;
170  }
171  else if (eventOnMe) {
172  if (!isMouseOver) {
173  isMouseOver = true;
174  processAction(Action(Action::MouseEntered, event.localMousePos));
175  }
176  return true;
177  }
178  else if (isMouseOver) {
179  isMouseOver = false;
180  processAction(Action(Action::MouseLeft, event.localMousePos));
181  }
182  return false;
183 
184  default:
185  return false;
186  }
187 }
