212void Value::validateArgs(const std::string& func, const std::vector<Value>& args) {
213 constexpr PrimitiveValue::Type types[] = {Types...};
214 constexpr unsigned int nargs = sizeof...(Types);
215
216 if (args.size() != nargs)
217 throw Error(func + "() takes " + std::to_string(nargs) + " arguments");
218 for (unsigned int i = 0; i < nargs; ++i) {
219 const Value& v = args[i].value().deref();
220 if ((v.value().getType() & types[i]) == 0) {
221 throw Error(func + "() argument " + std::to_string(i) + " must be a " +
222 PrimitiveValue::typeToString(types[i]) + " but " +
223 PrimitiveValue::typeToString(v.value().getType()) + " was passed");
224 }
225 }
