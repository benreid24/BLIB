129void Actor::update() {
130 for (unsigned int i = 0; i < (*activeControls).size(); ++i) {
131 const Control& ctrl = (*activeControls)[i];
132
133 switch (ctrl.type) {
134 case Control::Type::SingleTrigger:
135 if (ctrl.triggerControl.active()) {
136 dispatch(i, DispatchType::TriggerActivated, false);
137 }
138 break;
139 case Control::Type::Movement:
140 if (ctrl.movementControl.type == MovementControl::Type::Triggers) {
141 const auto& keys = ctrl.movementControl.keys;
142 if (keys.up.active()) { dispatch(i, DispatchType::MovementUp, false); }
143 if (keys.right.active()) { dispatch(i, DispatchType::MovementRight, false); }
144 if (keys.down.active()) { dispatch(i, DispatchType::MovementDown, false); }
145 if (keys.left.active()) { dispatch(i, DispatchType::MovementLeft, false); }
146 }
147 else {
148 const sf::Vector2f& pos = ctrl.movementControl.joystick.cachedPosition;
149 if (pos.x >= StickThreshold) { dispatch(i, DispatchType::MovementRight, false); }
150 if (pos.x <= -StickThreshold) { dispatch(i, DispatchType::MovementLeft, false); }
151 if (pos.y >= StickThreshold) { dispatch(i, DispatchType::MovementUp, false); }
152 if (pos.y <= -StickThreshold) { dispatch(i, DispatchType::MovementDown, false); }
153 }
154 break;
155 default:
156 break;
157 }
158 }
159}

196void Actor::process(const sf::Event& event) {
197 // Determine if keyboard or controller is being used, then process controls
198 bool shouldProcess = false;
199
200 event.visit(util::Visitor{
201 [this, &shouldProcess](const sf::Event::JoystickButtonPressed& jbEvent) {
202 if (joystick >= sf::Joystick::Count || jbEvent.joystickId == joystick) {
203 activeControls = &joystickControls;
204 shouldProcess = true;
205 }
206 },
207 [this, &shouldProcess](const sf::Event::JoystickButtonReleased& jbEvent) {
208 if (joystick >= sf::Joystick::Count || jbEvent.joystickId == joystick) {
209 activeControls = &joystickControls;
210 shouldProcess = true;
211 }
212 },
213 [this, &shouldProcess](const sf::Event::JoystickMoved& jmEvent) {
214 if (joystick >= sf::Joystick::Count || jmEvent.joystickId == joystick) {
215 activeControls = &joystickControls;
216 shouldProcess = true;
217 }
218 },
219 [this, &shouldProcess](const sf::Event::KeyPressed&) {
220 activeControls = &kbmControls;
221 shouldProcess = true;
222 },
223 [this, &shouldProcess](const sf::Event::KeyReleased&) {
224 activeControls = &kbmControls;
225 shouldProcess = true;
226 },
227 [this, &shouldProcess](const sf::Event::MouseButtonPressed&) {
228 activeControls = &kbmControls;
229 shouldProcess = true;
230 },
231 [this, &shouldProcess](const sf::Event::MouseButtonReleased&) {
232 activeControls = &kbmControls;
233 shouldProcess = true;
234 },
235 [this, &shouldProcess](const sf::Event::MouseWheelScrolled&) {
236 activeControls = &kbmControls;
237 shouldProcess = true;
238 },
239 [this, &shouldProcess](const sf::Event::MouseMoved&) {
240 activeControls = &kbmControls;
241 shouldProcess = true;
242 },
243 [](const auto&) {
244 // Handle any other event types - do nothing
245 }});
246
247 if (!shouldProcess) { return; }
248
249 // Process control
250 for (unsigned int i = 0; i < (*activeControls).size(); ++i) {
251 Control& ctrl = (*activeControls)[i];
252
253 switch (ctrl.type) {
254 case Control::Type::SingleTrigger:
255 if (ctrl.triggerControl.process(event)) {
256 dispatch(i, DispatchType::TriggerActivated, true);
257 }
258 break;
259 case Control::Type::Movement: {
260 const DispatchType dt = ctrl.movementControl.process(event);
261 if (dt != DispatchType::_INVALID) { dispatch(i, dt, true); }
262 } break;
263 case Control::Type::Directional:
264 if (ctrl.directionalControl.process(owner, event)) {
265 dispatch(i, DispatchType::DirectionChanged, true);
266 }
267 break;
268 default:
269 break;
270 }
271 }
272}
