43 VmaAllocationCreateInfo allocInfo{};
44 allocInfo.requiredFlags = memPool;

92bool Buffer::ensureSize(VkDeviceSize newSize, bool skipCopy) {
93 if (newSize > size) {
94 doDefer();
95
96 const VkDeviceSize oldSize = size;
97 const VkBuffer oldBuffer = buffer;
98 size = newSize;
99 if (!doCreate()) { throw std::runtime_error("Failed to resize buffer"); }
100
101 if (!skipCopy) {
102 auto commandBuffer = renderer->getSharedCommandPool().createBuffer();
103
104 // barrier to ensure writes to old buffer are complete
105 VkBufferMemoryBarrier barrier{};
106 barrier.sType = VK_STRUCTURE_TYPE_BUFFER_MEMORY_BARRIER;
107 barrier.buffer = oldBuffer;
108 barrier.offset = 0;
109 barrier.size = oldSize;
110 barrier.srcAccessMask = VK_ACCESS_MEMORY_WRITE_BIT;
111 barrier.dstAccessMask = VK_ACCESS_MEMORY_READ_BIT;
112 vkCmdPipelineBarrier(commandBuffer,
113 VK_PIPELINE_STAGE_TRANSFER_BIT,
114 VK_PIPELINE_STAGE_TRANSFER_BIT,
115 VK_DEPENDENCY_BY_REGION_BIT,
116 0,
117 nullptr,
118 1,
119 &barrier,
120 0,
121 nullptr);
122
123 // copy old buffer into new
124 VkBufferCopy copyCmd{};
125 copyCmd.dstOffset = 0;
126 copyCmd.size = oldSize;
127 copyCmd.srcOffset = 0;
128 vkCmdCopyBuffer(commandBuffer, oldBuffer, buffer, 1, &copyCmd);
129
130 // insert pipeline barrier
131 VkBufferMemoryBarrier bufBarrier{};
132 bufBarrier.sType = VK_STRUCTURE_TYPE_BUFFER_MEMORY_BARRIER;
133 bufBarrier.srcAccessMask = VK_ACCESS_MEMORY_WRITE_BIT;
134 bufBarrier.dstAccessMask = VK_ACCESS_MEMORY_READ_BIT;
135 bufBarrier.buffer = buffer;
136 bufBarrier.offset = 0;
137 bufBarrier.size = oldSize;
138 vkCmdPipelineBarrier(commandBuffer,
139 VK_PIPELINE_STAGE_TRANSFER_BIT,
140 VK_PIPELINE_STAGE_VERTEX_INPUT_BIT,
141 VK_DEPENDENCY_BY_REGION_BIT,
142 0,
143 nullptr,
144 1,
145 &bufBarrier,
146 0,
147 nullptr);
148
149 commandBuffer.submit();
150
151 return true;
152 }
153 }
