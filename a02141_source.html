96void Swapchain::beginFrame(AttachmentSet*& renderFrame, VkCommandBuffer& cb) {
97 // wait for prior frame
98 vkCheck(vkWaitForFences(renderer.vulkanState().getDevice(),
99 1,
100 &frameData.current().commandBufferFence,
101 VK_TRUE,
102 UINT64_MAX));
103 vkCheck(vkResetFences(
104 renderer.vulkanState().getDevice(), 1, &frameData.current().commandBufferFence));
105
106 // recreate if out of date
107 if (outOfDate) { recreate(); }
108
109 // acquire next image
110 VkResult acquireResult;
111 do {
112 acquireResult = vkAcquireNextImageKHR(renderer.vulkanState().getDevice(),
113 swapchain,
114 UINT64_MAX,
115 imageSemaphores.getNext(),
116 VK_NULL_HANDLE,
117 &currentImageIndex);
118 if (acquireResult == VK_ERROR_OUT_OF_DATE_KHR) { recreate(); }
119 } while (acquireResult == VK_ERROR_OUT_OF_DATE_KHR);
120 if (acquireResult != VK_SUCCESS && acquireResult != VK_SUBOPTIMAL_KHR) {
121 throw std::runtime_error("Failed to acquire swapchain image");
122 }
123
124 // reset prior command buffer & fence
125 vkCheck(vkResetCommandBuffer(frameData.current().commandBuffer, 0));
126 vkCheck(vkResetFences(
127 renderer.vulkanState().getDevice(), 1, &frameData.current().commandBufferFence));
128
129 // begin command buffer
130 VkCommandBufferBeginInfo beginInfo{};
131 beginInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
132 beginInfo.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
133 if (vkBeginCommandBuffer(frameData.current().commandBuffer, &beginInfo) != VK_SUCCESS) {
134 throw std::runtime_error("Failed to begin recording command buffer");
135 }
136
137 // copy output
138 renderFrame = &renderFrames[currentImageIndex];
139 cb = frameData.current().commandBuffer;
140}

142void Swapchain::completeFrame() {
143 // end command buffer
144 vkCheck(vkEndCommandBuffer(frameData.current().commandBuffer));
145
146 // submit to queue
147 VkSemaphore waitSemaphores[] = {imageSemaphores.current()};
148 VkSemaphore signalSemaphores[] = {frameData.current().renderFinishedSemaphore};
149 VkPipelineStageFlags waitStages[] = {VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT};
150
151 VkSubmitInfo submitInfo{};
152 submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
153 submitInfo.waitSemaphoreCount = std::size(waitSemaphores);
154 submitInfo.pWaitSemaphores = waitSemaphores;
155 submitInfo.pWaitDstStageMask = waitStages;
156 submitInfo.signalSemaphoreCount = std::size(signalSemaphores);
157 submitInfo.pSignalSemaphores = signalSemaphores;
158 submitInfo.commandBufferCount = 1;
159 submitInfo.pCommandBuffers = &frameData.current().commandBuffer;
160 renderer.vulkanState().submitCommandBuffer(submitInfo, frameData.current().commandBufferFence);
161
162 // trigger swap chain
163 VkPresentInfoKHR presentInfo{};
164 presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
165 presentInfo.waitSemaphoreCount = 1;
166 presentInfo.pWaitSemaphores = &frameData.current().renderFinishedSemaphore;
167 presentInfo.swapchainCount = 1;
168 presentInfo.pSwapchains = &swapchain;
169 presentInfo.pImageIndices = &currentImageIndex;
170 presentInfo.pResults = nullptr; // Optional
171
172 const VkResult presentResult =
173 vkQueuePresentKHR(renderer.vulkanState().getPresentQueue(), &presentInfo);
174 switch (presentResult) {
175 case VK_ERROR_OUT_OF_DATE_KHR:
176 case VK_SUBOPTIMAL_KHR:
177 recreate();
178 break;
179 case VK_SUCCESS:
180 break;
181 default:
182 throw std::runtime_error("Failed to present swap chain image");
183 }
184}

230void Swapchain::createSwapchain() {
231 // cache the surface that we created with
232 oldSurface = renderer.vulkanState().getSurface();
233
234 // get supported swap chain details
235 SwapChainSupportDetails swapChainSupport;
236 swapChainSupport.populate(renderer.vulkanState().getPhysicalDevice(),
237 renderer.vulkanState().getSurface());
238 const VkSurfaceFormatKHR& surfaceFormat = swapChainSupport.swapSurfaceFormat();
239
240 // image count in swap chain
241 std::uint32_t imageCount = swapChainSupport.capabilities.minImageCount + 1;
242 if (swapChainSupport.capabilities.maxImageCount > 0 &&
243 imageCount > swapChainSupport.capabilities.maxImageCount) {
244 imageCount = swapChainSupport.capabilities.maxImageCount;
245 }
246
247 // swap chain create params
248 VkSwapchainCreateInfoKHR createInfo{};
249 createInfo.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
250 createInfo.surface = renderer.vulkanState().getSurface();
251 createInfo.minImageCount = imageCount;
252 createInfo.imageFormat = surfaceFormat.format;
253 createInfo.imageColorSpace = surfaceFormat.colorSpace;
254 createInfo.imageExtent = swapChainSupport.swapExtent(window.getSize());
255 createInfo.imageArrayLayers = 1;
256 createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT | VK_IMAGE_USAGE_TRANSFER_SRC_BIT;
257
258 // queue chain config
259 QueueFamilyLocator indices;
260 indices.populate(renderer.vulkanState().getPhysicalDevice(),
261 renderer.vulkanState().getSurface());
262 std::uint32_t queueFamilies[] = {indices.graphicsFamily.value(), indices.presentFamily.value()};
263
264 if (indices.graphicsFamily != indices.presentFamily) {
265 createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
266 createInfo.queueFamilyIndexCount = std::size(queueFamilies);
267 createInfo.pQueueFamilyIndices = queueFamilies;
268 }
269 else {
270 createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
271 createInfo.queueFamilyIndexCount = 0; // Optional
272 createInfo.pQueueFamilyIndices = nullptr; // Optional
273 }
274
275 // more create params
276 createInfo.preTransform = swapChainSupport.capabilities.currentTransform;
277 createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR; // for other windows/applications
278 createInfo.presentMode = swapChainSupport.presentationMode(!windowSettings.vsyncEnabled());
279 createInfo.clipped = VK_TRUE;
280 createInfo.oldSwapchain = oldSwapchain;
281
282 const VkResult result =
283 vkCreateSwapchainKHR(renderer.vulkanState().getDevice(), &createInfo, nullptr, &swapchain);
284 if (result != VK_SUCCESS) {
285 BL_LOG_ERROR << "Swap chain creation failed: " << result;
286 throw std::runtime_error("Failed to create swap chain");
287 }
288
289 // Fetch images
290 VkImage images[8];
291 vkCheck(vkGetSwapchainImagesKHR(
292 renderer.vulkanState().getDevice(), swapchain, &imageCount, images));
293 imageFormat = surfaceFormat.format;
294
295 // assign attachment sets & transition images
296 auto cb = renderer.getSharedCommandPool().createBuffer();
297 renderFrames.clear();
298 renderFrames.reserve(imageCount);
299 std::array<VkImageAspectFlags, 1> aspects{VK_IMAGE_ASPECT_COLOR_BIT};
300 for (unsigned int i = 0; i < imageCount; ++i) {
301 VkImageView view = renderer.vulkanState().createImageView(images[i], imageFormat);
302 renderFrames.emplace_back(1, &images[i], &view, aspects.data(), createInfo.imageExtent);
303 renderer.vulkanState().transitionImageLayout(cb,
304 images[i],
305 VK_IMAGE_LAYOUT_UNDEFINED,
306 VK_IMAGE_LAYOUT_PRESENT_SRC_KHR,
307 1,
308 VK_IMAGE_ASPECT_COLOR_BIT);
309 }
310 cb.submit();
311}
