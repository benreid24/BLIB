50void Scene::syncObjects() {
51 std::unique_lock lock(objectMutex);
52 std::unique_lock queueLock(queueMutex);
53 isClearingQueues = true;
54
55 // do batch changes
56 for (const auto& change : queuedBatchChanges) {
57 auto& objectPipelines = change.changed->sceneKey.updateFreq == UpdateSpeed::Static ?
58 staticPipelines :
59 dynamicPipelines;
60 doBatchChange(change, objectPipelines[change.changed->sceneKey.sceneId]);
61 objectPipelines[change.changed->sceneKey.sceneId] = change.newPipeline;
62 }
63 queuedBatchChanges.clear();
64
65 // remove queued objects
66 for (auto* obj : queuedRemovals) { removeQueuedObject(obj); }
67 queuedRemovals.clear();

71 for (auto& add : queuedAdds) {
72 if (engine.ecs().entityExists(add.entity, tx)) { addQueuedObject(add); }
73 }
74 queuedAdds.clear();

76 isClearingQueues = false;
77}
