130void Renderer::renderFrame() {
131 std::unique_lock lock(renderMutex);
132
133 // begin frame
134 vk::StandardAttachmentSet* currentFrame = nullptr;
135 VkCommandBuffer commandBuffer = nullptr;
136 state.beginFrame(currentFrame, commandBuffer);
137 framebuffers.current().recreateIfChanged(*currentFrame);
138
139 // kick off transfers
140 textures.onFrameStart();
141 for (auto& rt : renderTextures) { rt->handleDescriptorSync(); }
142 if (commonObserver.hasScene()) { commonObserver.handleDescriptorSync(); }
143 else {
144 for (auto& o : observers) { o->handleDescriptorSync(); }
145 }
146 state.transferEngine.executeTransfers();
147
148 // render offscreen textures
149 for (auto& rt : renderTextures) { rt->render(); }
150
151 const auto clearDepthBuffer = [this, commandBuffer]() {
152 VkClearAttachment attachment{};
153 attachment.aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT;
154 attachment.clearValue = clearColors[1];
155
156 VkClearRect rect{};
157 rect.rect.extent.width = renderRegion.width;
158 rect.rect.extent.height = renderRegion.height;
159 rect.rect.offset.x = renderRegion.left;
160 rect.rect.offset.y = renderRegion.top;
161 rect.baseArrayLayer = 0;
162 rect.layerCount = 1;
163
164 vkCmdClearAttachments(commandBuffer, 1, &attachment, 1, &rect);
165 };
166
167 // record commands to render scenes
168 for (auto& o : observers) { o->renderScene(commandBuffer); }
169 if (!virtualObservers.empty()) {
170 clearDepthBuffer();
171 for (auto& o : virtualObservers) { o->renderScene(commandBuffer); }
172 }
173 if (commonObserver.hasScene()) {
174 clearDepthBuffer();
175 commonObserver.renderScene(commandBuffer);
176 }
177
178 // perform render pass for final scene renders and overlays
179 framebuffers.current().beginRender(commandBuffer,
180 {{0, 0}, currentFrame->renderExtent()},
181 clearColors,
182 std::size(clearColors),
183 false);
184
185 // render scene outputs
186 for (auto& o : observers) { o->compositeSceneAndOverlay(commandBuffer); }
187 if (!virtualObservers.empty()) {
188 clearDepthBuffer();
189 for (auto& o : virtualObservers) { o->compositeSceneAndOverlay(commandBuffer); }
190 }
191 if (commonObserver.hasScene()) {
192 clearDepthBuffer();
193 commonObserver.compositeSceneAndOverlay(commandBuffer);
194 }
195
196 // complete frame
197 framebuffers.current().finishRender(commandBuffer);
198 state.completeFrame();
199
200 // submit texture exports
201 imageExporter.onFrameEnd();
202}
