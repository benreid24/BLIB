91std::vector<std::string> recurseParams(const parser::Node::Ptr& plist) {
92 using G = Parser::Grammar;
93 if constexpr (ExtraChecks) {
94 if (plist->type != G::ParamList) throw Error("Internal error: Expected ParamList", plist);
95 if (plist->children.size() != 3)
96 throw Error("Internal error: Invalid ParamList children", plist);
97 }
98
99 if (plist->children[0]->type == G::ParamList) {
100 if constexpr (ExtraChecks) {
101 if (plist->children[2]->type != G::Id)
102 throw Error("Internal error: Expected Id in ParamList", plist->children[2]);
103 }
104 std::vector<std::string> params = recurseParams(plist->children[0]);
105 params.push_back(plist->children[2]->data);
106 return params;
107 }
108 else if (plist->children[0]->type == G::Id && plist->children[2]->type == G::Id)
109 return {plist->children[0]->data, plist->children[2]->data};
110 throw Error("Internal error: Invalid ParamList children types", plist);
111}

113std::vector<std::string> parseParams(const parser::Node::Ptr& fhead) {
114 using G = Parser::Grammar;
