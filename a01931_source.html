152void Menu::refreshScroll() {
153 if (!selectedItem) return;
154
155 if (maxSize.x > 0.f) {
156 // left side
157 if (selectedItem->position.x < offset.x) {
158 offset.x = selectedItem->position.x - bgndPadding.left;
159 }
160 // right side
161 if (selectedItem->position.x + selectedItem->getSize().x > offset.x + maxSize.x) {
162 offset.x = selectedItem->getSize().x + selectedItem->position.y + bgndPadding.width -
163 maxSize.x;
164 }
165 }
166 if (maxSize.y > 0.f) {
167 // top
168 if (selectedItem->position.y < offset.y) {
169 offset.y = selectedItem->position.y - bgndPadding.top;
170 }
171 // bottom
172 if (selectedItem->position.y + selectedItem->getSize().y > offset.y + maxSize.y) {
173 offset.y = selectedItem->getSize().y + selectedItem->position.y + bgndPadding.height -
174 maxSize.y;
175 }
176 }
177
178 for (auto& item : items) { item->notifyOffset(offset); }
179}

207void Menu::removeItem(Item* item, bool c) {
208 unsigned int i = items.size();
209 for (unsigned int j = 0; j < items.size(); ++j) {
210 if (items[j].get() == item) { i = j; }
211 else {
212 for (unsigned int k = 0; k < Item::AttachPoint::_NUM_ATTACHPOINTS; ++k) {
213 if (items[j]->attachments[k] == item) {
214 if (c) { items[j]->attachments[k] = item->attachments[k]; }
215 else { items[j]->attachments[k] = nullptr; }
216 }
217 }
218 }
219 }
220 if (i == items.size()) return;
221 items[i]->doSceneRemove();
222 items.erase(items.begin() + i);
223 if (selectedItem == item) {
224 selectedItem = nullptr;
225 if (!items.empty()) {
226 for (unsigned int k = 0; k < Item::AttachPoint::_NUM_ATTACHPOINTS; ++k) {
227 if (item->attachments[k]) {
228 setSelectedItem(item->attachments[k]);
229 break;
230 }
231 }
232 }
233 if (!selectedItem) { setSelectedItem(items.front().get()); }
234 }
235 refreshPositions();
236}

238void Menu::processEvent(const Event& event) {
239 switch (event.type) {
240 case Event::SelectorMove: {
241 if (selectedItem->attachments[event.moveEvent.direction]) {
242 const Item* ogSel = selectedItem;
243 Item* item = selectedItem->attachments[event.moveEvent.direction];
244 while (item) {
245 if (item->isSelectable()) {
246 setSelectedItem(item);
247 break;
248 }
249 if (!item->allowsSelectionCrossing()) break;
250 item = item->attachments[event.moveEvent.direction];
251 }
252 playSound(ogSel != selectedItem ? moveSound : failSound);
253 }
254 else { playSound(failSound); }
255 } break;
256 case Event::Activate:
257 selectedItem->getSignal(Item::Activated)();
258 playSound(selectSound);
259 break;
260 case Event::SelectorLocation: {
261 const glm::vec2 os = observer->transformToOverlaySpace(
262 {event.locationEvent.position.x, event.locationEvent.position.y});
263 const glm::vec2 pos = os - position - offset;
264 const sf::Vector2f sfpos(pos.x, pos.y);
265 for (const auto& item : items) {
266 const sf::FloatRect rect(
267 item->position.x, item->position.y, item->getSize().x, item->getSize().y);
268 if (rect.contains(sfpos)) {
269 if (item->isSelectable() && selectedItem != item.get()) {
270 setSelectedItem(item.get());
271 playSound(moveSound);
272 }
273 }
274 }
275 } break;
276 default:
277 BL_LOG_ERROR << "Menu::processEvent received invalid Event::Type: " << event.type;
278 break;
279 }

282void Menu::refreshPositions() {
283 if (items.empty()) return;
284
285 std::queue<Item*> toVisit;
286 std::unordered_set<Item*> visited;
287 toVisit.emplace(items.front().get());
288 visited.insert(items.front().get());
289 items.front()->position = {0.f, 0.f};
290 sf::FloatRect bounds = {0.f, 0.f, 0.f, 0.f};
291
292 while (!toVisit.empty()) {
293 Item* item = toVisit.front();
294 toVisit.pop();
295
296 const glm::vec2 size = item->getSize();
297 if (item->position.x < bounds.left) bounds.left = item->position.x;
298 if (item->position.x + size.x > bounds.width) bounds.width = item->position.x + size.x;
299 if (item->position.y < bounds.top) bounds.top = item->position.y;
300 if (item->position.y + size.y > bounds.height) bounds.height = item->position.y + size.y;
301
302 for (unsigned int i = 0; i < Item::AttachPoint::_NUM_ATTACHPOINTS; ++i) {
303 const Item::AttachPoint ap = static_cast<Item::AttachPoint>(i);
304
305 Item* v = item->attachments[i];
306 if (v && v->parent == ap && visited.find(v) == visited.end()) {
307 if (!v->positionOverridden) {
308 v->position = move(item->position, size, v->getSize(), ap);
309 }
310 else { v->position = v->posOverride; }
311 toVisit.emplace(v);
312 visited.insert(v);
313 }
314 }
315 }
316
317 totalSize = {bounds.width - bounds.left, bounds.height - bounds.top};
318 const glm::vec2 finalOffset(bounds.left - bgndPadding.left, bounds.top - bgndPadding.top);
319 for (auto& item : items) { item->notifyPosition(item->position - finalOffset); }
320
321 if (selectedItem != nullptr) {
322 com::Transform2D* tform =
323 world->engine().ecs().getComponent<com::Transform2D>(selectedItem->getEntity());
324 const glm::vec2& pos = tform->getLocalPosition();
325 selector->notifySelection(
326 selectedItem->getEntity(),
327 {pos.x, pos.y, selectedItem->getSize().x, selectedItem->getSize().y});
328 }
329
330 refreshScroll();
331 refreshBackground();
332}

340glm::vec2 Menu::move(const glm::vec2& pos, const glm::vec2& psize, const glm::vec2& esize,
341 Item::AttachPoint ap) {
342 switch (ap) {
343 case Item::AttachPoint::Top:
344 return {pos.x, pos.y - std::max(minSize.y, esize.y) - padding.y};
345 case Item::AttachPoint::Right:
346 return {pos.x + std::max(minSize.x, psize.x) + padding.x, pos.y};
347 case Item::AttachPoint::Bottom:
348 return {pos.x, pos.y + std::max(minSize.y, psize.y) + padding.y};
349 case Item::AttachPoint::Left:
350 return {pos.x - std::max(minSize.x, esize.x) - padding.x, pos.y};
351 default:
352 return pos;
353 }
