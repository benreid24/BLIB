342 void Registry::finishComponentAdd(Entity ent, unsigned int cIndex, T* component) {
343  bl::event::Dispatcher::dispatch<event::ComponentAdded<T>>({ent, *component});
344  ComponentMask::SimpleMask& mask = entityMasks[ent.getIndex()];
345  const ComponentMask::SimpleMask ogMask = mask;
346  ComponentMask::add(mask, cIndex);
347  for (auto& view : views) {
348  if (view->mask.passes(mask)) { view->tryAddEntity(ent); }
349  else if (view->mask.passes(ogMask) && ComponentMask::has(view->mask.excluded, cIndex)) {
350  view->removeEntity(ent);
351  }
352  }
353 
354  // ensure parent traits are up to date
355  auto& pool = getPool<T>();
356  for (Entity child : parentGraph.getChildren(ent)) {
357  const std::uint64_t ic = child.getIndex();
358  const ComponentMask mask{.required = entityMasks[ic]};
359  if (mask.contains(pool.ComponentIndex)) { pool.onParentSet(child, ent); }
360  }
361  const Entity parent = parentGraph.getParent(ent);
362  if (parent != InvalidEntity) {
363  const std::uint64_t ip = parent.getIndex();
364  const ComponentMask mask{.required = entityMasks[ip]};
365  if (mask.contains(pool.ComponentIndex)) { pool.onParentSet(ent, parent); }
366  }

380 void Registry::removeComponent(Entity ent) {
381  std::lock_guard lock(entityLock);
382 
383  auto& pool = getPool<T>();
384  const auto i = ent.getIndex();
385  if (i >= entityMasks.size()) { return; }
386  ComponentMask::SimpleMask& mask = entityMasks[i];
387  if (!ComponentMask::contains(mask, pool.ComponentIndex)) { return; }
388 
389  // notify pools of parent remove
390  for (Entity child : parentGraph.getChildren(ent)) {
391  const std::uint32_t ic = child.getIndex();
392  const ComponentMask mask{.required = entityMasks[ic]};
393  if (mask.contains(pool.ComponentIndex)) { pool.onParentRemove(ent, child); }
394  }
395  const Entity parent = parentGraph.getParent(ent);
396  if (parent != InvalidEntity) {
397  const std::uint32_t ip = parent.getIndex();
398  const ComponentMask mask{.required = entityMasks[ip]};
399  if (mask.contains(pool.ComponentIndex)) { pool.onParentRemove(parent, ent); }
400  }
401 
402  // do remove
403  pool.remove(ent);
404  for (auto& view : views) {
405  if (view->mask.passes(mask)) { view->removeEntity(ent); }
406  }
407  ComponentMask::remove(mask, pool.ComponentIndex);
408 }
