166inline void EntityComponentShaderResourceBase::markObjectDirty(scene::Key key) {
167 auto& range = key.updateFreq == UpdateSpeed::Dynamic ? dirtyDynamic : dirtyStatic;
168 if (range.start > range.end) {
169 range.start = key.sceneId;
170 range.end = key.sceneId;
171 }
172 else {
173 range.start = key.sceneId < range.start ? key.sceneId : range.start;
174 range.end = key.sceneId > range.end ? key.sceneId : range.end;
175 }

212template<typename TCom, typename TPayload, typename TDynamicStorage, typename TStaticStorage>
213bool EntityComponentShaderResource<TCom, TPayload, TDynamicStorage, TStaticStorage>::allocateObject(
214 ecs::Entity entity, scene::Key key) {
215 TCom* component = registry->getComponent<TCom>(entity);
216 if (!component) { return false; }
217
218 if (key.updateFreq == UpdateSpeed::Dynamic) {
219 if (dynamicBuffer.ensureSize(key.sceneId + 1)) {
220 dynamicRefresh = 0x1 << cfg::Limits::MaxConcurrentFrames;
221 }
222 component->link(this, key);
223 if (srcDynamicComponents.size() <= key.sceneId) {
224 srcDynamicComponents.resize(key.sceneId + 1, nullptr);
225 }
226 srcDynamicComponents[key.sceneId] = component;
227 }
228 else {
229 if (staticBuffer.ensureSize(key.sceneId + 1)) {
230 staticRefresh = 0x1 << cfg::Limits::MaxConcurrentFrames;
231 }
232 component->link(this, key);
233 if (srcStaticComponents.size() <= key.sceneId) {
234 srcStaticComponents.resize(key.sceneId + 1, nullptr);
235 }
236 srcStaticComponents[key.sceneId] = component;
237 }
238 markObjectDirty(key);
239 return true;
240}

254template<typename TCom, typename TPayload, typename TDynamicStorage, typename TStaticStorage>
255void EntityComponentShaderResource<TCom, TPayload, TDynamicStorage,
256 TStaticStorage>::performTransfer() {
257 if (dirtyStatic.end >= dirtyStatic.start) {
258 staticBuffer.markDirty(dirtyStatic.start, dirtyStatic.end - dirtyStatic.start + 1);
259 staticBuffer.queueTransfer();
260 dirtyStatic = DirtyRange();
261 }
262 if (dirtyDynamic.end >= dirtyDynamic.start) {
263 dynamicBuffer.markDirty(dirtyDynamic.start, dirtyDynamic.end - dirtyDynamic.start + 1);
264 dirtyDynamic = DirtyRange();
265 }
266 dynamicRefresh = dynamicRefresh >> 1;
267 staticRefresh = staticRefresh >> 1;
268}

271void EntityComponentShaderResource<TCom, TPayload, TDynamicStorage,
272 TStaticStorage>::copyFromSource() {
273 for (TCom* com : srcDynamicComponents) {
274 if (com && com->isDirty()) {
275 auto& payload = dynamicBuffer[com->getSceneKey().sceneId];
276 using PayloadType = std::remove_reference_t<decltype(payload)>;
277 com->template refresh<PayloadType>(payload);
278 markObjectDirty(com->getSceneKey());
279 }
280 }

285 com->template refresh<PayloadType>(payload);
