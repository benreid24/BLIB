111 void Animation2DSystem::doSlideshowAdd(com::Animation2DPlayer& player) {
112  // determine frame offset and player index
113  const auto it = slideshowFrameMap.find(player.animation.get());
114  const bool uploadFrames = it == slideshowFrameMap.end();
115  const std::uint32_t offset =
116  uploadFrames ?
117  slideshowFrameRangeAllocator.alloc(player.animation->frameCount()).range.start :
118  it->second;
119  const std::uint32_t index = slideshowPlayerIds.allocate();
120  player.playerIndex = index;
121 
122  // add frame offset and current frame to SSBO's
123  slideshowFrameOffsetSSBO.ensureSize(index + 1);
124  if (slideshowPlayerCurrentFrameSSBO.ensureSize(index + 1)) {
125  players->forEach([this](ecs::Entity, com::Animation2DPlayer& player) {
126  slideshowPlayerCurrentFrameSSBO.assignRef(player.framePayload, player.playerIndex);
127  });
128  }
129  else { slideshowPlayerCurrentFrameSSBO.assignRef(player.framePayload, player.playerIndex); }
130  slideshowFrameOffsetSSBO[index] = offset;
131  slideshowPlayerCurrentFrameSSBO[index] = player.currentFrame;
132  slideshowOffsetUploadRange.addRange(index, 1);
133 
134  // add frames to SSBO if required
135  if (uploadFrames) {
136  slideshowFrameMap[player.animation.get()] = offset;
137  slideshowFramesSSBO.ensureSize(offset + player.animation->frameCount());
138  for (std::size_t i = 0; i < player.animation->frameCount(); ++i) {
139  const auto& src = player.animation->getFrame(i).shards.front();
140  auto& frame = slideshowFramesSSBO[offset + i];
141  const sf::FloatRect& tex = src.normalizedSource;
142  frame.opacity = static_cast<float>(src.alpha) / 255.f;
143  frame.texCoords[0] = {tex.left, tex.top};
144  frame.texCoords[1] = {tex.left + tex.width, tex.top};
145  frame.texCoords[2] = {tex.left + tex.width, tex.top + tex.height};
146  frame.texCoords[3] = {tex.left, tex.top + tex.height};
147  }
148  slideshowFrameUploadRange.addRange(offset, player.animation->frameCount());
149  }
150  slideshowDataRefCounts[player.animation.get()] += 1;
151 
152  // mark that descriptors need to be reset
153  slideshowRefreshRequired = rc::Config::MaxConcurrentFrames;
154 }

156 void Animation2DSystem::doSlideshowFree(const com::Animation2DPlayer& player) {
157  slideshowPlayerIds.release(player.playerIndex);
158 
159  const auto refCountIt = slideshowDataRefCounts.find(player.animation.get());
160  if (refCountIt == slideshowDataRefCounts.end()) {
161  throw std::runtime_error("Player component removed without having been added");
162  }
163  refCountIt->second -= 1;
164 
165  if (refCountIt->second == 0) {
166  slideshowDataRefCounts.erase(refCountIt);
167  const auto offsetIt = slideshowFrameMap.find(player.animation.get());
168  if (offsetIt == slideshowFrameMap.end()) {
169  throw std::runtime_error("Failed to find cached frame offset for animation");
170  }
171 
172  slideshowFrameRangeAllocator.release(
173  {offsetIt->second, static_cast<std::uint32_t>(player.animation->frameCount())});
174  slideshowFrameMap.erase(offsetIt);
