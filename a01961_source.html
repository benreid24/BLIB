33DispatchType MovementControl::process(const sf::Event& event) {
34 if (type == Type::Triggers) {
35 if (keys.up.process(event)) return DispatchType::MovementUp;
36 if (keys.right.process(event)) return DispatchType::MovementRight;
37 if (keys.down.process(event)) return DispatchType::MovementDown;
38 if (keys.left.process(event)) return DispatchType::MovementLeft;
39 }
40 else if (type == Type::Joystick) {
41 const sf::Vector2f old = joystick.cachedPosition;
42
43 if (joystick.process(event)) {
44 if (event.joystickMove.axis == joystick.verticalAxis) {
45 if (joystick.cachedPosition.y >= StickThreshold && old.y < StickThreshold)
46 return DispatchType::MovementUp;
47 if (joystick.cachedPosition.y <= -StickThreshold && old.y > -StickThreshold)
48 return DispatchType::MovementDown;
49 }
50 else {
51 if (joystick.cachedPosition.x >= StickThreshold && old.x < StickThreshold)
52 return DispatchType::MovementRight;
53 if (joystick.cachedPosition.x <= -StickThreshold && old.y > -StickThreshold)
54 return DispatchType::MovementLeft;
55 }
56 }
57 }
58 return DispatchType::_INVALID;
59}

83sf::Vector2f MovementControl::readValue() const {
84 if (type == Type::Triggers) {
85 sf::Vector2f dir;
86 if (keys.up.active()) dir.y += 1.f;
87 if (keys.right.active()) dir.x += 1.f;
88 if (keys.down.active()) dir.y -= 1.f;
89 if (keys.left.active()) dir.x -= 1.f;
90 return math::normalized(dir);
91 }
92 else {
93 return math::normalized(joystick.cachedPosition);
94 }
95}
