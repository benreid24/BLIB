104void FadeEffectTask::execute(const rg::ExecutionContext& ctx) {
105 if (!ctx.isFinalStep) {
106 output->currentFramebuffer().beginRender(ctx.commandBuffer,
107 output->scissor,
108 output->clearColors,
109 output->clearColorCount,
110 false,
111 output->getRenderPass().rawPass());
112 }
113 else {
114 VkClearAttachment attachments[2]{};
115 attachments[0].aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
116 attachments[0].colorAttachment = 0;
117 attachments[0].clearValue = output->clearColors[0];
118 attachments[1].aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT;
119 attachments[1].clearValue = output->clearColors[1];
120
121 VkClearRect rects[2]{};
122 rects[0].rect = output->scissor;
123 rects[0].baseArrayLayer = 0;
124 rects[0].layerCount = 1;
125 rects[1] = rects[0];
126
127 vkCmdClearAttachments(ctx.commandBuffer, 2, attachments, 2, rects);
128 }
129
130 vkCmdSetScissor(ctx.commandBuffer, 0, 1, &output->scissor);
131 vkCmdSetViewport(ctx.commandBuffer, 0, 1, &output->viewport);
132
133 VkBuffer vb = indexBuffer.vertexBufferHandle();
134 VkDeviceSize offsets[] = {0};
135 pipeline->bind(ctx.commandBuffer, output->renderPassId);
136 vkCmdBindVertexBuffers(ctx.commandBuffer, 0, 1, &vb, offsets);
137 vkCmdBindIndexBuffer(
138 ctx.commandBuffer, indexBuffer.indexBufferHandle(), 0, buf::IndexBuffer::IndexType);
139 vkCmdBindDescriptorSets(ctx.commandBuffer,
140 VK_PIPELINE_BIND_POINT_GRAPHICS,
141 pipeline->pipelineLayout().rawLayout(),
142 0,
143 1,
144 &descriptorSets.current(),
145 0,
146 nullptr);
147 vkCmdPushConstants(ctx.commandBuffer,
148 pipeline->pipelineLayout().rawLayout(),
149 VK_SHADER_STAGE_FRAGMENT_BIT,
150 0,
151 sizeof(float),
152 &factor);
153 vkCmdDrawIndexed(ctx.commandBuffer,
154 indexBuffer.getDrawParameters().indexCount,
155 1,
156 0,
157 indexBuffer.getDrawParameters().vertexOffset,
158 indexBuffer.getDrawParameters().firstInstance);
159
160 if (!ctx.isFinalStep) { output->currentFramebuffer().finishRender(ctx.commandBuffer); }

163void FadeEffectTask::update(float dt) {
164 if (factor < fadeEnd) {
165 factor += fadeSpeed * dt;
166 if (factor >= fadeEnd) {
167 fadeSpeed = 0.f;
168 factor = fadeEnd;
169 }
170 }
171 else if (factor > fadeEnd) {
172 factor += fadeSpeed * dt;
173 if (factor <= fadeEnd) {
174 fadeSpeed = 0.f;
175 factor = fadeEnd;
176 }
177 }
