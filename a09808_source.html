1243 Group g;
1244 g.addField("type", Serializer<script::PrimitiveValue::Type>::serialize(val.getType()));
1245
1246 Value v(false);
1247 switch (val.getType()) {
1248 case script::PrimitiveValue::TBool:
1249 v = val.getAsBool();
1250 break;
1251 case script::PrimitiveValue::TInteger:
1252 v = val.getAsInt();
1253 break;
1254 case script::PrimitiveValue::TFloat:
1255 v = val.getAsFloat();
1256 break;
1257 case script::PrimitiveValue::TString:
1258 v = val.getAsString();
1259 break;

1295 return false;

1297

1299 if (!v) return false;
1300
1301 switch (type) {
1302 case script::PrimitiveValue::TBool:
1303 if (!v->getAsBool()) return false;
1304 result.value() = *v->getAsBool();
1305 break;
1306
1307 case script::PrimitiveValue::TInteger:
1308 if (!v->getAsInteger() && !v->getAsFloat()) return false;
1309 result.value() = v->getNumericAsInteger();
1310 break;
1311
1312 case script::PrimitiveValue::TFloat:
