27template<typename T>
28void ParticleManager<T>::update(util::ThreadPool& threadPool, float dt, float realDt) {
29 std::unique_lock lock(mutex);
30
31 // call updaters before anything else
32 if (!updaters.empty()) {
33 typename TUpdater::Proxy proxy(*this, affectors, emitters, sinks);
34 for (std::size_t i = updaters.size(); i > 0; --i) {
35 const std::size_t idx = i - 1;
36 updaters[idx]->update(proxy, dt, realDt);
37 if (proxy.erased) {
38 if (i != updaters.size()) { updaters[idx] = std::move(updaters.back()); }
39 updaters.pop_back();
40 }
41 proxy.reset();
42 }
43 }
44
45 if (!sinks.empty() && !particles.empty()) {
46 // reset free list
47 freeList.clear();
48 freed.resize(particles.size());
49 std::fill(freed.begin(), freed.end(), false);
50
51 // update sinks first to populate freelist
52 for (std::size_t i = sinks.size(); i > 0; --i) {
53 updateSink(i - 1, threadPool, dt, realDt);
54 }
55 }
56
57 // run emitters to fill holes
58 if (!emitters.empty()) {
59 typename TEmitter::Proxy proxy(*this, particles, freeList);
60 for (std::size_t i = emitters.size(); i > 0; --i) {
61 const std::size_t idx = i - 1;
62 emitters[idx]->update(proxy, dt, realDt);
63 if (proxy.erased) {
64 if (i != emitters.size()) { emitters[idx] = std::move(emitters.back()); }
65 emitters.pop_back();
66 }
67 proxy.reset();
68 }
69 }
70
71 // remove particles that did not get re-emitted
72 for (std::size_t i : freeList) {
73 if (particles.size() > 1 && i < particles.size() - 1) { particles[i] = particles.back(); }
74 particles.pop_back();
75 }
76 freeList.clear();
77
78 // run affectors over all particles
79 if (!affectors.empty()) {
80 std::vector<std::uint8_t> erased;
81 erased.resize(affectors.size(), 0);
82
83 futures.reserve(particles.size() / ParticlesPerThread + 1);
84 auto it = particles.begin();
85 while (it != particles.end()) {
86 const std::size_t len =
87 std::min<std::size_t>(std::distance(it, particles.end()), ParticlesPerThread);
88 futures.emplace_back(threadPool.queueTask([this, it, len, dt, realDt, &erased]() {
89 typename TAffector::Proxy proxy(*this, std::span<T>(&*it, len));
90 unsigned int i = 0;
91 for (auto& affector : affectors) {
92 affector->update(proxy, dt, realDt);
93 erased[i] = proxy.erased ? 1 : 0;
94 proxy.reset();
95 ++i;
96 }
97 }));
98 it += len;
99 }
100
101 for (auto& f : futures) { f.wait(); }
102 futures.clear();
103
104 for (std::size_t i = affectors.size(); i > 0; --i) {
105 const std::size_t idx = i - 1;
106 if (erased[idx] != 0) {
107 if (i != affectors.size()) { affectors[idx] = std::move(affectors.back()); }
108 affectors.pop_back();
109 }
110 }
111 }
112
113 // update global info
114 globalInfo.cameraToWindowScale =
115 world->engine().renderer().getObserver().getRegionSize().x /
116 world->engine().renderer().getObserver().getCurrentCamera()->getViewerSize().x;
117
118 // update renderer data
119 renderer.notifyData(particles.data(), particles.size());
120}

243template<typename T>
244void ParticleManager<T>::updateSink(std::size_t i, util::ThreadPool& pool, float dt, float realDt) {
245 auto* sink = sinks[i].get();
246 typename TSink::Proxy proxy(releaseMutex, *this, &particles.front(), freeList, freed);
247 auto it = particles.begin();
248
249 futures.reserve(particles.size() / ParticlesPerThread + 1);
250 while (it != particles.end()) {
251 const std::size_t len =
252 std::min<std::size_t>(std::distance(it, particles.end()), ParticlesPerThread);
253 futures.emplace_back(pool.queueTask([this, &proxy, it, len, sink, dt, realDt]() {
254 sink->update(proxy, std::span<T>(&*it, len), dt, realDt);
255 }));
256 it += len;
257 }
258
259 for (auto& f : futures) { f.wait(); }
260 futures.clear();
261
262 if (proxy.erased) {
263 if (i != sinks.size() - 1) { sinks[i] = std::move(sinks.back()); }
264 sinks.pop_back();
265 }
