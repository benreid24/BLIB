48bool InputStream::open(const std::string& path) {
49 auto& file = stream.emplace<std::ifstream>(path, std::ios::binary);
50 if (!file.good()) { return false; }
51 file.seekg(0, std::ios::end);
52 knownSize = file.tellg();
53 file.seekg(0, std::ios::beg);
54 return true;

57bool InputStream::open(const std::string& path, std::size_t offset, std::size_t length) {
58 auto& section = stream.emplace<FileSection>(path, offset);
59 if (!section.file.good()) { return false; }
60 section.file.seekg(offset + length);
61 if (!section.file.good()) { return false; }
62 section.file.seekg(offset);
63 if (!section.file.good()) { return false; }

83std::size_t InputStream::read(void* data, std::size_t len) {
84 return std::visit(
85 util::Visitor{[](const std::monostate&) { return static_cast<std::size_t>(0); },
86 [data, len](std::ifstream& stream) -> std::size_t {
87 stream.read(static_cast<char*>(data), len);
88 return stream.gcount();
89 },
90 [this, data, len](FileSection& section) -> std::size_t {
91 const std::size_t globalPos = section.file.tellg();
92 const std::size_t localPos = globalPos - section.offset;
93 const std::size_t toRead = std::min(len, knownSize - localPos);
94 section.file.read(static_cast<char*>(data), toRead);
95 return section.file.gcount();
96 },
97 [data, len](Buffer& buf) -> std::size_t {
98 const std::size_t toRead = std::min(len, buf.data.size() - buf.pos);
99 std::memcpy(data, buf.data.data() + buf.pos, toRead);
100 buf.pos += toRead;

102 },
103 [data, len](std::istream* s) -> std::size_t {
104 if (!s || !s->good()) { return static_cast<std::size_t>(0); }
105 s->read(static_cast<char*>(data), len);
106 return s->gcount();
107 }},
108 stream);
