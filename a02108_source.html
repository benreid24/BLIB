55void Scene::syncObjects() {
56 std::unique_lock lock(objectMutex);
57 std::unique_lock queueLock(queueMutex);
58 isClearingQueues = true;
59
60 // do batch changes
61 for (const auto& change : queuedBatchChanges) {
62 auto& objectPipelines = change.changed->sceneKey.updateFreq == UpdateSpeed::Static ?
63 staticPipelines :
64 dynamicPipelines;
65 doBatchChange(change, objectPipelines[change.changed->sceneKey.sceneId]);
66 objectPipelines[change.changed->sceneKey.sceneId] = change.newPipeline;
67 }
68 queuedBatchChanges.clear();
69
70 // remove queued objects
71 for (auto* obj : queuedRemovals) { removeQueuedObject(obj); }
72 queuedRemovals.clear();
73
74 // add queued objects
75 ecs::Transaction<ecs::tx::EntityRead> tx(engine.ecs());
76 for (auto& add : queuedAdds) {
77 if (engine.ecs().entityExists(add.entity, tx)) { addQueuedObject(add); }
78 }
79 queuedAdds.clear();
80
81 isClearingQueues = false;
82}

103void Scene::removeQueuedObject(scene::SceneObject* obj) {
104 std::unique_lock lock(queueMutex);
105
106 auto& objectPipelines =
107 obj->sceneKey.updateFreq == UpdateSpeed::Static ? staticPipelines : dynamicPipelines;
108 const auto cachedKey = obj->sceneKey.sceneId;
109 std::uint32_t pipeline =
110 cachedKey < objectPipelines.size() ? objectPipelines[cachedKey] : NoPipeline;
111 if (pipeline != NoPipeline) {
112 doObjectRemoval(obj, pipeline);
113 objectPipelines[cachedKey] = NoPipeline;
