22void TexturePool::init() {
23 // create descriptor layout
24 VkDescriptorSetLayoutBinding setBindings[] = {textures.getLayoutBinding()};
25 VkDescriptorSetLayoutCreateInfo descriptorCreateInfo{};
26 descriptorCreateInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_LAYOUT_CREATE_INFO;
27 descriptorCreateInfo.bindingCount = std::size(setBindings);
28 descriptorCreateInfo.pBindings = setBindings;
29 if (VK_SUCCESS !=
30 vkCreateDescriptorSetLayout(
31 vulkanState.device, &descriptorCreateInfo, nullptr, &descriptorSetLayout)) {
32 throw std::runtime_error("Failed to create texture pool descriptor set layout");
33 }
34
35 // create descriptor pool
36 VkDescriptorPoolSize poolSize{};
37 poolSize.type = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
38 poolSize.descriptorCount = Config::MaxTextureCount * 2 * Config::MaxConcurrentFrames;
39
40 VkDescriptorPoolCreateInfo poolCreate{};
41 poolCreate.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_POOL_CREATE_INFO;
42 poolCreate.maxSets = 2 * Config::MaxConcurrentFrames;
43 poolCreate.poolSizeCount = 1;
44 poolCreate.pPoolSizes = &poolSize;
45 if (VK_SUCCESS !=
46 vkCreateDescriptorPool(vulkanState.device, &poolCreate, nullptr, &descriptorPool)) {
47 throw std::runtime_error("Failed to create texture descriptor pool");
48 }
49
50 // allocate descriptor set
51 VkDescriptorSet allocatedSets[2 * Config::MaxConcurrentFrames];
52 std::array<VkDescriptorSetLayout, 2 * Config::MaxConcurrentFrames> setLayouts;
53 setLayouts.fill(descriptorSetLayout);
54
55 VkDescriptorSetAllocateInfo setAlloc{};
56 setAlloc.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
57 setAlloc.descriptorPool = descriptorPool;
58 setAlloc.descriptorSetCount = setLayouts.size();
59 setAlloc.pSetLayouts = setLayouts.data();
60 if (VK_SUCCESS != vkAllocateDescriptorSets(vulkanState.device, &setAlloc, allocatedSets)) {
61 throw std::runtime_error("Failed to allocate texture descriptor set");
62 }
63
64 unsigned int i = 0;
65 descriptorSets.init(vulkanState, [&i, &allocatedSets](auto& set) { set = allocatedSets[i++]; });
66 rtDescriptorSets.init(vulkanState,
67 [&i, &allocatedSets](auto& set) { set = allocatedSets[i++]; });
68
69 // create error texture pattern and init texture array
70 constexpr unsigned int ErrorSize = 1024;
71 constexpr unsigned int ErrorBoxSize = 64;
72 textures.getErrorPattern().create(ErrorSize, ErrorSize);
73 for (unsigned int x = 0; x < ErrorSize; ++x) {
74 for (unsigned int y = 0; y < ErrorSize; ++y) {
75 const unsigned int xi = x / ErrorBoxSize;
76 const unsigned int yi = y / ErrorBoxSize;
77 if ((xi % 2) == (yi % 2)) {
78 textures.getErrorPattern().setPixel(x, y, sf::Color(230, 66, 245));
79 }
80 else { textures.getErrorPattern().setPixel(x, y, sf::Color(255, 254, 196)); }
81 }
82 }
83 textures.init(descriptorSets, rtDescriptorSets);
84}

117void TexturePool::doRelease(std::uint32_t i) {
118 refCounts[i] = 0;
119
120 if (i < reverseFileMap.size()) {
121 freeSlots.release(i);
122 if (reverseFileMap[i]) {
123 fileMap.erase(*reverseFileMap[i]);
124 reverseFileMap[i] = nullptr;
125 }
126 if (reverseImageMap[i]) {
127 imageMap.erase(reverseImageMap[i]);
128 reverseImageMap[i] = nullptr;
129 }
130 }
131 else { freeRtSlots.release(i - reverseFileMap.size()); }
132
133 textures.resetTexture(i);
134}

148TextureRef TexturePool::allocateTexture() {
149 if (!freeSlots.available()) {
150 releaseUnusedLocked();
151 if (!freeSlots.available()) { throw std::runtime_error("All texture slots in use!"); }
152 }
153
154 const std::uint32_t i = freeSlots.allocate();
155 refCounts[i].store(0);
156 reverseFileMap[i] = nullptr;
157 reverseImageMap[i] = nullptr;
158
159 return TextureRef{*this, textures.getTexture(i), i};

216 if (!sampler) { sampler = vulkanState.samplerCache.filteredBorderClamped(); }
