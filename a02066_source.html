108 CodeSceneObject* obj = static_cast<CodeSceneObject*>(object.getSceneRef().object);

110#ifdef BLIB_DEBUG

116 renderContext.bindDescriptors(
117 obj->pipeline->getPipeline(cfg::RenderPhases::Forward)->pipelineLayout().rawLayout(),
118 obj->sceneKey.updateFreq,
119 obj->descriptors.get(renderContext.currentObserverIndex()).data(),

126 obj->pipeline->getPipeline(cfg::RenderPhases::Forward)

129 i,
130 obj->sceneKey);
