242bool IndexBufferT<T>::ensureSize(std::uint32_t vertexCount, std::uint32_t indexCount) {
243 bool r = false;
244 if (vertexCount > cpuVertexBuffer.size()) {
245 cpuVertexBuffer.resize(vertexCount);
246 gpuVertexBuffer.ensureSize(vertexCount * sizeof(T));
247 vertexWriteStart = 0;
248 vertexWriteCount = vertexCount;
249 r = true;
250 }
251 if (indexCount > cpuIndexBuffer.size()) {
252 cpuIndexBuffer.resize(indexCount);
253 gpuIndexBuffer.ensureSize(indexCount * sizeof(IndexType));
254 indexWriteStart = 0;
255 indexWriteCount = indexCount;
256 r = true;
257 }
258 return r;
259}

270template<typename T>
271void IndexBufferT<T>::executeTransfer(VkCommandBuffer commandBuffer,
272 tfr::TransferContext& context) {
273 VkBuffer stagingBuf;
274 void* stagingMem;
275 VkBufferCopy copyCmd{};
276 VkBufferMemoryBarrier barrier{};
277
278 // vertex buffer
279 if (vertexWriteStart + vertexWriteCount <= cpuVertexBuffer.size() && !cpuVertexBuffer.empty()) {
280 const VkDeviceSize vertexOffset = vertexWriteStart * sizeof(T);
281 const VkDeviceSize vertexSize = vertexWriteCount * sizeof(T);
282
283 if (vertexSize > 0) {
284 context.createTemporaryStagingBuffer(vertexSize, stagingBuf, &stagingMem);
285 std::memcpy(stagingMem, &cpuVertexBuffer[vertexWriteStart], vertexSize);
286
287 copyCmd.srcOffset = 0;
288 copyCmd.dstOffset = vertexOffset;
289 copyCmd.size = vertexSize;
290 vkCmdCopyBuffer(commandBuffer, stagingBuf, gpuVertexBuffer.getBuffer(), 1, &copyCmd);
291
292 barrier.sType = VK_STRUCTURE_TYPE_BUFFER_MEMORY_BARRIER;
293 barrier.buffer = gpuVertexBuffer.getBuffer();
294 barrier.offset = vertexOffset;
295 barrier.size = vertexSize;
296 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
297 barrier.dstAccessMask = VK_ACCESS_VERTEX_ATTRIBUTE_READ_BIT;
298 context.registerBufferBarrier(barrier);
299 }
300 }
301 else { BL_LOG_WARN << "Invalid vertex buffer write, skipping"; }
302
303 // index buffer
304 if (indexWriteStart + indexWriteCount <= cpuIndexBuffer.size() && !cpuIndexBuffer.empty()) {
305 const VkDeviceSize indexOffset = indexWriteStart * sizeof(IndexType);
306 const VkDeviceSize indexSize = indexWriteCount * sizeof(IndexType);
307
308 if (indexSize > 0) {
309 context.createTemporaryStagingBuffer(indexSize, stagingBuf, &stagingMem);
310 std::memcpy(stagingMem, &cpuIndexBuffer[indexWriteStart], indexSize);
311
312 copyCmd.size = indexSize;
313 copyCmd.dstOffset = indexOffset;
314 vkCmdCopyBuffer(commandBuffer, stagingBuf, gpuIndexBuffer.getBuffer(), 1, &copyCmd);
315
316 barrier.buffer = gpuIndexBuffer.getBuffer();
317 barrier.size = indexSize;
318 context.registerBufferBarrier(barrier);
319 }
320 }
321 else { BL_LOG_WARN << "Invalid index buffer write, skipping"; }
322}
