146  prevSpace = EndIter;

149  maxLineHeight = it.getText().cachedLineHeight;

154  it.getText().cachedLineHeight = maxLineHeight;
155  glm::vec2 advance =
156  it.getText().advanceCharacterPos(*font, nextPos, it.getChar(), prevChar);
157  prevChar = it.getChar();
158

160  if (lineOnNextSpace) {
161  it.makeNewline();

163  wordStartPos = glm::vec2{0.f, advance.y + maxLineHeight};
164  }
165  else {

169  }

173  }

176  if (prevSpace != EndIter) {

178  resetLine(it);
