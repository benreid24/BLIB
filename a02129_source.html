25scene::SceneObject* BatchedScene::doAdd(ecs::Entity entity, rcom::DrawableBase& obj,
26 UpdateSpeed updateFreq) {
27 const auto alloc = objects.allocate(updateFreq, entity);
28 const std::uint32_t sceneId = alloc.newObject->sceneKey.sceneId;
29 auto& transCache = updateFreq == UpdateSpeed::Static ? staticTransCache : dynamicTransCache;
30
31 if (alloc.addressesChanged) {
32 transCache.resize(alloc.newCapacity);
33 handleAddressChange(updateFreq, alloc.originalBaseAddress);
34 }
35
36 transCache[sceneId] = obj.getContainsTransparency();
37 auto& batch = obj.getContainsTransparency() ? transparentObjects : opaqueObjects;
38 vk::Pipeline& pipeline = renderer.pipelineCache().getPipeline(obj.getCurrentPipeline());
39
40 // find or create layout batch
41 LayoutBatch* layoutBatch = nullptr;
42 for (LayoutBatch& lb : batch.batches) {
43 if (lb.layout.rawLayout() == pipeline.pipelineLayout().rawLayout()) {
44 layoutBatch = &lb;
45 break;
46 }
47 }
48 if (!layoutBatch) {
49 batch.batches.emplace_back(descriptorSets, pipeline.pipelineLayout());
50 layoutBatch = &batch.batches.back();
51 }
52
53 // init descriptor sets, return on fail
54 for (std::uint8_t i = 0; i < layoutBatch->descriptorCount; ++i) {
55 if (!layoutBatch->descriptors[i]->allocateObject(entity, alloc.newObject->sceneKey)) {
56 for (std::int8_t j = i - 1; j >= 0; --j) {
57 layoutBatch->descriptors[j]->releaseObject(entity, alloc.newObject->sceneKey);
58 }
59 return nullptr;
60 }
61 }
62
63 // find or create pipeline batch
64 auto& pipelineBatches = updateFreq == UpdateSpeed::Static ? layoutBatch->staticBatches :
65 layoutBatch->dynamicBatches;
66 PipelineBatch* pipelineBatch = nullptr;
67 for (PipelineBatch& pb : pipelineBatches) {
68 if (&pb.pipeline == &pipeline) {
69 pipelineBatch = &pb;
70 break;
71 }
72 }
73 if (!pipelineBatch) {
74 pipelineBatches.emplace_back(pipeline);
75 pipelineBatch = &pipelineBatches.back();
76 }
77
78 // add to pipeline batch
79 pipelineBatch->objects.emplace_back(alloc.newObject);
80
81 // add scene link
82 engine.ecs().emplaceComponent<com::BatchSceneLink>(entity, alloc.newObject->sceneKey);
83
84 return alloc.newObject;
85}

107void BatchedScene::releaseObject(SceneObject* object, std::uint32_t pipelineId) {
108 // lookup object
109 const ecs::Entity entity = objects.getObjectEntity(object->sceneKey);
110 auto& transCache =
111 object->sceneKey.updateFreq == UpdateSpeed::Static ? staticTransCache : dynamicTransCache;
112 auto& batch = transCache[object->sceneKey.sceneId] ? transparentObjects : opaqueObjects;
113
114 // remove from batch
115 vk::Pipeline& pipeline = renderer.pipelineCache().getPipeline(pipelineId);
116 for (LayoutBatch& p : batch.batches) {
117 if (p.layout.rawLayout() == pipeline.pipelineLayout().rawLayout()) {
118 auto& pipelineBatches = object->sceneKey.updateFreq == UpdateSpeed::Static ?
119 p.staticBatches :
120 p.dynamicBatches;
121 for (PipelineBatch& pb : pipelineBatches) {
122 if (&pb.pipeline == &pipeline) {
123 for (auto it = pb.objects.begin(); it != pb.objects.end(); ++it) {
124 if (*it == object) {
125 pb.objects.erase(it);
126 goto removeDescriptors;
127 }
128 }
129 break;
130 }
131 }
132 removeDescriptors:
133 for (std::uint8_t i = 0; i < p.descriptorCount; ++i) {
134 p.descriptors[i]->releaseObject(entity, object->sceneKey);
135 }
136 break;
137 }
138 }
139
140 // cleanup scene link
141 engine.ecs().removeComponent<com::BatchSceneLink>(entity);

144void BatchedScene::doBatchChange(const BatchChange& change, std::uint32_t ogPipeline) {
145 const bool isStatic = change.changed->sceneKey.updateFreq == UpdateSpeed::Static;
146 auto& transCache = isStatic ? staticTransCache : dynamicTransCache;
147 const bool wasTrans = transCache[change.changed->sceneKey.sceneId];
148 transCache[change.changed->sceneKey.sceneId] = change.newTrans;
149
150 vk::Pipeline& newPipeline = renderer.pipelineCache().getPipeline(change.newPipeline);
151 vk::Pipeline& oldPipeline = renderer.pipelineCache().getPipeline(ogPipeline);
152
153 // determine what changed
154 const bool transChanged = wasTrans != change.newTrans;
155 const bool pipelineChanged = ogPipeline != change.newPipeline;
156 bool layoutChanged = false;
157
158 // determine if layout changed
159 if (pipelineChanged) {
160 layoutChanged =
161 newPipeline.pipelineLayout().rawLayout() != oldPipeline.pipelineLayout().rawLayout();
162 }
163
164 // re-bucket if trans, layout, or pipeline changed
165 if (transChanged || layoutChanged || pipelineChanged) {
166 // remove from old bucket
167 LayoutBatch* oldLayoutBatch = nullptr;
168 auto& oldObjectBatch = wasTrans ? transparentObjects : opaqueObjects;
169 for (LayoutBatch& lb : oldObjectBatch.batches) {
170 if (lb.layout.rawLayout() == oldPipeline.pipelineLayout().rawLayout()) {
171 oldLayoutBatch = &lb;
172 auto& pipelineBatches = isStatic ? lb.staticBatches : lb.dynamicBatches;
173 for (PipelineBatch& pb : pipelineBatches) {
174 if (&pb.pipeline == &oldPipeline) {
175 for (auto it = pb.objects.begin(); it != pb.objects.end(); ++it) {
176 if (*it == change.changed) {
177 pb.objects.erase(it);
178 break;
179 }
180 }
181 break;
182 }
183 }
184 break;
185 }
186 }
187
188 auto& batch = change.newTrans ? transparentObjects : opaqueObjects;
189
190 // find or create new layout batch
191 LayoutBatch* layoutBatch = nullptr;
192 for (LayoutBatch& lb : batch.batches) {
193 if (lb.layout.rawLayout() == newPipeline.pipelineLayout().rawLayout()) {
194 layoutBatch = &lb;
195 break;
196 }
197 }
198 if (!layoutBatch) {
199 layoutBatch = &batch.batches.emplace_back(descriptorSets, newPipeline.pipelineLayout());
200 }
201
202 // find or create pipeline batch
203 auto& pipelineBatches = isStatic ? layoutBatch->staticBatches : layoutBatch->dynamicBatches;
204 PipelineBatch* pipelineBatch = nullptr;
205 for (PipelineBatch& pb : pipelineBatches) {
206 if (&pb.pipeline == &newPipeline) {
207 pipelineBatch = &pb;
208 break;
209 }
210 }
211 if (!pipelineBatch) { pipelineBatch = &pipelineBatches.emplace_back(newPipeline); }
212
213 // add to pipeline batch
214 pipelineBatch->objects.emplace_back(change.changed);
215
216 // update descriptors if layout changed
217 if (layoutChanged) {
218 SceneObject& object = *change.changed;
219 const ecs::Entity entity = objects.getObjectEntity(object.sceneKey);
220 newPipeline.pipelineLayout().updateDescriptorSets(descriptorSets,
221 oldLayoutBatch->descriptors.data(),
222 oldLayoutBatch->descriptorCount,
223 entity,
224 object.sceneKey.sceneId,
225 object.sceneKey.updateFreq);
226 }
227 }

230void BatchedScene::renderScene(scene::SceneRenderContext& ctx) {
231 for (ObjectBatch* batch : std::array<ObjectBatch*, 2>{&opaqueObjects, &transparentObjects}) {
232 for (LayoutBatch& lb : batch->batches) {
233 const VkPipelineLayout layout = lb.layout.rawLayout();
234 UpdateSpeed speed = UpdateSpeed::Dynamic;
235
236 for (auto* pipelineBatches : std::array<std::vector<PipelineBatch>*, 2>{
237 &lb.dynamicBatches, &lb.staticBatches}) {
238 // bind layout descriptors
239 ctx.bindDescriptors(layout, speed, lb.descriptors.data(), lb.descriptorCount);
240
241 // render each pipeline
242 if (!lb.bindless) {
243 for (PipelineBatch& pb : *pipelineBatches) {
244 ctx.bindPipeline(pb.pipeline);
245 for (SceneObject* obj : pb.objects) {
246 if (obj->hidden) { continue; }
247 for (std::uint8_t i = lb.perObjStart; i < lb.descriptorCount; ++i) {
248 lb.descriptors[i]->bindForObject(ctx, layout, i, obj->sceneKey);
249 }
250 ctx.renderObject(*obj);
251 }
252 }
253 }
254 else {
255 for (PipelineBatch& pb : *pipelineBatches) {
256 ctx.bindPipeline(pb.pipeline);
257 for (SceneObject* obj : pb.objects) {
258 if (obj->hidden) { continue; }
259 ctx.renderObject(*obj);
260 }
261 }
262 }
263
264 speed = UpdateSpeed::Static;
265 }
266 }
267 }
