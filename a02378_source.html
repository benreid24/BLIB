111 ds::SetWriteHelper setWriter;
112 setWriter.hintWriteCount(bufferWriteCount);
113 setWriter.hintBufferInfoCount(bufferWriteCount);
114
115 // write descriptors

117 const auto set = descriptorSets.getRaw(j);
118
119 VkDescriptorBufferInfo& cameraBufferInfo = setWriter.getNewBufferInfo();
120 cameraBufferInfo.buffer = cameraBuffer->getBuffer().getRawBuffer(j);
121 cameraBufferInfo.offset = 0;
122 cameraBufferInfo.range = cameraBuffer->getBuffer().getTotalAlignedSize();
123
124 VkWriteDescriptorSet& cameraWrite = setWriter.getNewSetWrite(set);
125 cameraWrite.descriptorCount = 1;
126 cameraWrite.dstBinding = 0;
127 cameraWrite.dstArrayElement = 0;
128 cameraWrite.pBufferInfo = &cameraBufferInfo;
129 cameraWrite.descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
130
131 VkDescriptorBufferInfo& lightInfoBufferInfo = setWriter.getNewBufferInfo();
132 lightInfoBufferInfo.buffer = lightBuffer->getBuffer().getCurrentFrameRawBuffer();
133 lightInfoBufferInfo.offset = 0;
134 lightInfoBufferInfo.range = lightBuffer->getBuffer().getTotalAlignedSize();
135
136 VkWriteDescriptorSet& lightInfoWrite = setWriter.getNewSetWrite(set);
137 lightInfoWrite.descriptorCount = 1;
138 lightInfoWrite.dstBinding = 1;
139 lightInfoWrite.dstArrayElement = 0;
140 lightInfoWrite.pBufferInfo = &lightInfoBufferInfo;
141 lightInfoWrite.descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
142 }
143
144 setWriter.performWrite(renderer.vulkanState().getDevice());
145
146 cameraBuffer->getBuffer().transferEveryFrame();
147 cameraBuffer->getBuffer().setCopyFullRange(true);
148 lightBuffer->getBuffer().transferEveryFrame();
149 lightBuffer->getBuffer().setCopyFullRange(true);
150
151 updateImageDescriptors();
152
153 subscribe(renderer.getSignalChannel());

163void Scene3DInstance::updateImageDescriptors(unsigned int frameIndex) {
164 const std::uint32_t imageWriteCount =
165 (cfg::Limits::MaxSpotShadows + cfg::Limits::MaxPointShadows + 1); // +1 for ssao buffer
166 ds::SetWriteHelper setWriter;
167 setWriter.hintWriteCount(imageWriteCount);
168 setWriter.hintImageInfoCount(imageWriteCount);
169
170 const auto set = descriptorSets.getRaw(frameIndex);
171
172 VkSampler sampler = renderer.samplerCache().shadowMap();
173 for (unsigned int k = 0; k < cfg::Limits::MaxSpotShadows; ++k) {
174 VkDescriptorImageInfo& imageInfo = setWriter.getNewImageInfo();
175 imageInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
176 imageInfo.imageView =
177 shadowMaps ? shadowMaps->getSpotShadowImage(k).getView() : emptySpotShadowMap.getView();
178 imageInfo.sampler = sampler;
179
180 VkWriteDescriptorSet& write = setWriter.getNewSetWrite(set);
181 write.descriptorCount = 1;
182 write.dstBinding = 2;
183 write.dstArrayElement = k;
184 write.pImageInfo = &imageInfo;
185 write.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
186 }
187
188 for (unsigned int k = 0; k < cfg::Limits::MaxPointShadows; ++k) {
189 VkDescriptorImageInfo& imageInfo = setWriter.getNewImageInfo();
190 imageInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
191 imageInfo.imageView = shadowMaps ? shadowMaps->getPointShadowImage(k).getView() :
192 emptyPointShadowMap.getView();
193 imageInfo.sampler = sampler;
194
195 VkWriteDescriptorSet& write = setWriter.getNewSetWrite(set);
196 write.descriptorCount = 1;
197 write.dstBinding = 3;
198 write.dstArrayElement = k;
199 write.pImageInfo = &imageInfo;
200 write.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
201 }
202
203 // ssao buffer
204 const bool useSSAO = ssaoBuffer && renderer.getSettings().getSSAO() != Settings::SSAO::None;
205 VkDescriptorImageInfo& ssaoInfo = setWriter.getNewImageInfo();
206 ssaoInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
207 ssaoInfo.imageView = useSSAO ? ssaoBuffer->getImages().attachmentSet().getImageView(0) :
208 emptySSAOImage.getView();
209 ssaoInfo.sampler = renderer.samplerCache().noFilterEdgeClamped();
210
211 VkWriteDescriptorSet& ssaoWrite = setWriter.getNewSetWrite(set);
212 ssaoWrite.descriptorCount = 1;
213 ssaoWrite.dstBinding = 4;
214 ssaoWrite.dstArrayElement = 0;
215 ssaoWrite.pImageInfo = &ssaoInfo;
216 ssaoWrite.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
217
218 setWriter.performWrite(renderer.vulkanState().getDevice());
219}

236void Scene3DInstance::updateDescriptors() {
237 auto& cameras = shadowMapCameras->getBuffer();
238 auto& data = lightBuffer->getBuffer()[0];
239
240 cameras[0].viewProj[0] = data.sun.viewProjectionMatrix;
241
242 for (unsigned int i = 0; i < data.nSpotShadows; ++i) {
243 cameras[i + 1].viewProj[0] = data.spotlightsWithShadows[i].getViewProjectionMatrix();
244 }
245
246 for (unsigned int i = 0; i < data.nPointShadows; ++i) {
247 auto& light = data.pointLightsWithShadows[i];
248 auto& cam = cameras[i + cfg::Limits::MaxSpotShadows];
249 cam.posAndFarPlane = glm::vec4(light.pos, light.planes.farPlane);
250 for (unsigned int j = 0; j < 6; ++j) { cam.viewProj[j] = light.viewProjectionMatrices[j]; }
251 }
252
253 if (deferredImageUpdates != 0) {
254 deferredImageUpdates = deferredImageUpdates << 1;
255 updateImageDescriptors(renderer.vulkanState().currentFrameIndex());
256 }
257}
