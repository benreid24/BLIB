54bool DependencySingleBase::matchAndLoad(const std::vector<RepoDependency>& dependencies) {
55 const RepoDependency* match = nullptr;
56 for (const auto& dep : dependencies) {
57 if (dep.tag == tag) {
58 if (!match) { match = &dep; }
59 else {
60 BL_LOG_ERROR << "Multiple dependencies with tag '" << tag << "' found for asset "
61 << owner.getAsset().getUUID().toString();
62 }
63 }
64 }
65 if (!match) {
66 BL_LOG_ERROR << "Failed to find dependency with tag '" << tag << "' for asset "
67 << owner.getAsset().getUUID().toString();
68 return depPolicy != DependencyPolicy::Required;
69 }
70 uuid = match->uuid;
71 if (!load()) {
72 BL_LOG_ERROR << "Failed to load dependency with UUID " << uuid.toString() << " for asset "
73 << owner.getAsset().getUUID().toString();
74 return depPolicy != DependencyPolicy::Required;
75 }
