166void Animation2DSystem::doSlideshowAdd(com::Animation2DPlayer& player) {
167 // determine frame offset and player index
168 const auto it = slideshowFrameMap.find(player.animation.get());
169 const bool uploadFrames = it == slideshowFrameMap.end();
170 const std::uint32_t offset =
171 uploadFrames ?
172 slideshowFrameRangeAllocator.alloc(player.animation->frameCount()).range.start :
173 it->second;
174 const std::uint32_t index = slideshowPlayerIds.allocate();
175 player.playerIndex = index;
176
177 // add frame offset and current frame to SSBO's
178 slideshowFrameOffsetSSBO.ensureSize(index + 1);
179 if (slideshowPlayerCurrentFrameSSBO.ensureSize(index + 1)) {
180 players->forEach([this](ecs::Entity, com::Animation2DPlayer& p) {
181 if (p.framePayload.valid()) {
182 slideshowPlayerCurrentFrameSSBO.assignRef(p.framePayload, p.playerIndex);
183 }
184 });
185 }
186 else { slideshowPlayerCurrentFrameSSBO.assignRef(player.framePayload, player.playerIndex); }
187 slideshowFrameOffsetSSBO[index] = offset;
188 slideshowPlayerCurrentFrameSSBO[index] = player.currentFrame;
189 slideshowFrameOffsetSSBO.markDirty(index, 1);
190
191 // fetch texture to convert texCoords and set texture id
192 rc::res::TextureRef texture = renderer.texturePool().getOrLoadTexture(
193 player.animation->resolvedSpritesheet(),
194 {.format = rc::vk::CommonTextureFormats::SRGBA32Bit,
195 .sampler = rc::vk::SamplerOptions::Type::NoFilterEdgeClamped}); // TODO - parameterize?
196 player.texture = texture;
197 slideshowTextureSSBO.ensureSize(index + 1);
198 slideshowTextureSSBO[index] = texture.id();
199 slideshowTextureSSBO.markDirty(index, 1);
200
201 // add frames to SSBO if required
202 if (uploadFrames) {
203 slideshowFrameMap[player.animation.get()] = offset;
204 slideshowFramesSSBO.ensureSize(offset + player.animation->frameCount());
205 for (std::size_t i = 0; i < player.animation->frameCount(); ++i) {
206 const auto& src = player.animation->getFrame(i).shards.front();
207 auto& frame = slideshowFramesSSBO[offset + i];
208 const sf::FloatRect& tex = src.normalizedSource;
209 frame.opacity = static_cast<float>(src.alpha) / 255.f;
210 frame.texCoords[0] = texture->convertCoord({tex.position.x, tex.position.y});
211 frame.texCoords[1] =
212 texture->convertCoord({tex.position.x + tex.size.x, tex.position.y});
213 frame.texCoords[2] =
214 texture->convertCoord({tex.position.x + tex.size.x, tex.position.y + tex.size.y});
215 frame.texCoords[3] =
216 texture->convertCoord({tex.position.x, tex.position.y + tex.size.y});
217 }
218 slideshowFramesSSBO.markDirty(offset, player.animation->frameCount());
219 }
220 slideshowDataRefCounts[player.animation.get()] += 1;
221
222 // mark that descriptors need to be reset
223 slideshowRefreshRequired = rc::cfg::Limits::MaxConcurrentFrames;
224}

226void Animation2DSystem::doSlideshowFree(const com::Animation2DPlayer& player) {
227 slideshowPlayerIds.release(player.playerIndex);
228
229 const auto refCountIt = slideshowDataRefCounts.find(player.animation.get());
230 if (refCountIt == slideshowDataRefCounts.end()) {
231 throw std::runtime_error("Player component removed without having been added");
232 }
233 refCountIt->second -= 1;
234
235 if (refCountIt->second == 0) {
236 slideshowDataRefCounts.erase(refCountIt);
237 const auto offsetIt = slideshowFrameMap.find(player.animation.get());
238 if (offsetIt == slideshowFrameMap.end()) {
239 throw std::runtime_error("Failed to find cached frame offset for animation");
240 }
241
242 slideshowFrameRangeAllocator.release(
243 {offsetIt->second, static_cast<std::uint32_t>(player.animation->frameCount())});
244 slideshowFrameMap.erase(offsetIt);
245 }

353Animation2DSystem::VertexAnimation::VertexAnimation(rc::Renderer& renderer,
354 const gfx::a2d::AnimationData& anim)
355: useCount(0) {
356 unsigned int shardCount = 0;
357 for (unsigned int i = 0; i < anim.frameCount(); ++i) {
358 shardCount += anim.getFrame(i).shards.size();
359 }
360 if (anim.frameCount() > 0) {
361 frameToIndices.resize(anim.frameCount(), {});
362
363 // allocate buffer
364 indexBuffer.create(renderer, shardCount * 4, shardCount * 6);
365
366 // fetch texture to convert texCoords
367 rc::res::TextureRef texture =
368 renderer.texturePool().getOrLoadTexture(anim.resolvedSpritesheet());
369
370 // populate buffer
371 std::uint32_t vi = 0;
372 std::uint32_t ii = 0;
373 for (unsigned int i = 0; i < anim.frameCount(); ++i) {
374 const auto& frame = anim.getFrame(i);
375
376 // cache draw params
377 frameToIndices[i].vertexStart = vi;
378 frameToIndices[i].indexStart = ii;
379 frameToIndices[i].indexCount = frame.shards.size() * 6;
380
381 for (unsigned int j = 0; j < frame.shards.size(); ++j) {
382 const auto& shard = frame.shards[j];
383
384 // compute shard transform
385 const glm::vec2 size(static_cast<float>(shard.source.size.x),
386 static_cast<float>(shard.source.size.y));
387 com::Transform2D transform;
388 transform.setOrigin(anim.shardsAreCentered() ? size * 0.5f : glm::vec2{0.f, 0.f});
389 transform.setRotation(shard.rotation);
390 transform.setPosition({shard.offset.x, shard.offset.y});
391 transform.setScale({shard.scale.x, shard.scale.y});
392 const glm::mat4 tmat = transform.getLocalTransform();
393
394 // set draw indices (CW winding)
395 const auto ibase = ii + j * 6;
396 const auto vbase = vi + j * 4;
397 indexBuffer.indices()[ibase + 0] = vbase + 0;
398 indexBuffer.indices()[ibase + 1] = vbase + 1;
399 indexBuffer.indices()[ibase + 2] = vbase + 2;
400 indexBuffer.indices()[ibase + 3] = vbase + 0;
401 indexBuffer.indices()[ibase + 4] = vbase + 2;
402 indexBuffer.indices()[ibase + 5] = vbase + 3;
403
404 // set draw vertices
405 // color
406 for (unsigned int k = 0; k < 4; ++k) {
407 indexBuffer.vertices()[vbase + k].color =
408 glm::vec4{1.f, 1.f, 1.f, static_cast<float>(shard.alpha) / 255.f};
409 }
410
411 // pos
412 glm::vec4 src{0.f, 0.f, 0.f, 1.f};
413 indexBuffer.vertices()[vbase + 0].pos = tmat * src;
414 src.x = static_cast<float>(shard.source.size.x);
415 indexBuffer.vertices()[vbase + 1].pos = tmat * src;
416 src.y = static_cast<float>(shard.source.size.y);
417 indexBuffer.vertices()[vbase + 2].pos = tmat * src;
418 src.x = 0.f;
419 indexBuffer.vertices()[vbase + 3].pos = tmat * src;
420 if (anim.shardsAreCentered()) {
421 for (unsigned int k = 0; k < 4; ++k) {
422 indexBuffer.vertices()[vbase + k].pos -=
423 glm::vec3{size * glm::vec2{shard.scale.x, shard.scale.y} * 0.5f, 0.f};
424 }
425 }
426
427 // texture coord
428 indexBuffer.vertices()[vbase + 0].texCoord = {shard.normalizedSource.position.x,
429 shard.normalizedSource.position.y};
430 indexBuffer.vertices()[vbase + 1].texCoord = {shard.normalizedSource.position.x +
431 shard.normalizedSource.size.x,
432 shard.normalizedSource.position.y};
433 indexBuffer.vertices()[vbase + 2].texCoord = {
434 shard.normalizedSource.position.x + shard.normalizedSource.size.x,
435 shard.normalizedSource.position.y + shard.normalizedSource.size.y};
436 indexBuffer.vertices()[vbase + 3].texCoord = {shard.normalizedSource.position.x,
437 shard.normalizedSource.position.y +
438 shard.normalizedSource.size.y};
439 for (unsigned int i = 0; i < 4; ++i) {
440 indexBuffer.vertices()[vbase + i].texCoord =
441 texture->convertCoord(indexBuffer.vertices()[vbase + i].texCoord);
442 }
443 }
444
445 vi += frame.shards.size() * 4;
446 ii += frameToIndices[i].indexCount;
447 }
448 indexBuffer.queueTransfer();
449 }
