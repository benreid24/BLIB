92void TextEntry::recalcNewlines() {
93 newlines.clear();
94 currentLine = 0;
95 if (cursorPos > input.size()) cursorPos = input.size();
96
97 newlines.push_back(-1);
98 for (unsigned int i = 0; i < input.size(); ++i) {
99 if (i == cursorPos) currentLine = newlines.size() - 1;
100 if (input[i] == '\n') newlines.push_back(i);
101 }
102 if (cursorPos == input.size()) currentLine = newlines.size() - 1;
103 newlines.push_back(input.size());
104}

106void TextEntry::onInput(const Event& action) {
107 if (action.type() != Event::TextEntered) return;
108
109 const uint32_t c = action.character();
110 if (c == 8) { // backspace
111 if (cursorPos > 0) {
112 input.erase(cursorPos - 1, 1);
113 --cursorPos;
114 }
115 }
116 else if (c >= ' ') {
117 if (!maxInputLen.has_value() || input.size() < maxInputLen.value()) {
118 input.insert(cursorPos, 1, c);
119 ++cursorPos;
120 }
121 }
122
123 filter();
124 recalcNewlines();
125 refreshComponent();
126 fireChanged();
127}

137void TextEntry::onKeypress(const Event& action) {
138 if (action.type() != Event::KeyPressed) return;
139
140 if (action.key().code == sf::Keyboard::Key::Right) {
141 if (cursorPos < input.size()) ++cursorPos;
142 showAndResetCursor();
143 }
144 else if (action.key().code == sf::Keyboard::Key::Left) {
145 if (cursorPos > 0) --cursorPos;
146 showAndResetCursor();
147 }
148 else if (action.key().code == sf::Keyboard::Key::Up) {
149 if (currentLine > 0) cursorUp();
150 showAndResetCursor();
151 }
152 else if (action.key().code == sf::Keyboard::Key::Down) {
153 if (currentLine < newlines.size() - 2) cursorDown();
154 showAndResetCursor();
155 }
156 else if (action.key().code == sf::Keyboard::Key::Home) {
157 cursorPos = newlines[currentLine] + 1;
158 showAndResetCursor();
159 }
160 else if (action.key().code == sf::Keyboard::Key::End) {
161 cursorPos = newlines[currentLine + 1];
162 showAndResetCursor();
163 }
164 else if (action.key().code == sf::Keyboard::Key::Delete) {
165 if (cursorPos < input.size()) input.erase(cursorPos, 1);
166 filter();
167 fireChanged();
168 refreshComponent();
169 }
170 else if (action.key().code == sf::Keyboard::Key::Enter) {
171 if (newlines.size() - 1 < lineCount || allowMoreLines) {
172 if (cursorPos < input.size())
173 input.insert(cursorPos, 1, '\n');
174 else
175 input.push_back('\n');
176 ++cursorPos;
177 }
178 filter();
179 fireChanged();
180 refreshComponent();
181 }
182 else if (action.key().control && action.key().code == sf::Keyboard::Key::V) {
183 const std::string c = sf::Clipboard::getString().toAnsiString();
184 if (cursorPos < input.size())
185 input.insert(cursorPos, c);
186 else
187 input += c;
188 cursorPos += c.size();
189 filter();
190 fireChanged();
191 refreshComponent();
192 }
193
194 recalcNewlines();
195}

227void TextEntry::filter() {
228 if (mode == Mode::Any) return;
229
230 if (modeEnabled(mode, Mode::Integer) || modeEnabled(mode, Mode::Float)) {
231 bool decimalFound = false;
232 for (int i = 0; i < static_cast<int>(input.size()); ++i) {
233 if (!std::isdigit(input[i])) {
234 if (i == 0 && input[i] == '-' && modeEnabled(mode, Mode::Signed)) continue;
235 if (input[i] == '.' && modeEnabled(mode, Mode::Float) && !decimalFound) {
236 decimalFound = true;
237 continue;
238 }
239 input.erase(i, 1);
240 --i;
241 --cursorPos;
242 }
243 }
244 if (input.empty()) { input = "0"; }
245 if (cursorPos > input.size()) cursorPos = input.size();
246 }
247}
