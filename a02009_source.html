48// FreeType callbacks that operate on a sf::InputStream
49unsigned long read(FT_Stream rec, unsigned long offset, unsigned char* buffer,
50 unsigned long count) {
51 sf::Int64 convertedOffset = static_cast<sf::Int64>(offset);
52 sf::InputStream* stream = static_cast<sf::InputStream*>(rec->descriptor.pointer);
53 if (stream->seek(convertedOffset) == convertedOffset) {
54 if (count > 0)
55 return static_cast<unsigned long>(
56 stream->read(reinterpret_cast<char*>(buffer), static_cast<sf::Int64>(count)));
57 else
58 return 0;
59 }
60 else
61 return count > 0 ? 0 : 1; // error code is 0 if we're reading, or nonzero if we're seeking
62}
63void close(FT_Stream) {}

65// Helper to interpret memory as a specific type
66template<typename T, typename U>
67inline T reinterpret(const U& input) {
68 T output;
69 std::memcpy(&output, &input, sizeof(U));
70 return output;
71}

73// Combine outline thickness, boldness and font glyph index into a single 64-bit key
74sf::Uint64 combine(sf::Uint8 size, float outlineThickness, bool bold, sf::Uint32 index) {
75 return (static_cast<sf::Uint64>(reinterpret<sf::Uint32>(outlineThickness)) << 32) |
76 (static_cast<sf::Uint64>(bold) << 31) | (static_cast<sf::Uint64>(size) << 32) | index;
77}

142bool VulkanFont::loadFromFile(const std::string& filename) {
143 needsUpload = true;
144
145#ifndef SFML_SYSTEM_ANDROID
146
147 // Cleanup the previous resources
148 cleanup();
149 m_refCount = new int(1);
150
151 // Initialize FreeType
152 // Note: we initialize FreeType for every font instance in order to avoid having a single
153 // global manager that would create a lot of issues regarding creation and destruction order.
154 FT_Library library;
155 if (FT_Init_FreeType(&library) != 0) {
156 err() << "Failed to load font \"" << filename << "\" (failed to initialize FreeType)"
157 << std::endl;
158 return false;
159 }
160 m_library = library;
161
162 // Load the new font face from the specified file
163 FT_Face face;
164 if (FT_New_Face(static_cast<FT_Library>(m_library), filename.c_str(), 0, &face) != 0) {
165 err() << "Failed to load font \"" << filename << "\" (failed to create the font face)"
166 << std::endl;
167 return false;
168 }
169
170 // Load the stroker that will be used to outline the font
171 FT_Stroker stroker;
172 if (FT_Stroker_New(static_cast<FT_Library>(m_library), &stroker) != 0) {
173 err() << "Failed to load font \"" << filename << "\" (failed to create the stroker)"
174 << std::endl;
175 FT_Done_Face(face);
176 return false;
177 }
178
179 // Select the unicode character map
180 if (FT_Select_Charmap(face, FT_ENCODING_UNICODE) != 0) {
181 err() << "Failed to load font \"" << filename
182 << "\" (failed to set the Unicode character set)" << std::endl;
183 FT_Stroker_Done(stroker);
184 FT_Done_Face(face);
185 return false;
186 }
187
188 // Store the loaded font in our ugly void* :)
189 m_stroker = stroker;
190 m_face = face;
191
192 // Store the font information
193 m_info.family = face->family_name ? face->family_name : std::string();
194
195 return true;
196
197#else
198
199 if (m_stream) delete static_cast<priv::ResourceStream*>(m_stream);
200
201 m_stream = new priv::ResourceStream(filename);
202 return loadFromStream(*static_cast<priv::ResourceStream*>(m_stream));
203
204#endif
205}

268bool VulkanFont::loadFromStream(InputStream& stream) {
269 needsUpload = true;
270
271 // Cleanup the previous resources
272 cleanup();
273 m_refCount = new int(1);
274
275 // Initialize FreeType
276 // Note: we initialize FreeType for every font instance in order to avoid having a single
277 // global manager that would create a lot of issues regarding creation and destruction order.
278 FT_Library library;
279 if (FT_Init_FreeType(&library) != 0) {
280 err() << "Failed to load font from stream (failed to initialize FreeType)" << std::endl;
281 return false;
282 }
283 m_library = library;
284
285 // Make sure that the stream's reading position is at the beginning
286 stream.seek(0);
287
288 // Prepare a wrapper for our stream, that we'll pass to FreeType callbacks
289 FT_StreamRec* rec = new FT_StreamRec;
290 std::memset(rec, 0, sizeof(*rec));
291 rec->base = NULL;
292 rec->size = static_cast<unsigned long>(stream.getSize());
293 rec->pos = 0;
294 rec->descriptor.pointer = &stream;
295 rec->read = &read;
296 rec->close = &close;
297
298 // Setup the FreeType callbacks that will read our stream
299 FT_Open_Args args;
300 args.flags = FT_OPEN_STREAM;
301 args.stream = rec;
302 args.driver = 0;
303
304 // Load the new font face from the specified stream
305 FT_Face face;
306 if (FT_Open_Face(static_cast<FT_Library>(m_library), &args, 0, &face) != 0) {
307 err() << "Failed to load font from stream (failed to create the font face)" << std::endl;
308 delete rec;
309 return false;
310 }
311
312 // Load the stroker that will be used to outline the font
313 FT_Stroker stroker;
314 if (FT_Stroker_New(static_cast<FT_Library>(m_library), &stroker) != 0) {
315 err() << "Failed to load font from stream (failed to create the stroker)" << std::endl;
316 FT_Done_Face(face);
317 delete rec;
318 return false;
319 }
320
321 // Select the Unicode character map
322 if (FT_Select_Charmap(face, FT_ENCODING_UNICODE) != 0) {
323 err() << "Failed to load font from stream (failed to set the Unicode character set)"
324 << std::endl;
325 FT_Done_Face(face);
326 FT_Stroker_Done(stroker);
327 delete rec;
328 return false;
329 }
330
331 // Store the loaded font in our ugly void* :)
332 m_stroker = stroker;
333 m_face = face;
334 m_streamRec = rec;
335
336 // Store the font information
337 m_info.family = face->family_name ? face->family_name : std::string();
338
339 return true;
340}

346const Glyph& VulkanFont::getGlyph(Uint32 codePoint, unsigned int characterSize, bool bold,
347 float outlineThickness) const {
348 // Build the key by combining the glyph index (based on code point), bold flag, and outline
349 // thickness
350 Uint64 key = combine(characterSize,
351 outlineThickness,
352 bold,
353 FT_Get_Char_Index(static_cast<FT_Face>(m_face), codePoint));
354
355 // Search the glyph into the cache
356 GlyphTable::const_iterator it = glyphs.find(key);
357 if (it != glyphs.end()) {
358 // Found: just return it
359 return it->second;
360 }
361 else {
362 // Not found: we have to load it
363 const_cast<sf::VulkanFont*>(this)->needsUpload = true;
364 Glyph glyph = loadGlyph(codePoint, characterSize, bold, outlineThickness);
365 return glyphs.insert(std::make_pair(key, glyph)).first->second;
366 }
367}

375float VulkanFont::getKerning(Uint32 first, Uint32 second, unsigned int characterSize,
376 bool bold) const {
377 // Special case where first or second is 0 (null character)
378 if (first == 0 || second == 0) return 0.f;
379
380 FT_Face face = static_cast<FT_Face>(m_face);
381
382 if (face && setCurrentSize(characterSize)) {
383 // Convert the characters to indices
384 FT_UInt index1 = FT_Get_Char_Index(face, first);
385 FT_UInt index2 = FT_Get_Char_Index(face, second);
386
387 // Retrieve position compensation deltas generated by FT_LOAD_FORCE_AUTOHINT flag
388 float firstRsbDelta = static_cast<float>(getGlyph(first, characterSize, bold).rsbDelta);
389 float secondLsbDelta = static_cast<float>(getGlyph(second, characterSize, bold).lsbDelta);
390
391 // Get the kerning vector if present
392 FT_Vector kerning;
393 kerning.x = kerning.y = 0;
394 if (FT_HAS_KERNING(face))
395 FT_Get_Kerning(face, index1, index2, FT_KERNING_UNFITTED, &kerning);
396
397 // X advance is already in pixels for bitmap fonts
398 if (!FT_IS_SCALABLE(face)) return static_cast<float>(kerning.x);
399
400 // Combine kerning with compensation deltas and return the X advance
401 // Flooring is required as we use FT_KERNING_UNFITTED flag which is not quantized in 64
402 // based grid
403 return std::floor((secondLsbDelta - firstRsbDelta + static_cast<float>(kerning.x) + 32) /
404 static_cast<float>(1 << 6));
405 }
406 else {
407 // Invalid font
408 return 0.f;
409 }
410}

423float VulkanFont::getUnderlinePosition(unsigned int characterSize) const {
424 FT_Face face = static_cast<FT_Face>(m_face);
425
426 if (face && setCurrentSize(characterSize)) {
427 // Return a fixed position if font is a bitmap font

480void VulkanFont::cleanup() {
481 // Check if we must destroy the FreeType pointers
482 if (m_refCount) {
483 // Decrease the reference counter
484 (*m_refCount)--;
485
486 // Free the resources only if we are the last owner
487 if (*m_refCount == 0) {
488 // Delete the reference counter
489 delete m_refCount;
490
491 // Destroy the stroker
492 if (m_stroker) FT_Stroker_Done(static_cast<FT_Stroker>(m_stroker));
493
494 // Destroy the font face
495 if (m_face) FT_Done_Face(static_cast<FT_Face>(m_face));
496
497 // Destroy the stream rec instance, if any (must be done after FT_Done_Face!)
498 if (m_streamRec) delete static_cast<FT_StreamRec*>(m_streamRec);
499
500 // Close the library
501 if (m_library) FT_Done_FreeType(static_cast<FT_Library>(m_library));
502 }
503 }
504
505 // Reset members
506 m_library = NULL;
507 m_face = NULL;
508 m_stroker = NULL;
509 m_streamRec = NULL;
510 m_refCount = NULL;
511 glyphs.clear();
512 rows.clear();
513 nextRow = 3;
514 needsUpload = true;
515 vulkanTexture.release();
516 std::vector<Uint8>().swap(m_pixelBuffer);
517 buffer.clear();
518}

521Glyph VulkanFont::loadGlyph(Uint32 codePoint, unsigned int characterSize, bool bold,
522 float outlineThickness) const {
523 // The glyph to return
524 Glyph glyph;
525
526 // First, transform our ugly void* to a FT_Face
527 FT_Face face = static_cast<FT_Face>(m_face);
528 if (!face) return glyph;
529
530 // Set the character size
531 if (!setCurrentSize(characterSize)) return glyph;
532
533 // Load the glyph corresponding to the code point
534 FT_Int32 flags = FT_LOAD_TARGET_NORMAL | FT_LOAD_FORCE_AUTOHINT;
535 if (outlineThickness != 0) flags |= FT_LOAD_NO_BITMAP;
536 if (FT_Load_Char(face, codePoint, flags) != 0) return glyph;
537
538 // Retrieve the glyph
539 FT_Glyph glyphDesc;
540 if (FT_Get_Glyph(face->glyph, &glyphDesc) != 0) return glyph;
541
542 // Apply bold and outline (there is no fallback for outline) if necessary -- first technique
543 // using outline (highest quality)
544 FT_Pos weight = 1 << 6;
545 bool outline = (glyphDesc->format == FT_GLYPH_FORMAT_OUTLINE);
546 if (outline) {
547 if (bold) {
548 FT_OutlineGlyph outlineGlyph = reinterpret_cast<FT_OutlineGlyph>(glyphDesc);
549 FT_Outline_Embolden(&outlineGlyph->outline, weight);
550 }
551
552 if (outlineThickness != 0) {
553 FT_Stroker stroker = static_cast<FT_Stroker>(m_stroker);
554
555 FT_Stroker_Set(stroker,
556 static_cast<FT_Fixed>(outlineThickness * static_cast<float>(1 << 6)),
557 FT_STROKER_LINECAP_ROUND,
558 FT_STROKER_LINEJOIN_ROUND,
559 0);
560 FT_Glyph_Stroke(&glyphDesc, stroker, true);
561 }
562 }
563
564 // Convert the glyph to a bitmap (i.e. rasterize it)
565 // Warning! After this line, do not read any data from glyphDesc directly, use
566 // bitmapGlyph.root to access the FT_Glyph data.
567 FT_Glyph_To_Bitmap(&glyphDesc, FT_RENDER_MODE_NORMAL, 0, 1);
568 FT_BitmapGlyph bitmapGlyph = reinterpret_cast<FT_BitmapGlyph>(glyphDesc);
569 FT_Bitmap& bitmap = bitmapGlyph->bitmap;
570
571 // Apply bold if necessary -- fallback technique using bitmap (lower quality)
572 if (!outline) {
573 if (bold) FT_Bitmap_Embolden(static_cast<FT_Library>(m_library), &bitmap, weight, weight);
574
575 if (outlineThickness != 0)
576 err() << "Failed to outline glyph (no fallback available)" << std::endl;
577 }
578
579 // Compute the glyph's advance offset
580 glyph.advance = static_cast<float>(bitmapGlyph->root.advance.x >> 16);
581 if (bold) glyph.advance += static_cast<float>(weight) / static_cast<float>(1 << 6);
582
583 glyph.lsbDelta = static_cast<int>(face->glyph->lsb_delta);
584 glyph.rsbDelta = static_cast<int>(face->glyph->rsb_delta);
585
586 unsigned int width = bitmap.width;
587 unsigned int height = bitmap.rows;
588
589 if ((width > 0) && (height > 0)) {
590 // Leave a small padding around characters, so that filtering doesn't
591 // pollute them with pixels from neighbors
592 const unsigned int padding = 2;
593
594 width += 2 * padding;
595 height += 2 * padding;
596
597 // Find a good position for the new glyph into the texture
598 glyph.textureRect = findGlyphRect(width, height);
599
600 // Make sure the texture data is positioned in the center
601 // of the allocated texture rectangle
602 glyph.textureRect.left += static_cast<int>(padding);
603 glyph.textureRect.top += static_cast<int>(padding);
604 glyph.textureRect.width -= static_cast<int>(2 * padding);
605 glyph.textureRect.height -= static_cast<int>(2 * padding);
606
607 // Compute the glyph's bounding box
608 glyph.bounds.left = static_cast<float>(bitmapGlyph->left);
609 glyph.bounds.top = static_cast<float>(-bitmapGlyph->top);
610 glyph.bounds.width = static_cast<float>(bitmap.width);
611 glyph.bounds.height = static_cast<float>(bitmap.rows);
612
613 // Resize the pixel buffer to the new size and fill it with transparent white pixels
614 m_pixelBuffer.resize(width * height * 4);
615
616 Uint8* current = &m_pixelBuffer[0];
617 Uint8* end = current + width * height * 4;
618
619 while (current != end) {
620 (*current++) = 255;
621 (*current++) = 255;
622 (*current++) = 255;
623 (*current++) = 0;
624 }
625
626 // Extract the glyph's pixels from the bitmap
627 const Uint8* pixels = bitmap.buffer;
628 if (bitmap.pixel_mode == FT_PIXEL_MODE_MONO) {
629 // Pixels are 1 bit monochrome values
630 for (unsigned int y = padding; y < height - padding; ++y) {
631 for (unsigned int x = padding; x < width - padding; ++x) {
632 // The color channels remain white, just fill the alpha channel
633 std::size_t index = x + y * width;
634 m_pixelBuffer[index * 4 + 3] =
635 ((pixels[(x - padding) / 8]) & (1 << (7 - ((x - padding) % 8)))) ? 255 : 0;
636 }
637 pixels += bitmap.pitch;
638 }
639 }
640 else {
641 // Pixels are 8 bits gray levels
642 for (unsigned int y = padding; y < height - padding; ++y) {
643 for (unsigned int x = padding; x < width - padding; ++x) {
644 // The color channels remain white, just fill the alpha channel
645 std::size_t index = x + y * width;
646 m_pixelBuffer[index * 4 + 3] = pixels[x - padding];
647 }
648 pixels += bitmap.pitch;
649 }
650 }
651
652 // Write the pixels to the texture
653 unsigned int x = static_cast<unsigned int>(glyph.textureRect.left) - padding;
654 unsigned int y = static_cast<unsigned int>(glyph.textureRect.top) - padding;
655 unsigned int w = static_cast<unsigned int>(glyph.textureRect.width) + 2 * padding;
656 unsigned int h = static_cast<unsigned int>(glyph.textureRect.height) + 2 * padding;
657 for (unsigned int xi = 0; xi < w; ++xi) {
658 for (unsigned int yi = 0; yi < h; ++yi) {
659 std::size_t index = xi + yi * width;
660 texture.setPixel(
661 x + xi, y + yi, sf::Color{255, 255, 255, m_pixelBuffer[index * 4 + 3]});
662 }
663 }
664 }
665
666 // Delete the FT glyph
667 FT_Done_Glyph(glyphDesc);
668
669 // Done :)
670 return glyph;
671}

674IntRect VulkanFont::findGlyphRect(unsigned int width, unsigned int height) const {
675 // Find the line that fits well the glyph
676 Row* row = NULL;
677 float bestRatio = 0;
678 for (std::vector<Row>::iterator it = rows.begin(); it != rows.end() && !row; ++it) {
679 float ratio = static_cast<float>(height) / static_cast<float>(it->height);
680
681 // Ignore rows that are either too small or too high
682 if ((ratio < 0.7f) || (ratio > 1.f)) continue;
683
684 // Check if there's enough horizontal space left in the row
685 if (width > texture.getSize().x - it->width) continue;
686
687 // Make sure that this new row is the best found so far
688 if (ratio < bestRatio) continue;
689
690 // The current row passed all the tests: we can select it
691 row = &*it;
692 bestRatio = ratio;
693 }
694
695 // If we didn't find a matching row, create a new one (10% taller than the glyph)
696 if (!row) {
697 unsigned int rowHeight = height + height / 10;
698 while ((nextRow + rowHeight >= texture.getSize().y) || (width >= texture.getSize().x)) {
699 // Not enough space: resize the texture if possible
700 unsigned int textureWidth = texture.getSize().x;
701 unsigned int textureHeight = texture.getSize().y;
702
703 // Make the texture 2 times bigger
704 Image newTexture;
705 newTexture.create(textureWidth * 2, textureHeight * 2, sf::Color(255, 255, 255, 0));
706 newTexture.copy(texture, 0, 0);
707 texture = std::move(newTexture);
708 }
709
710 // We can now create the new row
711 rows.emplace_back(nextRow, rowHeight);
712 nextRow += rowHeight;
713 row = &rows.back();
714 }
715
716 // Find the glyph's rectangle on the selected row
717 IntRect rect(Rect<unsigned int>(row->width, row->top, width, height));
718
719 // Update the row informations
720 row->width += width;
721
722 return rect;
723}

726bool VulkanFont::setCurrentSize(unsigned int characterSize) const {
727 // FT_Set_Pixel_Sizes is an expensive function, so we must call it
728 // only when necessary to avoid killing performances
729
730 FT_Face face = static_cast<FT_Face>(m_face);
731 FT_UShort currentSize = face->size->metrics.x_ppem;
732
733 if (currentSize != characterSize) {
734 FT_Error result = FT_Set_Pixel_Sizes(face, 0, characterSize);
735
736 if (result == FT_Err_Invalid_Pixel_Size) {
737 // In the case of bitmap fonts, resizing can
738 // fail if the requested size is not available
739 if (!FT_IS_SCALABLE(face)) {
740 err() << "Failed to set bitmap font size to " << characterSize << std::endl;
741 err() << "Available sizes are: ";
742 for (int i = 0; i < face->num_fixed_sizes; ++i) {
743 const long size = (face->available_sizes[i].y_ppem + 32) >> 6;
744 err() << size << " ";
745 }
746 err() << std::endl;
747 }
748 else { err() << "Failed to set font size to " << characterSize << std::endl; }
749 }
750
751 return result == FT_Err_Ok;
752 }
753
754 return true;
755}
