68bool Registry::destroyEntity(Entity start, const Transaction<tx::EntityRead>& tx) {
69 if (!entityExists(start, tx)) { return false; }

80bool Registry::queueEntityDestroy(Entity start) {
81 // check if we can remove due to dependencies
82 if (dependencyGraph.hasDependencies(start)) {
83 markEntityForRemoval(start);
84 return false;
85 }
86
87 // determine list of entities to remove due to parenting and dependencies
88 std::unique_lock deleteLock(deletionState.mutex);
89 std::vector<Entity>& toRemove = deletionState.toRemove;
90 std::vector<Entity>& toVisit = deletionState.toVisit;
91 std::vector<Entity>& toUnparent = deletionState.toUnparent;
92 toRemove.reserve(16);
93 toVisit.reserve(16);
94 toVisit.emplace_back(start);
95
96 while (!toVisit.empty()) {
97 const Entity visiting = toVisit.back();
98 toVisit.pop_back();
99
100 if (!entityExistsLocked(visiting)) { continue; }
101
102 toRemove.emplace_back(visiting);
103
104 // increment version here
105 ++entityVersions[visiting.getIndex()];
106
107 // send events now
108 const std::uint32_t index = visiting.getIndex();
109 const ComponentMask::SimpleMask mask = entityMasks[index];
110 bl::event::Dispatcher::dispatch<event::EntityDestroyed>({visiting});
111 for (ComponentPoolBase* pool : componentPools) {
112 if (ComponentMask::has(mask, pool->ComponentIndex)) {
113 pool->fireRemoveEventOnly(visiting);
114 }
115 }
116
117 // add marked dependencies
118 for (Entity resource : dependencyGraph.getResources(visiting)) {
119 dependencyGraph.removeDependency(resource, visiting);
120 if (!dependencyGraph.hasDependencies(resource)) {
121 const std::uint32_t i = resource.getIndex();
122 if (i < markedForRemoval.size() && markedForRemoval[i]) {
123 toVisit.emplace_back(resource);
124 }
125 }
126 }
127
128 // add children
129 for (Entity child : parentGraph.getChildren(visiting)) {
130 const std::uint32_t j = child.getIndex();
131 switch (parentDestructionBehaviors[j]) {
132 case ParentDestructionBehavior::DestroyedWithParent:
133 toVisit.emplace_back(child);
134 break;
135 case ParentDestructionBehavior::OrphanedByParent:
136 toUnparent.reserve(16);
137 toUnparent.emplace_back(child);
138 break;
139 }
140 }
141 }
142
143 // unparent orphans
144 std::unique_lock parentLock(entityLock);
145 for (Entity child : deletionState.toUnparent) { removeEntityParentLocked(child, false); }
146 deletionState.toUnparent.clear();
147
148 return true;
149}

151void Registry::doEntityDestroyLocked(Entity ent) {
152 removeEntityParentLocked(ent, true);
153
154 const std::uint32_t index = ent.getIndex();
155 const ComponentMask::SimpleMask mask = entityMasks[index];
156
157 // reset metadata
158 entityAllocator.release(index);
159 entityMasks[index] = ComponentMask::EmptyMask;
160 entityFlags[index] = Flags::None;
161 entityWorlds[index] = 0;
162
163 // remove from views
164 for (auto& view : views) {
165 if (view->mask.passes(mask)) { view->removeEntity(ent); }
166 }
167
168 // destroy components
169 for (ComponentPoolBase* pool : componentPools) {
170 if (ComponentMask::has(mask, pool->ComponentIndex)) { pool->remove(ent, false); }
171 }
172
173 // remove parenting info
174 parentGraph.removeEntity(ent);
175
176 // remove dependency info
177 if (index < markedForRemoval.size()) { markedForRemoval[index] = false; }
178}

180void Registry::flushDeletions() {

182 std::unique_lock queueLock(deletionState.mutex);

267void Registry::setEntityParent(Entity child, Entity parent, const Transaction<tx::EntityRead>&) {

297void Registry::removeEntityParentLocked(Entity child, bool fromDestroy) {
298 const Entity parent = parentGraph.getParent(child);
299 if (parent == InvalidEntity) { return; }
300 const std::uint32_t ic = child.getIndex();
301 const std::uint32_t ip = parent.getIndex();
302 if (!entityAllocator.isAllocated(ic) || !entityAllocator.isAllocated(ip)) {
303 BL_LOG_WARN << "Tried to un-parent bad entities: " << child << " <- " << parent;
304 return;
305 }
306
307 const ComponentMask mask{.required = entityMasks[ic]};
308 parentGraph.unParent(child);
309 for (ComponentPoolBase* pool : componentPools) {
310 if (mask.contains(pool->ComponentIndex)) { pool->onParentRemove(parent, child); }
311 }
312
313 bl::event::Dispatcher::dispatch<event::EntityParentRemoved>({child, fromDestroy});
314}
