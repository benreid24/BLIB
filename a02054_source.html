27scene::SceneObject* BatchedScene::doAdd(ecs::Entity entity, rcom::DrawableBase& obj,
28 UpdateSpeed updateFreq) {
29 const auto alloc = objects.allocate(updateFreq, entity);
30 const std::uint32_t sceneId = alloc.newObject->sceneKey.sceneId;
31 auto& cache = updateFreq == UpdateSpeed::Static ? staticCache : dynamicCache;
32
33 if (alloc.addressesChanged) {
34 cache.ensureSize(alloc.newCapacity);
35 handleAddressChange(updateFreq, alloc.originalBaseAddress);
36 }
37
38 cache.transparency[sceneId] = obj.getContainsTransparency();
39 cache.specializations[sceneId] = obj.getPipelineSpecialization();
40 auto& batch = obj.getContainsTransparency() ? transparentObjects : opaqueObjects;
41 mat::MaterialPipeline* pipeline = obj.getCurrentPipeline();
42
43 PipelineBatch& pipelineBatch = batch.getOrCreateBatch(this, *pipeline);
44 if (!pipelineBatch.addObject(entity, alloc.newObject, obj.getPipelineSpecialization())) {
45 return nullptr;
46 }
47 pipelineBatch.initObserversMaybe(targetTable);
48
49 // add scene link
50 engine.ecs().emplaceComponent<com::BatchSceneLink>(entity, alloc.newObject->sceneKey);
51
52 return alloc.newObject;
53}

88void BatchedScene::doBatchChange(const BatchChange& change, mat::MaterialPipeline* oldPipeline) {
89 const ecs::Entity entity = change.changed->entity;
90 const bool isStatic = change.changed->sceneKey.updateFreq == UpdateSpeed::Static;
91 auto& cache = isStatic ? staticCache : dynamicCache;
92 const bool wasTrans = cache.transparency[change.changed->sceneKey.sceneId];
93 cache.transparency[change.changed->sceneKey.sceneId] = change.newTrans;
94 const std::uint32_t priorSpec = cache.specializations[change.changed->sceneKey.sceneId];
95 cache.specializations[change.changed->sceneKey.sceneId] = change.newSpecialization;
96
97 auto& oldObjectBatch = wasTrans ? transparentObjects : opaqueObjects;
98 auto& newObjectBatch = change.newTrans ? transparentObjects : opaqueObjects;
99
100 PipelineBatch* oldPipelineBatch = oldObjectBatch.getBatch(oldPipeline);
101 if (!oldPipelineBatch) {
102 BL_LOG_ERROR << "Failed to re-bucket entity " << entity
103 << " - Old pipeline batch not found";
104 return;
105 }
106
107 // determine what changed
108 const bool transChanged = wasTrans != change.newTrans;
109 const bool pipelineChanged = oldPipeline != change.newPipeline;
110 const bool specializationChanged = priorSpec != change.newSpecialization;
111
112 if (pipelineChanged) {
113 PipelineBatch& newPipelineBatch =
114 newObjectBatch.getOrCreateBatch(this, *change.newPipeline);
115 oldPipelineBatch->removeForRebatch(change.changed, priorSpec);
116 newPipelineBatch.addForRebatch(change.changed, change.newSpecialization);
117 newPipelineBatch.updateDescriptors(entity, change.changed, *oldPipelineBatch);
118 newPipelineBatch.initObserversMaybe(targetTable);
119 }
120 else if (transChanged) {
121 PipelineBatch& newPipelineBatch = newObjectBatch.getBatch(*oldPipelineBatch);
122 oldPipelineBatch->removeForRebatch(change.changed, priorSpec);
123 newPipelineBatch.addForRebatch(change.changed, change.newSpecialization);
124 }
125 else if (specializationChanged) {
126 PipelineBatch& newPipelineBatch =
127 newObjectBatch.getOrCreateBatch(this, *change.newPipeline);
128 newPipelineBatch.removeForRebatch(change.changed, priorSpec);
129 newPipelineBatch.addForRebatch(change.changed, change.newSpecialization);
130 newPipelineBatch.initObserversMaybe(targetTable);
131 }

134void BatchedScene::renderBatch(scene::SceneRenderContext& ctx, ObjectBatch& batch) {
135 for (auto& pipelineBatch : batch.batches) {
136 for (auto& specBatch : pipelineBatch.specBatches) {
137 if (!pipelineBatch.pipeline.bind(ctx.getCommandBuffer(),
138 ctx.getRenderPhase(),
139 ctx.currentRenderPass(),
140 specBatch.specializationId)) {
141 continue;
142 }
143
144 vk::Pipeline* pipeline = pipelineBatch.pipeline.getPipeline(ctx.getRenderPhase());
145 ds::InstanceTable& descriptors =
146 pipelineBatch.perPhaseDescriptors[ctx.getRenderPhase()];
147
148 UpdateSpeed speed = UpdateSpeed::Dynamic;
149 for (auto* objectBatch : {&specBatch.objectsDynamic, &specBatch.objectsStatic}) {
150 ctx.bindDescriptors(pipeline->pipelineLayout().rawLayout(),
151 speed,
152 descriptors.get(ctx.currentObserverIndex()).data(),
153 descriptors.getDescriptorSetCount());
154
155 if (!descriptors.isBindless()) {
156 for (SceneObject* obj : *objectBatch) {
157 if (obj->component->isHidden()) { continue; }
158 for (std::uint8_t i = descriptors.getPerObjectStart();
159 i < descriptors.getDescriptorSetCount();
160 ++i) {
161 descriptors.get(ctx.currentObserverIndex())[i]->bindForObject(
162 ctx, pipeline->pipelineLayout().rawLayout(), i, obj->sceneKey);
163 }
164 ctx.renderObject(*obj);
165 }
166 }
167 else {
168 for (SceneObject* obj : *objectBatch) {
169 if (obj->component->isHidden()) { continue; }
170 ctx.renderObject(*obj);
171 }
172 }
173
174 speed = UpdateSpeed::Static;
175 }
176 }
177 }

234void BatchedScene::PipelineBatch::registerObserver(unsigned int index, RenderTarget& observer) {
235 const unsigned int startIndex = allDescriptors.size();
236 const auto addInstanceMaybe = [this, startIndex](ds::DescriptorSetInstance* instance) {
237 for (unsigned int i = startIndex; i < allDescriptors.size(); ++i) {
238 if (allDescriptors[i] == instance) { return; }
239 }
240 allDescriptors.emplace_back(instance);
241 };
242
243 // create descriptor sets (happens in addObserver) for all new layouts
244 for (RenderPhase phase = 0; phase < cfg::Limits::MaxRenderPhaseId; ++phase) {
245 vk::Pipeline* phasePipeline = pipeline.getPipeline(phase);
246 if (phasePipeline) {
247 perPhaseDescriptors[phase].addObserver(index, observer);
248 auto& descriptors = perPhaseDescriptors[phase];
249 for (unsigned int i = 0; i < descriptors.getDescriptorSetCount(); ++i) {
250 addInstanceMaybe(descriptors.get(index)[i]);
251 }
252 }
253 }
254
255 // call allocateObject on all new sets for all existing objects
256 for (SpecializationBatch& specBatch : specBatches) {
257 for (unsigned int i = startIndex; i < allDescriptors.size(); ++i) {
258 for (SceneObject* obj : specBatch.objectsStatic) {
259 if (!allDescriptors[i]->allocateObject(obj->entity, obj->sceneKey)) {
260 BL_LOG_ERROR << "Failed to add entity to scene descriptors for new observer: "
261 << obj->entity;
262 }
263 }
264 for (SceneObject* obj : specBatch.objectsDynamic) {
265 if (!allDescriptors[i]->allocateObject(obj->entity, obj->sceneKey)) {
266 BL_LOG_ERROR << "Failed to add entity to scene descriptors for new observer: "
267 << obj->entity;
268 }
269 }
270 }
271 }

274void BatchedScene::PipelineBatch::unregisterObserver(unsigned int index, RenderTarget&) {
275 const auto removeFromAll = [this, index](ds::DescriptorSetInstance* instance) {
276 for (unsigned int i = 0; i < allDescriptors.size(); ++i) {
277 if (allDescriptors[i] == instance) {
278 allDescriptors.erase(i);
279 return;
280 }
281 }
282 };
283
284 for (RenderPhase i = 0; i < cfg::Limits::MaxRenderPhaseId; ++i) {
285 vk::Pipeline* phasePipeline = pipeline.getPipeline(i);
286 if (phasePipeline) {
287 const auto& layoutSet = perPhaseDescriptors[i].get(index);
288 for (auto* set : layoutSet) { removeFromAll(set); }
289 perPhaseDescriptors[i].removeObserver(index);
290 }
291 }

305 std::uint32_t specialization) {
306 for (unsigned int i = 0; i < allDescriptors.size(); ++i) {
307 if (!allDescriptors[i]->allocateObject(entity, sceneObject->sceneKey)) {
308 BL_LOG_ERROR << "Failed to add entity " << entity

312 }
313 return false;
314 }
315 }

348void BatchedScene::PipelineBatch::updateDescriptors(ecs::Entity entity, SceneObject* object,
349 PipelineBatch& prevBatch) {
350 using AllCtr = decltype(allDescriptors);
351 const auto find = [](const AllCtr& all, ds::DescriptorSetInstance* set) -> bool {
352 for (ds::DescriptorSetInstance* s : all) {
353 if (s == set) { return true; }
354 }
355 return false;
356 };
357
358 // remove from old descriptors not in new
359 for (ds::DescriptorSetInstance* set : prevBatch.allDescriptors) {
360 if (!find(allDescriptors, set)) { set->releaseObject(entity, object->sceneKey); }
361 }
362
363 // add to new descriptors not in old
364 for (ds::DescriptorSetInstance* set : allDescriptors) {
365 if (!find(prevBatch.allDescriptors, set)) {
366 if (!set->allocateObject(entity, object->sceneKey)) {
367 BL_LOG_ERROR << "Failed to update descriptors for entity " << entity
368 << " in scene due to descriptor failure";
369 }
370 }
371 }
