46com::Physics2D* Physics2D::findEntityAtPosition(const engine::World2D& world, const glm::vec2& pos,
47 b2QueryFilter filter) const {
48 const float s = world.getWorldToBoxScale();
49 const b2Circle circle{.center = {s, s}, .radius = s};
50 b2Transform transform = b2Transform_identity;
51 transform.p = {pos.x * s, pos.y * s};
52 b2ShapeId found = b2_nullShapeId;
53 b2World_OverlapCircle(
54 world.getBox2dWorldId(), &circle, transform, filter, &worldOverlapCallback, &found);
55 if (B2_IS_NULL(found)) { return nullptr; }
56 const b2BodyId body = b2Shape_GetBody(found);
57 return static_cast<com::Physics2D*>(b2Body_GetUserData(body));
58}

84void Physics2D::update(std::mutex&, float dt, float, float, float) {
85 channel.syncDeferred();
86
87 for (auto world : worlds) {
88 if (!world) { continue; }
89
90 const b2WorldId worldId = world->getBox2dWorldId();
91 const float worldToBoxScale = world->getWorldToBoxScale();
92
93 b2World_Step(worldId, dt, 4);
94
95 b2BodyEvents moveEvents = b2World_GetBodyEvents(worldId);
96 for (int i = 0; i < moveEvents.moveCount; ++i) {
97 auto& event = moveEvents.moveEvents[i];
98 com::Physics2D& pc = *static_cast<com::Physics2D*>(event.userData);
99 pc.transform->setPosition(event.transform.p.x / worldToBoxScale,
100 event.transform.p.y / worldToBoxScale);
101 if (!b2Body_IsFixedRotation(event.bodyId)) {
102 pc.transform->setRotation(
103 math::radiansToDegrees(b2Rot_GetAngle(event.transform.q)));
104 }
105 }
106
107 const auto getComponent = [this](b2ShapeId shape) -> com::Physics2D& {
108 const b2BodyId body = b2Shape_GetBody(shape);
109 return *static_cast<com::Physics2D*>(b2Body_GetUserData(body));
110 };
111
112 b2ContactEvents contacts = b2World_GetContactEvents(worldId);
113 for (int i = 0; i < contacts.beginCount; ++i) {
114 com::Physics2D& a = getComponent(contacts.beginEvents[i].shapeIdA);
115 com::Physics2D& b = getComponent(contacts.beginEvents[i].shapeIdB);
116 emitter.emit<EntityCollisionBeginEvent>({a.entity, b.entity});
117 }
118 for (int i = 0; i < contacts.endCount; ++i) {
119 com::Physics2D& a = getComponent(contacts.endEvents[i].shapeIdA);
120 com::Physics2D& b = getComponent(contacts.endEvents[i].shapeIdB);
121 emitter.emit<EntityCollisionEndEvent>({a.entity, b.entity});
122 }
123 for (int i = 0; i < contacts.hitCount; ++i) {
124 com::Physics2D& a = getComponent(contacts.hitEvents[i].shapeIdA);
125 com::Physics2D& b = getComponent(contacts.hitEvents[i].shapeIdB);
126 emitter.emit<EntityCollisionHitEvent>({a.entity, b.entity, contacts.hitEvents[i]});
127 }
128
129 b2SensorEvents sensors = b2World_GetSensorEvents(worldId);
130 for (int i = 0; i < sensors.beginCount; ++i) {
131 com::Physics2D& entity = getComponent(sensors.beginEvents[i].visitorShapeId);
132 com::Physics2D& sensor = getComponent(sensors.beginEvents[i].sensorShapeId);
133 emitter.emit<SensorEntered>({entity.entity, sensor.entity});
134 }
135 for (int i = 0; i < sensors.endCount; ++i) {
136 com::Physics2D& entity = getComponent(sensors.endEvents[i].visitorShapeId);
137 com::Physics2D& sensor = getComponent(sensors.endEvents[i].sensorShapeId);
138 emitter.emit<SensorExited>({entity.entity, sensor.entity});
139 }
140 }

176std::optional<b2BodyId> Physics2D::createBody(
177 b2BodyDef& bodyDef, b2ShapeDef& shapeDef,
178 ecs::ComponentSet<ecs::Require<com::Hitbox2D, com::Transform2D>>& set) {
179 if (!set.isValid()) {
180 BL_LOG_ERROR << "Failed to add physics to entity (missing components): " << set.entity();
181 return {};
182 }
183
184 engine::World2D* world = worlds[set.entity().getWorldIndex()];
185 if (!world) {
186 BL_LOG_ERROR << "Failed to add physics to entity (wrong world type): " << set.entity();
187 return {};
188 }
189
190 const b2WorldId worldId = world->getBox2dWorldId();
191 const float worldToBoxScale = world->getWorldToBoxScale();
192
193 b2BodyId bodyId;
194 const auto createBody =
195 [this, worldId, worldToBoxScale, &set, &bodyId, &bodyDef]() -> b2BodyId {
196 const glm::vec2 pos = set.get<com::Transform2D>()->getGlobalPosition() * worldToBoxScale;
197 bodyDef.position.x = pos.x;
198 bodyDef.position.y = pos.y;
199 bodyDef.rotation =
200 b2MakeRot(math::degreesToRadians(set.get<com::Transform2D>()->getRotation()));
201 bodyId = b2CreateBody(worldId, &bodyDef);
202 return bodyId;
203 };
204
205 const glm::vec2 origin = set.get<com::Transform2D>()->getOrigin() * worldToBoxScale;
206 switch (set.get<com::Hitbox2D>()->getType()) {
207 case com::Hitbox2D::Circle: {
208 b2Circle circle;
209 circle.radius = set.get<com::Hitbox2D>()->getRadius() * worldToBoxScale;
210 circle.center.x = circle.radius - origin.x;
211 circle.center.y = circle.radius - origin.y;
212 b2CreateCircleShape(createBody(), &shapeDef, &circle);
213 } break;
214
215 case com::Hitbox2D::Rectangle: {
216 const glm::vec2 hsize = set.get<com::Hitbox2D>()->getSize() * worldToBoxScale * 0.5f;
217 const glm::vec2 diff = hsize - origin;
218 b2Polygon box = b2MakeOffsetBox(hsize.x, hsize.y, {diff.x, diff.y}, 0.f);
219 b2CreatePolygonShape(createBody(), &shapeDef, &box);
220 } break;
221
222 default:
223 BL_LOG_ERROR << "Invalid hitbox type for physics simulation: "
224 << set.get<com::Hitbox2D>()->getType();
225 return {};
226 }
227
228 return bodyId;
229}
