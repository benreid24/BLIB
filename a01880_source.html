158void Slider::onAcquisition() {
159 if (value < 0) value = 0;
160 if (value > 1) value = 1;
161
162 constexpr float ButPadding = 0.15f;
163 const float butSize = std::min(getAcquisition().width, getAcquisition().height);
164 freeSpace = (dir == Horizontal) ? getAcquisition().width : getAcquisition().height;
165 float offset = 0;
166 if (decreaseBut->visible()) {
167 decreaseBut->setChildPadding(butSize * ButPadding, false);
168 Packer::manuallyPackElement(
169 decreaseBut, {getAcquisition().left, getAcquisition().top, butSize, butSize});
170 offset += butSize;
171 freeSpace -= butSize;
172 }
173 if (increaseBut->visible()) {
174 increaseBut->setChildPadding(butSize * ButPadding, false);
175 const float x = (dir == Horizontal) ? (getAcquisition().width - butSize) : (0);
176 const float y = (dir == Vertical) ? (getAcquisition().height - butSize) : (0);
177
178 Packer::manuallyPackElement(
179 increaseBut, {x + getAcquisition().left, y + getAcquisition().top, butSize, butSize});
180 freeSpace -= butSize;
181 }
182
183 sliderSize = std::floor(freeSpace * sliderSizeRatio);
184 freeSpace -= sliderSize;
185 const float pos = offset + freeSpace * value;
186
187 if (dir == Horizontal)
188 Packer::manuallyPackElement(
189 slider, {pos + getAcquisition().left, getAcquisition().top, sliderSize, butSize});
190 else
191 Packer::manuallyPackElement(
192 slider, {getAcquisition().left, pos + getAcquisition().top, butSize, sliderSize});
193}
