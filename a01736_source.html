173 void ComboBoxComponent::updateOptions() {
174  ComboBox& owner = getOwnerAs<ComboBox>();
175  const RenderSettings& settings = owner.renderSettings();
176  const glm::vec2 boxSize(owner.getOptionSize().x, owner.getOptionSize().y);
177  const float outline = settings.outlineThickness.value_or(1.f);
178  const sf::Vector2f innerSize(boxSize.x - outline * 2.f, boxSize.y - outline * 2.f);
179 
180  openOptions.resize(owner.getAllOptions().size());
181  auto it = openOptions.begin();
182  for (unsigned int i = 0; i < owner.getAllOptions().size(); ++i) {
183  Option& o = *(it++);
184  const float y = static_cast<float>(i) * boxSize.y;
185 
186  if (!o.created) {
187  o.created = true;
188  o.background.create(*enginePtr, openOptionBoxes, boxSize);
189  o.text.create(*enginePtr, *settings.font.value_or(Font::get()));
190  o.text.setParent(openOptionBoxes);
191  if (currentOverlay) { o.text.addToScene(currentOverlay, rc::UpdateSpeed::Static); }
192  }
193 
194  o.text.getSection().setString(owner.getAllOptions()[i]);
195  configureText(o.text, settings);
196  const sf::Vector2f tpos =
197  RenderSettings::calculatePosition(RenderSettings::Left,
198  RenderSettings::Center,
199  sf::FloatRect{sf::Vector2f{0.f, y}, innerSize},
200  {o.text.getLocalSize().x, o.text.getLocalSize().y});
201  o.text.getTransform().setPosition({tpos.x, tpos.y + y});
202 
203  o.background.setSize(boxSize);
204  if (static_cast<int>(i) == owner.getMousedOption()) {
205  o.background.setFillColor(
206  sfcol(settings.secondaryFillColor.value_or(sf::Color(80, 80, 250))));
207  }
208  else {
209  o.background.setFillColor(sfcol(settings.fillColor.value_or(sf::Color(100, 100, 100))));
210  }
211  o.background.setOutlineColor(sfcol(settings.outlineColor.value_or(sf::Color::Black)));
212  o.background.setOutlineThickness(-outline);
213  o.background.getLocalTransform().setPosition({0.f, y});
214  }
