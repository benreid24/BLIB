148 void Renderer::renderFrame() {
149  // kick off transfers
150  for (vk::RenderTexture* rt : renderTextures) { rt->handleDescriptorSync(); }
151  if (commonObserver.hasScene()) { commonObserver.handleDescriptorSync(); }
152  else {
153  for (auto& o : observers) { o->handleDescriptorSync(); }
154  }
155  state.transferEngine.executeTransfers();
156 
157  // begin frame
158  vk::StandardAttachmentSet* currentFrame = nullptr;
159  VkCommandBuffer commandBuffer = nullptr;
160  state.beginFrame(currentFrame, commandBuffer);
161  framebuffers.current().recreateIfChanged(*currentFrame);
162 
163  // begin render texture rendering
164  for (vk::RenderTexture* rt : renderTextures) { rt->renderScene(commandBuffer); }
165 
166  const auto clearDepthBuffer = [this, commandBuffer]() {
167  VkClearAttachment attachment{};
168  attachment.aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT;
169  attachment.clearValue = clearColors[1];
170 
171  VkClearRect rect{};
172  rect.rect.extent.width = renderRegion.width;
173  rect.rect.extent.height = renderRegion.height;
174  rect.rect.offset.x = renderRegion.left;
175  rect.rect.offset.y = renderRegion.top;
176  rect.baseArrayLayer = 0;
177  rect.layerCount = 1;
178 
179  vkCmdClearAttachments(commandBuffer, 1, &attachment, 1, &rect);
180  };
181 
182  // record commands to render scenes
183  for (auto& o : observers) { o->renderScene(commandBuffer); }
184  if (commonObserver.hasScene()) {
185  clearDepthBuffer();
186  commonObserver.renderScene(commandBuffer);
187  }
188 
189  // perform render pass for final scene renders and overlays
190  framebuffers.current().beginRender(commandBuffer,
191  {{0, 0}, currentFrame->renderExtent()},
192  clearColors,
193  std::size(clearColors),
194  false);
195 
196  // render scene outputs
197  for (auto& o : observers) { o->compositeSceneAndOverlay(commandBuffer); }
198  if (commonObserver.hasScene()) {
199  clearDepthBuffer();
200  commonObserver.compositeSceneAndOverlay(commandBuffer);
201  }
202 
203  // complete frame
204  framebuffers.current().finishRender(commandBuffer);
205  state.completeFrame();
