102ReferenceValue Value::getProperty(const std::string& name, bool create) {
103 auto& props = deref().properties;
104
105 const auto bit = builtins.find(name);
106 if (bit != builtins.end()) { return ReferenceValue((this->*bit->second)()); }
107
108 if (!props && !create) { throw Error("Cannot access undefined property '" + name + "'"); }
109 else if (!props) {
110 props.reset(new std::unordered_map<std::string, ReferenceValue>());
111 }
112
113 const auto it = props->find(name);
114 if (it == props->end()) {
115 if (create) { return props->emplace(name, Value()).first->second; }
116 throw Error("Cannot access undefined property '" + name + "'");
117 }
118 return it->second;
119}
