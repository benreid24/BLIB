104 bool Element::handleEvent(const RawEvent& event) {
105  if (!visible()) return false;
106 
107  if (event.event.type == sf::Event::MouseWheelScrolled) {
108  if (getAcquisition().contains(sf::Vector2i(event.localMousePos))) {
109  return handleScroll(event);
110  }
111  return false;
112  }
113 
114  if (handleRawEvent(event)) return true;
115 
116  const bool eventOnMe =
117  acquisition.contains(sf::Vector2i(event.localMousePos.x, event.localMousePos.y));
118 
119  switch (event.event.type) {
120  case sf::Event::TextEntered:
121  case sf::Event::KeyPressed:
122  case sf::Event::KeyReleased:
123  case sf::Event::MouseWheelScrolled:
124  return processAction(Action::fromRaw(event));
125 
126  case sf::Event::MouseButtonPressed:
127  if (eventOnMe) {
128  if (!active()) return true;
129 
130  if (takeFocus()) {
131  if (event.event.mouseButton.button == sf::Mouse::Left) {
132  dragStart = event.localMousePos;
133  isLeftPressed = true;
134  return processAction(Action(Action::Pressed, event.localMousePos));
135  }
136  else if (event.event.mouseButton.button == sf::Mouse::Right) {
137  isRightPressed = true;
138  return true;
139  }
140  }
141  return true; // click blocked
142  }
143  return false;
144 
145  case sf::Event::MouseButtonReleased:
146  if (!active()) {
147  isLeftPressed = isRightPressed = false;
148  return eventOnMe;
149  }
150  if (event.event.mouseButton.button == sf::Mouse::Left) {
151  if (isLeftPressed) {
152  isLeftPressed = false;
153  processAction(Action(Action::Released, event.localMousePos));
154  if (eventOnMe) {
155  processAction(Action(Action::LeftClicked, event.localMousePos));
156  return true;
157  }
158  }
159  }
160  else if (event.event.mouseButton.button == sf::Mouse::Right) {
161  if (isRightPressed) {
162  isRightPressed = false;
163  if (eventOnMe) {
164  processAction(Action(Action::RightClicked, event.localMousePos));
165  return true;
166  }
167  }
168  }
169  return false;
170 
171  case sf::Event::MouseMoved:
172  if (!active()) { return eventOnMe; }
173  if (eventOnMe) { fireSignal(Action(Action::MouseMoved, event.localMousePos)); }
174  if (isLeftPressed) {
175  isMouseOver = eventOnMe;
176  return processAction(Action(Action::Dragged, dragStart, event.localMousePos));
177  }
178  else if (eventOnMe) {
179  if (!isMouseOver) {
180  isMouseOver = true;
181  processAction(Action(Action::MouseEntered, event.localMousePos));
182  }
183  return true;
184  }
185  else if (isMouseOver) {
186  isMouseOver = false;
187  processAction(Action(Action::MouseLeft, event.localMousePos));
188  }
189  return false;
190 
191  default:
192  return false;
193  }
194 }
