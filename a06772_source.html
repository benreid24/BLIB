37com::Physics2D* Physics2D::findEntityAtPosition(const engine::World2D& world, const glm::vec2& pos,
38 b2QueryFilter filter) const {
39 const float s = world.getWorldToBoxScale();
40 const b2Circle circle{.center = {s, s}, .radius = s};
41 b2Transform transform = b2Transform_identity;
42 transform.p = {pos.x * s, pos.y * s};
43 b2ShapeId found = b2_nullShapeId;
44 b2World_OverlapCircle(
45 world.getBox2dWorldId(), &circle, transform, filter, &worldOverlapCallback, &found);
46 if (B2_IS_NULL(found)) { return nullptr; }
47 const b2BodyId body = b2Shape_GetBody(found);
48 return static_cast<com::Physics2D*>(b2Body_GetUserData(body));
49}

72void Physics2D::update(std::mutex&, float dt, float, float, float) {
73 for (auto world : worlds) {
74 if (!world) { continue; }
75
76 const b2WorldId worldId = world->getBox2dWorldId();
77 const float worldToBoxScale = world->getWorldToBoxScale();
78
79 b2World_Step(worldId, dt, 4);
80
81 b2BodyEvents moveEvents = b2World_GetBodyEvents(worldId);
82 for (int i = 0; i < moveEvents.moveCount; ++i) {
83 auto& event = moveEvents.moveEvents[i];
84 com::Physics2D& pc = *static_cast<com::Physics2D*>(event.userData);
85 pc.transform->setPosition(event.transform.p.x / worldToBoxScale,
86 event.transform.p.y / worldToBoxScale);
87 if (!b2Body_IsFixedRotation(event.bodyId)) {
88 pc.transform->setRotation(
89 math::radiansToDegrees(b2Rot_GetAngle(event.transform.q)));
90 }
91 }
92
93 const auto getComponent = [this](b2ShapeId shape) -> com::Physics2D& {
94 const b2BodyId body = b2Shape_GetBody(shape);
95 return *static_cast<com::Physics2D*>(b2Body_GetUserData(body));
96 };
97
98 b2ContactEvents contacts = b2World_GetContactEvents(worldId);
99 for (int i = 0; i < contacts.beginCount; ++i) {
100 com::Physics2D& a = getComponent(contacts.beginEvents[i].shapeIdA);
101 com::Physics2D& b = getComponent(contacts.beginEvents[i].shapeIdB);
102 event::Dispatcher::dispatch<EntityCollisionBeginEvent>({a.entity, b.entity});
103 }
104 for (int i = 0; i < contacts.endCount; ++i) {
105 com::Physics2D& a = getComponent(contacts.endEvents[i].shapeIdA);
106 com::Physics2D& b = getComponent(contacts.endEvents[i].shapeIdB);
107 event::Dispatcher::dispatch<EntityCollisionEndEvent>({a.entity, b.entity});
108 }
109 for (int i = 0; i < contacts.hitCount; ++i) {
110 com::Physics2D& a = getComponent(contacts.hitEvents[i].shapeIdA);
111 com::Physics2D& b = getComponent(contacts.hitEvents[i].shapeIdB);
112 event::Dispatcher::dispatch<EntityCollisionHitEvent>(
113 {a.entity, b.entity, contacts.hitEvents[i]});
114 }
115
116 b2SensorEvents sensors = b2World_GetSensorEvents(worldId);
117 for (int i = 0; i < sensors.beginCount; ++i) {
118 com::Physics2D& entity = getComponent(sensors.beginEvents[i].visitorShapeId);
119 com::Physics2D& sensor = getComponent(sensors.beginEvents[i].sensorShapeId);
120 event::Dispatcher::dispatch<SensorEntered>({entity.entity, sensor.entity});
121 }
122 for (int i = 0; i < sensors.endCount; ++i) {
123 com::Physics2D& entity = getComponent(sensors.endEvents[i].visitorShapeId);
124 com::Physics2D& sensor = getComponent(sensors.endEvents[i].sensorShapeId);
125 event::Dispatcher::dispatch<SensorExited>({entity.entity, sensor.entity});
126 }
127 }

163std::optional<b2BodyId> Physics2D::createBody(
164 b2BodyDef& bodyDef, b2ShapeDef& shapeDef,
165 ecs::ComponentSet<ecs::Require<com::Hitbox2D, com::Transform2D>>& set) {
166 if (!set.isValid()) {
167 BL_LOG_ERROR << "Failed to add physics to entity (missing components): " << set.entity();
168 return {};
169 }
170
171 engine::World2D* world = worlds[set.entity().getWorldIndex()];
172 if (!world) {
173 BL_LOG_ERROR << "Failed to add physics to entity (wrong world type): " << set.entity();
174 return {};
175 }
176
177 const b2WorldId worldId = world->getBox2dWorldId();
178 const float worldToBoxScale = world->getWorldToBoxScale();
179
180 b2BodyId bodyId;
181 const auto createBody =
182 [this, worldId, worldToBoxScale, &set, &bodyId, &bodyDef]() -> b2BodyId {
183 const glm::vec2 pos = set.get<com::Transform2D>()->getGlobalPosition() * worldToBoxScale;
184 bodyDef.position.x = pos.x;
185 bodyDef.position.y = pos.y;
186 bodyDef.rotation =
187 b2MakeRot(math::degreesToRadians(set.get<com::Transform2D>()->getRotation()));
188 bodyId = b2CreateBody(worldId, &bodyDef);
189 return bodyId;
190 };
191
192 const glm::vec2 origin = set.get<com::Transform2D>()->getOrigin() * worldToBoxScale;
193 switch (set.get<com::Hitbox2D>()->getType()) {
194 case com::Hitbox2D::Circle: {
195 b2Circle circle;
196 circle.radius = set.get<com::Hitbox2D>()->getRadius() * worldToBoxScale;
197 circle.center.x = circle.radius - origin.x;
198 circle.center.y = circle.radius - origin.y;
199 b2CreateCircleShape(createBody(), &shapeDef, &circle);
200 } break;
201
202 case com::Hitbox2D::Rectangle: {
203 const glm::vec2 hsize = set.get<com::Hitbox2D>()->getSize() * worldToBoxScale * 0.5f;
204 const glm::vec2 diff = hsize - origin;
205 b2Polygon box = b2MakeOffsetBox(hsize.x, hsize.y, {diff.x, diff.y}, 0.f);
206 b2CreatePolygonShape(createBody(), &shapeDef, &box);
207 } break;
208
209 default:
210 BL_LOG_ERROR << "Invalid hitbox type for physics simulation: "
211 << set.get<com::Hitbox2D>()->getType();
212 return {};
213 }
214
215 return bodyId;
216}
