71bool Trigger::process(const sf::Event& event) {
72 switch (type) {
73 case Type::Key:
74 if (event.key.code == key) {
75 if (event.type == sf::Event::KeyPressed) { return makeOrtoggleActive(); }
76 else if (event.type == sf::Event::KeyReleased) {
77 makeInactive();
78 }
79 }
80 return false;
81
82 case Type::MouseButton:
83 if (event.mouseButton.button == mouseButton) {
84 if (event.type == sf::Event::MouseButtonPressed) { return makeOrtoggleActive(); }
85 else if (event.type == sf::Event::MouseButtonReleased) {
86 makeInactive();
87 }
88 }
89 return false;
90
91 case Type::MouseWheelUp:
92 if (event.type == sf::Event::MouseWheelScrolled) {
93 if (event.mouseWheelScroll.wheel == mouseWheel && event.mouseWheelScroll.delta > 0.f) {
94 if (toggle) {
95 nowActive = !nowActive;
96 return nowActive;
97 }
98 return true;
99 }
100 }
101 return false;
102
103 case Type::MouseWheelDown:
104 if (event.type == sf::Event::MouseWheelScrolled) {
105 if (event.mouseWheelScroll.wheel == mouseWheel && event.mouseWheelScroll.delta < 0.f) {
106 if (toggle) {
107 nowActive = !nowActive;
108 return nowActive;
109 }
110 return true;
111 }
112 }
113 return false;
114
115 case Type::JoystickButton:
116 if (event.joystickButton.button == joystickButton) {
117 if (event.type == sf::Event::JoystickButtonPressed) { return makeOrtoggleActive(); }
118 else if (event.type == sf::Event::JoystickButtonReleased) {
119 makeInactive();
120 }
121 }
122 return false;
123
124 case Type::JoystickPositive:
125 if (event.type == sf::Event::JoystickMoved) {
126 if (event.joystickMove.axis == joystickAxis) {
127 if (event.joystickMove.position >= StickThreshold) { return makeOrtoggleActive(); }
128 makeInactive();
129 }
130 }
131 return false;
132
133 case Type::JoystickNegative:
134 if (event.type == sf::Event::JoystickMoved) {
135 if (event.joystickMove.axis == joystickAxis) {
136 if (event.joystickMove.position <= -StickThreshold) { return makeOrtoggleActive(); }
137 makeInactive();
138 }
139 }
140 return false;
141
142 case Type::Invalid:
143 default:
144 return false;
145 }
146}

148bool Trigger::configureFromEvent(const sf::Event& event) {
149 switch (event.type) {
150 case sf::Event::KeyPressed:
151 if (joystickMode) return false;
152 type = Type::Key;
153 key = event.key.code;
154 return true;
155
156 case sf::Event::MouseButtonPressed:
157 if (joystickMode) return false;
158 type = Type::MouseButton;
159 mouseButton = event.mouseButton.button;
160 return true;
161
162 case sf::Event::MouseWheelScrolled:
163 if (joystickMode) return false;
164 type = event.mouseWheelScroll.delta > 0.f ? Type::MouseWheelUp : Type::MouseWheelDown;
165 mouseWheel = event.mouseWheelScroll.wheel;
166 return true;
167
168 case sf::Event::JoystickButtonPressed:
169 if (!joystickMode) return false;
170 type = Type::JoystickButton;
171 joystickButton = event.joystickButton.button;
172 return true;
173
174 case sf::Event::JoystickMoved:
175 if (!joystickMode) return false;
176 if (event.joystickMove.position >= StickThreshold) {
177 type = Type::JoystickPositive;
178 joystickAxis = event.joystickMove.axis;
179 return true;
180 }
181 else if (event.joystickMove.position <= -StickThreshold) {
182 type = Type::JoystickNegative;
183 joystickAxis = event.joystickMove.axis;
184 return true;
185 }
186 return false;
187
188 default:
189 return false;
190 }
