69bool Registry::destroyEntity(Entity start, const Transaction<tx::EntityRead>& tx) {

81bool Registry::queueEntityDestroy(Entity start) {
82 // check if we can remove due to dependencies
83 if (dependencyGraph.hasDependencies(start)) {
84 markEntityForRemoval(start);
85 return false;
86 }
87
88 // determine list of entities to remove due to parenting and dependencies
89 std::unique_lock deleteLock(deletionState.mutex);
90 std::vector<Entity>& toRemove = deletionState.toRemove;
91 std::vector<Entity>& toVisit = deletionState.toVisit;
92 std::vector<Entity>& toUnparent = deletionState.toUnparent;
93 toRemove.reserve(16);
94 toVisit.reserve(16);
95 toVisit.emplace_back(start);
96
97 while (!toVisit.empty()) {
98 const Entity visiting = toVisit.back();
99 toVisit.pop_back();
100
101 if (!entityExistsLocked(visiting)) { continue; }
102
103 toRemove.emplace_back(visiting);
104
105 // increment version here
106 ++entityVersions[visiting.getIndex()];
107
108 // send events now
109 const std::uint32_t index = visiting.getIndex();
110 const ComponentMask::SimpleMask mask = entityMasks[index];
111 emitter.emit<event::EntityDestroyed>({visiting});
112 for (ComponentPoolBase* pool : componentPools) {
113 if (ComponentMask::has(mask, pool->ComponentIndex)) {
114 pool->fireRemoveEventOnly(visiting);
115 }
116 }
117
118 // add marked dependencies
119 for (Entity resource : dependencyGraph.getResources(visiting)) {
120 dependencyGraph.removeDependency(resource, visiting);
121 if (!dependencyGraph.hasDependencies(resource)) {
122 const std::uint32_t i = resource.getIndex();
123 if (i < markedForRemoval.size() && markedForRemoval[i]) {
124 toVisit.emplace_back(resource);
125 }
126 }
127 }
128
129 // add children
130 for (Entity child : parentGraph.getChildren(visiting)) {
131 const std::uint32_t j = child.getIndex();
132 switch (parentDestructionBehaviors[j]) {
133 case ParentDestructionBehavior::DestroyedWithParent:
134 toVisit.emplace_back(child);
135 break;
136 case ParentDestructionBehavior::OrphanedByParent:
137 toUnparent.reserve(16);
138 toUnparent.emplace_back(child);
139 break;
140 }
141 }
142 }
143
144 // unparent orphans
145 std::unique_lock parentLock(entityLock);
146 for (Entity child : deletionState.toUnparent) { removeEntityParentLocked(child, false); }
147 deletionState.toUnparent.clear();
148
149 return true;
150}

152void Registry::doEntityDestroyLocked(Entity ent) {
153 removeEntityParentLocked(ent, true);
154
155 const std::uint32_t index = ent.getIndex();
156 const ComponentMask::SimpleMask mask = entityMasks[index];
157
158 // reset metadata
159 entityAllocator.release(index);
160 entityMasks[index] = ComponentMask::EmptyMask;
161 entityFlags[index] = Flags::None;
162 entityWorlds[index] = 0;
163
164 // remove from views
165 for (auto& view : views) {
166 if (view->mask.passes(mask)) { view->removeEntity(ent); }
167 }
168
169 // destroy components
170 for (ComponentPoolBase* pool : componentPools) {
171 if (ComponentMask::has(mask, pool->ComponentIndex)) { pool->remove(ent, false); }
172 }
173
174 // remove parenting info
175 parentGraph.removeEntity(ent);
176
177 // remove dependency info
178 if (index < markedForRemoval.size()) { markedForRemoval[index] = false; }
179}

181void Registry::flushDeletions() {
182 std::unique_lock lock(entityLock);
183 std::unique_lock queueLock(deletionState.mutex);
184
185 // destroy queued components
186 for (auto& pool : componentPools) { pool->flushRemovals(); }
187
188 // destroy queued entities
189 for (const Entity& ent : deletionState.toRemove) { doEntityDestroyLocked(ent); }
190 deletionState.toRemove.clear();

267void Registry::setEntityParent(Entity child, Entity parent, const Transaction<tx::EntityRead>&) {
268 // remove parent first
269 removeEntityParentLocked(child, false);

271 // check both valid
272 const std::uint32_t ic = child.getIndex();
273 const std::uint32_t ip = parent.getIndex();
274 if (!entityAllocator.isAllocated(ic) || !entityAllocator.isAllocated(ip)) {

276 return;

297void Registry::removeEntityParentLocked(Entity child, bool fromDestroy) {
298 const Entity parent = parentGraph.getParent(child);
299 if (parent == InvalidEntity) { return; }
300 const std::uint32_t ic = child.getIndex();
301 const std::uint32_t ip = parent.getIndex();
302 if (!entityAllocator.isAllocated(ic) || !entityAllocator.isAllocated(ip)) {
303 BL_LOG_WARN << "Tried to un-parent bad entities: " << child << " <- " << parent;
304 return;
305 }
306
307 const ComponentMask mask{.required = entityMasks[ic]};
308 parentGraph.unParent(child);
309 for (ComponentPoolBase* pool : componentPools) {
310 if (mask.contains(pool->ComponentIndex)) { pool->onParentRemove(parent, child); }
311 }
312
313 emitter.emit<event::EntityParentRemoved>({child, fromDestroy});
314}
