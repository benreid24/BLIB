175 void setParent(Entity child, Entity parent) {
176 T* childCom = get(child);
177 T* parentCom = get(parent);
178
179 if (childCom && parentCom) { childCom->parent = parentCom; }
180 else {
181 // only child is checked before this is called so no log for bad parent
182 if (!childCom) { BL_LOG_ERROR << "Invalid child entity: " << child; }
183 }
184 }
185
186 void addChild(Entity child, Entity parent) {
187 T* parentCom = get(parent);
188 if (!parentCom) { return; }
189 T* childCom = get(child);
190 if (childCom) {
191 auto& c = parentCom->children;
192 const auto it = std::find(c.begin(), c.end(), childCom);
193 if (it == c.end()) { parentCom->children.emplace_back(childCom); }
194 }
195 else { BL_LOG_ERROR << "Invalid child entity " << child << " for parent " << parent; }
196 }
197
198 virtual void onParentRemove(Entity parent, Entity orphan) override {
199 constexpr bool needsLock = IsParentAware || IsChildAware;
200 if constexpr (needsLock) { poolLock.lockWrite(); }
201
202 if constexpr (std::is_base_of_v<trait::ParentAware<T>, T>) { removeParent(orphan); }
203 if constexpr (std::is_base_of_v<trait::ChildAware<T>, T>) { removeChild(parent, orphan); }
204
205 if constexpr (needsLock) { poolLock.unlockWrite(); }
206 }
207
208 void removeParent(Entity orphan) {
209 T* com = get(orphan);
210 if (com) { com->parent = nullptr; }
211 else { BL_LOG_WARN << "Invalid orphan entity: " << orphan; }
212 }
213
214 void removeChild(Entity parent, Entity orphan) {
215 T* pcom = get(parent);
216 if (!pcom) {
217 // only child is checked before this is called so no log for bad parent
218 return;
219 }
220 T* com = get(orphan);
221 if (com) {
222 auto& c = pcom->children;
223 const auto it = std::find(c.begin(), c.end(), com);
224 if (it != c.end()) { c.erase(it); }
225 }
226 else { BL_LOG_ERROR << "Invalid orphan entity: " << orphan; }
227 }

323template<typename T>
324void* ComponentPool<T>::doRemoveLocked(Entity ent, bool fireEvent) {
325 // determine if present

327 if (entIndex >= entityToComponent.size()) return nullptr;
328 T* com = entityToComponent[entIndex];

360 std::unique_lock lock(removalQueueMutex);
361 util::ReadWriteLock::WriteScopeGuard writeLock(poolLock);
