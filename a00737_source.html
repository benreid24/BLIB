141 sf::Vector2f Notebook::minimumRequisition() const {
142  sf::Vector2f tabReq = tabArea->getRequisition();
143  tabReq.x = maxWidth > 0.f ? std::min(maxWidth, tabReq.x) : tabReq.x;
144  tabReq.y += TabTopPadding;
145  sf::Vector2f contentReq(0, 0);
146  for (const Page& p : pages) {
147  const sf::Vector2f creq = p.content->getRequisition();
148  contentReq.x = std::max(contentReq.x, creq.x);
149  contentReq.y = std::max(contentReq.y, creq.y);
150  }
151  contentReq += 2.f * sf::Vector2f(ContentPadding, ContentPadding);
152  return {std::max(tabReq.x, contentReq.x), tabReq.y + contentReq.y};
153 }

185 void Notebook::makePageActiveDirect(Page* page) {
186  if (activePage) {
187  activePage->content->setVisible(false, false);
188  if (page != activePage && activePage) activePage->onClose();
189  }
190  page->content->setVisible(true, false);
191  page->content->moveToTop();
192  Packer::manuallyPackElement(page->content, contentArea(), true);
193  if (page != activePage) page->onOpen();
194  activePage = page;
195 
196  if (maxWidth > 0.f) {
197  const float x = page->label->getPosition().x - scroll;
198  if (x < getPosition().x) { scroll = page->label->getPosition().x - getPosition().x; }
199  else {
200  const float right = getAcquisition().left + getAcquisition().width;
201  const float xr = x + page->label->getAcquisition().width;
202  if (xr > right) {
203  scroll = (page->label->getPosition().x - getPosition().x) - getAcquisition().width +
204  page->label->getAcquisition().width;
205  }
206  }
207  constrainScroll();
208  }
