124 void Actor::update() {
125  for (unsigned int i = 0; i < (*activeControls).size(); ++i) {
126  const Control& ctrl = (*activeControls)[i];
127 
128  switch (ctrl.type) {
129  case Control::Type::SingleTrigger:
130  if (ctrl.triggerControl.active()) {
131  dispatch(i, DispatchType::TriggerActivated, false);
132  }
133  break;
134  case Control::Type::Movement:
135  if (ctrl.movementControl.type == MovementControl::Type::Triggers) {
136  const auto& keys = ctrl.movementControl.keys;
137  if (keys.up.active()) { dispatch(i, DispatchType::MovementUp, false); }
138  if (keys.right.active()) { dispatch(i, DispatchType::MovementRight, false); }
139  if (keys.down.active()) { dispatch(i, DispatchType::MovementDown, false); }
140  if (keys.left.active()) { dispatch(i, DispatchType::MovementLeft, false); }
141  }
142  else {
143  const sf::Vector2f& pos = ctrl.movementControl.joystick.cachedPosition;
144  if (pos.x >= StickThreshold) { dispatch(i, DispatchType::MovementRight, false); }
145  if (pos.x <= -StickThreshold) { dispatch(i, DispatchType::MovementLeft, false); }
146  if (pos.y >= StickThreshold) { dispatch(i, DispatchType::MovementUp, false); }
147  if (pos.y <= -StickThreshold) { dispatch(i, DispatchType::MovementDown, false); }
148  }
149  break;
150  default:
151  break;
152  }

188 void Actor::process(const sf::Event& event) {
189  // determine if keyboard or controller is being used
190  switch (event.type) {
191  case sf::Event::JoystickButtonPressed:
192  case sf::Event::JoystickButtonReleased:
193  if (joystick < sf::Joystick::Count && event.joystickButton.joystickId != joystick) return;
194  activeControls = &joystickControls;
195  break;
196  case sf::Event::JoystickMoved:
197  if (joystick < sf::Joystick::Count && event.joystickMove.joystickId != joystick) return;
198  activeControls = &joystickControls;
199  break;
200  case sf::Event::KeyPressed:
201  case sf::Event::KeyReleased:
202  case sf::Event::MouseButtonPressed:
203  case sf::Event::MouseButtonReleased:
204  case sf::Event::MouseWheelScrolled:
205  case sf::Event::MouseMoved:
206  activeControls = &kbmControls;
207  break;
208  default:
209  return; // can exit early, event does not trigger any controls
210  }
211 
212  // process control
213  for (unsigned int i = 0; i < (*activeControls).size(); ++i) {
214  Control& ctrl = (*activeControls)[i];
215 
216  switch (ctrl.type) {
217  case Control::Type::SingleTrigger:
218  if (ctrl.triggerControl.process(event)) {
219  dispatch(i, DispatchType::TriggerActivated, true);
220  }
221  break;
222  case Control::Type::Movement: {
223  const DispatchType dt = ctrl.movementControl.process(event);
224  if (dt != DispatchType::_INVALID) { dispatch(i, dt, true); }
225  } break;
226  case Control::Type::Directional:
227  if (ctrl.directionalControl.process(owner, event)) {
228  dispatch(i, DispatchType::DirectionChanged, true);
229  }
230  break;
231  default:
232  break;
233  }
