179bool DescriptorPool::Subpool::canAllocate(const SetBindingInfo& allocInfo,
180 std::size_t setCount) const {
181 if (freeSets < setCount) { return false; }
182
183 for (std::size_t i = 0; i < setCount; ++i) {
184 for (std::size_t j = 0; j < allocInfo.bindingCount; ++j) {
185 const auto& binding = allocInfo.bindings[j];
186 if (available[binding.descriptorType] < binding.descriptorCount * setCount) {
187 return false;
188 }
189 }
190 }
191
192 return true;
193}

195void DescriptorPool::Subpool::allocate(const SetBindingInfo& allocInfo,
196 VkDescriptorSetLayout layout, VkDescriptorSet* sets,
197 std::size_t setCount) {
198 // update metadata
199 freeSets -= setCount;
200 for (std::size_t j = 0; j < allocInfo.bindingCount; ++j) {
201 const auto& binding = allocInfo.bindings[j];
202 available[binding.descriptorType] -= binding.descriptorCount * setCount;
203 }
204
205 // duplicate layout for vulkan api
206 std::array<VkDescriptorSetLayout, 128> layoutsArray;
207 std::vector<VkDescriptorSetLayout> layoutsVector;
208 VkDescriptorSetLayout* pLayouts;
209 if (setCount <= layoutsArray.size()) {
210 for (unsigned int i = 0; i < setCount; ++i) { layoutsArray[i] = layout; }
211 pLayouts = layoutsArray.data();
212 }
213 else {
214 layoutsVector.resize(setCount, layout);
215 pLayouts = layoutsVector.data();
216 }
217
218 // allocate sets
219 VkDescriptorSetAllocateInfo setAllocInfo{};
220 setAllocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
221 setAllocInfo.descriptorPool = pool;
222 setAllocInfo.descriptorSetCount = setCount;
223 setAllocInfo.pSetLayouts = pLayouts;
224 if (vkAllocateDescriptorSets(vulkanState.device, &setAllocInfo, sets) != VK_SUCCESS) {
