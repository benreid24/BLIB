147 void Text::computeWordWrap() {
148  for (auto& section : sections) {
149  section.wordWrappedContent = section.content;
150  section.cachedLineHeight = section.computeLineSpacing(*font);
151  }
152  if (wrapType == WrapType::None || wordWrapWidth <= 0.f) { return; }
153 
154  float pw = 1.f;
155  if (wrapType == WrapType::Relative && getTransform().hasParent()) {
156  const ecs::Entity parent = engine().ecs().getEntityParent(entity());
157  auto cset =
158  engine().ecs().getComponentSet<ecs::Require<com::OverlayScaler, com::Transform2D>>(
159  parent);
160  if (cset.isValid()) {
161  pw = cset.get<com::OverlayScaler>()->getEntityBounds().width *
162  cset.get<com::Transform2D>()->getScale().x;
163  }
164  }
165  const float maxWidth = wordWrapWidth * pw / getTransform().getScale().x;
166  const Iter EndIter = Iter::end(sections);
167 
168  glm::vec2 nextPos(0.f, 0.f);
169  float maxLineHeight = 0.f;
170  Iter prevSpace = EndIter;
171  glm::vec2 wordStartPos(0.f, 0.f);
172  bool lineOnNextSpace = false;
173  std::uint32_t prevChar = 0;
174 
175  const auto resetLine = [this, &maxLineHeight, &prevSpace, &lineOnNextSpace, EndIter](Iter it) {
176  prevSpace = EndIter;
177  lineOnNextSpace = false;
178  it.getText().cachedLineHeight = it.getText().computeLineSpacing(*font);
179  maxLineHeight = it.getText().cachedLineHeight;
180  };
181 
182  for (Iter it = Iter::begin(sections); it != EndIter; ++it) {
183  maxLineHeight = std::max(it.getText().computeLineSpacing(*font), maxLineHeight);
184  it.getText().cachedLineHeight = maxLineHeight;
185  glm::vec2 advance =
186  it.getText().advanceCharacterPos(*font, nextPos, it.getChar(), prevChar);
187  prevChar = it.getChar();
188 
189  if (it.getChar() == ' ') {
190  if (lineOnNextSpace) {
191  it.makeNewline();
192  resetLine(it);
193  wordStartPos = glm::vec2{0.f, advance.y + maxLineHeight};
194  }
195  else {
196  prevSpace = it;
197  wordStartPos = advance;
198  }
199  }
200  else if (it.getChar() == '\n') {
201  resetLine(it);
202  wordStartPos = advance;
203  }
204  else {
205  if (nextPos.x > maxWidth) { // nextPos == prevPos here
206  if (prevSpace != EndIter) {
207  prevSpace.makeNewline();
208  resetLine(it);
209  const float wordWidth = advance.x - wordStartPos.x;
210  advance = glm::vec2{wordWidth, advance.y + maxLineHeight};
211  wordStartPos = advance;
212  }
213  else { lineOnNextSpace = true; }
214  }
215  }
216 
217  nextPos = advance;
218  }
219 }
