52void SSAOBufferShaderResource::populateAndTransfer() {
53 auto& params = buffer[0];
54
55 // copy parameters
56 params.bias = settings->getSSAOBias();
57 params.radius = settings->getSSAORadius();
58 params.exponent = settings->getSSAOExponent();
59
60 // create sample direction vectors
61 params.sampleCount = getSampleCount(settings->getSSAO());
62 for (std::uint32_t i = 0; i < params.sampleCount; ++i) {
63 // make random unit vector with positive z (facing outside of surface)
64 glm::vec3 s(util::Random::get<float>(-1.f, 1.f),
65 util::Random::get<float>(-1.f, 1.f),
66 util::Random::get<float>(0.f, 1.f));
67 s = glm::normalize(s);
68
69 // scale by random factor
70 s *= util::Random::get<float>(0.f, 1.f);
71
72 // scale by lerp of i to gen more samples near to the surface
73 float scale = static_cast<float>(i) / static_cast<float>(params.sampleCount);
74 scale = lerp(0.1f, 1.0f, scale * scale);
75 s *= scale;
76
77 params.samples[i] = glm::vec4(s, 0.f);
78 }
79
80 // create random noise rotation vectors
81 for (std::uint32_t x = 0; x < 4; ++x) {
82 for (std::uint32_t y = 0; y < 4; ++y) {
83 params.randomRotations[x][y] = glm::vec4(
84 util::Random::get<float>(-1.f, 1.f), util::Random::get<float>(-1.f, 1.f), 0.f, 0.f);
85 }
86 }
87
88 buffer.queueTransfer();
