32void constrainScissor(VkRect2D& scissor, const VkRect2D& limits) {
33 scissor.offset.x = std::max(scissor.offset.x, limits.offset.x);
34 scissor.offset.y = std::max(scissor.offset.y, limits.offset.y);
35
36 const int limitRight = limits.offset.x + limits.extent.width;
37 if (scissor.offset.x + static_cast<int>(scissor.extent.width) > limitRight) {
38 if (limitRight >= scissor.offset.x) {
39 scissor.extent.width = limitRight - scissor.offset.x;
40 }
41 else { scissor.extent.width = 0; }
42 }
43
44 const int limitBottom = limits.offset.y + limits.extent.height;
45 if (scissor.offset.y + static_cast<int>(scissor.extent.height) > limitBottom) {
46 if (limitBottom >= scissor.offset.y) {
47 scissor.extent.height = limitBottom - scissor.offset.y;
48 }
49 else { scissor.extent.height = 0; }
50 }
51}

93void OverlayScalerSystem::refreshEntity(Result& cset) {
94 com::OverlayScaler& scaler = *cset.get<com::OverlayScaler>();
95 com::Transform2D& transform = *cset.get<com::Transform2D>();
96 rc::ovy::OverlayObject& obj = *cset.get<rc::ovy::OverlayObject>();
97
98 scaler.dirty = false;
99 scaler.transformVersion = transform.getVersion();
100 scaler.cachedScaledOrigin = transform.getOrigin() * transform.getScale();
101
102 // dummy entities only get scissor, all else is skipped
103 if (cset.entity().flagSet(ecs::Flags::Dummy)) {
104 updateScissor(cset, 1.f, 1.f);
105 return;
106 }
107
108 if (!obj.overlayViewport) { return; }
109
110 const VkViewport& viewport = *obj.overlayViewport;
111 glm::vec2 parentSize = cam::OverlayCamera::getOverlayCoordinateSpace();
112 glm::vec2 parentOrigin(0.f, 0.f);
113 if (scaler.hasParent()) {
114 parentSize.x =
115 scaler.getParent().cachedObjectBounds.width * transform.getParent().getScale().x;
116 parentSize.y =
117 scaler.getParent().cachedObjectBounds.height * transform.getParent().getScale().y;
118 parentOrigin = scaler.getParent().cachedScaledOrigin;
119 }
120
121 float xScale = 1.f;
122 float yScale = 1.f;
123
124 // scale
125 switch (scaler.scaleType) {
126 case com::OverlayScaler::WidthPercent:
127 xScale = scaler.widthPercent * parentSize.x / scaler.cachedObjectBounds.width;
128 yScale = xScale;
129 break;
130
131 case com::OverlayScaler::HeightPercent:
132 yScale = scaler.heightPercent * parentSize.y / scaler.cachedObjectBounds.height;
133 xScale = yScale;
134 break;
135
136 case com::OverlayScaler::SizePercent:
137 xScale = scaler.widthPercent * parentSize.x / scaler.cachedObjectBounds.width;
138 yScale = scaler.heightPercent * parentSize.y / scaler.cachedObjectBounds.height;
139 break;
140
141 case com::OverlayScaler::PixelRatio:
142 xScale = scaler.cachedObjectBounds.width * scaler.pixelRatio / viewport.width;
143 yScale = scaler.cachedObjectBounds.height * scaler.pixelRatio / viewport.height;
144 break;
145
146 case com::OverlayScaler::LineHeight:
147 yScale = scaler.overlayRatio * parentSize.y;
148 xScale = yScale;
149 break;
150
151 case com::OverlayScaler::None:
152 default:
153 break;
154 }
155
156 if (scaler.scaleType != com::OverlayScaler::None) {
157 if (transform.getScale().x != xScale || transform.getScale().y != yScale) {
158 transform.setScale({xScale, yScale});
159 if (scaler.onScale) { scaler.onScale(); }
160 }
161 }
162 else {
163 xScale = transform.getScale().x;
164 yScale = transform.getScale().y;
165 }
166
167 // position
168 switch (scaler.posType) {
169 case com::OverlayScaler::ParentSpace:
170 transform.setPosition(scaler.parentPosition * parentSize - parentOrigin);
171 break;
172
173 case com::OverlayScaler::NoPosition:
174 default:
175 break;
176 }
177
178 // scissor
179 updateScissor(cset, xScale, yScale);
180
181 // update global size
182 const glm::vec2 pos = transform.getGlobalPosition();
183 const glm::vec2& overlaySize = cam::OverlayCamera::getOverlayCoordinateSpace();
184 scaler.cachedTargetRegion = {
185 viewport.x + viewport.width * pos.x / overlaySize.x,
186 viewport.y + viewport.height * pos.y / overlaySize.y,
187 viewport.width * (scaler.cachedObjectBounds.width * xScale) / overlaySize.x,
188 viewport.height * (scaler.cachedObjectBounds.height * yScale) / overlaySize.y};
189}

191void OverlayScalerSystem::updateScissor(Result& cset, float xScale, float yScale) {
192 rc::ovy::OverlayObject& obj = *cset.get<rc::ovy::OverlayObject>();
193 VkRect2D& scissor = obj.cachedScissor;
194 com::Transform2D& transform = *cset.get<com::Transform2D>();
195 com::OverlayScaler& scaler = *cset.get<com::OverlayScaler>();
196 const glm::vec2 pos = transform.getGlobalPosition();
197
198 switch (scaler.scissorMode) {
199 case com::OverlayScaler::ScissorSelf:
200 case com::OverlayScaler::ScissorSelfConstrained:
201 if (obj.overlayViewport) {
202 const VkViewport& viewport = *obj.overlayViewport;
203 const glm::vec2& origin = transform.getOrigin();
204 const glm::vec2 offset((origin.x - scaler.cachedObjectBounds.left) * xScale,
205 (origin.y - scaler.cachedObjectBounds.top) * yScale);
206 const glm::vec2 corner =
207 (pos - offset) / cam::OverlayCamera::getOverlayCoordinateSpace();
208
209 scissor.offset.x = viewport.x + viewport.width * corner.x;
210 scissor.offset.y = viewport.y + viewport.height * corner.y;
211 scissor.extent.width = viewport.width * (scaler.cachedObjectBounds.width * xScale) /
212 cam::OverlayCamera::getOverlayCoordinateSpace().x;
213 scissor.extent.height = viewport.height * (scaler.cachedObjectBounds.height * yScale) /
214 cam::OverlayCamera::getOverlayCoordinateSpace().y;
215
216 VkRect2D limits = makeScissor(viewport);
217 if (scaler.scissorMode == com::OverlayScaler::ScissorSelfConstrained &&
218 obj.hasParent()) {
219 limits = obj.getParent().cachedScissor;
220 }
221 constrainScissor(scissor, limits);
222 }
223 else { BL_LOG_ERROR << "Dummy entities do not support ScissorSelf modes"; }
224 break;
225
226 case com::OverlayScaler::ScissorObserver:
227 if (obj.overlayViewport) { scissor = makeScissor(*obj.overlayViewport); }
228 else { BL_LOG_ERROR << "Dummy entities do not support ScissorObserver mode"; }
229 break;
230
231 case com::OverlayScaler::ScissorInherit:
232 if (!obj.hasParent()) {
233 if (obj.overlayViewport) { obj.cachedScissor = makeScissor(*obj.overlayViewport); }
234 }
235 else { obj.cachedScissor = obj.getParent().cachedScissor; }
236 break;
237
238 case com::OverlayScaler::ScissorFixed:
239 scissor.offset.x = scaler.fixedScissor.left;
240 scissor.offset.y = scaler.fixedScissor.top;
241 scissor.extent.width = scaler.fixedScissor.width;
242 scissor.extent.height = scaler.fixedScissor.height;
243 break;
244 }
