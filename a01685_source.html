102std::vector<std::string> FileUtil::listDirectory(const std::string& path, const std::string& ext,
103 bool recursive) {
104 if (path.empty()) return {};
105
106 std::vector<std::string> list;
107
108 std::filesystem::path extPath(ext);
109 if (ext.size() >= 1 && ext[0] != '.') { extPath = "." + ext; }
110
111 const auto append = [&list, &ext, &path, &extPath](const std::filesystem::path& cpath) {
112 bool add = std::filesystem::is_regular_file(cpath);
113 if (add && !ext.empty()) { add = cpath.extension() == extPath; }
114 if (add) { list.emplace_back(cpath.generic_string()); }
115 };
116
117 if (recursive) {
118 for (const auto& cpath : std::filesystem::recursive_directory_iterator(path)) {
119 append(cpath);
120 }
121 }
122 else {
123 for (const auto& cpath : std::filesystem::directory_iterator(path)) { append(cpath); }
124 }
125
126 return list;
127}
