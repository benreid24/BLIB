121void Texture::executeTransfer(VkCommandBuffer cb, tfr::TransferContext& engine) {
122 const auto transitionLayout = [this, cb, &engine]() {
123 VkImageMemoryBarrier barrier{};
124 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
125 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
126 barrier.newLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
127 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
128 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
129 barrier.image = image.getImage();
130 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
131 barrier.subresourceRange.baseMipLevel = 0;
132 barrier.subresourceRange.levelCount = image.getLevelCount();
133 barrier.subresourceRange.baseArrayLayer = 0;
134 barrier.subresourceRange.layerCount = image.getLayerCount();
135 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
136 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
137 engine.registerImageBarrier(barrier);
138 image.notifyNewLayout(VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL);
139 };
140
141 const auto copyImageToStaging = [](const sf::Image& src, void* dst, const sf::IntRect& region) {
142 for (int x = region.position.x; x < region.position.x + region.size.x; ++x) {
143 for (int y = region.position.y; y < region.position.y + region.size.y; ++y) {
144 const std::ptrdiff_t offset = (x + y * src.getSize().x) * sizeof(sf::Color);
145 std::uint8_t* d = static_cast<std::uint8_t*>(dst) + offset;
146 const std::uint8_t* s = src.getPixelsPtr() + offset;
147 std::memcpy(d, s, sizeof(sf::Color));
148 }
149 }
150 };
151
152 // copy image contents if required
153 if (altImg || transferImg) {
154 const sf::Image& src = altImg ? *altImg : *transferImg;
155
156 bool fullImage = false;
157 if (source.size.x == 0 || source.size.y == 0) {
158 fullImage = true;
159 source.position.x = 0;
160 source.position.y = 0;
161 source.size.x = src.getSize().x;
162 source.size.y = src.getSize().y;
163 source.size.y /= image.getLayerCount();
164 }
165
166 // create staging buffer
167 const VkDeviceSize stageSize = source.size.x * source.size.y * 4 * image.getLayerCount();
168 VkBuffer stagingBuffer;
169 void* data;
170 engine.createTemporaryStagingBuffer(stageSize, stagingBuffer, &data);
171 if (!fullImage) { copyImageToStaging(src, data, source); }
172 else { std::memcpy(data, src.getPixelsPtr(), stageSize); }
173
174 // transition to transfer dst prior to copy
175 image.transitionLayout(cb, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, true);
176
177 // TODO - support loading existing mip maps
178 // issue copy command
179 VkDeviceSize layerSize = image.getSize().width * image.getSize().height * sizeof(sf::Color);
180 VkBufferImageCopy copyInfos[6]{};
181 for (unsigned int face = 0; face < image.getLayerCount(); ++face) {
182 auto& copyInfo = copyInfos[face];
183 copyInfo.bufferOffset = face * layerSize;
184 copyInfo.bufferRowLength = 0;
185 copyInfo.bufferImageHeight = 0;
186 copyInfo.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
187 copyInfo.imageSubresource.mipLevel = 0;
188 copyInfo.imageSubresource.baseArrayLayer = face;
189 copyInfo.imageSubresource.layerCount = 1;
190 copyInfo.imageOffset.x = destPos.x;
191 copyInfo.imageOffset.y = destPos.y;
192 copyInfo.imageExtent.width = source.size.x;
193 copyInfo.imageExtent.height = source.size.y;
194 copyInfo.imageExtent.depth = 1;
195 }
196 vkCmdCopyBufferToImage(cb,
197 stagingBuffer,
198 image.getImage(),
199 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
200 image.getLayerCount(),
201 copyInfos);
202
203 bool needsTransferToShaderSrcLayout = true;
204
205 // generate mip maps if requested
206 if (createOptions.genMipmaps) {
207 if (image.canGenMipMapsOnGpu()) {
208 image.genMipMapsOnGpu(cb, VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL);
209 needsTransferToShaderSrcLayout = false;
210 }
211 else if (image.getLevelCount() > 1) {
212 const std::uint32_t mipLevels = image.getLevelCount();
213 std::vector<VkBufferImageCopy> copyInfos;
214 copyInfos.resize(mipLevels - 1, VkBufferImageCopy{});
215
216 // pre-pass to determine total size
217 std::size_t totalSize = 0;
218 sf::IntRect bounds =
219 Image::getMipLevelBounds({src.getSize().x, src.getSize().y}, 1);
220 for (unsigned int i = 1; i < mipLevels; ++i) {
221 totalSize += bounds.size.x * bounds.size.y * sizeof(sf::Color);
222 bounds = Image::getNextMipLevelBounds(bounds, i);
223 }
224
225 // get one combined staging buffer
226 VkBuffer mipStaging;
227 void* stagingDst = nullptr;
228 engine.createTemporaryStagingBuffer(totalSize, mipStaging, &stagingDst);
229
230 // gen on cpu
231 sf::Image mipSrc = Image::genMipMapsOnCpu(src);
232
233 // copy mip levels
234 std::uint32_t baseOffset = 0;
235 bounds = Image::getMipLevelBounds({src.getSize().x, src.getSize().y}, 1);
236 for (unsigned int i = 1; i < mipLevels; ++i) {
237 // copy from staging to mip image
238 auto& copy = copyInfos[i - 1];
239 copy.bufferOffset = baseOffset;
240 copy.bufferImageHeight = 0;
241 copy.bufferRowLength = 0;
242 copy.imageExtent.width = bounds.size.x;
243 copy.imageExtent.height = bounds.size.y;
244 copy.imageExtent.depth = 1;
245 copy.imageOffset.x = copy.imageOffset.y = copy.imageOffset.z = 0;
246 copy.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
247 copy.imageSubresource.mipLevel = i;
248 copy.imageSubresource.baseArrayLayer = 0;
249 copy.imageSubresource.layerCount = 1;
250
251 // copy from cpu to staging
252 const std::uint32_t copySize =
253 bounds.size.x * bounds.size.y * sizeof(sf::Color);
254 const std::uint32_t srcOffset =
255 (bounds.position.y * mipSrc.getSize().x + bounds.position.x) *
256 sizeof(sf::Color);
257 void* copyDst = static_cast<char*>(stagingDst) + baseOffset;
258 std::memcpy(copyDst, mipSrc.getPixelsPtr() + srcOffset, copySize);
259
260 baseOffset += copySize;
261 bounds = Image::getNextMipLevelBounds(bounds, i);
262 }
263
264 // issue copy commands
265 vkCmdCopyBufferToImage(cb,
266 mipStaging,
267 image.getImage(),
268 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
269 copyInfos.size(),
270 copyInfos.data());
271 }
272 }
273
274 // insert pipeline barrier if required
275 if (needsTransferToShaderSrcLayout) { transitionLayout(); }
276
277 // cleanup
278 transferImg.release();
279 altImg = nullptr;
280 destPos.x = 0;
281 destPos.y = 0;
282 source.position.x = 0;
283 source.position.y = 0;
284 source.size.x = 0;
285 source.size.y = 0;
286 }
287 else if (image.getCurrentLayout() != VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL) {
288 transitionLayout();
289 }

308void Texture::updateTrans(const sf::Image& content) {
309 const std::uint32_t s = content.getSize().x * content.getSize().y;
310 const std::uint32_t thresh = s / 10;
311 std::uint32_t t = 0;
312
313 // jump directly from alpha to alpha
314 const std::uint8_t* end = content.getPixelsPtr() + s * 4;
315 for (const std::uint8_t* a = content.getPixelsPtr() + 3; a < end; a += 4) {
316 if (*a > 0 || *a < 255) {
317 ++t;
318 if (t >= thresh) {
319 hasTransparency = true;
320 return;
321 }
322 }
323 }
324
325 hasTransparency = false;
326}
