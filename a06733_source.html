146 void Text::computeWordWrap() {
147  for (auto& section : sections) {
148  section.wordWrappedContent = section.content;
149  section.cachedLineHeight = section.computeLineSpacing(*font);
150  }
151  if (wrapType == WrapType::None || wordWrapWidth <= 0.f) { return; }
152 
153  float pw = 1.f;
154  if (wrapType == WrapType::Relative && getTransform().hasParent()) {
155  const ecs::Entity parent = engine().ecs().getEntityParent(entity());
156  auto cset =
157  engine().ecs().getComponentSet<ecs::Require<com::OverlayScaler, com::Transform2D>>(
158  parent);
159  if (cset.isValid()) {
160  pw = cset.get<com::OverlayScaler>()->getEntityBounds().width *
161  cset.get<com::Transform2D>()->getScale().x;
162  }
163  }
164  const float maxWidth = wordWrapWidth * pw / getTransform().getScale().x;
165  const Iter EndIter = Iter::end(sections);
166 
167  glm::vec2 nextPos(0.f, 0.f);
168  float maxLineHeight = 0.f;
169  Iter prevSpace = EndIter;
170  glm::vec2 wordStartPos(0.f, 0.f);
171  bool lineOnNextSpace = false;
172  std::uint32_t prevChar = 0;
173 
174  const auto resetLine = [this, &maxLineHeight, &prevSpace, &lineOnNextSpace, EndIter](Iter it) {
175  prevSpace = EndIter;
176  lineOnNextSpace = false;
177  it.getText().cachedLineHeight = it.getText().computeLineSpacing(*font);
178  maxLineHeight = it.getText().cachedLineHeight;
179  };
180 
181  for (Iter it = Iter::begin(sections); it != EndIter; ++it) {
182  maxLineHeight = std::max(it.getText().computeLineSpacing(*font), maxLineHeight);
183  it.getText().cachedLineHeight = maxLineHeight;
184  glm::vec2 advance =
185  it.getText().advanceCharacterPos(*font, nextPos, it.getChar(), prevChar);
186  prevChar = it.getChar();
187 
188  if (it.getChar() == ' ') {
189  if (lineOnNextSpace) {
190  it.makeNewline();
191  resetLine(it);
192  wordStartPos = glm::vec2{0.f, advance.y + maxLineHeight};
193  }
194  else {
195  prevSpace = it;
196  wordStartPos = advance;
197  }
198  }
199  else if (it.getChar() == '\n') {
200  resetLine(it);
201  wordStartPos = advance;
202  }
203  else {
204  if (nextPos.x > maxWidth) { // nextPos == prevPos here
205  if (prevSpace != EndIter) {
206  prevSpace.makeNewline();
207  resetLine(it);
208  const float wordWidth = advance.x - wordStartPos.x;
209  advance = glm::vec2{wordWidth, advance.y + maxLineHeight};
210  wordStartPos = advance;
211  }
212  else { lineOnNextSpace = true; }
213  }
214  }
215 
216  nextPos = advance;
217  }
218 }
