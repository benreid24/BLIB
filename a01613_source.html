178 void Menu::removeItem(Item* item, bool c) {
179  unsigned int i = items.size();
180  for (unsigned int j = 0; j < items.size(); ++j) {
181  if (items[j].get() == item) { i = j; }
182  else {
183  for (unsigned int k = 0; k < Item::AttachPoint::_NUM_ATTACHPOINTS; ++k) {
184  if (items[j]->attachments[k] == item) {

209 void Menu::processEvent(const Event& event) {
210  switch (event.type) {
211  case Event::SelectorMove: {
212  if (selectedItem->attachments[event.moveEvent.direction]) {
213  const Item* ogSel = selectedItem;
214  Item* item = selectedItem->attachments[event.moveEvent.direction];
215  while (item) {
216  if (item->isSelectable()) {
217  setSelectedItem(item);
218  break;
219  }
220  if (!item->allowsSelectionCrossing()) break;
221  item = item->attachments[event.moveEvent.direction];
222  }
223  playSound(ogSel != selectedItem ? moveSound : failSound);
224  }
225  else { playSound(failSound); }
226  } break;
227  case Event::Activate:
228  selectedItem->getSignal(Item::Activated)();
229  playSound(selectSound);
230  break;
231  case Event::SelectorLocation: {
232  const glm::vec2 os = observer->transformToOverlaySpace(
233  {event.locationEvent.position.x, event.locationEvent.position.y});
234  const glm::vec2 pos = os - position - offset;
235  const sf::Vector2f sfpos(pos.x, pos.y);
236  for (const auto& item : items) {
237  const sf::FloatRect rect(
238  item->position.x, item->position.y, item->getSize().x, item->getSize().y);
239  if (rect.contains(sfpos)) {
240  if (item->isSelectable() && selectedItem != item.get()) {
241  setSelectedItem(item.get());
242  playSound(moveSound);
243  }
244  }
245  }
246  } break;
247  default:
248  BL_LOG_ERROR << "Menu::processEvent received invalid Event::Type: " << event.type;
249  break;
250  }

253 void Menu::refreshPositions() {
254  if (items.empty()) return;
255 
256  std::queue<Item*> toVisit;
257  std::unordered_set<Item*> visited;
258  toVisit.emplace(items.front().get());
259  visited.insert(items.front().get());
260  items.front()->position = {0.f, 0.f};
261  sf::FloatRect bounds = {0.f, 0.f, 0.f, 0.f};
262 
263  while (!toVisit.empty()) {
264  Item* item = toVisit.front();
265  toVisit.pop();
266 
267  const glm::vec2 size = item->getSize();
268  if (item->position.x < bounds.left) bounds.left = item->position.x;
269  if (item->position.x + size.x > bounds.width) bounds.width = item->position.x + size.x;
270  if (item->position.y < bounds.top) bounds.top = item->position.y;
271  if (item->position.y + size.y > bounds.height) bounds.height = item->position.y + size.y;
272 
273  for (unsigned int i = 0; i < Item::AttachPoint::_NUM_ATTACHPOINTS; ++i) {
274  const Item::AttachPoint ap = static_cast<Item::AttachPoint>(i);
275 
276  Item* v = item->attachments[i];
277  if (v && v->parent == ap && visited.find(v) == visited.end()) {
278  if (!v->positionOverridden) {
279  v->position = move(item->position, size, v->getSize(), ap);
280  }
281  else { v->position = v->posOverride; }
282  toVisit.emplace(v);
283  visited.insert(v);
284  }
285  }
286  }
287 
288  totalSize = {bounds.width - bounds.left, bounds.height - bounds.top};
289  for (auto& item : items) {
290  item->notifyPosition(item->position - glm::vec2{bounds.left, bounds.top} +
291  glm::vec2{bgndPadding.left, bgndPadding.top});
292  }
293 
294  const glm::vec2 thick(background.getOutlineThickness(), background.getOutlineThickness());
295  background.setSize(visibleSize() - thick * 2.f);
296  background.getTransform().setPosition(position + thick);
297 
298  if (selectedItem != nullptr) {
299  const glm::vec2& pos = selectedItem->transform->getLocalPosition();
300  selector->notifySelection(
301  selectedItem->getEntity(),
302  {pos.x, pos.y, selectedItem->getSize().x, selectedItem->getSize().y});
303  }
304 
305  refreshScroll();
306 }
