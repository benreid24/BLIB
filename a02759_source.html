164bool Engine::loop() {
165 BL_LOG_INFO << "Starting engine with state: " << states.top()->name();
166
167 sf::Clock loopTimer; // measures duration of entire loop top to bottom
168 sf::Clock updateOuterTimer; // measure duration from update tick start to start
169 const float minFrameLengthSeconds =
170 engineSettings.maximumFramerate() > 0.f ? 1.f / engineSettings.maximumFramerate() : 0.f;
171 // always limit to at least 100us to avoid issues with SFML time resolution
172 const std::uint64_t minFrameLength =
173 std::max(static_cast<std::uint64_t>(minFrameLengthSeconds * 1'000'000.f), MinFrameLengthMicroseconds);
174 std::uint64_t lag = 0;
175
176 sf::Clock updateMeasureTimer; // measures duration of update ticks
177 std::uint64_t updateTimestep = engineSettings.updateTimestepMicroseconds();
178 std::uint64_t averageUpdateTime = engineSettings.updateTimestepMicroseconds();
179
180 float lastWarnTime = -6.f; // always log first warning
181 bool followupLog = false;
182 auto fallBehindWarning =
183 [&lastWarnTime, &updateMeasureTimer, &followupLog](std::uint64_t behind) {
184 if (updateMeasureTimer.getElapsedTime().asSeconds() - lastWarnTime >= 5.f) {
185 lastWarnTime = updateMeasureTimer.getElapsedTime().asSeconds();
186 followupLog = true;
187 BL_LOG_WARN << "Can't catch up, running " << toSeconds(behind) << " seconds behind";
188 }
189 };
190 auto skipUpdatesInfo = [&followupLog](int frameCount) {
191 if (followupLog) {
192 followupLog = false;
193 BL_LOG_INFO << "Skipping " << frameCount << " updates";
194 }
195 };
196
197 sf::Clock fpsTimer;
198 float frameCount = 0.f;
199
200 workers.start();
201 backgroundWorkers.start(2);
202 states.top()->activate(*this);
203 eventEmitter.emit<event::Startup>({states.top()});
204
205 while (true) {
206 // Clear flags from last loop
207 engineFlags.clear();
208
209 if (rendererInstance.has_value()) {
210 std::optional<sf::Event> event = rendererInstance->getWindow().pollEvent();
211 while (event.has_value()) {
212 eventEmitter.emit<sf::Event>(event.value());
213
214 bool shouldExit = false;
215 event->visit(util::Visitor{
216 [this, &shouldExit](const sf::Event::Closed&) {
217 eventEmitter.emit<event::Shutdown>({event::Shutdown::WindowClosed});
218 renderThreadShouldRun = false;
219 renderingCv.notify_one();
220 shouldExit = true;
221 },
222 [this](const sf::Event::Resized& resize) { handleResize(resize, true); },
223 [this, &updateOuterTimer, &loopTimer, &shouldExit](
224 const sf::Event::FocusLost&) {
225 eventEmitter.emit<event::Paused>({});
226 if (!awaitFocus()) {
227 eventEmitter.emit<event::Shutdown>({event::Shutdown::WindowClosed});
228 shouldExit = true;
229 }
230 eventEmitter.emit<event::Resumed>({});
231 updateOuterTimer.restart();
232 loopTimer.restart();
233 },
234 [](const auto&) {
235 // catch all
236 }});
237 if (shouldExit) { return true; }
238 event = rendererInstance->getWindow().pollEvent();
239 }
240 }
241
242 ecsSystems.notifyFrameStart();
243
244 // Update and render
245 lag += scaleTime(updateOuterTimer.getElapsedTime().asMicroseconds(), timeScale);
246 updateOuterTimer.restart();
247 const std::uint64_t startingLag = lag;
248 std::uint64_t totalDt = 0;
249 updateMeasureTimer.restart();
250
251 // update until caught up
252 while (lag >= updateTimestep) {
253 // track tick time
254 totalDt += updateTimestep;
255 lag -= updateTimestep;
256
257 const std::uint64_t updateStart = updateMeasureTimer.getElapsedTime().asMicroseconds();
258 const float dt = toSeconds(updateTimestep);
259 const float realDt = toRealSeconds(updateTimestep, timeScale);
260 const float lagSeconds = toSeconds(lag);
261 const float realLagSeconds = toRealSeconds(lag, timeScale);
262
263 // core update game logic
264 input.update();
265 states.top()->update(*this, dt, realDt);
266 ecsSystems.update(FrameStage::FrameStart,
267 FrameStage::MARKER_OncePerFrame,
268 states.top()->systemsMask(),
269 dt,
270 realDt,
271 lagSeconds,
272 realLagSeconds);
273 signalChannel.syncDeferred();
274
275 // check if we should end early
276 if (engineFlags.stateChangeReady()) {
277 lag = 0.f;
278 break;
279 }
280
281 // handle timing
282 averageUpdateTime =
283 (8 * averageUpdateTime +
284 2 * (updateMeasureTimer.getElapsedTime().asMicroseconds() - updateStart)) /
285 10;
286 if (updateMeasureTimer.getElapsedTime().asMicroseconds() > startingLag * 11 / 10) {
287 fallBehindWarning(updateMeasureTimer.getElapsedTime().asMicroseconds() -
288 startingLag);
289 if (engineSettings.allowVariableTimestep()) {
290 const float newTs = updateTimestep * 21 / 20;
291 BL_LOG_INFO << "Adjusting update timestep from " << updateTimestep << "s to "
292 << newTs << "s";
293 updateTimestep = newTs;
294 averageUpdateTime = updateTimestep;
295 }
296 else {
297 skipUpdatesInfo(
298 std::ceil(static_cast<float>(lag) / static_cast<float>(updateTimestep)));
299 lag = 0;
300 }
301 }
302 }
303
304 // update timing
305 if (averageUpdateTime < updateTimestep * 9 / 10 &&
306 updateTimestep > engineSettings.updateTimestepMicroseconds()) {
307 const std::uint64_t newTs =
308 std::max(engineSettings.updateTimestepMicroseconds(), updateTimestep * 19 / 20);
309 BL_LOG_INFO << "Performance improved, adjusting timestep from "
310 << toSeconds(updateTimestep) << "s to " << toSeconds(newTs) << "s";
311 updateTimestep = newTs;
312 }
313
314 // wait for render thread to finish current frame & block it
315 std::unique_lock renderLock(renderingMutex);
316
317 // Free world slots
318 for (auto& world : worlds) {
319 if (world && world.refCount() == 1) {
320 eventEmitter.emit<event::WorldDestroyed>({*world});
321 world.release();
322 }
323 }
324
325 // Process flags
326 bool stateChanged = false;
327 while (engineFlags.stateChangeReady()) {
328 stateChanged = true;
329 preStateChange();
330
331 if (engineFlags.active(Flags::Terminate)) {
332 eventEmitter.emit<event::Shutdown>({event::Shutdown::Terminated});
333 return true;
334 }
335 else if (engineFlags.active(Flags::PopState)) {
336 BL_LOG_INFO << "Popping state: " << states.top()->name();
337 State::Ptr prev = states.top();
338 prev->deactivate(*this);
339 states.pop();
340 if (states.empty()) { // exit if no states left
341 BL_LOG_INFO << "Final state popped, exiting";
342 eventEmitter.emit<event::Shutdown>({event::Shutdown::FinalStatePopped});
343 return true;
344 }
345 BL_LOG_INFO << "New engine state (popped): " << states.top()->name();
346 postStateChange(prev);
347 }
348 else if (engineFlags.active(Flags::_priv_ReplaceState)) {
349 BL_LOG_INFO << "New engine state (replaced): " << newState->name();
350 State::Ptr prev = states.top();
351 prev->deactivate(*this);
352 states.pop();
353 states.push(newState);
354 postStateChange(prev);
355 }
356 else if (engineFlags.active(Flags::_priv_PushState)) {
357 BL_LOG_INFO << "New engine state (pushed): " << newState->name();
358 State::Ptr& prev = states.top();
359 prev->deactivate(*this);
360 states.push(newState);
361 postStateChange(prev);
362 }
363
364 updateOuterTimer.restart();
365 loopTimer.restart();
366 }
367
368 if (!stateChanged) {
369 // signal render next frame
370 if (rendererInstance.has_value()) {
371 // run once per frame systems
372 ecsSystems.update(FrameStage::MARKER_OncePerFrame,
373 FrameStage::COUNT,
374 states.top()->systemsMask(),
375 toSeconds(totalDt),
376 toRealSeconds(totalDt, timeScale),
377 toSeconds(lag),
378 toRealSeconds(lag, timeScale));
379
380 // flush scene object changes
381 rendererInstance->syncSceneObjects();
382
383 // Copy shared data to renderer while in locked section
384 rendererInstance->copyDataFromSources();
385
386 // Flush ECS deletion queues
387 entityRegistry.flushDeletions();
388
389 // signal rendering thread to start
390 renderLock.unlock();
391 renderingCv.notify_one();
392 }
393
394 // Adhere to FPS cap
395 const std::uint64_t st = minFrameLength - loopTimer.getElapsedTime().asMicroseconds();
396 if (st > 0) { sf::sleep(sf::microseconds(st)); }
397 loopTimer.restart();
398
399 frameCount += 1.f;
400 if (rendererInstance && fpsTimer.getElapsedTime().asSeconds() >= 1.f &&
401 engineSettings.logFps()) {
402 const float fps = frameCount / fpsTimer.getElapsedTime().asSeconds();
403 rendererInstance->getWindow().getSfWindow().setTitle(
404 rendererInstance->getSettings().getWindowSettings().title() + " (" +
405 std::to_string(int(std::roundf(fps))) + " fps)");
406 frameCount = 0.f;
407 fpsTimer.restart();
408 }
409 }
410 else {
411 // Flush ECS deletion queues
412 entityRegistry.flushDeletions();
413 }
414 }
415
416 return false; // shouldn't be able to get here
417}
