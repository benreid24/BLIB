148bool Engine::loop() {
149 BL_LOG_INFO << "Starting engine with state: " << states.top()->name();
150
151 sf::Clock loopTimer;
152 sf::Clock updateOuterTimer;
153 const float minFrameLength =
154 engineSettings.maximumFramerate() > 0 ? 1.f / engineSettings.maximumFramerate() : 0.f;
155 float lag = 0.f;
156
157 sf::Clock updateMeasureTimer;
158 float updateTimestep = engineSettings.updateTimestep();
159 float averageUpdateTime = engineSettings.updateTimestep();
160
161 float lastWarnTime = -6.f; // always log first warning
162 bool followupLog = false;
163 auto fallBehindWarning = [&lastWarnTime, &updateMeasureTimer, &followupLog](float behind) {
164 if (updateMeasureTimer.getElapsedTime().asSeconds() - lastWarnTime >= 5.f) {
165 lastWarnTime = updateMeasureTimer.getElapsedTime().asSeconds();
166 followupLog = true;
167 BL_LOG_WARN << "Can't catch up, running " << behind << " seconds behind";
168 }
169 };
170 auto skipUpdatesInfo = [&followupLog](int frameCount) {
171 if (followupLog) {
172 followupLog = false;
173 BL_LOG_INFO << "Skipping " << frameCount << " updates";
174 }
175 };
176
177 sf::Clock fpsTimer;
178 float frameCount = 0.f;
179
180 workers.start();
181 backgroundWorkers.start(2);
182 states.top()->activate(*this);
183 eventEmitter.emit<event::Startup>({states.top()});
184
185 while (true) {
186 // Clear flags from last loop
187 engineFlags.clear();
188
189 if (rendererInstance.has_value()) {
190 sf::Event event;
191 while (rendererInstance->getWindow().pollEvent(event)) {
192 eventEmitter.emit<sf::Event>(event);
193
194 switch (event.type) {
195 case sf::Event::Closed:
196 eventEmitter.emit<event::Shutdown>({event::Shutdown::WindowClosed});
197 return true;
198
199 case sf::Event::LostFocus:
200 eventEmitter.emit<event::Paused>({});
201 if (!awaitFocus()) {
202 eventEmitter.emit<event::Shutdown>({event::Shutdown::WindowClosed});
203 return true;
204 }
205 eventEmitter.emit<event::Resumed>({});
206 updateOuterTimer.restart();
207 loopTimer.restart();
208 break;
209
210 case sf::Event::Resized:
211 handleResize(event.size, true);
212 break;
213
214 default:
215 break;
216 }
217 }
218 }
219
220 ecsSystems.notifyFrameStart();
221
222 // Update and render
223 lag += updateOuterTimer.getElapsedTime().asSeconds() * timeScale;
224 updateOuterTimer.restart();
225 const float startingLag = lag;
226 float totalDt = 0.f;
227 updateMeasureTimer.restart();
228
229 // update until caught up
230 while (lag >= updateTimestep) {
231 const float updateStart = updateMeasureTimer.getElapsedTime().asSeconds();
232
233 // core update game logic
234 totalDt += updateTimestep;
235 input.update();
236 states.top()->update(*this, updateTimestep, updateTimestep / timeScale);
237 ecsSystems.update(FrameStage::FrameStart,
238 FrameStage::MARKER_OncePerFrame,
239 states.top()->systemsMask(),
240 updateTimestep,
241 updateTimestep / timeScale,
242 lag,
243 lag / timeScale);
244 signalChannel.syncDeferred();
245
246 // check if we should end early
247 if (engineFlags.stateChangeReady()) {
248 lag = 0.f;
249 break;
250 }
251
252 // handle timing
253 lag -= updateTimestep;
254 averageUpdateTime =
255 0.8f * averageUpdateTime +
256 0.2f * (updateMeasureTimer.getElapsedTime().asSeconds() - updateStart);
257 if (updateMeasureTimer.getElapsedTime().asSeconds() > startingLag * 1.1f) {
258 fallBehindWarning(updateMeasureTimer.getElapsedTime().asSeconds() - startingLag);
259 if (engineSettings.allowVariableTimestep()) {
260 const float newTs = updateTimestep * 1.05f;
261 BL_LOG_INFO << "Adjusting update timestep from " << updateTimestep << "s to "
262 << newTs << "s";
263 updateTimestep = newTs;
264 averageUpdateTime = updateTimestep;
265 }
266 else {
267 skipUpdatesInfo(std::ceil(lag / updateTimestep));
268 lag = 0.f;
269 }
270 }
271 }
272
273 // update timing
274 if (averageUpdateTime < updateTimestep * 0.9f &&
275 updateTimestep > engineSettings.updateTimestep()) {
276 const float newTs = std::max(engineSettings.updateTimestep(), updateTimestep * 0.95f);
277 BL_LOG_INFO << "Performance improved, adjusting timestep from " << updateTimestep
278 << "s to " << newTs << "s";
279 updateTimestep = newTs;
280 }
281
282 // wait for render thread to finish current frame & block it
283 std::unique_lock renderLock(renderingMutex);
284
285 // Free world slots
286 for (auto& world : worlds) {
287 if (world && world.refCount() == 1) {
288 eventEmitter.emit<event::WorldDestroyed>({*world});
289 world.release();
290 }
291 }
292
293 // Process flags
294 bool stateChanged = false;
295 while (engineFlags.stateChangeReady()) {
296 stateChanged = true;
297
298 if (engineFlags.active(Flags::Terminate)) {
299 eventEmitter.emit<event::Shutdown>({event::Shutdown::Terminated});
300 return true;
301 }
302 else if (engineFlags.active(Flags::PopState)) {
303 BL_LOG_INFO << "Popping state: " << states.top()->name();
304 State::Ptr prev = states.top();
305 prev->deactivate(*this);
306 states.pop();
307 if (states.empty()) { // exit if no states left
308 BL_LOG_INFO << "Final state popped, exiting";
309 eventEmitter.emit<event::Shutdown>({event::Shutdown::FinalStatePopped});
310 return true;
311 }
312 BL_LOG_INFO << "New engine state (popped): " << states.top()->name();
313 postStateChange(prev);
314 }
315 else if (engineFlags.active(Flags::_priv_ReplaceState)) {
316 BL_LOG_INFO << "New engine state (replaced): " << newState->name();
317 State::Ptr prev = states.top();
318 prev->deactivate(*this);
319 states.pop();
320 states.push(newState);
321 postStateChange(prev);
322 }
323 else if (engineFlags.active(Flags::_priv_PushState)) {
324 BL_LOG_INFO << "New engine state (pushed): " << newState->name();
325 State::Ptr& prev = states.top();
326 prev->deactivate(*this);
327 states.push(newState);
328 postStateChange(prev);
329 }
330
331 updateOuterTimer.restart();
332 loopTimer.restart();
333 }
334
335 if (!stateChanged) {
336 // signal render next frame
337 if (rendererInstance.has_value()) {
338 // run once per frame systems
339 ecsSystems.update(FrameStage::MARKER_OncePerFrame,
340 FrameStage::COUNT,
341 states.top()->systemsMask(),
342 totalDt,
343 totalDt / timeScale,
344 lag,
345 lag / timeScale);
346
347 // flush scene object changes
348 rendererInstance->syncSceneObjects();
349
350 // Copy shared data to renderer while in locked section
351 rendererInstance->copyDataFromSources();
352
353 // Flush ECS deletion queues
354 entityRegistry.flushDeletions();
355
356 // signal rendering thread to start
357 renderLock.unlock();
358 renderingCv.notify_one();
359 }
360
361 // Adhere to FPS cap
362 if (minFrameLength > 0) {
363 const float st = minFrameLength - loopTimer.getElapsedTime().asSeconds();
364 if (st > 0) sf::sleep(sf::seconds(st));
365 loopTimer.restart();
366 }
367
368 frameCount += 1.f;
369 if (rendererInstance && fpsTimer.getElapsedTime().asSeconds() >= 1.f &&
370 engineSettings.logFps()) {
371 const float fps = frameCount / fpsTimer.getElapsedTime().asSeconds();
372 rendererInstance->getWindow().getSfWindow().setTitle(
373 rendererInstance->getSettings().getWindowSettings().title() + " (" +
374 std::to_string(int(std::roundf(fps))) + " fps)");
375 frameCount = 0.f;
376 fpsTimer.restart();
377 }
378 }
379 else {
380 // Flush ECS deletion queues
381 entityRegistry.flushDeletions();
382 }
383 }
384
385 return false; // shouldn't be able to get here
386}
