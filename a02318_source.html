13void computeSingleTBN(Vertex3D& v1, Vertex3D& v2, Vertex3D& v3) {
14 const glm::vec3 edge1 = v2.pos - v1.pos;
15 const glm::vec3 edge2 = v3.pos - v1.pos;
16 const glm::vec2 deltaUV1 = v2.texCoord - v1.texCoord;
17 const glm::vec2 deltaUV2 = v3.texCoord - v1.texCoord;
18
19 const float f = 1.0f / (deltaUV1.x * deltaUV2.y - deltaUV2.x * deltaUV1.y);
20
21 glm::vec3 tangent;
22 tangent.x = f * (deltaUV2.y * edge1.x - deltaUV1.y * edge2.x);
23 tangent.y = f * (deltaUV2.y * edge1.y - deltaUV1.y * edge2.y);
24 tangent.z = f * (deltaUV2.y * edge1.z - deltaUV1.y * edge2.z);
25
26 const glm::vec3 normal = glm::normalize(glm::cross(edge1, edge2));
27
28 v1.normal = normal;
29 v2.normal = normal;
30 v3.normal = normal;
31
32 v1.tangent = tangent;
33 v2.tangent = tangent;
34 v3.tangent = tangent;
35}
