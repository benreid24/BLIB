122template<typename Payload, typename DefaultLoader, typename... Versions>
123bool VersionedSerializer<Payload, DefaultLoader, Versions...>::write(OutputStream& output,
124 const Payload& value) {
125 if (!output.write<std::uint32_t>(Header)) return false;
126 if (!output.write<std::uint32_t>(CurrentVersionNumber)) return false;
127 return LatestVersion::write(value, output);
128}

130template<typename Payload, typename DefaultLoader, typename... Versions>
131bool VersionedSerializer<Payload, DefaultLoader, Versions...>::read(InputStream& input,
132 Payload& value) {
133 constexpr bool (*loaders[])(Payload&, InputStream&) = {&Versions::read...};
134
135 std::uint32_t header = 0;
136 if (!input.peek<uint32_t>(header)) return false;
137
138 if (header == Header) {
139 input.skip(sizeof(std::uint32_t)); // skip header
140
141 std::uint32_t version;
142 if (!input.read<std::uint32_t>(version)) return false;
143 if (version >= VersionCount) {
144 BL_LOG_ERROR << "Invalid file version for payload" << typeid(Payload).name() << ": "
145 << " version=" << version << " max version=" << CurrentVersionNumber;
146 return false;
147 }
148 return loaders[version](value, input);
149 }
150 return DefaultLoader::read(value, input);
151}
