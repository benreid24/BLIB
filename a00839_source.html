228  buffer[offset + 2].texCoords = {s.left + s.width, s.top + s.height};
229  buffer[offset + 3].texCoords = {s.left, s.top + s.height};

232  halfSize.x *= scale.x;

235  {halfSize.x, -halfSize.y},
