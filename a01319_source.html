255template<typename T>
256typename BatchIndexBufferT<T>::AllocHandle BatchIndexBufferT<T>::allocate(std::uint32_t reqv,
257 std::uint32_t reqi) {
258 const bool needVertices = usedVertices + reqv > vertexCapacity();
259 const bool needIndices = usedIndices + reqi > indexCapacity();
260 if (needVertices || needIndices) {
261 const std::uint32_t vc =
262 needVertices ? std::max(vertexCapacity() * 2, usedVertices + reqv) : vertexCapacity();
263 const std::uint32_t ic =
264 needIndices ? std::max(indexCapacity() * 2, usedIndices + reqi) : indexCapacity();
265 storage.ensureSize(vc, ic);
266 }
267
268 auto it = allocations.emplace(allocations.end(), AllocInfo{});
269 AllocInfo& alloc = *it;
270 alloc.indexStart = usedIndices;
271 alloc.vertexStart = usedVertices;
272 alloc.vertexSize = reqv;
273 alloc.indexSize = reqi;
274
275 usedVertices += reqv;
276 usedIndices += reqi;
277 storage.configureWriteRange(0, usedVertices, 0, usedIndices);
278
279 return {*this, alive, it};
280}

344template<typename T>
345void BatchIndexBufferT<T>::release(typename std::list<AllocInfo>::iterator it) {
346 const AllocInfo alloc = *it;
347 allocations.erase(it);
348
349 // copy vertices to the left
350 const auto vertexStartIt = storage.vertices().begin() + alloc.vertexStart;
351 const auto nextVertexIt = vertexStartIt + alloc.vertexSize;
352 std::copy(nextVertexIt, storage.vertices().end(), vertexStartIt);
353
354 // copy indices to the left
355 const auto indexStartIt = storage.indices().begin() + alloc.indexStart;
356 const auto nextIndexIt = indexStartIt + alloc.indexSize;
357 std::copy(nextIndexIt, storage.indices().end(), indexStartIt);
358
359 // update indices of affected allocations
360 for (AllocInfo& a : allocations) {
361 if (a.vertexStart > alloc.vertexStart) { a.vertexStart -= alloc.vertexSize; }
362 if (a.indexStart > alloc.indexStart) { a.indexStart -= alloc.indexSize; }
363 }
364
365 // update values of affected indices
366 for (auto& i : storage.indices()) {
367 if (i >= alloc.vertexStart) { i -= alloc.vertexSize; }
368 }
369
370 // update metadata and commit
371 usedVertices -= alloc.vertexSize;
372 usedIndices -= alloc.indexSize;
373 storage.configureWriteRange(0, usedVertices, 0, usedIndices);
374 commit();
375}
