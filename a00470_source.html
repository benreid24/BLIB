151PipelineSpecialization& PipelineSpecialization::setShaderSpecializationValue(
152 VkShaderStageFlags stage, std::uint32_t entry, std::uint32_t offset, T value) {
153 for (auto& spec : shaderSpecializations) {
154 if (spec.stage == stage) {
155 if (offset + sizeof(T) > spec.storage.size()) {
156 BL_LOG_ERROR << "Shader specialization value exceeds buffer. Ignoring";
157 return *this;
158 }
159 if (entry >= spec.entries.size()) {
160 BL_LOG_ERROR << "Shader specialization entry exceeds declared map size. Ignoring";
161 return *this;
162 }
163 *static_cast<T*>(static_cast<void*>(spec.storage.data() + offset)) = value;
164 spec.entries[entry].offset = offset;
165 spec.entries[entry].size = sizeof(T);
166 spec.entries[entry].constantID = entry;
167 return *this;
168 }
169 }

171 BL_LOG_ERROR << "Specialization created for undeclared shader stage";
