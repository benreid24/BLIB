115  if (!inited) init();
116 
117  const auto keyIt = strKeymap.find(s);
118  if (keyIt != strKeymap.end()) { return ControlInfo{keyIt->second}; }
119 
120  const auto mbutIt = strMouseButtonMap.find(s);
121  if (mbutIt != strMouseButtonMap.end()) { return ControlInfo{mbutIt->second}; }
122 
123  auto axIt = strAxisMap.find(s);
124  if (axIt != strAxisMap.end()) { return ControlInfo{axIt->second}; }
125 
126  axIt = strAxisMap.find(s.substr(1));
127  if (axIt != strAxisMap.end()) { return ControlInfo(axIt->second, s[0] == '+'); }
128 
129  if (s == VerticalWheelUp) { return ControlInfo{sf::Mouse::VerticalWheel, true}; }
130  if (s == VerticalWheelDown) { return ControlInfo{sf::Mouse::VerticalWheel, false}; }
131  if (s == HorizontalWheelLeft) { return ControlInfo{sf::Mouse::HorizontalWheel, false}; }
132  if (s == HorizontalWheelRight) { return ControlInfo{sf::Mouse::HorizontalWheel, true}; }
133 
134  std::string_view view(s);
135  if (view.substr(0, JoystickButtonPrefix.size()) == JoystickButtonPrefix) {
136  std::string_view num = view.substr(JoystickButtonPrefix.size());
137  const unsigned int but = std::atoi(std::string(num).c_str());
138  if (but >= static_cast<unsigned int>(sf::Joystick::ButtonCount)) { return {}; }
139  return ControlInfo{but};
140  }
141  return {};
