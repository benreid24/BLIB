165void Animation2DSystem::doSlideshowAdd(com::Animation2DPlayer& player) {
166 // determine frame offset and player index
167 const auto it = slideshowFrameMap.find(player.animation.get());
168 const bool uploadFrames = it == slideshowFrameMap.end();
169 const std::uint32_t offset =
170 uploadFrames ?
171 slideshowFrameRangeAllocator.alloc(player.animation->frameCount()).range.start :
172 it->second;
173 const std::uint32_t index = slideshowPlayerIds.allocate();
174 player.playerIndex = index;
175
176 // add frame offset and current frame to SSBO's
177 slideshowFrameOffsetSSBO.ensureSize(index + 1);
178 if (slideshowPlayerCurrentFrameSSBO.ensureSize(index + 1)) {
179 players->forEach([this](ecs::Entity, com::Animation2DPlayer& p) {
180 if (p.framePayload.valid()) {
181 slideshowPlayerCurrentFrameSSBO.assignRef(p.framePayload, p.playerIndex);
182 }
183 });
184 }
185 else { slideshowPlayerCurrentFrameSSBO.assignRef(player.framePayload, player.playerIndex); }
186 slideshowFrameOffsetSSBO[index] = offset;
187 slideshowPlayerCurrentFrameSSBO[index] = player.currentFrame;
188 slideshowFrameOffsetSSBO.expandTransferRange(index, 1);
189
190 // fetch texture to convert texCoords and set texture id
191 rc::res::TextureRef texture = renderer.texturePool().getOrLoadTexture(
192 player.animation->resolvedSpritesheet(),
193 renderer.vulkanState().samplerCache.noFilterEdgeClamped()); // TODO - parameterize?
194 player.texture = texture;
195 slideshowTextureSSBO.ensureSize(index + 1);
196 slideshowTextureSSBO[index] = texture.id();
197 slideshowTextureSSBO.expandTransferRange(index, 1);
198
199 // add frames to SSBO if required
200 if (uploadFrames) {
201 slideshowFrameMap[player.animation.get()] = offset;
202 slideshowFramesSSBO.ensureSize(offset + player.animation->frameCount());
203 for (std::size_t i = 0; i < player.animation->frameCount(); ++i) {
204 const auto& src = player.animation->getFrame(i).shards.front();
205 auto& frame = slideshowFramesSSBO[offset + i];
206 const sf::FloatRect& tex = src.normalizedSource;
207 frame.opacity = static_cast<float>(src.alpha) / 255.f;
208 frame.texCoords[0] = texture->convertCoord({tex.left, tex.top});
209 frame.texCoords[1] = texture->convertCoord({tex.left + tex.width, tex.top});
210 frame.texCoords[2] =
211 texture->convertCoord({tex.left + tex.width, tex.top + tex.height});
212 frame.texCoords[3] = texture->convertCoord({tex.left, tex.top + tex.height});
213 }
214 slideshowFramesSSBO.expandTransferRange(offset, player.animation->frameCount());
215 }
216 slideshowDataRefCounts[player.animation.get()] += 1;
217
218 // mark that descriptors need to be reset
219 slideshowRefreshRequired = rc::Config::MaxConcurrentFrames;
220}

222void Animation2DSystem::doSlideshowFree(const com::Animation2DPlayer& player) {
223 slideshowPlayerIds.release(player.playerIndex);
224
225 const auto refCountIt = slideshowDataRefCounts.find(player.animation.get());
226 if (refCountIt == slideshowDataRefCounts.end()) {
227 throw std::runtime_error("Player component removed without having been added");
228 }
229 refCountIt->second -= 1;
230
231 if (refCountIt->second == 0) {
232 slideshowDataRefCounts.erase(refCountIt);
233 const auto offsetIt = slideshowFrameMap.find(player.animation.get());
234 if (offsetIt == slideshowFrameMap.end()) {
235 throw std::runtime_error("Failed to find cached frame offset for animation");
236 }
237
238 slideshowFrameRangeAllocator.release(
239 {offsetIt->second, static_cast<std::uint32_t>(player.animation->frameCount())});
240 slideshowFrameMap.erase(offsetIt);
241 }

350Animation2DSystem::VertexAnimation::VertexAnimation(rc::Renderer& renderer,
351 const gfx::a2d::AnimationData& anim)
352: useCount(0) {
353 unsigned int shardCount = 0;
354 for (unsigned int i = 0; i < anim.frameCount(); ++i) {
355 shardCount += anim.getFrame(i).shards.size();
356 }
357 if (anim.frameCount() > 0) {
358 frameToIndices.resize(anim.frameCount(), {});
359
360 // allocate buffer
361 indexBuffer.create(renderer.vulkanState(), shardCount * 4, shardCount * 6);
362
363 // fetch texture to convert texCoords
364 rc::res::TextureRef texture =
365 renderer.texturePool().getOrLoadTexture(anim.resolvedSpritesheet());
366
367 // populate buffer
368 std::uint32_t vi = 0;
369 std::uint32_t ii = 0;
370 for (unsigned int i = 0; i < anim.frameCount(); ++i) {
371 const auto& frame = anim.getFrame(i);
372
373 // cache draw params
374 frameToIndices[i].vertexStart = vi;
375 frameToIndices[i].indexStart = ii;
376 frameToIndices[i].indexCount = frame.shards.size() * 6;
377
378 for (unsigned int j = 0; j < frame.shards.size(); ++j) {
379 const auto& shard = frame.shards[j];
380
381 // compute shard transform
382 const glm::vec2 size(static_cast<float>(shard.source.width),
383 static_cast<float>(shard.source.height));
384 com::Transform2D transform;
385 transform.setOrigin(anim.shardsAreCentered() ? size * 0.5f : glm::vec2{0.f, 0.f});
386 transform.setRotation(shard.rotation);
387 transform.setPosition({shard.offset.x, shard.offset.y});
388 transform.setScale({shard.scale.x, shard.scale.y});
389 const glm::mat4 tmat = transform.getLocalTransform();
390
391 // set draw indices (CW winding)
392 const auto ibase = ii + j * 6;
393 const auto vbase = vi + j * 4;
394 indexBuffer.indices()[ibase + 0] = vbase + 0;
395 indexBuffer.indices()[ibase + 1] = vbase + 1;
396 indexBuffer.indices()[ibase + 2] = vbase + 2;
397 indexBuffer.indices()[ibase + 3] = vbase + 0;
398 indexBuffer.indices()[ibase + 4] = vbase + 2;
399 indexBuffer.indices()[ibase + 5] = vbase + 3;
400
401 // set draw vertices
402 // color
403 for (unsigned int k = 0; k < 4; ++k) {
404 indexBuffer.vertices()[vbase + k].color =
405 glm::vec4{1.f, 1.f, 1.f, static_cast<float>(shard.alpha) / 255.f};
406 }
407
408 // pos
409 glm::vec4 src{0.f, 0.f, 0.f, 1.f};
410 indexBuffer.vertices()[vbase + 0].pos = tmat * src;
411 src.x = static_cast<float>(shard.source.width);
412 indexBuffer.vertices()[vbase + 1].pos = tmat * src;
413 src.y = static_cast<float>(shard.source.height);
414 indexBuffer.vertices()[vbase + 2].pos = tmat * src;
415 src.x = 0.f;
416 indexBuffer.vertices()[vbase + 3].pos = tmat * src;
417 if (anim.shardsAreCentered()) {
418 for (unsigned int k = 0; k < 4; ++k) {
419 indexBuffer.vertices()[vbase + k].pos -=
420 glm::vec3{size * glm::vec2{shard.scale.x, shard.scale.y} * 0.5f, 0.f};
421 }
422 }
423
424 // texture coord
425 indexBuffer.vertices()[vbase + 0].texCoord = {shard.normalizedSource.left,
426 shard.normalizedSource.top};
427 indexBuffer.vertices()[vbase + 1].texCoord = {shard.normalizedSource.left +
428 shard.normalizedSource.width,
429 shard.normalizedSource.top};
430 indexBuffer.vertices()[vbase + 2].texCoord = {
431 shard.normalizedSource.left + shard.normalizedSource.width,
432 shard.normalizedSource.top + shard.normalizedSource.height};
433 indexBuffer.vertices()[vbase + 3].texCoord = {shard.normalizedSource.left,
434 shard.normalizedSource.top +
435 shard.normalizedSource.height};
436 for (unsigned int i = 0; i < 4; ++i) {
437 indexBuffer.vertices()[vbase + i].texCoord =
438 texture->convertCoord(indexBuffer.vertices()[vbase + i].texCoord);
439 }
440 }
441
442 vi += frame.shards.size() * 4;
443 ii += frameToIndices[i].indexCount;
444 }
445 indexBuffer.queueTransfer();
446 }
