91void FinalSwapframeAsset::updateAttachments(std::uint32_t i) {
92 const bool useMsaa =
93 engine->renderer().getSettings().getAntiAliasing() != Settings::AntiAliasing::None;
94 auto& set = attachmentSets.getRaw(i);
95 auto& chain = engine->renderer().getSwapchain();
96 const unsigned int swapIndex = useMsaa ? 1 : 0;
97 const unsigned int depthIndex = useMsaa ? 2 : 1;
98 const unsigned int attachmentCount = useMsaa ? 3 : 2;
99
100 ensureSampledImage();
101 set.setAttachment(swapIndex,
102 chain.swapFrameAtIndex(i).getImage(0),
103 chain.swapFrameAtIndex(i).getImageView(0));
104 set.setAttachmentAspect(swapIndex, VK_IMAGE_ASPECT_COLOR_BIT);
105 set.setAttachment(depthIndex, depthBufferAsset->getBuffer());
106 if (useMsaa) { set.setAttachment(0, sampledImage); }
107 set.setAttachmentCount(attachmentCount);
108 set.setRenderExtent(chain.swapFrameAtIndex(i).getRenderExtent());
109 set.setOutputIndex(swapIndex);
110}

124void FinalSwapframeAsset::ensureSampledImage() {
125 const bool useMsaa =
126 engine->renderer().getSettings().getAntiAliasing() != Settings::AntiAliasing::None;
127
128 if (useMsaa) {
129 const auto size = window->getSfWindow().getSize();
130 bool transition = false;
131 if (!sampledImage.isCreated() ||
132 sampledImage.getSampleCount() !=
133 engine->renderer().getSettings().getMSAASampleCount()) {
134 sampledImage.create(
135 engine->renderer(),
136 {.type = vk::ImageOptions::Type::Image2D,
137 .format = engine->renderer().getSwapchain().swapImageFormat(),
138 .usage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT | VK_IMAGE_USAGE_SAMPLED_BIT,
139 .extent = {size.x, size.y},
140 .aspect = VK_IMAGE_ASPECT_COLOR_BIT,
141 .allocFlags = VMA_ALLOCATION_CREATE_DEDICATED_MEMORY_BIT,
142 .samples = engine->renderer().getSettings().getMSAASampleCount()});
143 transition = true;
144 }
145 else if (sampledImage.getSize().width != size.x ||
146 sampledImage.getSize().height != size.y) {
147 sampledImage.resize({size.x, size.y}, false);
148 transition = true;
149 }
150
151 if (transition) { sampledImage.transitionLayout(VK_IMAGE_LAYOUT_PRESENT_SRC_KHR); }
152 }
153 else { sampledImage.deferDestroy(); }
154}
