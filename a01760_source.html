56unsigned int TextEntryComponent::findCursorPosition(const sf::Vector2f& overlayPos) const {
57 const auto result = text.findCharacterAtPosition({overlayPos.x, overlayPos.y});
58 if (result.found && result.sectionIndex == 0) { return result.characterIndex; }
59
60 // determine clicked line and if at start or end
61 const TextEntry& owner = getOwnerAs<TextEntry>();
62 const glm::vec2 gp = text.getTransform().getGlobalPosition();
63 const sf::Vector2f lp = overlayPos - sf::Vector2f(gp.x, gp.y);
64 const float lineHeight = text.getFont().getLineSpacing(text.getSection().getCharacterSize());
65
66 if (lp.y < 0.f) { return 0; }
67 if (lp.y > lineHeight * static_cast<float>(owner.getLineCount())) {
68 return owner.getInput().size();
69 }
70
71 const unsigned int line = std::floor(lp.y / lineHeight) + 1.f;
72 if (lp.x < owner.getAcquisition().width * 0.5f) { return owner.getLineStartIndex(line); }
73 else { return owner.getLineEndIndex(line); }
74}

128void TextEntryComponent::positionItems() {
129 constexpr float Padding = 2.f;
130
131 TextEntry& owner = getOwnerAs<TextEntry>();
132 const sf::FloatRect& acq = owner.getAcquisition();
133 const sf::FloatRect bounds = text.getLocalBounds();
134 const glm::vec2 cpos = text.findCharacterPosition(0, owner.getCursorPosition());
135
136 glm::vec2 textOffset(0.f, 0.f);
137 if ((bounds.width + bounds.left) > acq.width || (bounds.top + bounds.height) > acq.height) {
138 const glm::vec2 gcpos = cpos + text.getTransform().getLocalPosition();
139 const sf::Glyph& g = text.getFont().getGlyph(
140 text.getSection().getString()[owner.getCursorPosition()],
141 text.getSection().getCharacterSize(),
142 (text.getSection().getStyle() & sf::Text::Style::Bold) == sf::Text::Style::Bold,
143 text.getSection().getOutlineThickness());
144
145 const float right = acq.width - Padding * 2.f;
146 if (gcpos.x < 0.f) { textOffset.x = -cpos.x; }
147 else if (cpos.x > right) {
148 textOffset.x = right - cpos.x - g.bounds.width - g.advance - CaratWidth;
149 }
150
151 const float lineHeight =
152 text.getFont().getLineSpacing(text.getSection().getCharacterSize());
153 const float y = static_cast<float>(owner.getCurrentLine()) * lineHeight;
154 const float by = y + lineHeight;
155 const float bottom = acq.height - Padding * 2.f;
156 if (y < 0.f) { textOffset.y = -y; }
157 else if (by > bottom) { textOffset.y = bottom - by; }
158 }
159
160 text.getTransform().setPosition(glm::vec2{Padding, Padding} + textOffset);
161 carat.getTransform().setPosition(cpos);
162}
