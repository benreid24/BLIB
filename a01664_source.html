46Settings& Settings::fromConfig() {
47 updateTime = Configuration::getOrDefault<float>(UpdatePeriodKey, updateTime);
48 maxFps = Configuration::getOrDefault<float>(MaxFpsKey, maxFps);
49 allowVariableInterval =
50 Configuration::getOrDefault<bool>(VariableTimestepKey, allowVariableInterval);
51 loggingFps = Configuration::getOrDefault<bool>(LogFpsKey, loggingFps);
52
53 if (Configuration::getOrDefault<unsigned int>(WindowParameters::WindowWidthKey, 0) != 0 ||
54 windowParams.has_value()) {
55 if (!windowParams.has_value()) { windowParams.emplace(); }
56 windowParams.value().fromConfig();
57 }
58
59 return *this;
60}
