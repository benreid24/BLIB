49template<typename TShaderResource>
50sr::Key<TShaderResource> makeShaderResourceKey(std::string_view assetTag, std::string_view purpose,
51 unsigned int i) {
52 std::string key = std::string(assetTag);
53 if (!purpose.empty()) { key += "_" + std::string(purpose); }
54 if (i > 0) { key += "_" + std::to_string(i); }
55 return sr::Key<TShaderResource>({key.data(), key.size()});

197 virtual void onResize(glm::u32vec2 newSize) override {
198 switch (Traits::Size.type) {
199 case sri::TargetSize::FixedSize:
200 // bail if our size already matches
201 if (cachedScissor.extent.width == Traits::Size.size.x &&
202 cachedScissor.extent.height == Traits::Size.size.y) {
203 return;
204 }

208 cachedViewport.height = static_cast<float>(Traits::Size.size.y);

213 cachedScissor.extent.height = newSize.y;

222 void createAttachments() {
223 if (renderer) {
224 const bool createResolve =
225 ResolveAttachmentCount > 0 &&
226 renderer->getSettings().getAntiAliasing() != Settings::AntiAliasing::None;
227 const VkSampleCountFlagBits sampleCount =
228 UsesMSAA ? renderer->getSettings().getMSAASampleCount() : VK_SAMPLE_COUNT_1_BIT;
229
230 attachmentSet.setRenderExtent(cachedScissor.extent);
231
232 if (createResolve) {
233 resolveImages.create(*renderer,
234 ResolveAttachmentCount,
235 {cachedScissor.extent.width, cachedScissor.extent.height},
236 Traits::Formats.data(),
237 Traits::Usages.data(),
238 VK_SAMPLE_COUNT_1_BIT);
239 attachmentSet.setOutputIndex(ResolveAttachmentCount);
240 }
241 else if (UsesMSAA) {
242 resolveImages.deferDestroy();
243 attachmentSet.setOutputIndex(0);
244 }
245
246 std::uint32_t ac = Traits::AttachmentCount;
247 attachmentSet.copy(images->getImages().attachmentSet(), Traits::AttachmentCount);
248 if (createResolve) {
249 attachmentSet.setAttachments(ac, resolveImages);
250 ac += ResolveAttachmentCount;
251 }
252 if constexpr (DepthAttachment == DepthAttachmentType::SharedDepthBuffer) {
253 depthBufferAsset->ensureValid(
254 {cachedScissor.extent.width, cachedScissor.extent.height}, sampleCount);
255 attachmentSet.setAttachment(ac, depthBufferAsset->getBuffer());
256 depthBufferView = depthBufferAsset->getBuffer().getView();
257 ++ac;
258 }
259 attachmentSet.setAttachmentCount(ac);
260 framebuffer.create(*renderer, renderPass, attachmentSet);
261 }
262 }

264 virtual void process(const event::SettingsChanged& event) override {
265 if (event.setting == event::SettingsChanged::Setting::AntiAliasing && UsesMSAA) {
266 createAttachments();
267 }
268 }
269
270 virtual void process(const event::TextureFormatChanged& event) override {
271 bool needsUpdate = false;
272 for (std::uint32_t i = 0; i < Traits::AttachmentCount; ++i) {
273 if (Traits::Formats[i] == event.semanticFormat) {
274 needsUpdate = true;
275 break;
276 }
277 }
278 if (depthBufferAsset && (event.semanticFormat == vk::SemanticTextureFormat::Depth ||
279 event.semanticFormat == vk::SemanticTextureFormat::DepthStencil)) {
280 needsUpdate = true;
281 }
282 if (needsUpdate) { createAttachments(); }
283 }
