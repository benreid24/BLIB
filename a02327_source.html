110 playing = true;
111 if (!paused) {

113 currentIndex = 0;

123 paused = false;

190void Playlist::shuffle() {
191 const std::vector<std::string> order(std::move(songs));
192 ctr::FastEraseVector<std::size_t> indices;
193 indices.reserve(songs.size());
194 songs.reserve(order.size());
195 songs.clear();
196
197 for (std::size_t i = 0; i < order.size(); ++i) { indices.push_back(i); }
198
199 while (!indices.empty()) {
200 const std::size_t i = util::Random::get<std::size_t>(0, indices.size() - 1);
201 songs.emplace_back(std::move(order[indices[i]]));
202 indices.erase(i);
203 }
204}
