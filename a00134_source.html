309 void Registry::finishComponentAdd(Entity ent, unsigned int cIndex, T* component) {
310  bl::event::Dispatcher::dispatch<event::ComponentAdded<T>>({ent, *component});
311  ComponentMask::SimpleMask& mask = entityMasks[ent.getIndex()];
312  const ComponentMask::SimpleMask ogMask = mask;
313  ComponentMask::add(mask, cIndex);
314  for (auto& view : views) {
315  if (view->mask.passes(mask)) { view->tryAddEntity(ent); }
316  else if (view->mask.passes(ogMask) && ComponentMask::has(view->mask.excluded, cIndex)) {
317  view->removeEntity(ent);
318  }
319  }
320 
321  // notify pools of parent set
322  auto& pool = getPool<T>();
323  for (Entity child : parentGraph.getChildren(ent)) {
324  const std::uint32_t ic = child.getIndex();
325  const ComponentMask mask{.required = entityMasks[ic]};
326  if (mask.contains(pool.ComponentIndex)) { pool.onParentSet(child, ent); }

343 
344  auto& pool = getPool<T>();
345 
346  // notify pools of parent remove on children
347  for (Entity child : parentGraph.getChildren(ent)) {
348  const std::uint32_t ic = child.getIndex();
349  const ComponentMask mask{.required = entityMasks[ic]};
350  if (mask.contains(pool.ComponentIndex)) {
