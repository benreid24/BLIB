376 Partition* p = findLeaf(oldPos);
377 for (unsigned int i = 0; i < p->entities.size(); ++i) {
378 if (p->entities[i].first == oldPos) {

417template<typename TCoord, typename TPayload>
418typename QuadTree<TCoord, TPayload>::ResultSet QuadTree<TCoord, TPayload>::getInArea(
419 const Area& area) {
420 ResultSet result;
421 result.leafs.reserve(10);
422
423 std::vector<Partition*> toVisit;
424 toVisit.reserve(10);
425 toVisit.push_back(top);
426
427 while (!toVisit.empty()) {
428 Partition* current = toVisit.back();
429 toVisit.pop_back();
430
431 if (current->isLeaf()) {
432 if (!current->entities.empty()) result.leafs.push_back(current);
433 }
434 else {
435 toVisit.reserve(toVisit.size() + 4);
436 for (unsigned int i = 0; i < 4; ++i) {
437 if (current->children[i]->area.intersects(area)) {
438 toVisit.push_back(current->children[i]);
439 }
440 }
441 }
442 }
443
444 return result;
445}

537template<typename TCoord, typename TPayload>
538void QuadTree<TCoord, TPayload>::Partition::tryCollapseChildren(std::size_t maxLoad) {
539 const std::size_t load = children[0]->entities.size() + children[1]->entities.size() +
540 children[2]->entities.size() + children[3]->entities.size();
541 if (load <= maxLoad) {
542 entities.reserve(load);
543 entities.insert(entities.end(),
544 std::make_move_iterator(children[0]->entities.begin()),
545 std::make_move_iterator(children[0]->entities.end()));
546 entities.insert(entities.end(),
547 std::make_move_iterator(children[1]->entities.begin()),
548 std::make_move_iterator(children[1]->entities.end()));
549 entities.insert(entities.end(),
550 std::make_move_iterator(children[2]->entities.begin()),
551 std::make_move_iterator(children[2]->entities.end()));
552 entities.insert(entities.end(),
553 std::make_move_iterator(children[3]->entities.begin()),
554 std::make_move_iterator(children[3]->entities.end()));
555 delete children[0];
556 delete children[1];
557 delete children[2];
558 delete children[3];
559 children[0] = nullptr;
560 children[1] = nullptr;
561 children[2] = nullptr;
562 children[3] = nullptr;

567template<typename TCoord, typename TPayload>
568void QuadTree<TCoord, TPayload>::Partition::partitionChildren(std::size_t maxLoad) {
569 children[TopLeft] = new Partition(this, area.topLeft());
570 children[BottomLeft] = new Partition(this, area.bottomLeft());
571 children[TopRight] = new Partition(this, area.topRight());
572 children[BottomRight] = new Partition(this, area.bottomRight());
573
574 for (auto&& pair : entities) {
575 children[QuadTree<TCoord, TPayload>::index(area, pair.first)]->entities.emplace_back(
576 std::move(pair));
577 }
578 entities.clear();
579
580 if (children[0]->entities.size() > maxLoad) children[0]->partitionChildren(maxLoad);
581 if (children[1]->entities.size() > maxLoad) children[1]->partitionChildren(maxLoad);
582 if (children[2]->entities.size() > maxLoad) children[2]->partitionChildren(maxLoad);
583 if (children[3]->entities.size() > maxLoad) children[3]->partitionChildren(maxLoad);
584}
