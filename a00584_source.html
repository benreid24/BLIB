495 void DefaultRenderer::renderTooltip(sf::RenderTarget& target, sf::RenderStates states,
496  const Element* e, const sf::Vector2f& mousePos) const {
497  constexpr float MinWidth = 140.f;
498  constexpr float Padding = 3.f;
499  static const sf::Vector2f PaddingOffset(Padding, Padding);
500  constexpr float MaxWidth = 250.f;
501 
502  const std::string& t = e->getTooltip();

504  const sf::View& v = target.getView();
505 
506  float left = v.getCenter().x - v.getSize().x * 0.5f;
507  float hs = v.getSize().x - (mousePos.x - left);
508  left = mousePos.x;
509  if (hs < MinWidth) {
510  const float d = MinWidth - hs;
511  left -= d;
512  hs += d;
513  }
514  hs = std::min(hs, MaxWidth);
515 
516  std::string content;
517  std::string word;
518  content.reserve(t.size() + 4);
519  word.reserve(12);
520  rt.setPosition(0.f, 0.f);
521  const auto addWord = [&word, &content, &rt, hs]() {
522  const std::string attempt = content + " " + word;
523  rt.setString(attempt);
524  char sep = ' ';
525  if (rt.getGlobalBounds().width + rt.getGlobalBounds().left > hs - Padding * 2.f) {
526  sep = '\n';
527  }
528  if (!content.empty()) content.push_back(sep);
529  content += word;
530  word.clear();
531  };
532  for (char c : t) {
533  if (std::isspace(c)) { addWord(); }
534  else {
535  word.push_back(c);
536  }
537  }
538  if (!word.empty()) { addWord(); }
539  rt.setString(content);
540 
541  const sf::Vector2f size(rt.getGlobalBounds().left + rt.getGlobalBounds().width + 2.f * Padding,
542  rt.getGlobalBounds().top + rt.getGlobalBounds().height + 2.f * Padding);
543  const float top = v.getCenter().y - v.getSize().y * 0.5f;
544  float y = mousePos.y - rt.getGlobalBounds().top - rt.getGlobalBounds().height - Padding * 2.f;
545  if (y < top) { y = top + Padding; }
546  const sf::Vector2f pos(left, y);
547 
548  static sf::RectangleShape rect;

550  rect.setPosition(pos);
551  rect.setFillColor(sf::Color(230, 230, 170));
552  rect.setOutlineColor(sf::Color::Black);
