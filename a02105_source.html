56 std::unique_lock lock(progressMutex);

88void TextureExport::performCopy() {
89 // create resources
90 commandPool = vulkanState.createCommandPool(VK_COMMAND_POOL_CREATE_TRANSIENT_BIT);
91 VkCommandBufferAllocateInfo cbAllocInfo{};
92 cbAllocInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_ALLOCATE_INFO;
93 cbAllocInfo.level = VK_COMMAND_BUFFER_LEVEL_PRIMARY;
94 cbAllocInfo.commandPool = commandPool;
95 cbAllocInfo.commandBufferCount = 1;
96 vkCheck(vkAllocateCommandBuffers(vulkanState.device, &cbAllocInfo, &commandBuffer));
97
98 VkFenceCreateInfo createFence{};
99 createFence.sType = VK_STRUCTURE_TYPE_FENCE_CREATE_INFO;
100 createFence.flags = 0;
101 vkCheck(vkCreateFence(vulkanState.device, &createFence, nullptr, &fence));
102
103 // create staging image to copy to
104 vulkanState.createImage(srcExtent.width,
105 srcExtent.height,
106 OutputFormat,
107 VK_IMAGE_TILING_LINEAR,
108 VK_IMAGE_USAGE_TRANSFER_SRC_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT,
109 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT |
110 VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
111 &destImage,
112 &destAlloc,
113 &destAllocInfo,
114 VMA_ALLOCATION_CREATE_HOST_ACCESS_RANDOM_BIT);
115 // Begin recording commands
116 VkCommandBufferBeginInfo begin{};
117 begin.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
118 begin.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
119 vkCheck(vkBeginCommandBuffer(commandBuffer, &begin));
120
121 // transition image to copy source layout
122 vulkanState.transitionImageLayout(
123 commandBuffer, srcImage, srcLayout, VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL);
124
125 // transition dest image to dest layout
126 vulkanState.transitionImageLayout(
127 commandBuffer, destImage, VK_IMAGE_LAYOUT_UNDEFINED, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL);
128
129 const VkFormatProperties srcFormatProps = vulkanState.getFormatProperties(srcFormat);
130 const VkFormatProperties convFormatProps = vulkanState.getFormatProperties(OutputFormat);
131
132 // blit if supported otherwise copy
133 if ((srcFormatProps.optimalTilingFeatures & VK_FORMAT_FEATURE_BLIT_SRC_BIT) != 0 &&
134 (convFormatProps.linearTilingFeatures & VK_FORMAT_FEATURE_BLIT_DST_BIT) != 0) {
135 const VkOffset3D blitSize{static_cast<std::int32_t>(srcExtent.width),
136 static_cast<std::int32_t>(srcExtent.height),
137 1};
138 VkImageBlit imageBlitRegion{};
139 imageBlitRegion.srcSubresource.aspectMask = srcAspect;
140 imageBlitRegion.srcSubresource.layerCount = 1;
141 imageBlitRegion.srcOffsets[1] = blitSize;
142 imageBlitRegion.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
143 imageBlitRegion.dstSubresource.layerCount = 1;
144 imageBlitRegion.dstOffsets[1] = blitSize;
145
146 vkCmdBlitImage(commandBuffer,
147 srcImage,
148 VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
149 destImage,
150 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
151 1,
152 &imageBlitRegion,
153 VK_FILTER_NEAREST);
154 }
155 else {
156 // check if the format requires swizzling
157 requiresManualConversion = requiresSwizzle(srcFormat);
158
159 VkImageCopy imageCopyRegion{};
160 imageCopyRegion.srcSubresource.aspectMask = srcAspect;
161 imageCopyRegion.srcSubresource.layerCount = 1;
162 imageCopyRegion.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
163 imageCopyRegion.dstSubresource.layerCount = 1;
164 imageCopyRegion.extent = srcExtent;
165
166 vkCmdCopyImage(commandBuffer,
167 srcImage,
168 VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
169 destImage,
170 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
171 1,
172 &imageCopyRegion);
173 }
174
175 // insert barrier to block until copy is complete
176 VkImageMemoryBarrier barrier{};
177 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
178 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL;
179 barrier.newLayout = srcLayout;
180 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
181 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
182 barrier.image = srcImage;
183 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
184 barrier.subresourceRange.baseMipLevel = 0;
185 barrier.subresourceRange.levelCount = 1;
186 barrier.subresourceRange.baseArrayLayer = 0;
187 barrier.subresourceRange.layerCount = 1;
188 barrier.srcAccessMask = VK_ACCESS_TRANSFER_READ_BIT;
189 barrier.dstAccessMask = VK_ACCESS_MEMORY_WRITE_BIT;
190 vkCmdPipelineBarrier(commandBuffer,
191 VK_PIPELINE_STAGE_TRANSFER_BIT,
192 VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT,
193 0,
194 0,
195 nullptr,
196 0,
197 nullptr,
198 1,
199 &barrier);
200
201 // transition dest image to general format
202 VkImageMemoryBarrier destBarrier{};
203 destBarrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
204 destBarrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
205 destBarrier.newLayout = VK_IMAGE_LAYOUT_GENERAL;
206 destBarrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
207 destBarrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
208 destBarrier.image = destImage;
209 destBarrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
210 destBarrier.subresourceRange.baseMipLevel = 0;
211 destBarrier.subresourceRange.levelCount = 1;
212 destBarrier.subresourceRange.baseArrayLayer = 0;
213 destBarrier.subresourceRange.layerCount = 1;
214 destBarrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
215 destBarrier.dstAccessMask = VK_ACCESS_MEMORY_READ_BIT;
216 vkCmdPipelineBarrier(commandBuffer,
217 VK_PIPELINE_STAGE_TRANSFER_BIT,
218 VK_PIPELINE_STAGE_TRANSFER_BIT,
219 0,
220 0,
221 nullptr,
222 0,
223 nullptr,
224 1,
225 &destBarrier);
226
227 // submit commands
228 vkCheck(vkEndCommandBuffer(commandBuffer));
229 VkSubmitInfo submit{};
230 submit.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
231 submit.pCommandBuffers = &commandBuffer;
232 submit.commandBufferCount = 1;
233 vulkanState.submitCommandBuffer(submit, fence);
234}
