54void Canvas::setScale() {
55 const sf::Vector2f origSize(textureSize);
56 sf::Vector2f area = size.value_or(origSize);
57 if (fillAcq) {
58 area.x = getAcquisition().width;
59 area.y = getAcquisition().height;
60 }
61 scale.x = area.x / origSize.x;
62 scale.y = area.y / origSize.y;
63 if (fillAcq && maintainAR) {
64 scale.x = std::min(scale.x, scale.y);
65 scale.y = scale.x;
66 }
67 offset = RenderSettings::calculatePosition(
68 getRenderSettings().horizontalAlignment.value_or(RenderSettings::Center),
69 getRenderSettings().horizontalAlignment.value_or(RenderSettings::Center),
70 getAcquisition(),
71 {origSize.x * scale.x, origSize.y * scale.y});
72
73 if (getComponent()) { getComponent()->onElementUpdated(); }
74}
