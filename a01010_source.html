660template<typename T>
661void Registry::finishComponentAdd(Entity ent, unsigned int cIndex, T* component) {
662 bl::event::Dispatcher::dispatch<event::ComponentAdded<T>>({ent, *component});
663 ComponentMask::SimpleMask& mask = entityMasks[ent.getIndex()];
664 const ComponentMask::SimpleMask ogMask = mask;
665 ComponentMask::add(mask, cIndex);
666 for (auto& view : views) {
667 if (view->mask.passes(mask)) { view->tryAddEntity(ent); }
668 else if (view->mask.passes(ogMask) && ComponentMask::has(view->mask.excluded, cIndex)) {
669 view->removeEntity(ent);
670 }
671 }
672
673 // ensure parent traits are up to date
674 auto& pool = getPool<T>();
675 for (Entity child : parentGraph.getChildren(ent)) {
676 const std::uint64_t ic = child.getIndex();
677 const ComponentMask mask{.required = entityMasks[ic]};
678 if (mask.contains(pool.ComponentIndex)) { pool.onParentSet(child, ent); }
679 }
680 const Entity parent = parentGraph.getParent(ent);
681 if (parent != InvalidEntity) {
682 const std::uint64_t ip = parent.getIndex();
683 const ComponentMask mask{.required = entityMasks[ip]};
684 if (mask.contains(pool.ComponentIndex)) { pool.onParentSet(ent, parent); }
685 }
686}

716template<typename T>
717void Registry::removeComponent(
718 Entity ent,
719 const Transaction<tx::EntityUnlocked, tx::ComponentRead<>, tx::ComponentWrite<T>>&) {
720 if (!entityExistsLocked(ent)) { return; }
721
722 auto& pool = getPool<T>();
723 const auto i = ent.getIndex();
724 if (i >= entityMasks.size()) { return; }
725 ComponentMask::SimpleMask& mask = entityMasks[i];
726 if (!ComponentMask::contains(mask, pool.ComponentIndex)) { return; }
727
728 // notify pools of parent remove
729 for (Entity child : parentGraph.getChildren(ent)) {
730 const std::uint32_t ic = child.getIndex();
731 const ComponentMask mask{.required = entityMasks[ic]};
732 if (mask.contains(pool.ComponentIndex)) { pool.onParentRemove(ent, child); }
733 }
734 const Entity parent = parentGraph.getParent(ent);
735 if (parent != InvalidEntity) {
736 const std::uint32_t ip = parent.getIndex();
737 const ComponentMask mask{.required = entityMasks[ip]};
738 if (mask.contains(pool.ComponentIndex)) { pool.onParentRemove(parent, ent); }
739 }
740
741 // do remove
742 void* com = pool.queueRemove(ent);
743 for (auto& view : views) {
744 if (ComponentMask::contains(view->mask.required, pool.ComponentIndex)) {
745 view->removeEntity(ent);
746 }
747 else if (ComponentMask::contains(view->mask.optional, pool.ComponentIndex)) {
748 view->nullEntityComponent(ent, com);
749 }
750 }
751 ComponentMask::remove(mask, pool.ComponentIndex);
752}

772template<typename TRequire, typename TOptional, typename TExclude>
773View<TRequire, TOptional, TExclude>* Registry::getOrCreateView() {
774 using TView = View<TRequire, TOptional, TExclude>;
775 std::unique_lock lock(viewMutex);
776
777 // find existing view
778 const std::type_index viewId = typeid(TView);
779 for (auto& view : views) {
780 if (view->id == viewId) return static_cast<TView*>(view.get());
781 }
782
783 // create new view
784 views.emplace_back(new TView(*this));
785 return static_cast<TView*>(views.back().get());
786}

788template<typename T>
789ComponentPool<T>& Registry::getPool() {
790 const std::type_index tid = typeid(T);
791 auto it = poolMap.find(tid);
792 if (it == poolMap.end()) {
793#ifdef BLIB_DEBUG
794 if (poolMap.size() >= ComponentMask::MaxComponentTypeCount) {
795 BL_LOG_CRITICAL << "Maximum component types reached with component: " << tid.name()
796 << ". Specify BLIB_ECS_USE_WIDE_MASK to increase max allowed.";
797 std::exit(1);
798 }
799#endif
800 it = poolMap.try_emplace(tid, new ComponentPool<T>(*this, poolMap.size())).first;
801 componentPools.emplace_back(it->second.get());
802 }
803
804 return static_cast<ComponentPool<T>&>(*it->second);
805}
