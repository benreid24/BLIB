168 void Menu::processEvent(const Event& event) {
169  switch (event.type) {
170  case Event::SelectorMove: {
171  if (selectedItem->attachments[event.moveEvent.direction]) {
172  Item* item = selectedItem->attachments[event.moveEvent.direction];
173  while (item) {
174  if (item->isSelectable()) {
175  selectedItem->getSignal(Item::Deselected)();
176  selectedItem = item;
177  selectedItem->getSignal(Item::Selected)();
178  refreshScroll();
179  break;
180  }
181  if (!item->allowsSelectionCrossing()) break;
182  item = item->attachments[event.moveEvent.direction];
183  }
184  }
185  } break;
186  case Event::Activate:
187  selectedItem->getSignal(Item::Activated)();
188  break;
189  case Event::SelectorLocation: {
190  const sf::Vector2f pos = event.locationEvent.position - position - offset;
191  for (const auto& item : items) {
192  if (sf::FloatRect(item->position, item->getSize()).contains(pos)) {
193  if (item->isSelectable()) {
194  selectedItem->getSignal(Item::Deselected)();
195  selectedItem = item.get();
196  selectedItem->getSignal(Item::Selected)();
197  refreshScroll();
198  }
199  }
200  }
201  } break;
202  default:
203  BL_LOG_ERROR << "Menu::processEvent received invalid Event::Type: " << event.type;
204  break;
205  }

208 void Menu::refreshPositions() {
209  if (items.empty()) return;
210 
211  std::queue<Item*> toVisit;
212  std::unordered_set<Item*> visited;
213  toVisit.emplace(items.front().get());
214  visited.insert(items.front().get());
215  items.front()->position = {0.f, 0.f};
216  sf::FloatRect bounds = {0.f, 0.f, 0.f, 0.f};
217 
218  while (!toVisit.empty()) {
219  Item* item = toVisit.front();
220  toVisit.pop();
221 
222  const sf::Vector2f size = item->getSize();
223  if (item->position.x < bounds.left) bounds.left = item->position.x;
224  if (item->position.x + size.x > bounds.width) bounds.width = item->position.x + size.x;
225  if (item->position.y < bounds.top) bounds.top = item->position.y;
226  if (item->position.y + size.y > bounds.height) bounds.height = item->position.y + size.y;
227 
228  for (unsigned int i = 0; i < Item::AttachPoint::_NUM_ATTACHPOINTS; ++i) {
229  const Item::AttachPoint ap = static_cast<Item::AttachPoint>(i);
230 
231  Item* v = item->attachments[i];
232  if (v && v->parent == ap && visited.find(v) == visited.end()) {
233  if (!v->positionOverridden) {
234  v->position = move(item->position, size, v->getSize(), ap);
235  }
236  toVisit.emplace(v);
237  visited.insert(v);
238  }
239  }
240  }
241 
242  totalSize = {bounds.width - bounds.left, bounds.height - bounds.top};
243  for (auto& item : items) {
244  item->position.x -= bounds.left;
245  item->position.y -= bounds.top;
246  }
247

249  background.setSize(visibleSize() - thick * 2.f);
250  background.setPosition(thick);

253 }
