67 void Swapchain::beginFrame(StandardAttachmentSet*& renderFrame, VkCommandBuffer& cb) {
68  // wait for prior frame
69  vkCheck(vkWaitForFences(
70  vulkanState.device, 1, &frameData.current().commandBufferFence, VK_TRUE, UINT64_MAX));
71  vkCheck(vkResetFences(vulkanState.device, 1, &frameData.current().commandBufferFence));
72 
73  // recreate if out of date
74  if (outOfDate) { recreate(); }
75 
76  // acquire next image
77  VkResult acquireResult;
78  do {
79  acquireResult = vkAcquireNextImageKHR(vulkanState.device,
80  swapchain,
81  UINT64_MAX,
82  frameData.current().imageAvailableSemaphore,
83  VK_NULL_HANDLE,
84  &currentImageIndex);
85  if (acquireResult == VK_ERROR_OUT_OF_DATE_KHR) { recreate(); }
86  } while (acquireResult == VK_ERROR_OUT_OF_DATE_KHR);
87  if (acquireResult != VK_SUCCESS && acquireResult != VK_SUBOPTIMAL_KHR) {
88  throw std::runtime_error("Failed to acquire swapchain image");
89  }
90 
91  // reset prior command buffer
92  vkCheck(vkResetCommandBuffer(frameData.current().commandBuffer, 0));
93 
94  // begin command buffer
95  VkCommandBufferBeginInfo beginInfo{};
96  beginInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
97  beginInfo.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
98  if (vkBeginCommandBuffer(frameData.current().commandBuffer, &beginInfo) != VK_SUCCESS) {
99  throw std::runtime_error("Failed to begin recording command buffer");
100  }
101 
102  // copy output
103  renderFrame = &renderFrames[currentImageIndex];
104  cb = frameData.current().commandBuffer;
105 }

107 void Swapchain::completeFrame() {
108  // end command buffer
109  vkCheck(vkEndCommandBuffer(frameData.current().commandBuffer));
110 
111  // submit to queue
112  VkSemaphore waitSemaphores[] = {frameData.current().imageAvailableSemaphore};
113  VkSemaphore signalSemaphores[] = {frameData.current().renderFinishedSemaphore};
114  VkPipelineStageFlags waitStages[] = {VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT};
115 
116  VkSubmitInfo submitInfo{};
117  submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
118  submitInfo.waitSemaphoreCount = std::size(waitSemaphores);
119  submitInfo.pWaitSemaphores = waitSemaphores;
120  submitInfo.pWaitDstStageMask = waitStages;
121  submitInfo.signalSemaphoreCount = std::size(signalSemaphores);
122  submitInfo.pSignalSemaphores = signalSemaphores;
123  submitInfo.commandBufferCount = 1;
124  submitInfo.pCommandBuffers = &frameData.current().commandBuffer;
125  if (vkQueueSubmit(
126  vulkanState.graphicsQueue, 1, &submitInfo, frameData.current().commandBufferFence) !=
127  VK_SUCCESS) {
128  throw std::runtime_error("Failed to submit draw command buffer");
129  }
130 
131  // trigger swap chain
132  VkPresentInfoKHR presentInfo{};
133  presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
134  presentInfo.waitSemaphoreCount = 1;
135  presentInfo.pWaitSemaphores = &frameData.current().renderFinishedSemaphore;
136  presentInfo.swapchainCount = 1;
137  presentInfo.pSwapchains = &swapchain;
138  presentInfo.pImageIndices = &currentImageIndex;
139  presentInfo.pResults = nullptr; // Optional
140 
141  const VkResult presentResult = vkQueuePresentKHR(vulkanState.presentQueue, &presentInfo);
142  switch (presentResult) {
143  case VK_ERROR_OUT_OF_DATE_KHR:
144  case VK_SUBOPTIMAL_KHR:
145  recreate();
146  break;
147  case VK_SUCCESS:
148  break;
149  default:
150  throw std::runtime_error("Failed to present swap chain image");
151  }
152 }

178 void Swapchain::createSwapchain() {
179  // get supported swap chain details
180  SwapChainSupportDetails swapChainSupport;
181  swapChainSupport.populate(vulkanState.physicalDevice, surface);
182  const VkSurfaceFormatKHR& surfaceFormat = swapChainSupport.swapSurfaceFormat();
183 
184  // image count in swap chain
185  std::uint32_t imageCount = swapChainSupport.capabilities.minImageCount + 1;
186  if (swapChainSupport.capabilities.maxImageCount > 0 &&
187  imageCount > swapChainSupport.capabilities.maxImageCount) {
188  imageCount = swapChainSupport.capabilities.maxImageCount;
189  }
190 
191  // swap chain create params
192  VkSwapchainCreateInfoKHR createInfo{};
193  createInfo.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
194  createInfo.surface = surface;
195  createInfo.minImageCount = imageCount;
196  createInfo.imageFormat = surfaceFormat.format;
197  createInfo.imageColorSpace = surfaceFormat.colorSpace;
198  createInfo.imageExtent = swapChainSupport.swapExtent(window.getSize());
199  createInfo.imageArrayLayers = 1;
200  createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT;
201 
202  // queue chain config
203  QueueFamilyLocator indices;
204  indices.populate(vulkanState.physicalDevice, surface);
205  std::uint32_t queueFamilies[] = {indices.graphicsFamily.value(), indices.presentFamily.value()};
206 
207  if (indices.graphicsFamily != indices.presentFamily) {
208  createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
209  createInfo.queueFamilyIndexCount = std::size(queueFamilies);
210  createInfo.pQueueFamilyIndices = queueFamilies;
211  }
212  else {
213  createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
214  createInfo.queueFamilyIndexCount = 0; // Optional
215  createInfo.pQueueFamilyIndices = nullptr; // Optional
216  }
217 
218  // more create params
219  createInfo.preTransform = swapChainSupport.capabilities.currentTransform;
220  createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR; // for other windows/applications
221  createInfo.presentMode =
222  swapChainSupport.presentationMode(!engine::Configuration::getOrDefault<bool>(
223  engine::Settings::WindowParameters::VSyncKey, true));
224  createInfo.clipped = VK_TRUE;
225  createInfo.oldSwapchain = VK_NULL_HANDLE;
226 
227  const VkResult result =
228  vkCreateSwapchainKHR(vulkanState.device, &createInfo, nullptr, &swapchain);
229  if (result != VK_SUCCESS) {
230  BL_LOG_ERROR << "Swap chain creation failed: " << result;
231  throw std::runtime_error("Failed to create swap chain");
232  }
233 
234  // Fetch images
235  VkImage images[8];
236  vkCheck(vkGetSwapchainImagesKHR(vulkanState.device, swapchain, &imageCount, images));
237  imageFormat = surfaceFormat.format;
238 
239  // create depth buffers
240  depthBuffers.resize(imageCount);
241  for (unsigned int i = 0; i < imageCount; ++i) {
242  depthBuffers[i].create(vulkanState,
243  StandardAttachmentBuffers::findDepthFormat(vulkanState),
244  VK_IMAGE_USAGE_DEPTH_STENCIL_ATTACHMENT_BIT,
245  createInfo.imageExtent);
246  }
247 
248  // assign attachment sets
249  renderFrames.resize(imageCount);
250  for (unsigned int i = 0; i < imageCount; ++i) {
251  renderFrames[i].setRenderExtent(createInfo.imageExtent);
252  renderFrames[i].setAttachments(images[i],
253  vulkanState.createImageView(images[i], imageFormat),
254  depthBuffers[i].image(),
255  depthBuffers[i].view());
256  }
257 }
