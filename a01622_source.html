33void constrainScissor(VkRect2D& scissor, const VkRect2D& limits) {
34 scissor.offset.x = std::max(scissor.offset.x, limits.offset.x);
35 scissor.offset.y = std::max(scissor.offset.y, limits.offset.y);
36
37 const int limitRight = limits.offset.x + limits.extent.width;
38 if (scissor.offset.x + scissor.extent.width > limitRight) {
39 if (limitRight >= scissor.offset.x) {
40 scissor.extent.width = limitRight - scissor.offset.x;
41 }
42 else { scissor.extent.width = 0; }
43 }
44
45 const int limitBottom = limits.offset.y + limits.extent.height;
46 if (scissor.offset.y + scissor.extent.height > limitBottom) {
47 if (limitBottom >= scissor.offset.y) {
48 scissor.extent.height = limitBottom - scissor.offset.y;
49 }
50 else { scissor.extent.height = 0; }
51 }
52}

94void OverlayScalerSystem::refreshEntity(Result& cset) {
95 com::OverlayScaler& scaler = *cset.get<com::OverlayScaler>();
96 com::Transform2D& transform = *cset.get<com::Transform2D>();
97 rc::ovy::OverlayObject& obj = *cset.get<rc::ovy::OverlayObject>();
98
99 scaler.dirty = false;
100 scaler.transformVersion = transform.getVersion();
101 scaler.cachedScaledOrigin = transform.getOrigin() * transform.getScale();
102
103 // dummy entities only get scissor, all else is skipped
104 if (cset.entity().flagSet(ecs::Flags::Dummy)) {
105 updateScissor(cset, 1.f, 1.f);
106 return;
107 }
108
109 if (!obj.overlayViewport) { return; }
110
111 const VkViewport& viewport = *obj.overlayViewport;
112 glm::vec2 parentSize = cam::OverlayCamera::getOverlayCoordinateSpace();
113 glm::vec2 parentOrigin(0.f, 0.f);
114 if (scaler.hasParent()) {
115 parentSize.x =
116 scaler.getParent().cachedObjectBounds.width * transform.getParent().getScale().x;
117 parentSize.y =
118 scaler.getParent().cachedObjectBounds.height * transform.getParent().getScale().y;
119 parentOrigin = scaler.getParent().cachedScaledOrigin;
120 }
121
122 float xScale = 1.f;
123 float yScale = 1.f;
124
125 // scale
126 switch (scaler.scaleType) {
127 case com::OverlayScaler::WidthPercent:
128 xScale = scaler.widthPercent * parentSize.x / scaler.cachedObjectBounds.width;
129 yScale = xScale;
130 break;
131
132 case com::OverlayScaler::HeightPercent:
133 yScale = scaler.heightPercent * parentSize.y / scaler.cachedObjectBounds.height;
134 xScale = yScale;
135 break;
136
137 case com::OverlayScaler::SizePercent:
138 xScale = scaler.widthPercent * parentSize.x / scaler.cachedObjectBounds.width;
139 yScale = scaler.heightPercent * parentSize.y / scaler.cachedObjectBounds.height;
140 break;
141
142 case com::OverlayScaler::PixelRatio:
143 xScale = scaler.cachedObjectBounds.width * scaler.pixelRatio / viewport.width;
144 yScale = scaler.cachedObjectBounds.height * scaler.pixelRatio / viewport.height;
145 break;
146
147 case com::OverlayScaler::LineHeight:
148 yScale = scaler.overlayRatio * parentSize.y;
149 xScale = yScale;
150 break;
151
152 case com::OverlayScaler::None:
153 default:
154 break;
155 }
156
157 if (scaler.scaleType != com::OverlayScaler::None) {
158 if (transform.getScale().x != xScale || transform.getScale().y != yScale) {
159 transform.setScale({xScale, yScale});
160 if (scaler.onScale) { scaler.onScale(); }
161 }
162 }
163 else {
164 xScale = transform.getScale().x;
165 yScale = transform.getScale().y;
166 }
167
168 // position
169 switch (scaler.posType) {
170 case com::OverlayScaler::ParentSpace:
171 transform.setPosition(scaler.parentPosition * parentSize - parentOrigin);
172 break;
173
174 case com::OverlayScaler::NoPosition:
175 default:
176 break;
177 }
178
179 // scissor
180 updateScissor(cset, xScale, yScale);
181
182 // update global size
183 const glm::vec2 pos = transform.getGlobalPosition();
184 const glm::vec2& overlaySize = cam::OverlayCamera::getOverlayCoordinateSpace();
185 scaler.cachedTargetRegion = {
186 viewport.x + viewport.width * pos.x / overlaySize.x,
187 viewport.y + viewport.height * pos.y / overlaySize.y,
188 viewport.width * (scaler.cachedObjectBounds.width * xScale) / overlaySize.x,
189 viewport.height * (scaler.cachedObjectBounds.height * yScale) / overlaySize.y};
190}

192void OverlayScalerSystem::updateScissor(Result& cset, float xScale, float yScale) {
193 rc::ovy::OverlayObject& obj = *cset.get<rc::ovy::OverlayObject>();
194 VkRect2D& scissor = obj.cachedScissor;
195 com::Transform2D& transform = *cset.get<com::Transform2D>();
196 com::OverlayScaler& scaler = *cset.get<com::OverlayScaler>();
197 const glm::vec2 pos = transform.getGlobalPosition();
198
199 switch (scaler.scissorMode) {
200 case com::OverlayScaler::ScissorSelf:
201 case com::OverlayScaler::ScissorSelfConstrained:
202 if (obj.overlayViewport) {
203 const VkViewport& viewport = *obj.overlayViewport;
204 const glm::vec2& origin = transform.getOrigin();
205 const glm::vec2 offset((origin.x - scaler.cachedObjectBounds.left) * xScale,
206 (origin.y - scaler.cachedObjectBounds.top) * yScale);
207 const glm::vec2 corner =
208 (pos - offset) / cam::OverlayCamera::getOverlayCoordinateSpace();
209
210 scissor.offset.x = viewport.x + viewport.width * corner.x;
211 scissor.offset.y = viewport.y + viewport.height * corner.y;
212 scissor.extent.width = viewport.width * (scaler.cachedObjectBounds.width * xScale) /
213 cam::OverlayCamera::getOverlayCoordinateSpace().x;
214 scissor.extent.height = viewport.height * (scaler.cachedObjectBounds.height * yScale) /
215 cam::OverlayCamera::getOverlayCoordinateSpace().y;
216
217 VkRect2D limits = makeScissor(viewport);
218 if (scaler.scissorMode == com::OverlayScaler::ScissorSelfConstrained &&
219 obj.hasParent()) {
220 limits = obj.getParent().cachedScissor;
221 }
222 constrainScissor(scissor, limits);
223 }
224 else { BL_LOG_ERROR << "Dummy entities do not support ScissorSelf modes"; }
225 break;
226
227 case com::OverlayScaler::ScissorObserver:
228 if (obj.overlayViewport) { scissor = makeScissor(*obj.overlayViewport); }
229 else { BL_LOG_ERROR << "Dummy entities do not support ScissorObserver mode"; }
230 break;
231
232 case com::OverlayScaler::ScissorInherit:
233 if (!obj.hasParent()) {
234 if (obj.overlayViewport) { obj.cachedScissor = makeScissor(*obj.overlayViewport); }
235 }
236 else { obj.cachedScissor = obj.getParent().cachedScissor; }
237 break;
238
239 case com::OverlayScaler::ScissorFixed:
240 scissor.offset.x = scaler.fixedScissor.left;
241 scissor.offset.y = scaler.fixedScissor.top;
242 scissor.extent.width = scaler.fixedScissor.width;
243 scissor.extent.height = scaler.fixedScissor.height;
244 break;
245 }
