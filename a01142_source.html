48 // FreeType callbacks that operate on a sf::InputStream
49 unsigned long read(FT_Stream rec, unsigned long offset, unsigned char* buffer,
50  unsigned long count) {
51  sf::Int64 convertedOffset = static_cast<sf::Int64>(offset);
52  sf::InputStream* stream = static_cast<sf::InputStream*>(rec->descriptor.pointer);
53  if (stream->seek(convertedOffset) == convertedOffset) {
54  if (count > 0)
55  return static_cast<unsigned long>(
56  stream->read(reinterpret_cast<char*>(buffer), static_cast<sf::Int64>(count)));
57  else
58  return 0;
59  }
60  else
61  return count > 0 ? 0 : 1; // error code is 0 if we're reading, or nonzero if we're seeking
62 }

73 // Combine outline thickness, boldness and font glyph index into a single 64-bit key
74 sf::Uint64 combine(sf::Uint8 size, float outlineThickness, bool bold, sf::Uint32 index) {
75  return (static_cast<sf::Uint64>(reinterpret<sf::Uint32>(outlineThickness)) << 32) |
76  (static_cast<sf::Uint64>(bold) << 31) | (static_cast<sf::Uint64>(size) << 32) | index;

142 bool VulkanFont::loadFromFile(const std::string& filename) {
143  needsUpload = true;
144 
145 #ifndef SFML_SYSTEM_ANDROID
146 
147  // Cleanup the previous resources
148  cleanup();
149  m_refCount = new int(1);
150 
151  // Initialize FreeType
152  // Note: we initialize FreeType for every font instance in order to avoid having a single
153  // global manager that would create a lot of issues regarding creation and destruction order.
154  FT_Library library;
155  if (FT_Init_FreeType(&library) != 0) {
156  err() << "Failed to load font \"" << filename << "\" (failed to initialize FreeType)"
157  << std::endl;
158  return false;
159  }
160  m_library = library;
161 
162  // Load the new font face from the specified file
163  FT_Face face;
164  if (FT_New_Face(static_cast<FT_Library>(m_library), filename.c_str(), 0, &face) != 0) {
165  err() << "Failed to load font \"" << filename << "\" (failed to create the font face)"
166  << std::endl;
167  return false;
168  }
169 
170  // Load the stroker that will be used to outline the font
171  FT_Stroker stroker;
172  if (FT_Stroker_New(static_cast<FT_Library>(m_library), &stroker) != 0) {
173  err() << "Failed to load font \"" << filename << "\" (failed to create the stroker)"
174  << std::endl;
175  FT_Done_Face(face);
176  return false;
177  }
178 
179  // Select the unicode character map
180  if (FT_Select_Charmap(face, FT_ENCODING_UNICODE) != 0) {
181  err() << "Failed to load font \"" << filename
182  << "\" (failed to set the Unicode character set)" << std::endl;
183  FT_Stroker_Done(stroker);
184  FT_Done_Face(face);
185  return false;
186  }
187 
188  // Store the loaded font in our ugly void* :)
189  m_stroker = stroker;
190  m_face = face;
191 
192  // Store the font information
193  m_info.family = face->family_name ? face->family_name : std::string();
194 
195  return true;
196 
197 #else
198 
199  if (m_stream) delete static_cast<priv::ResourceStream*>(m_stream);
200 
201  m_stream = new priv::ResourceStream(filename);
202  return loadFromStream(*static_cast<priv::ResourceStream*>(m_stream));
203 
204 #endif
205 }

342 const Glyph& VulkanFont::getGlyph(Uint32 codePoint, unsigned int characterSize, bool bold,
343  float outlineThickness) const {
344  // Build the key by combining the glyph index (based on code point), bold flag, and outline
345  // thickness
346  Uint64 key = combine(characterSize,
347  outlineThickness,
348  bold,
349  FT_Get_Char_Index(static_cast<FT_Face>(m_face), codePoint));
350 
351  // Search the glyph into the cache
352  GlyphTable::const_iterator it = glyphs.find(key);
353  if (it != glyphs.end()) {
354  // Found: just return it
355  return it->second;
356  }
357  else {
358  // Not found: we have to load it
359  const_cast<sf::VulkanFont*>(this)->needsUpload = true;
360  Glyph glyph = loadGlyph(codePoint, characterSize, bold, outlineThickness);
361  return glyphs.insert(std::make_pair(key, glyph)).first->second;
362  }
363 }

371 float VulkanFont::getKerning(Uint32 first, Uint32 second, unsigned int characterSize,
372  bool bold) const {
373  // Special case where first or second is 0 (null character)
374  if (first == 0 || second == 0) return 0.f;
375 
376  FT_Face face = static_cast<FT_Face>(m_face);
377 
378  if (face && setCurrentSize(characterSize)) {
379  // Convert the characters to indices
380  FT_UInt index1 = FT_Get_Char_Index(face, first);
381  FT_UInt index2 = FT_Get_Char_Index(face, second);
382 
383  // Retrieve position compensation deltas generated by FT_LOAD_FORCE_AUTOHINT flag
384  float firstRsbDelta = static_cast<float>(getGlyph(first, characterSize, bold).rsbDelta);
385  float secondLsbDelta = static_cast<float>(getGlyph(second, characterSize, bold).lsbDelta);
386 
387  // Get the kerning vector if present
388  FT_Vector kerning;
389  kerning.x = kerning.y = 0;
390  if (FT_HAS_KERNING(face))
391  FT_Get_Kerning(face, index1, index2, FT_KERNING_UNFITTED, &kerning);
392 
393  // X advance is already in pixels for bitmap fonts
394  if (!FT_IS_SCALABLE(face)) return static_cast<float>(kerning.x);
395 
396  // Combine kerning with compensation deltas and return the X advance
397  // Flooring is required as we use FT_KERNING_UNFITTED flag which is not quantized in 64
398  // based grid
399  return std::floor((secondLsbDelta - firstRsbDelta + static_cast<float>(kerning.x) + 32) /
400  static_cast<float>(1 << 6));
401  }
402  else {
403  // Invalid font
404  return 0.f;
405  }
406 }

476 void VulkanFont::cleanup() {
477  // Check if we must destroy the FreeType pointers
478  if (m_refCount) {
479  // Decrease the reference counter
480  (*m_refCount)--;
481 
482  // Free the resources only if we are the last owner
483  if (*m_refCount == 0) {
484  // Delete the reference counter
485  delete m_refCount;
486 
487  // Destroy the stroker
488  if (m_stroker) FT_Stroker_Done(static_cast<FT_Stroker>(m_stroker));
489 
490  // Destroy the font face
491  if (m_face) FT_Done_Face(static_cast<FT_Face>(m_face));
492 
493  // Destroy the stream rec instance, if any (must be done after FT_Done_Face!)
494  if (m_streamRec) delete static_cast<FT_StreamRec*>(m_streamRec);
495 
496  // Close the library
497  if (m_library) FT_Done_FreeType(static_cast<FT_Library>(m_library));
498  }
499  }
500 
501  // Reset members
502  m_library = NULL;
503  m_face = NULL;
504  m_stroker = NULL;
505  m_streamRec = NULL;
506  m_refCount = NULL;
507  glyphs.clear();
508  rows.clear();
509  nextRow = 3;
510  needsUpload = true;
511  vulkanTexture.release();
512  std::vector<Uint8>().swap(m_pixelBuffer);
513 }

516 Glyph VulkanFont::loadGlyph(Uint32 codePoint, unsigned int characterSize, bool bold,
517  float outlineThickness) const {
518  // The glyph to return
519  Glyph glyph;
520 
521  // First, transform our ugly void* to a FT_Face
522  FT_Face face = static_cast<FT_Face>(m_face);
523  if (!face) return glyph;
524 
525  // Set the character size
526  if (!setCurrentSize(characterSize)) return glyph;
527 
528  // Load the glyph corresponding to the code point
529  FT_Int32 flags = FT_LOAD_TARGET_NORMAL | FT_LOAD_FORCE_AUTOHINT;
530  if (outlineThickness != 0) flags |= FT_LOAD_NO_BITMAP;
531  if (FT_Load_Char(face, codePoint, flags) != 0) return glyph;
532 
533  // Retrieve the glyph
534  FT_Glyph glyphDesc;
535  if (FT_Get_Glyph(face->glyph, &glyphDesc) != 0) return glyph;
536 
537  // Apply bold and outline (there is no fallback for outline) if necessary -- first technique
538  // using outline (highest quality)
539  FT_Pos weight = 1 << 6;
540  bool outline = (glyphDesc->format == FT_GLYPH_FORMAT_OUTLINE);
541  if (outline) {
542  if (bold) {
543  FT_OutlineGlyph outlineGlyph = reinterpret_cast<FT_OutlineGlyph>(glyphDesc);
544  FT_Outline_Embolden(&outlineGlyph->outline, weight);
545  }
546 
547  if (outlineThickness != 0) {
548  FT_Stroker stroker = static_cast<FT_Stroker>(m_stroker);
549 
550  FT_Stroker_Set(stroker,
551  static_cast<FT_Fixed>(outlineThickness * static_cast<float>(1 << 6)),
552  FT_STROKER_LINECAP_ROUND,
553  FT_STROKER_LINEJOIN_ROUND,
554  0);
555  FT_Glyph_Stroke(&glyphDesc, stroker, true);
556  }
557  }
558 
559  // Convert the glyph to a bitmap (i.e. rasterize it)
560  // Warning! After this line, do not read any data from glyphDesc directly, use
561  // bitmapGlyph.root to access the FT_Glyph data.
562  FT_Glyph_To_Bitmap(&glyphDesc, FT_RENDER_MODE_NORMAL, 0, 1);
563  FT_BitmapGlyph bitmapGlyph = reinterpret_cast<FT_BitmapGlyph>(glyphDesc);
564  FT_Bitmap& bitmap = bitmapGlyph->bitmap;
565 
566  // Apply bold if necessary -- fallback technique using bitmap (lower quality)
567  if (!outline) {
568  if (bold) FT_Bitmap_Embolden(static_cast<FT_Library>(m_library), &bitmap, weight, weight);
569 
570  if (outlineThickness != 0)
571  err() << "Failed to outline glyph (no fallback available)" << std::endl;
572  }
573 
574  // Compute the glyph's advance offset
575  glyph.advance = static_cast<float>(bitmapGlyph->root.advance.x >> 16);
576  if (bold) glyph.advance += static_cast<float>(weight) / static_cast<float>(1 << 6);
577 
578  glyph.lsbDelta = static_cast<int>(face->glyph->lsb_delta);
579  glyph.rsbDelta = static_cast<int>(face->glyph->rsb_delta);
580 
581  unsigned int width = bitmap.width;
582  unsigned int height = bitmap.rows;
583 
584  if ((width > 0) && (height > 0)) {
585  // Leave a small padding around characters, so that filtering doesn't
586  // pollute them with pixels from neighbors
587  const unsigned int padding = 2;
588 
589  width += 2 * padding;
590  height += 2 * padding;
591 
592  // Find a good position for the new glyph into the texture
593  glyph.textureRect = findGlyphRect(width, height);
594 
595  // Make sure the texture data is positioned in the center
596  // of the allocated texture rectangle
597  glyph.textureRect.left += static_cast<int>(padding);
598  glyph.textureRect.top += static_cast<int>(padding);
599  glyph.textureRect.width -= static_cast<int>(2 * padding);
600  glyph.textureRect.height -= static_cast<int>(2 * padding);
601 
602  // Compute the glyph's bounding box
603  glyph.bounds.left = static_cast<float>(bitmapGlyph->left);
604  glyph.bounds.top = static_cast<float>(-bitmapGlyph->top);
605  glyph.bounds.width = static_cast<float>(bitmap.width);
606  glyph.bounds.height = static_cast<float>(bitmap.rows);
607 
608  // Resize the pixel buffer to the new size and fill it with transparent white pixels
609  m_pixelBuffer.resize(width * height * 4);
610 
611  Uint8* current = &m_pixelBuffer[0];
612  Uint8* end = current + width * height * 4;
613 
614  while (current != end) {
615  (*current++) = 255;
616  (*current++) = 255;
617  (*current++) = 255;
618  (*current++) = 0;
619  }
620 
621  // Extract the glyph's pixels from the bitmap
622  const Uint8* pixels = bitmap.buffer;
623  if (bitmap.pixel_mode == FT_PIXEL_MODE_MONO) {
624  // Pixels are 1 bit monochrome values
625  for (unsigned int y = padding; y < height - padding; ++y) {
626  for (unsigned int x = padding; x < width - padding; ++x) {
627  // The color channels remain white, just fill the alpha channel
628  std::size_t index = x + y * width;
629  m_pixelBuffer[index * 4 + 3] =
630  ((pixels[(x - padding) / 8]) & (1 << (7 - ((x - padding) % 8)))) ? 255 : 0;
631  }
632  pixels += bitmap.pitch;
633  }
634  }
635  else {
636  // Pixels are 8 bits gray levels
637  for (unsigned int y = padding; y < height - padding; ++y) {
638  for (unsigned int x = padding; x < width - padding; ++x) {
639  // The color channels remain white, just fill the alpha channel
640  std::size_t index = x + y * width;
641  m_pixelBuffer[index * 4 + 3] = pixels[x - padding];
642  }
643  pixels += bitmap.pitch;
644  }
645  }
646 
647  // Write the pixels to the texture
648  unsigned int x = static_cast<unsigned int>(glyph.textureRect.left) - padding;
649  unsigned int y = static_cast<unsigned int>(glyph.textureRect.top) - padding;
650  unsigned int w = static_cast<unsigned int>(glyph.textureRect.width) + 2 * padding;
651  unsigned int h = static_cast<unsigned int>(glyph.textureRect.height) + 2 * padding;
652  for (unsigned int xi = 0; xi < w; ++xi) {
653  for (unsigned int yi = 0; yi < h; ++yi) {
654  std::size_t index = xi + yi * width;
655  texture.setPixel(
656  x + xi, y + yi, sf::Color{255, 255, 255, m_pixelBuffer[index * 4 + 3]});
657  }
658  }
659  }
660 
661  // Delete the FT glyph
662  FT_Done_Glyph(glyphDesc);
663 
664  // Done :)
665  return glyph;
666 }

669 IntRect VulkanFont::findGlyphRect(unsigned int width, unsigned int height) const {
670  // Find the line that fits well the glyph
671  Row* row = NULL;
672  float bestRatio = 0;
673  for (std::vector<Row>::iterator it = rows.begin(); it != rows.end() && !row; ++it) {
674  float ratio = static_cast<float>(height) / static_cast<float>(it->height);
675 
676  // Ignore rows that are either too small or too high
677  if ((ratio < 0.7f) || (ratio > 1.f)) continue;
678 
679  // Check if there's enough horizontal space left in the row
680  if (width > texture.getSize().x - it->width) continue;
681 
682  // Make sure that this new row is the best found so far
683  if (ratio < bestRatio) continue;
684 
685  // The current row passed all the tests: we can select it
686  row = &*it;
687  bestRatio = ratio;
688  }
689 
690  // If we didn't find a matching row, create a new one (10% taller than the glyph)
691  if (!row) {
692  unsigned int rowHeight = height + height / 10;
693  while ((nextRow + rowHeight >= texture.getSize().y) || (width >= texture.getSize().x)) {
694  // Not enough space: resize the texture if possible
695  unsigned int textureWidth = texture.getSize().x;
696  unsigned int textureHeight = texture.getSize().y;
697 
698  // Make the texture 2 times bigger
699  Image newTexture;
700  newTexture.create(textureWidth * 2, textureHeight * 2, sf::Color(255, 255, 255, 0));
701  newTexture.copy(texture, 0, 0);
702  texture = std::move(newTexture);
703  }
704 
705  // We can now create the new row
706  rows.emplace_back(nextRow, rowHeight);
707  nextRow += rowHeight;
708  row = &rows.back();
709  }
710 
711  // Find the glyph's rectangle on the selected row
712  IntRect rect(Rect<unsigned int>(row->width, row->top, width, height));
713 
714  // Update the row informations
715  row->width += width;
716 
717  return rect;
718 }

721 bool VulkanFont::setCurrentSize(unsigned int characterSize) const {
722  // FT_Set_Pixel_Sizes is an expensive function, so we must call it
723  // only when necessary to avoid killing performances
724 
725  FT_Face face = static_cast<FT_Face>(m_face);
726  FT_UShort currentSize = face->size->metrics.x_ppem;
727 
728  if (currentSize != characterSize) {
729  FT_Error result = FT_Set_Pixel_Sizes(face, 0, characterSize);
730 
731  if (result == FT_Err_Invalid_Pixel_Size) {
732  // In the case of bitmap fonts, resizing can
733  // fail if the requested size is not available
734  if (!FT_IS_SCALABLE(face)) {
735  err() << "Failed to set bitmap font size to " << characterSize << std::endl;
736  err() << "Available sizes are: ";
737  for (int i = 0; i < face->num_fixed_sizes; ++i) {
738  const long size = (face->available_sizes[i].y_ppem + 32) >> 6;
739  err() << size << " ";
740  }
741  err() << std::endl;
742  }
743  else { err() << "Failed to set font size to " << characterSize << std::endl; }
744  }
745 
746  return result == FT_Err_Ok;
747  }
748 
749  return true;
750 }
