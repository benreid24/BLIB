455 typename ObjectPool<T>::Entry* ObjectPool<T>::doAdd() {
456  ++trackedSize;
457  if (next >= 0) {
458  Entry* e = &pool[next];
459  next = e->next();
460  return e;
461  }
462  else {

464  pool.emplace_back();
