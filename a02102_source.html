85void TransferEngine::executeTransfers() {
86 std::unique_lock lock(mutex);
87
88 if (immediateBucket.hasTransfers()) { immediateBucket.executeTransfers(); }

101void TransferEngine::Bucket::executeTransfers() {
102 // wait for prior transfers to complete before freeing resources
103 resetResourcesWithSync();
104
105 VkCommandBufferBeginInfo begin{};
106 begin.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
107 begin.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
108 vkCheck(vkBeginCommandBuffer(commandBuffer.current(), &begin));
109
110 // queue transfer commands
111 TransferContext context(mutex,
112 vulkanState,
113 stagingBuffers.current(),
114 stagingAllocs.current(),
115 memoryBarriers,
116 bufferBarriers,
117 imageBarriers);
118 for (Transferable* item : everyFrameItems) {
119 item->executeTransfer(commandBuffer.current(), context);
120 item->queued = false;
121 }
122 for (Transferable* item : oneTimeItems) {
123 item->executeTransfer(commandBuffer.current(), context);
124 item->queued = false;
125 }
126 oneTimeItems.clear();
127
128 // one unified sync
129 vkCmdPipelineBarrier(commandBuffer.current(),
130 VK_PIPELINE_STAGE_TRANSFER_BIT,
131 VK_PIPELINE_STAGE_ALL_GRAPHICS_BIT,
132 VK_DEPENDENCY_BY_REGION_BIT,
133 memoryBarriers.size(),
134 memoryBarriers.data(),
135 bufferBarriers.size(),
136 bufferBarriers.data(),
137 imageBarriers.size(),
138 imageBarriers.data());
139
140 // submit transfer commands
141 vkCheck(vkEndCommandBuffer(commandBuffer.current()));
142 VkSubmitInfo submit{};
143 submit.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
144 submit.pCommandBuffers = &commandBuffer.current();
145 submit.commandBufferCount = 1;
146 vkCheck(vkResetFences(vulkanState.device, 1, &fence.current()));
147 vulkanState.submitCommandBuffer(submit, fence.current());
148
149 // cleanup
150 memoryBarriers.clear();
151 bufferBarriers.clear();
152 imageBarriers.clear();
153}
