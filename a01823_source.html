150void FilePicker::onFolderClick(const std::string& f) {
151 highlight(f);
152 fileEntry->setInput(f);
153
154 if (f != clickedFile) {
155 clickedFile = f;
156 fileClickTime = timer.getElapsedTime().asSeconds();
157 }
158 else {
159 const float n = timer.getElapsedTime().asSeconds();
160 if (n - fileClickTime < DoubleClick) {
161 path.emplace_back(f);
162 populateFiles();
163 fileEntry->setInput("");

165 else { fileClickTime = n; }

184void FilePicker::onChooseClicked() {
185 if (fileEntry->getInput().empty()) {
186 dialog::tinyfd_messageBox("Error", "You must enter a filename", "ok", "error", 1);
187 return;
188 }
189
190 const std::string localpath = util::FileUtil::joinPath(buildPath(), fileEntry->getInput());
191 const std::string filename = util::FileUtil::joinPath(root, localpath);
192
193 if (util::FileUtil::directoryExists(filename)) {
194 fileClickTime = timer.getElapsedTime().asSeconds(); // trick
195 onFolderClick(fileEntry->getInput());
196 return;
197 }
198
199 if (mode == CreateNew) {
200 if (util::FileUtil::exists(filename)) {
201 if (dialog::tinyfd_messageBox("Overwrite file?",
202 (filename + " already exists, overwrite?").c_str(),
203 "yesno",
204 "warning",
205 0) != 1) {
206 return;
207 }
208 }
209 onChoose(localpath);
210 }
211 else {
212 if (mode == CreateOrPick || util::FileUtil::exists(filename)) { onChoose(localpath); }
213 else {
214 dialog::tinyfd_messageBox(
215 "Error", (fileEntry->getInput() + " does not exist").c_str(), "ok", "error", 1);
216 }
217 }

227void FilePicker::populateFiles() {
228 for (auto& pair : fileLabels) { pair.second->remove(); }
229 fileLabels.clear();
230 fileEntry->setInput("");
231 clickedFile = "";
232 fileClickTime = 0.f;
233
234 const std::string p = util::FileUtil::joinPath(root, buildPath());
235 std::vector<std::string> files, folders;
236 for (const std::string& ext : extensions) {
237 std::vector<std::string> f = util::FileUtil::listDirectory(p, ext, false);
238 files.insert(
239 files.end(), std::make_move_iterator(f.begin()), std::make_move_iterator(f.end()));
240 }
241 folders = util::FileUtil::listDirectoryFolders(p);
242 for (std::string& f : files) { f = util::FileUtil::getFilename(f); }
243 std::sort(files.begin(), files.end());
244 std::sort(folders.begin(), folders.end());
245
246 for (const std::string& f : folders) {
247 const auto onClick = [this, f](const Event&, Element*) { onFolderClick(f); };
248
249 Box::Ptr row = Box::create(LinePacker::create(LinePacker::Horizontal, 6));
250 row->getSignal(Event::LeftClicked).willAlwaysCall(onClick);

252 Icon::Ptr icon = Icon::create(Icon::Type::Folder, {IconSize, IconSize});
