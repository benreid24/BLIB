144void Outline3DTask::addObject(ecs::Entity entity, com::Rendered* drawable, com::Outline* outline) {
145 vk::Pipeline* pipeline =
146 drawable->getComponent()->getCurrentPipeline()->getPipeline(cfg::RenderPhases::Outline);
147 if (pipeline == standardPipeline.getGraphicsPipeline()) {
148 auto& list = drawable->getUpdateSpeed() == UpdateSpeed::Dynamic ? outlinedObjectsDynamic :
149 outlinedObjectsStatic;
150 list.emplace_back(Outlined{.drawable = drawable, .outline = outline});
151 }
152 else if (pipeline == skinnedPipeline.getGraphicsPipeline()) {
153 auto& list = drawable->getUpdateSpeed() == UpdateSpeed::Dynamic ?
154 outlinedSkinnedObjectsDynamic :
155 outlinedSkinnedObjectsStatic;
156 list.emplace_back(Outlined{.drawable = drawable, .outline = outline});
157 }
158 else {
159 BL_LOG_WARN << "Entity has outline but is using pipeline that does not support it: "
160 << entity;
161 }

164void Outline3DTask::removeObject(ecs::Entity entity) {
165 const auto remove = [&entity](std::vector<Outlined>& list) -> bool {
166 for (auto it = list.begin(); it != list.end(); ++it) {
167 if (it->drawable->getEntity() == entity) {
168 list.erase(it);
169 return true;
170 }
171 }
172 return false;
173 };
174
175 // fetching & checking likely does not save us any time in 90% of cases so just search
176 if (remove(outlinedObjectsDynamic)) { return; }
177 if (remove(outlinedSkinnedObjectsDynamic)) { return; }
178 if (remove(outlinedObjectsStatic)) { return; }
179 remove(outlinedSkinnedObjectsStatic);
180}
