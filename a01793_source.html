28 openBackground.scaleToSize({owner.getOptionRegion().width, owner.getOptionRegion().height});

157sf::Vector2f ComboBoxComponent::getRequisition() const {
158 if (optionsOutdated()) { const_cast<ComboBoxComponent*>(this)->updateOptions(); }
159
160 sf::Vector2f req(0.f, 0.f);
161 for (const auto& option : openOptions) {
162 const sf::FloatRect bounds = option.text.getLocalBounds();
163 const float w = bounds.left + bounds.width + BoxPad;
164 const float h = bounds.top + bounds.height + BoxPad;
165 req.x = std::max(req.x, w);
166 req.y = std::max(req.y, h);
167 }
168
169 return req;
170}

172void ComboBoxComponent::updateOptions() {
173 ComboBox& owner = getOwnerAs<ComboBox>();
174 const RenderSettings& settings = owner.renderSettings();
175 const glm::vec2 boxSize(owner.getOptionSize().x, owner.getOptionSize().y);
176 const float outline = settings.outlineThickness.value_or(1.f);
177 const sf::Vector2f innerSize(boxSize.x - outline * 2.f, boxSize.y - outline * 2.f);
178
179 openOptions.resize(owner.getAllOptions().size());
180 auto it = openOptions.begin();
181 for (unsigned int i = 0; i < owner.getAllOptions().size(); ++i) {
182 Option& o = *(it++);
183 const float y = static_cast<float>(i) * boxSize.y;
184
185 if (!o.created) {
186 o.created = true;
187 o.background.create(worldPtr->engine(), openOptionBoxes, boxSize);
188 o.text.create(*worldPtr, *settings.font.value_or(Font::get()));
189 o.text.setParent(openOptionBoxes);
190 if (currentOverlay) { o.text.addToScene(currentOverlay, rc::UpdateSpeed::Static); }
191 }
192
193 o.text.getSection().setString(owner.getAllOptions()[i]);
194 configureText(o.text, settings);
195 const sf::Vector2f tpos =
196 RenderSettings::calculatePosition(RenderSettings::Left,
197 RenderSettings::Center,
198 sf::FloatRect{sf::Vector2f{0.f, y}, innerSize},
199 {o.text.getLocalSize().x, o.text.getLocalSize().y});
200 o.text.getTransform().setPosition({tpos.x, tpos.y + y});
201
202 o.background.setSize(boxSize);
203 if (static_cast<int>(i) == owner.getMousedOption()) {
204 o.background.setFillColor(settings.secondaryFillColor.value_or(sf::Color(80, 80, 250)));
205 }
206 else { o.background.setFillColor(settings.fillColor.value_or(sf::Color(100, 100, 100))); }
207 o.background.setOutlineColor(settings.outlineColor.value_or(sf::Color::Black));
208 o.background.setOutlineThickness(-outline);
209 o.background.getLocalTransform().setPosition({0.f, y});
210 }
211}

228bool ComboBoxComponent::optionsOutdated() const {
229 const ComboBox& owner = getOwnerAs<ComboBox>();
230 if (owner.getAllOptions().size() != openOptions.size()) { return true; }
231 auto it = openOptions.begin();
232 for (unsigned int i = 0; i < openOptions.size(); ++i) {
233 const auto& option = *(it++);
234 const std::string& text = owner.getAllOptions()[i];
235 if (!option.created || option.text.getSection().getString() != text) { return true; }
236 }
237 return false;
238}
