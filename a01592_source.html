128void Actor::update() {
129 for (unsigned int i = 0; i < (*activeControls).size(); ++i) {
130 const Control& ctrl = (*activeControls)[i];
131
132 switch (ctrl.type) {
133 case Control::Type::SingleTrigger:
134 if (ctrl.triggerControl.active()) {
135 dispatch(i, DispatchType::TriggerActivated, false);
136 }
137 break;
138 case Control::Type::Movement:
139 if (ctrl.movementControl.type == MovementControl::Type::Triggers) {
140 const auto& keys = ctrl.movementControl.keys;
141 if (keys.up.active()) { dispatch(i, DispatchType::MovementUp, false); }
142 if (keys.right.active()) { dispatch(i, DispatchType::MovementRight, false); }
143 if (keys.down.active()) { dispatch(i, DispatchType::MovementDown, false); }
144 if (keys.left.active()) { dispatch(i, DispatchType::MovementLeft, false); }
145 }
146 else {
147 const sf::Vector2f& pos = ctrl.movementControl.joystick.cachedPosition;
148 if (pos.x >= StickThreshold) { dispatch(i, DispatchType::MovementRight, false); }
149 if (pos.x <= -StickThreshold) { dispatch(i, DispatchType::MovementLeft, false); }
150 if (pos.y >= StickThreshold) { dispatch(i, DispatchType::MovementUp, false); }
151 if (pos.y <= -StickThreshold) { dispatch(i, DispatchType::MovementDown, false); }
152 }
153 break;
154 default:
155 break;
156 }
157 }
158}

195void Actor::process(const sf::Event& event) {
196 // determine if keyboard or controller is being used
197 switch (event.type) {
198 case sf::Event::JoystickButtonPressed:
199 case sf::Event::JoystickButtonReleased:
200 if (joystick < sf::Joystick::Count && event.joystickButton.joystickId != joystick) return;
201 activeControls = &joystickControls;
202 break;
203 case sf::Event::JoystickMoved:
204 if (joystick < sf::Joystick::Count && event.joystickMove.joystickId != joystick) return;
205 activeControls = &joystickControls;
206 break;
207 case sf::Event::KeyPressed:
208 case sf::Event::KeyReleased:
209 case sf::Event::MouseButtonPressed:
210 case sf::Event::MouseButtonReleased:
211 case sf::Event::MouseWheelScrolled:
212 case sf::Event::MouseMoved:
213 activeControls = &kbmControls;
214 break;
215 default:
216 return; // can exit early, event does not trigger any controls
217 }
218
219 // process control
220 for (unsigned int i = 0; i < (*activeControls).size(); ++i) {
221 Control& ctrl = (*activeControls)[i];
222
223 switch (ctrl.type) {
224 case Control::Type::SingleTrigger:
225 if (ctrl.triggerControl.process(event)) {
226 dispatch(i, DispatchType::TriggerActivated, true);
227 }
228 break;
229 case Control::Type::Movement: {
230 const DispatchType dt = ctrl.movementControl.process(event);
231 if (dt != DispatchType::_INVALID) { dispatch(i, dt, true); }
232 } break;
233 case Control::Type::Directional:
234 if (ctrl.directionalControl.process(owner, event)) {
235 dispatch(i, DispatchType::DirectionChanged, true);
236 }
237 break;
238 default:
239 break;
240 }
241 }
