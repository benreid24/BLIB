312 void TextEntry::filter() {
313  if (mode == Mode::Any) return;
314 
315  if (modeEnabled(mode, Mode::Integer) || modeEnabled(mode, Mode::Float)) {
316  bool decimalFound = false;
317  for (int i = 0; i < static_cast<int>(input.size()); ++i) {
318  if (!std::isdigit(input[i])) {
319  if (i == 0 && input[i] == '-' && modeEnabled(mode, Mode::Signed)) continue;
320  if (input[i] == '.' && modeEnabled(mode, Mode::Float) && !decimalFound) {
321  decimalFound = true;
322  continue;
323  }
324  input.erase(i, 1);
325  --i;
326  --cursorPos;
327  }
328  }
329  if (input.empty()) { input = "0"; }
