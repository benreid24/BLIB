225template<typename TKey, typename TPayload>
226void Cache<TKey, TPayload>::checkEviction() {
227 if (cache.size() >= highWatermark) {
228 FastQueue<TKey>* queue = nullptr;
229 switch (evictionPolicy) {
230 case LeastRecentlyAccessed:
231 queue = &readQueue;
232 break;
233 case LeastRecentlyModified:
234 queue = &writeQueue;
235 break;
236
237 case FirstInFirstOut:
238 queue = &orderQueue;
239 break;
240 default:
241 BL_LOG_ERROR << "Invalid eviciton policy in queue: " << evictionPolicy;
242 return;
243 }
244
245 while (cache.size() > lowWatermark && !queue->empty()) {
246 TKey key;
247 queue->pop_back(key);
248 cache.erase(key);
249 }
250 }
