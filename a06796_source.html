82void Text::commit() {
83 if (commitTask.isQueued()) { commitTask.cancel(); }
84 commitTask = {};
85
86 // word wrap
87 computeWordWrap();
88
89 // count required vertex amount
90 std::uint32_t vertexCount = 0;
91 glm::vec2 trash;
92 for (auto& section : sections) {
93 vertexCount += section.refreshVertices(*font, nullptr, trash);
94 }
95
96 // create larger buffer if required
97 if (component().vertices.vertexCount() < vertexCount) {
98 component().vertices.create(engine().renderer().vulkanState(), vertexCount * 2);
99 }
100
101 // assign vertices
102 std::uint32_t vi = 0;
103 glm::vec2 cornerPos(0.f, 0.f);
104 for (auto& section : sections) {
105 vi += section.refreshVertices(*font, &component().vertices.vertices()[vi], cornerPos);
106 }
107
108 // upload vertices
109 component().vertices.queueTransfer(rc::tfr::Transferable::SyncRequirement::Immediate);
110
111 const auto bounds = getLocalBounds();
112 OverlayScalable::setLocalSize({bounds.width + bounds.left, bounds.height + bounds.top});
113
114 // update draw parameters
115 component().updateDrawParams(vertexCount);
116
117 // always upload new font atlas if required
118 font->syncTexture(engine().renderer());
119
120 boundsComputedWhileDirty = false;
121}

146void Text::computeWordWrap() {
147 for (auto& section : sections) {
148 section.wordWrappedContent = section.content;
149 section.cachedLineHeight = section.computeLineSpacing(*font);
150 }
151 if (wrapType == WrapType::None || wordWrapWidth <= 0.f) { return; }
152
153 float pw = 1.f;
154 if (wrapType == WrapType::Relative && getTransform().hasParent()) {
155 const ecs::Entity parent = engine().ecs().getEntityParent(entity());
156 auto cset =
157 engine().ecs().getComponentSet<ecs::Require<com::OverlayScaler, com::Transform2D>>(
158 parent);
159 if (cset.isValid()) {
160 pw = cset.get<com::OverlayScaler>()->getEntityBounds().width *
161 cset.get<com::Transform2D>()->getScale().x;
162 }
163 }
164 const float maxWidth = wordWrapWidth * pw / getTransform().getScale().x;
165 const Iter EndIter = Iter::end(sections);
166
167 glm::vec2 nextPos(0.f, 0.f);
168 float maxLineHeight = 0.f;
169 Iter prevSpace = EndIter;
170 glm::vec2 wordStartPos(0.f, 0.f);
171 bool lineOnNextSpace = false;
172 std::uint32_t prevChar = 0;
173
174 const auto resetLine = [this, &maxLineHeight, &prevSpace, &lineOnNextSpace, EndIter](Iter it) {
175 prevSpace = EndIter;
176 lineOnNextSpace = false;
177 it.getText().cachedLineHeight = it.getText().computeLineSpacing(*font);
178 maxLineHeight = it.getText().cachedLineHeight;
179 };
180
181 for (Iter it = Iter::begin(sections); it != EndIter; ++it) {
182 maxLineHeight = std::max(it.getText().computeLineSpacing(*font), maxLineHeight);
183 it.getText().cachedLineHeight = maxLineHeight;
184 glm::vec2 advance =
185 it.getText().advanceCharacterPos(*font, nextPos, it.getChar(), prevChar);
186 prevChar = it.getChar();
187
188 if (it.getChar() == ' ') {
189 if (lineOnNextSpace) {
190 it.makeNewline();
191 resetLine(it);
192 wordStartPos = glm::vec2{0.f, advance.y + maxLineHeight};
193 }
194 else {
195 prevSpace = it;
196 wordStartPos = advance;
197 }
198 }
199 else if (it.getChar() == '\n') {
200 resetLine(it);
201 wordStartPos = advance;
202 }
203 else {
204 if (nextPos.x > maxWidth) { // nextPos == prevPos here
205 if (prevSpace != EndIter) {
206 prevSpace.makeNewline();
207 resetLine(it);
208 const float wordWidth = advance.x - wordStartPos.x;
209 advance = glm::vec2{wordWidth, advance.y + maxLineHeight};
210 wordStartPos = advance;
211 }
212 else { lineOnNextSpace = true; }
213 }
214 }
215
216 nextPos = advance;
217 }
218}

260Text::CharSearchResult Text::findCharacterAtLocalPosition(const glm::vec2& position) const {
261 const sf::Vector2f localPos(position.x, position.y);
262
263 glm::vec2 nextPos(0.f, 0.f);
264 std::uint32_t prevChar = 0;
265 for (std::uint32_t si = 0; si < sections.size(); ++si) {
266 for (std::uint32_t i = 0; i < sections[si].wordWrappedContent.getSize(); ++i) {
267 const std::uint32_t curChar = sections[si].wordWrappedContent[i];
268 const glm::vec2 advance =
269 sections[si].advanceCharacterPos(*font, nextPos, curChar, prevChar);
270 const sf::FloatRect bounds(nextPos.x,
271 nextPos.y,
272 advance.x - nextPos.x,
273 sections[si].computeLineSpacing(*font));
274
275 if (bounds.contains(localPos)) { return {si, i}; }
276
277 nextPos = advance;
278 prevChar = curChar;
279 }
280
281 // transition to next section
282 switch (prevChar) {
283 case ' ':
284 case '\t':
285 case '\n':
286 break;
287 default:
288 nextPos.x += sections[si].computeWhitespaceWidth(*font);
289 break;
290 }
291 }
292
293 BL_LOG_DEBUG << "Could not find character position";
294 return {};
295}
