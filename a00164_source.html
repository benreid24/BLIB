174template<typename T>
175Ref<T> ResourceManager<T>::load(const std::string& uri) {
176 ResourceManager& m = get();
177 std::unique_lock lock(m.mapLock);
178 auto it = m.resources.find(uri);
179 if (it == m.resources.end()) {
180 it = m.resources.try_emplace(uri).first;
181
182 char* buffer = nullptr;
183 std::size_t len = 0;
184 if (!FileSystem::getData(uri, &buffer, len)) {
185 BL_LOG_ERROR << "Failed to find resource: " << uri;
186 return {&it->second};
187 }
188
189 // copy buffer for font
190 if constexpr (std::is_same_v<T, sf::Font>) {
191 std::vector<char>& rbuf = it->second.buffer;
192 rbuf.resize(len);
193 std::memcpy(rbuf.data(), buffer, len);
194 buffer = rbuf.data();
195 }
196
197 if (!m.doInit(uri, buffer, len, it->second.data)) {
198 BL_LOG_ERROR << "Failed to load resource: " << uri;
199 }
200 }
201
202 return {&it->second};
203}
