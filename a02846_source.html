75void BatchSlideshow::commit() {
76 if (dirty && owner) {
77 com::Animation2DPlayer* player = validatePlayer(playerEntity);
78 if (!player) {
79 BL_LOG_ERROR << "Animation player invalidated";
80 return;
81 }
82 if (player->getPlayerIndex() == com::Animation2DPlayer::InvalidIndex) {
83 BL_LOG_ERROR << "Animation player was not initialized by system";
84 return;
85 }
86 if (!player->getAnimation() || player->getAnimation()->frameCount() == 0) {
87 BL_LOG_WARN << "Ignoring empty animation";
88 return;
89 }
90 addPlayerDep();
91
92 if (!alloc.isValid()) {
93 alloc = owner->component().indexBuffer.allocate(4, 6);
94
95 alloc.getIndices()[0] = 0 + alloc.getInfo().vertexStart;
96 alloc.getIndices()[1] = 1 + alloc.getInfo().vertexStart;
97 alloc.getIndices()[2] = 2 + alloc.getInfo().vertexStart;
98 alloc.getIndices()[3] = 0 + alloc.getInfo().vertexStart;
99 alloc.getIndices()[4] = 2 + alloc.getInfo().vertexStart;
100 alloc.getIndices()[5] = 3 + alloc.getInfo().vertexStart;
101 }
102
103 rc::prim::SlideshowVertex* vertices = alloc.getVertices();
104 const sf::FloatRect src(player->getAnimation()->getFrame(0).shards.front().source);
105 vertices[0].pos = {0.f, 0.f, 0.f};
106 vertices[1].pos = {src.size.x, 0.f, 0.f};
107 vertices[2].pos = {src.size.x, src.size.y, 0.f};
108 vertices[3].pos = {0.f, src.size.y, 0.f};
109 for (unsigned int i = 0; i < 4; ++i) {
110 rc::prim::SlideshowVertex& vertex = vertices[i];
111 vertex.slideshowIndex = player->getPlayerIndex();
112 vertex.color = {1.f, 1.f, 1.f, 1.f};
113 vertex.pos = transform.transformPoint(vertex.pos);
114 }
115
116 alloc.commit();
117 owner->component().updateDrawParams();
118 }
119}
