47Settings& Settings::fromConfig() {
48 updateTime = Configuration::getOrDefault<float>(UpdatePeriodKey, updateTime);
49 maxFps = Configuration::getOrDefault<float>(MaxFpsKey, maxFps);
50 allowVariableInterval =
51 Configuration::getOrDefault<bool>(VariableTimestepKey, allowVariableInterval);
52 loggingFps = Configuration::getOrDefault<bool>(LogFpsKey, loggingFps);
53
54 if (Configuration::getOrDefault<bool>(RendererCreateKey, false) ||
55 rendererSettings.has_value()) {
56 if (!rendererSettings.has_value()) { rendererSettings.emplace(); }
57 rendererSettings.value().fromConfig(RendererPrefix);
