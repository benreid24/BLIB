38void Overlay::renderOpaqueObjects(scene::SceneRenderContext& ctx) {
39 std::unique_lock lock(objectMutex);
40 ecs::Transaction<ecs::tx::EntityUnlocked, ecs::tx::ComponentRead<ovy::OverlayObject>>
41 transaction(engine.ecs());
42
43 std::copy(roots.begin(), roots.end(), std::inserter(renderStack, renderStack.begin()));
44
45 if (needRefreshAll ||
46 static_cast<std::uint32_t>(ctx.parentViewport().width) != cachedTargetSize.x ||
47 static_cast<std::uint32_t>(ctx.parentViewport().height) != cachedTargetSize.y) {
48 cachedParentViewport = ctx.parentViewport();
49 cachedTargetSize.x = static_cast<std::uint32_t>(ctx.parentViewport().width);
50 cachedTargetSize.y = static_cast<std::uint32_t>(ctx.parentViewport().height);
51 needRefreshAll = false;
52 refreshAll();
53 }
54
55 VkPipelineLayout currentPipelineLayout = nullptr;
56 VkPipeline currentPipeline = nullptr;
57 UpdateSpeed currentSpeed{};
58 while (!renderStack.empty()) {
59 ovy::OverlayObject& obj = *renderStack.back();
60 renderStack.pop_back();
61
62 if (!obj.component || obj.component->isHidden()) { continue; }
63
64 const std::uint32_t spec = obj.component->getPipelineSpecialization();
65 if (!obj.entity.flagSet(ecs::Flags::Dummy)) {
66 vkCmdSetScissor(ctx.getCommandBuffer(), 0, 1, &obj.cachedScissor);
67
68 const vk::PipelineLayout& layout =
69 obj.pipeline->getPipeline(cfg::RenderPhases::Overlay)->pipelineLayout();
70 const VkPipelineLayout vkl = layout.rawLayout();
71 const VkPipeline vkp =
72 obj.pipeline->getRawPipeline(ctx.getRenderPhase(), ctx.currentRenderPass(), spec);
73 if (vkl != currentPipelineLayout || currentSpeed != obj.sceneKey.updateFreq) {
74 currentSpeed = obj.sceneKey.updateFreq;
75 currentPipelineLayout = vkl;
76 currentPipeline = vkp;
77 ctx.bindPipeline(*obj.pipeline, spec);
78 ctx.bindDescriptors(vkl,
79 obj.sceneKey.updateFreq,
80 obj.descriptors.get(ctx.currentObserverIndex()).data(),
81 obj.descriptors.getDescriptorSetCount());
82 }
83 else if (currentPipeline != vkp) {
84 currentPipeline = vkp;
85 ctx.bindPipeline(*obj.pipeline, spec);
86 }
87 for (std::uint8_t i = obj.descriptors.getPerObjectStart();
88 i < obj.descriptors.getDescriptorSetCount();
89 ++i) {
90 obj.descriptors.get(ctx.currentObserverIndex())[i]->bindForObject(
91 ctx, vkl, i, obj.sceneKey);
92 }
93 ctx.renderObject(obj);
94 }
95
96 std::copy(obj.getChildren().rbegin(),
97 obj.getChildren().rend(),
98 std::inserter(renderStack, renderStack.end()));
99 }

129void Overlay::doObjectRemoval(scene::SceneObject* object, mat::MaterialPipeline*) {
130 ovy::OverlayObject* obj = static_cast<ovy::OverlayObject*>(object);
131 obj->overlayViewport = nullptr;
132
133 removeFromAll(obj);
134
135 auto childCopy = obj->getChildren();
136 for (ovy::OverlayObject* child : childCopy) {
137 if (engine.ecs().getEntityParentDestructionBehavior(child->entity) !=
138 ecs::ParentDestructionBehavior::OrphanedByParent &&
139 child->sceneKey.sceneId != scene::Key::InvalidSceneId) {
140 removeObject(child);
141 }
142 }
143
144 const auto it = std::find(roots.begin(), roots.end(), obj);
145 if (it != roots.end()) {
146 roots.erase(it);
147 sortRoots();
148 }
149
150 obj->descriptors.releaseObject(obj->entity, obj->sceneKey);
151 objects.release(obj->sceneKey);
152 engine.ecs().removeComponent<ovy::OverlayObject>(obj->entity);
153 emitter.emit<rc::event::SceneObjectRemoved>({this, obj->entity});
154}

209void Overlay::sortRoots() {
210 std::sort(
211 roots.begin(), roots.end(), [this](ovy::OverlayObject* left, ovy::OverlayObject* right) {
212 com::Transform2D* lpos = engine.ecs().getComponent<com::Transform2D>(left->entity);
213 com::Transform2D* rpos = engine.ecs().getComponent<com::Transform2D>(right->entity);
214 if (!lpos || !rpos) { return left < right; }
215 return lpos->getGlobalDepth() < rpos->getGlobalDepth();
216 });
