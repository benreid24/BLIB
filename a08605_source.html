205template<typename T, typename... TArgs>
206T& Systems::registerSystem(FrameStage::V stage, StateMask::V stateMask, TArgs&&... args) {
207#ifdef BLIB_DEBUG
208 const auto it = typeMap.find(typeid(T));
209 if (it != typeMap.end()) {
210 BL_LOG_CRITICAL << "System duplicated: " << typeid(T).name();
211 throw std::runtime_error("Systems may only appear once per type");
212 }
213
214 if (stage >= systems.size()) {
215 BL_LOG_CRITICAL << "Out of range frame stage index: " << stage;
216 throw std::runtime_error("Out of range engine frame stage index");
217 }
218#endif
219
220 systems[stage].systems.emplace_back(
221 stateMask, std::move(std::make_unique<T>(std::forward<TArgs>(args)...)));
222 System* s = systems[stage].systems.back().system.get();
223 typeMap[typeid(T)] = s;
224 if (inited) { s->init(engine); }
225 return static_cast<T&>(*s);
226}
