344template<typename T>
345T& Engine::getPlayer(unsigned int i) {
346 static_assert(std::is_base_of_v<Player, T>, "T must derive from Player");
347
348 if constexpr (std::is_same_v<Player, T>) {
349 if (players.empty() && i == 0) {
350 BL_LOG_INFO << "No player exists, creating default";
351 addPlayer();
352 }
353 }
354
355#ifdef BLIB_DEBUG
356 T* p = dynamic_cast<T*>(players[i].get());
357 if (!p) { BL_LOG_ERROR << "Bad player cast"; }
358#else
359 T* p = static_cast<T*>(players[i].get());
360#endif
361
362 return *p;
363}

365template<typename T, typename... TArgs>
366T& Engine::addPlayer(TArgs&&... args) {
367 static_assert(std::is_base_of_v<Player, T>, "T must derive from Player");
368 static_assert(
369 std::is_constructible_v<T, Engine&, rc::Observer*, input::Actor*, TArgs...>,
370 "T must be constructible with T(Engine&, rc::Observer*, input::Actor*, TArgs...)");
371
372 auto& observer = rendererInstance->addObserver();
373 auto& actor = input.addActor();
374 T* np = new T(*this, &observer, &actor, args...);
375 auto& player = players.emplace_back(np);
376 eventEmitter.emit<event::PlayerAdded>({*player});
377 return *np;
378}

380template<typename TWorld, typename... TArgs>
381ctr::Ref<World, TWorld> Engine::createWorld(TArgs&&... args) {
382 static_assert(std::is_base_of_v<World, TWorld>, "TWorld must derive from World");
383 static_assert(std::is_constructible_v<TWorld, Engine&, TArgs...>,
384 "TWorld constructor must take Engine& as first parameter");
385
386 auto ref = worldPool.emplaceDerived<TWorld, TArgs...>(*this, std::forward<TArgs>(args)...);
387 for (unsigned int i = 0; i < worlds.size(); ++i) {
388 if (!worlds[i].isValid()) {
389 worlds[i] = ref;
390 ref->index = i;
391 eventEmitter.emit<event::WorldCreated>({*ref});
392 return ref;
393 }
394 }
395
396 throw std::runtime_error("A maximum of 8 worlds may exist concurrently");
397}
