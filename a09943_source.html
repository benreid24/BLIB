29bool Asset::initAfterDeserialize(Repository& r) {
30 repo = &r;
31
32 // already inited via dependency chain
33 if (state != State::Unknown) { return true; }
34
35 state = State::Unloaded;
36
37 for (const RepoDependency& dep : dependencies) {
38 Asset* depAsset = r.getDependencyForInit(dep.uuid);
39 if (!depAsset) {
40 state = State::Failed;
41 BL_LOG_ERROR << "Failed to find dependency with UUID " << dep.uuid.toString()

46 if (!depAsset->initAfterDeserialize(r)) {
47 state = State::Failed;
48 BL_LOG_ERROR << "Failed to initialize dependency with UUID " << dep.uuid.toString()

52 }

169bool Asset::reloadFromSource() {
170 if (state != State::Loaded) {
171 if (!load()) { return false; }
172 }
173
174 if (!metadata.hasSourceFileInfo()) { return true; }
175
176 util::FileUtil::FileInfo info;
177 if (!util::FileUtil::queryFileInfo(metadata.getSourceFileInfo()->path, info)) {
178 BL_LOG_WARN << "Failed to query source file '" << metadata.getSourceFileInfo()->path
179 << "' for asset " << uuid.toString();
180 return false;
181 }
182
183 if (metadata.getSourceFileInfo()->lastModified < info.modifiedTime) {
184 CreateContext::CreateData createData(metadata.getSourceFileInfo()->path);
185 if (!create(createData)) {
186 BL_LOG_ERROR << "Failed to reload asset " << uuid.toString() << " from source file "
187 << metadata.getSourceFileInfo()->path;
188 return false;
189 }
190 metadata.updateSourceFileModifiedTime();
191 }
192
193 return true;
