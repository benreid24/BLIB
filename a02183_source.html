119 for (GraphAsset* asset : task->assets.optionalInputs) {
120 if (asset) { createAsset(asset); }
121 }

124 // find & create sidecar assets here bc we know the task will get executed at this point
125 unsigned int i = 0;
126 for (std::string_view sidecar : task->assetTags.sidecarAssets) {
127 GraphAsset* sc = pool->createAsset(sidecar, "");
128 task->assets.sidecarAssets[i++] = sc;
129 createAsset(sc);

131 };
132
133 // find all tasks that can run in first step & reset asset indices
134 std::vector<VisitorStep> tasksToVisit;
135 tasksToVisit.reserve(tasks.size() * 2);
136 for (auto& task : tasks) {
137 resetAssetIndices(task.get());
138 if (taskIsFirst(task.get())) { tasksToVisit.emplace_back(task.get(), 0); }
139 }
140 if (finalAsset->firstAvailableStep != UnsetStep) {
141 throw std::runtime_error("RenderGraph does not contain expected final asset");
142 }
143
144 // traverse up graph from starting tasks and assign step indices to assets
145 unsigned int maxSteps = 0;
146 while (!tasksToVisit.empty()) {
147 VisitorStep current = tasksToVisit.back();
148 tasksToVisit.pop_back();
149
150 createAssets(current.task);
151 maxSteps = std::max(maxSteps, current.step);
152 const unsigned int nextStep = current.step + 1;
153
154 for (GraphAsset* output : current.task->assets.outputs) {
155 if (output->firstAvailableStep == UnsetStep ||
156 output->firstAvailableStep < current.step) {
157 output->firstAvailableStep = current.step;
158
159 for (auto& task : tasks) {
160 if (isInput(output, task.get())) {
161 tasksToVisit.emplace_back(task.get(), nextStep);
162 }
163 }
164 }
165 output->firstAvailableStep = std::max(output->firstAvailableStep, current.step);
166 }
167 }
168
169 if (finalAsset->firstAvailableStep == UnsetStep) {
170 throw std::runtime_error("RenderGraph timeline does not reach final asset");
171 }
172
173 // create timeline and add tasks
174 timeline.resize(maxSteps + 1);
175 for (auto& task : tasks) {
176 unsigned int i = 0;
177 for (GraphAsset* output : task->assets.outputs) {
178 TaskGroup* group = timeline[output->firstAvailableStep].getGroupForAsset(output);
179 group->addTask(task.get(), task->assetTags.outputs[i++].order);
180 }
181 }
182}
