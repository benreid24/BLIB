190  const std::vector<std::string> order(std::move(songs));
191  ctr::FastEraseVector<std::size_t> indices;
192  indices.reserve(songs.size());
193  songs.reserve(order.size());

199  const std::size_t i = util::Random::get<std::size_t>(0, indices.size() - 1);
200  songs.emplace_back(std::move(order[indices[i]]));
