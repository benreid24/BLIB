353template<typename T>
354T* RenderTarget::getDescriptorSet(Scene* scene) {
355 auto* descriptorSets = getDescriptorSetCache(scene);
356 if (!descriptorSets) {
357 BL_LOG_ERROR
358 << "Failed to get descriptor set cache for scene. RenderTarget does not render it";
359 return nullptr;
360 }
361
362 T* set = descriptorSets->getDescriptorSet<T>();
363 if (set) { return set; }
364
365 ds::DescriptorSetFactory* factory = descriptorFactories.getFactoryThatMakes<T>();
366 if (!factory) { throw std::runtime_error("Failed to find descriptor set"); }
367
368 return static_cast<T*>(descriptorSets->getDescriptorSet(factory));
369}
