96template<typename T>
97struct Serializer<T, true> {
98 static bool serialize(stream::OutputStream& output, const T& value) {
99 detail::OutputStreamWrapper wrapper(output);
100 if constexpr (std::is_enum<T>::value) {
101 return wrapper.write<std::underlying_type_t<T>>(
102 static_cast<std::underlying_type_t<T>>(value));
103 }
104 else if constexpr (std::is_same_v<T, bool>) { return wrapper.write<std::uint8_t>(value); }
105 else { return wrapper.write<T>(value); }
106 }
107
108 static bool deserialize(stream::InputStream& input, T& result) {
109 detail::InputStreamWrapper wrapper(input);
110 if constexpr (std::is_enum<T>::value) {
111 return wrapper.read<std::underlying_type_t<T>>(
112 *reinterpret_cast<std::underlying_type_t<T>*>(&result));
113 }
114 else if constexpr (std::is_same_v<T, bool>) {
115 std::uint8_t v;
116 if (!wrapper.read<std::uint8_t>(v)) return false;
117 result = v != 0;
118 return true;
119 }
120 else { return wrapper.read<T>(result); }
121 }
122
123 static std::uint32_t size(const T&) { return sizeof(T); }

602template<typename... Ts>
603struct Serializer<std::variant<Ts...>, false> {
604private:
605 struct HelperBase {
606 virtual bool deserialize(stream::InputStream& input, std::variant<Ts...>& v) const = 0;
607 };

609 template<typename U>
610 struct Helper : public HelperBase {
611 virtual bool deserialize(stream::InputStream& input,
612 std::variant<Ts...>& v) const override {
613 v.template emplace<U>();
614 return Serializer<U>::deserialize(input, std::get<U>(v));
615 }
616 };
617
618public:
619 static bool serialize(stream::OutputStream& output, const std::variant<Ts...>& v) {
620 const auto visitor = [&output](const auto& c) -> bool {
621 return Serializer<std::decay_t<decltype(c)>>::serialize(output, c);
622 };
623
624 detail::OutputStreamWrapper wrapper(output);
625 if (!wrapper.write<std::uint16_t>(v.index())) return false;
626 return std::visit(visitor, v);
627 }
628
629 static bool deserialize(stream::InputStream& input, std::variant<Ts...>& v) {
630 static const std::tuple<Helper<Ts>...> helperTuple(Helper<Ts>{}...);
631 static const HelperBase* helpers[] = {&std::get<Helper<Ts>>(helperTuple)...};
632
633 detail::InputStreamWrapper wrapper(input);
634 std::uint16_t i = 0;
635 if (!wrapper.read<std::uint16_t>(i)) return false;
636 if (i >= sizeof...(Ts)) return false;
637 return helpers[i]->deserialize(input, v);
638 }

641 static constexpr auto visitor = [](const auto& c) -> std::uint32_t {
642 return Serializer<std::decay_t<decltype(c)>>::size(c);

686 case script::PrimitiveValue::TFloat:
687 if (!Serializer<float>::serialize(output, value.getAsFloat())) return false;
688 break;
689 case script::PrimitiveValue::TString:
690 if (!Serializer<std::string>::serialize(output, value.getAsString())) return false;
691 break;
692 case script::PrimitiveValue::TArray:
693 if (!wrapper.write<std::uint32_t>(
694 static_cast<std::uint32_t>(value.getAsArray().size()))) {
695 return false;
696 }
697 for (const script::Value& av : value.getAsArray()) {
698 if (!serialize(output, av)) return false;
699 }
700 break;
701 default:
702 break;
703 }
704
705 if (!wrapper.write<std::uint32_t>(v.allProperties().size())) return false;
706 for (const auto& prop : v.allProperties()) {
707 if (!Serializer<std::string>::serialize(output, prop.first)) return false;
708 if (!Serializer<script::Value>::serialize(output, prop.second.deref())) return false;
709 }
710
711 return true;
712 }
713
714 static bool deserialize(stream::InputStream& input, script::Value& v) {
715 script::PrimitiveValue& value = v.value();
716 script::PrimitiveValue::Type type;
717 if (!Serializer<script::PrimitiveValue::Type>::deserialize(input, type)) return false;
718 if (!supported(type)) {
719 value = "<UNSUPPORTED TYPE>";
720 return true;
721 }
722
723 detail::InputStreamWrapper wrapper(input);
724
725 switch (type) {
726 case script::PrimitiveValue::TBool: {
727 bool bv;
728 if (!Serializer<bool>::deserialize(input, bv)) return false;
729 value = bv;
730 break;
731 }
732 case script::PrimitiveValue::TInteger: {
733 std::int32_t i;
734 if (!wrapper.read<std::int32_t>(i)) return false;
735 value = i;
736 break;
737 }
738 case script::PrimitiveValue::TFloat: {
739 float f;
740 if (!Serializer<float>::deserialize(input, f)) return false;
741 value = f;
742 break;
743 }
744 case script::PrimitiveValue::TString: {
745 std::string s;
746 if (!Serializer<std::string>::deserialize(input, s)) return false;
747 value = s;
748 break;
749 }
750 case script::PrimitiveValue::TArray: {
751 std::uint32_t s;
752 if (!wrapper.read<std::uint32_t>(s)) return false;
753 value = script::ArrayValue{};
754 value.getAsArray().resize(s);
755 for (unsigned int i = 0; i < s; ++s) {
756 if (!deserialize(input, value.getAsArray()[i])) return false;
757 }
758 break;
759 }
760 default:
761 break;
762 }
763
764 std::uint32_t s;
765 if (!wrapper.read<std::uint32_t>(s)) return false;
766 for (unsigned int i = 0; i < s; ++i) {
767 std::string key;
768 script::Value pv;
769 if (!Serializer<std::string>::deserialize(input, key)) return false;
770 if (!Serializer<script::Value>::deserialize(input, pv)) return false;
771 v.setProperty(key, {std::move(pv)});
772 }
773
774 return true;
775 }
776
777 static std::uint32_t size(const script::Value& v) {
778 std::uint32_t s = Serializer<script::PrimitiveValue::Type>::size(v.value().getType());
779 if (!supported(v.value().getType())) return s;
780
781 switch (v.value().getType()) {
782 case script::PrimitiveValue::TBool:
783 s += Serializer<bool>::size(true);
784 break;
785 case script::PrimitiveValue::TInteger:
786 s += sizeof(std::int32_t);
787 break;
788 case script::PrimitiveValue::TFloat:
789 s += Serializer<float>::size(v.value().getAsFloat());
790 break;
791 case script::PrimitiveValue::TString:
792 s += Serializer<std::string>::size(v.value().getAsString());
793 break;
794 case script::PrimitiveValue::TArray:
795 s += sizeof(std::uint32_t);
796 for (const script::Value& av : v.value().getAsArray()) { s += size(av); }
797 break;
798 default:
799 break;
800 }
801
802 s += sizeof(std::uint32_t);
803 for (const auto& prop : v.allProperties()) {
804 s += Serializer<std::string>::size(prop.first);
805 s += Serializer<script::Value>::size(prop.second.deref());
806 }
807
808 return s;
809 }

815 case script::PrimitiveValue::TInteger:
816 case script::PrimitiveValue::TFloat:
817 case script::PrimitiveValue::TString:
818 case script::PrimitiveValue::TArray:
819 return true;

821 return false;
822 }

855 static bool serialize(stream::OutputStream& output, const std::optional<T>& opt) {
856 detail::OutputStreamWrapper wrapper(output);
857 if (!wrapper.write<std::uint8_t>(opt.has_value() ? 1 : 0)) return false;
858 if (opt.has_value()) {
859 if (!Serializer<T>::serialize(output, *opt)) return false;
860 }
861 return true;
862 }
863
864 static bool deserialize(stream::InputStream& input, std::optional<T>& opt) {
865 detail::InputStreamWrapper wrapper(input);
866 std::uint8_t hasValue;
867 if (!wrapper.read<std::uint8_t>(hasValue)) return false;
868 if (hasValue) {
869 opt.emplace();
870 if (!Serializer<T>::deserialize(input, *opt)) return false;
871 }
872 else { opt.reset(); }
873 return true;
874 }

877 std::uint32_t s = sizeof(std::uint8_t);
878 if (opt.has_value()) { s += Serializer<T>::size(*opt); }
