305 void Registry::finishComponentAdd(Entity ent, unsigned int cIndex, T* component) {
306  bl::event::Dispatcher::dispatch<event::ComponentAdded<T>>({ent, *component});
307  ComponentMask::SimpleMask& mask = entityMasks[ent.getIndex()];
308  const ComponentMask::SimpleMask ogMask = mask;
309  ComponentMask::add(mask, cIndex);
310  for (auto& view : views) {
311  if (view->mask.passes(mask)) { view->tryAddEntity(ent); }
312  else if (view->mask.passes(ogMask) && ComponentMask::has(view->mask.excluded, cIndex)) {
313  view->removeEntity(ent);
314  }
315  }
316 
317  // ensure parent traits are up to date
318  auto& pool = getPool<T>();
319  for (Entity child : parentGraph.getChildren(ent)) {
320  const std::uint64_t ic = child.getIndex();
321  const ComponentMask mask{.required = entityMasks[ic]};
322  if (mask.contains(pool.ComponentIndex)) { pool.onParentSet(child, ent); }
323  }
324  const Entity parent = parentGraph.getParent(ent);
325  if (parent != InvalidEntity) {
326  const std::uint64_t ip = parent.getIndex();
327  const ComponentMask mask{.required = entityMasks[ip]};
328  if (mask.contains(pool.ComponentIndex)) { pool.onParentSet(ent, parent); }
329  }

345 
346  auto& pool = getPool<T>();
347 
348  // notify pools of parent remove on children
349  for (Entity child : parentGraph.getChildren(ent)) {
350  const std::uint32_t ic = child.getIndex();
351  const ComponentMask mask{.required = entityMasks[ic]};
352  if (mask.contains(pool.ComponentIndex)) {

354  if (parent == InvalidEntity) { continue; }

362  for (auto& view : views) {
