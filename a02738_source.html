36void Systems::update(FrameStage::V startStage, FrameStage::V endStage, StateMask::V stateMask,
37 float dt, float realDt, float lag, float realLag) {
38 const auto beg = systems.begin() + startStage;
39 const auto end = systems.begin() + endStage;
40 auto& tp = engine.threadPool();
41
42 for (auto it = beg; it != end; ++it) {
43 if (!it->tasks.empty()) {
44 tp.queueTask([it]() { it->drainTasks(); });
45 }
46 if (it->systems.size() > 1) {
47 for (auto& system : it->systems) {
48 if ((system.mask & stateMask) != 0) {
49 tp.queueTask([&system, it, dt, realDt, lag, realLag]() {
50 system.system->update(it->mutex, dt, realDt, lag, realLag);
51 });
52 }
53 }
54 }
55 else if (!it->systems.empty()) {
56 auto& system = it->systems.front();
57 if ((system.mask & stateMask) != 0) {
58 system.system->update(it->mutex, dt, realDt, lag, realLag);
59 }
60 }
61 tp.drain();
62 }
