31void BindlessTextureArray::init(vk::PerFrame<VkDescriptorSet>& ds,
32 vk::PerFrame<VkDescriptorSet>& rtds) {
33 descriptorSets = &ds;
34 rtDescriptorSets = &rtds;
35
36 // ensure an error pattern exists
37 if (errorPattern.getSize().x == 0) { errorPattern.create(32, 32, sf::Color(245, 66, 242)); }
38
39 // init error pattern texture
40 errorTexture.altImg = &errorPattern;
41 errorTexture.createFromContentsAndQueue();
42 errorTexture.sampler = vulkanState.samplerCache.filteredRepeated();
43 vulkanState.transferEngine.executeTransfers();
44
45 // init all textures to error pattern
46 for (vk::Texture& txtr : textures) { txtr = errorTexture; }
47 for (vk::TextureDoubleBuffered& txtr : renderTextures) { txtr = errorTexture; }
48
49 // fill descriptor set
50 const std::size_t len = textures.size() + renderTextures.size();
51 std::vector<VkDescriptorImageInfo> imageInfos(len * Config::MaxConcurrentFrames,
52 VkDescriptorImageInfo{});
53 for (unsigned int fi = 0; fi < Config::MaxConcurrentFrames; ++fi) {
54 std::size_t infoIndex = len * fi;
55
56 for (unsigned int i = 0; i < textures.size(); ++i, ++infoIndex) {
57 auto& info = imageInfos[infoIndex];
58 info.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
59 info.imageView = textures[i].view;
60 info.sampler = textures[i].getSampler();
61 }
62 for (unsigned int i = 0; i < renderTextures.size(); ++i, ++infoIndex) {
63 auto& info = imageInfos[infoIndex];
64 info.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
65 info.imageView = renderTextures[i].getImages().getRaw(fi).view;
66 info.sampler = renderTextures[i].getSampler();
67 }
68 }

73 const auto visitor = [this, &i, &fi, len, &setWrites, &imageInfos](auto& set) {
74 setWrites[i].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;

140void BindlessTextureArray::commitDescriptorUpdates() {
141 if (!queuedUpdates.current().empty()) {
142 // prepare descriptor updates before waiting
143 std::vector<VkWriteDescriptorSet> writes;
144 std::vector<VkDescriptorImageInfo> infos;
145 writes.resize(queuedUpdates.current().size() * 2, {});
146 infos.resize(queuedUpdates.current().size() * 2, {});
147
148 for (unsigned int j = 0; j < queuedUpdates.current().size(); ++j) {
149 vk::TextureBase* texture = queuedUpdates.current()[j];
150 const bool isRT = dynamic_cast<vk::TextureDoubleBuffered*>(texture) != nullptr;
151 const unsigned int k = queuedUpdates.current().size() + j;
152
153 std::uint32_t i;
154 VkImageView view;
155 if (!isRT) {
156 vk::Texture* cast = static_cast<vk::Texture*>(texture);
157 i = cast - textures.data();
158 view = cast->view;
159 }
160 else {
161 vk::TextureDoubleBuffered* cast = static_cast<vk::TextureDoubleBuffered*>(texture);
162 i = cast - renderTextures.data() + textures.size();
163 view = cast->getImages().current().view;
164 }
165
166 infos[j].imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
167 infos[j].imageView = view;
168 infos[j].sampler = texture->getSampler();
169
170 // write error texture to rt set if is rt itself
171 infos[k].imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
172 infos[k].imageView = isRT ? errorTexture.view : view;
173 infos[k].sampler = isRT ? errorTexture.getSampler() : texture->getSampler();
174
175 writes[j].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
176 writes[j].descriptorCount = 1;
177 writes[j].dstBinding = bindIndex;
178 writes[j].dstArrayElement = i;
179 writes[j].dstSet = descriptorSets->current();
180 writes[j].pImageInfo = &infos[j];
181 writes[j].descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
182
183 writes[k].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
184 writes[k].descriptorCount = 1;
185 writes[k].dstBinding = bindIndex;
186 writes[k].dstArrayElement = i;
187 writes[k].dstSet = rtDescriptorSets->current();
188 writes[k].pImageInfo = &infos[k];
189 writes[k].descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
190 }
191
192 queuedUpdates.current().clear();
193 vkUpdateDescriptorSets(vulkanState.device, writes.size(), writes.data(), 0, nullptr);
194 }
