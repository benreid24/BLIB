306template<typename T>
307T& Engine::getPlayer(unsigned int i) {
308 static_assert(std::is_base_of_v<Player, T>, "T must derive from Player");
309
310 if constexpr (std::is_same_v<Player, T>) {
311 if (players.empty() && i == 0) {
312 BL_LOG_INFO << "No player exists, creating default";
313 addPlayer();
314 }
315 }
316
317#ifdef BLIB_DEBUG
318 T* p = dynamic_cast<T*>(players[i].get());
319 if (!p) { BL_LOG_ERROR << "Bad player cast"; }
320#else
321 T* p = static_cast<T*>(players[i].get());
322#endif
323
324 return *p;
325}

327template<typename T, typename... TArgs>
328T& Engine::addPlayer(TArgs&&... args) {
329 static_assert(std::is_base_of_v<Player, T>, "T must derive from Player");
330 static_assert(
331 std::is_constructible_v<T, Engine&, rc::Observer*, input::Actor*, TArgs...>,
332 "T must be constructible with T(Engine&, rc::Observer*, input::Actor*, TArgs...)");
333
334 auto& observer = rendererInstance->addObserver();
335 auto& actor = input.addActor();
336 T* np = new T(*this, &observer, &actor, args...);
337 auto& player = players.emplace_back(np);
338 eventEmitter.emit<event::PlayerAdded>({*player});
339 return *np;
340}

342template<typename TWorld, typename... TArgs>
343ctr::Ref<World, TWorld> Engine::createWorld(TArgs&&... args) {
344 static_assert(std::is_base_of_v<World, TWorld>, "TWorld must derive from World");
345 static_assert(std::is_constructible_v<TWorld, Engine&, TArgs...>,
346 "TWorld constructor must take Engine& as first parameter");
347
348 auto ref = worldPool.emplaceDerived<TWorld, TArgs...>(*this, std::forward<TArgs>(args)...);
349 for (unsigned int i = 0; i < worlds.size(); ++i) {
350 if (!worlds[i].isValid()) {
351 worlds[i] = ref;
352 ref->index = i;
353 eventEmitter.emit<event::WorldCreated>({*ref});
354 return ref;
355 }
356 }
357
358 throw std::runtime_error("A maximum of 8 worlds may exist concurrently");
359}
