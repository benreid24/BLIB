73bool deviceHasRequiredExtensions(VkPhysicalDevice device, const char* name) {
74 std::uint32_t extensionCount;
75 vkCheck(vkEnumerateDeviceExtensionProperties(device, nullptr, &extensionCount, nullptr));
76
77 std::vector<VkExtensionProperties> availableExtensions(extensionCount);
78 vkCheck(vkEnumerateDeviceExtensionProperties(
79 device, nullptr, &extensionCount, availableExtensions.data()));
80
81 for (const char* ext : RequiredDeviceExtensions) {
82 if (std::find_if(availableExtensions.begin(),
83 availableExtensions.end(),
84 [ext](const VkExtensionProperties& c) {
85 return strcmp(c.extensionName, ext) == 0;
86 }) == availableExtensions.end()) {
87 BL_LOG_WARN << name << " is missing required extension " << ext;
88 return false;
89 }
90 }
91
92 return true;
93}

100 vkCheck(vkEnumerateDeviceExtensionProperties(

104 for (const char* ext : OptionalDeviceExtensions) {
105 if (std::find_if(availableExtensions.begin(),
106 availableExtensions.end(),

108 return strcmp(c.extensionName, ext) == 0;

112 else { score += 500; }
113 }

118int scorePhysicalDevice(VkPhysicalDevice device, const VkPhysicalDeviceProperties& deviceProperties,
119 const VkPhysicalDeviceFeatures& deviceFeatures,
120 const QueueFamilyLocator& queueFamilies, VkSurfaceKHR surface) {
121 // requirements
122 if (!deviceFeatures.samplerAnisotropy) return -1;
123 if (!queueFamilies.complete()) return -1;
124 if (!deviceHasRequiredExtensions(device, deviceProperties.deviceName)) return -1;
125 SwapChainSupportDetails swapChainDetails(device, surface);
126 if (swapChainDetails.formats.empty() || swapChainDetails.presentModes.empty()) return -1;
127
128 // preferred optional features
129 int score = 0;
130 score += deviceProperties.limits.maxImageDimension2D;
131 if (deviceProperties.deviceType == VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU) { score += 50000; }
132 score += scoreDeviceForOptionalExtensions(device, deviceProperties.deviceName);
133
134 return score;
135}

203void VulkanState::createInstance() {
204 // app info
205 const std::string appName =
206 engine::Configuration::getOrDefault<std::string>("blib.app.name", "BLIB Application");
207 VkApplicationInfo appInfo{};
208 appInfo.sType = VK_STRUCTURE_TYPE_APPLICATION_INFO;
209 appInfo.pApplicationName = appName.c_str();
210 appInfo.applicationVersion = VK_MAKE_VERSION(1, 0, 0);
211 appInfo.pEngineName = "BLIB";
212 appInfo.engineVersion = VK_MAKE_VERSION(3, 0, 0);
213 appInfo.apiVersion = VK_API_VERSION_1_3;
214
215 // instance data to populate
216 VkInstanceCreateInfo createInfo{};
217 createInfo.sType = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO;
218 createInfo.pApplicationInfo = &appInfo;
219
220// validation layers
221#ifdef BLIB_DEBUG
222 std::vector<VkLayerProperties> validLayers;
223 std::vector<const char*> validationLayers;
224 std::uint32_t layerCount = 0;
225 if (vkEnumerateInstanceLayerProperties(&layerCount, 0) != VK_SUCCESS) {
226 throw std::runtime_error("Failed to enumerate validation layers");
227 }
228 validLayers.resize(layerCount);
229 if (vkEnumerateInstanceLayerProperties(&layerCount, validLayers.data()) != VK_SUCCESS) {
230 throw std::runtime_error("Failed to retrieve validation layers");
231 }
232 validationLayers.reserve(RequestedValidationLayers.size());
233 for (const VkLayerProperties& layer : validLayers) {
234 if (RequestedValidationLayers.find(layer.layerName) != RequestedValidationLayers.end()) {
235 validationLayers.emplace_back(layer.layerName);
236 }
237 }
238 createInfo.enabledLayerCount = validationLayers.size();
239 createInfo.ppEnabledLayerNames = validationLayers.data();
240#else
241 createInfo.enabledLayerCount = 0;
242#endif
243
244 // query valid extensions
245 std::uint32_t extensionCount = 0;
246 vkCheck(vkEnumerateInstanceExtensionProperties(nullptr, &extensionCount, nullptr));
247 std::vector<VkExtensionProperties> extensions(extensionCount);
248 vkCheck(vkEnumerateInstanceExtensionProperties(nullptr, &extensionCount, extensions.data()));
249 std::unordered_set<std::string> extensionsSet;
250 for (const VkExtensionProperties& ext : extensions) {
251 extensionsSet.emplace(ext.extensionName);
252 }
253
254 // required extensions
255 std::vector<const char*> requiredExtentions =
256 engine::EngineWindow::getRequiredInstanceExtensions();
257#ifdef BLIB_DEBUG
258 requiredExtentions.emplace_back(VK_EXT_DEBUG_UTILS_EXTENSION_NAME);
259 requiredExtentions.emplace_back(VK_EXT_DEBUG_REPORT_EXTENSION_NAME);
260#endif
261
262 bool missing = false;
263 for (const char* ext : requiredExtentions) {
264 if (extensionsSet.find(ext) == extensionsSet.end()) {
265 BL_LOG_ERROR << "Missing required extension: " << ext;
266 missing = true;
267 }
268 }
269 if (missing) { throw std::runtime_error("Missing required extension"); }
270
271 createInfo.enabledExtensionCount = requiredExtentions.size();
272 createInfo.ppEnabledExtensionNames = requiredExtentions.data();
273
274#ifdef BLIB_MACOS
275 createInfo.flags |= VK_INSTANCE_CREATE_ENUMERATE_PORTABILITY_BIT_KHR;
276#endif
277
278 // actual instance creation
279 const VkResult result = vkCreateInstance(&createInfo, nullptr, &instance);
280 if (result != VK_SUCCESS) {
281 BL_LOG_ERROR << "Got instance creation error code: " << result;
282 throw std::runtime_error("Failed to create Vulkan instance");
283 }
284}

320void VulkanState::pickPhysicalDevice() {
321 std::uint32_t deviceCount = 0;
322 vkCheck(vkEnumeratePhysicalDevices(instance, &deviceCount, nullptr));
323 if (deviceCount == 0) { throw std::runtime_error("Failed to find GPUs with Vulkan support"); }
324 std::vector<VkPhysicalDevice> devices(deviceCount);
325 vkCheck(vkEnumeratePhysicalDevices(instance, &deviceCount, devices.data()));
326
327 int pscore = -1;
328 for (const auto& device : devices) {
329 VkPhysicalDeviceProperties deviceProperties;
330 VkPhysicalDeviceFeatures deviceFeatures;
331 vkGetPhysicalDeviceProperties(device, &deviceProperties);
332 vkGetPhysicalDeviceFeatures(device, &deviceFeatures);
333
334 QueueFamilyLocator queueFamilies;
335 queueFamilies.populate(device, surface);
336
337 const int s =
338 scorePhysicalDevice(device, deviceProperties, deviceFeatures, queueFamilies, surface);
339 if (s >= pscore) {
340 physicalDevice = device;
341 pscore = s;
342 }
343 }
344
345 if (pscore < 0 || physicalDevice == VK_NULL_HANDLE) {
346 throw std::runtime_error("Failed to find a suitable GPU");
347 }
348
349 vkGetPhysicalDeviceProperties(physicalDevice, &physicalDeviceProperties);
350
351 std::uint32_t extensionCount = 0;
352 vkCheck(
353 vkEnumerateDeviceExtensionProperties(physicalDevice, nullptr, &extensionCount, nullptr));
354 availableExtensions.resize(extensionCount);
355 vkCheck(vkEnumerateDeviceExtensionProperties(
356 physicalDevice, nullptr, &extensionCount, availableExtensions.data()));
357}

359void VulkanState::createLogicalDevice() {
360 QueueFamilyLocator indices;
361 indices.populate(physicalDevice, surface);
362
363 // queue(s) info
364 std::vector<VkDeviceQueueCreateInfo> queueCreateInfos;
365 std::set<std::uint32_t> uniqueQueueFamilies = {indices.graphicsFamily.value(),
366 indices.presentFamily.value()};
367 const float queuePriorities[] = {1.f};
368 queueCreateInfos.reserve(uniqueQueueFamilies.size());
369 for (std::uint32_t queueIndex : uniqueQueueFamilies) {
370 queueCreateInfos.emplace_back();
371 VkDeviceQueueCreateInfo& queueCreateInfo = queueCreateInfos.back();
372 queueCreateInfo.sType = VK_STRUCTURE_TYPE_DEVICE_QUEUE_CREATE_INFO;
373 queueCreateInfo.queueFamilyIndex = queueIndex;
374 queueCreateInfo.queueCount = 1;
375 queueCreateInfo.pQueuePriorities = queuePriorities;
376 }
377
378 // required features
379 VkPhysicalDeviceFeatures deviceFeatures{};
380 deviceFeatures.samplerAnisotropy = VK_TRUE;
381 deviceFeatures.wideLines =
382 physicalDeviceProperties.limits.lineWidthRange[1] > 1.f ? VK_TRUE : VK_FALSE;
383#ifdef BLIB_DEBUG
384 deviceFeatures.robustBufferAccess = VK_TRUE;
385#endif
386
387 VkDeviceCreateInfo createInfo{};
388 createInfo.sType = VK_STRUCTURE_TYPE_DEVICE_CREATE_INFO;
389 createInfo.pQueueCreateInfos = queueCreateInfos.data();
390 createInfo.queueCreateInfoCount = queueCreateInfos.size();
391 createInfo.pEnabledFeatures = &deviceFeatures;
392 createInfo.enabledExtensionCount = RequiredDeviceExtensions.size();
393 createInfo.ppEnabledExtensionNames = RequiredDeviceExtensions.data();
394 createInfo.enabledLayerCount = 0; // never used
395
396 if (vkCreateDevice(physicalDevice, &createInfo, nullptr, &device) != VK_SUCCESS) {
397 throw std::runtime_error("Failed to create logical device");
398 }
399
400 vkGetDeviceQueue(device, indices.graphicsFamily.value(), 0, &graphicsQueue);
401 vkGetDeviceQueue(device, indices.presentFamily.value(), 0, &presentQueue);
402}

467std::uint32_t VulkanState::findMemoryType(uint32_t typeFilter, VkMemoryPropertyFlags properties) {
468 VkPhysicalDeviceMemoryProperties memProperties;
469 vkGetPhysicalDeviceMemoryProperties(physicalDevice, &memProperties);
470 for (std::uint32_t i = 0; i < memProperties.memoryTypeCount; i++) {
471 if ((typeFilter & (1 << i)) &&
472 (memProperties.memoryTypes[i].propertyFlags & properties) == properties) {
473 return i;
474 }
475 }
476
477 throw std::runtime_error("failed to find suitable memory type!");
478}

564void VulkanState::transitionImageLayout(VkCommandBuffer commandBuffer, VkImage image,
565 VkImageLayout oldLayout, VkImageLayout newLayout) {
566 VkImageMemoryBarrier barrier{};
567 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
568 barrier.oldLayout = oldLayout;
569 barrier.newLayout = newLayout;
570 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
571 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
572
573 barrier.image = image;
574 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
575 barrier.subresourceRange.baseMipLevel = 0;
576 barrier.subresourceRange.levelCount = 1;
577 barrier.subresourceRange.baseArrayLayer = 0;
578 barrier.subresourceRange.layerCount = 1;
579
580 VkPipelineStageFlags sourceStage;
581 VkPipelineStageFlags destinationStage;
582
583 if (oldLayout == VK_IMAGE_LAYOUT_UNDEFINED &&
584 newLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL) {
585 barrier.srcAccessMask = 0;
586 barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
587
588 sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
589 destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
590 }
591 else if (newLayout == VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL) {
592 if (oldLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL) {
593 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
594 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
595
596 sourceStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
597 destinationStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
598 }
599 else {
600 barrier.srcAccessMask = 0;
601 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
602
603 sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
604 destinationStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
605 }
606 }
607 else if (newLayout == VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL) {
608 barrier.srcAccessMask = VK_ACCESS_SHADER_READ_BIT;
609 barrier.dstAccessMask = VK_ACCESS_MEMORY_READ_BIT;
610
611 sourceStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
612 destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
613 }
614 else { throw std::invalid_argument("Unsupported layout transition"); }
615
616 vkCmdPipelineBarrier(
617 commandBuffer, sourceStage, destinationStage, 0, 0, nullptr, 0, nullptr, 1, &barrier);
618}
