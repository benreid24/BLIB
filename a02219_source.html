122void RenderTarget::handleDescriptorSync() {
123 if (hasScene()) {
124 if (!scenes.back().camera) {
125 if (!dynamic_cast<Overlay*>(scenes.back().scene.get())) {
126 BL_LOG_WARN
127 << "Scene being rendered before having a camera set. Creating default camera";
128 }
129 scenes.back().camera = scenes.back().scene->createDefaultCamera();
130 scenes.back().scene->setDefaultNearAndFarPlanes(*scenes.back().camera);
131 }
132
133 const glm::mat4 projView = scenes.back().camera->getProjectionMatrix(viewport) *
134 scenes.back().camera->getViewMatrix();
135 scenes.back().scene->updateObserverCamera(scenes.back().observerIndex, projView);
136 scenes.back().scene->handleDescriptorSync();
137 if (scenes.back().overlay) {
138 scenes.back().overlay->updateObserverCamera(scenes.back().overlayIndex,
139 overlayProjView);

156glm::vec2 RenderTarget::transformToWorldSpace(const glm::vec2& sp) const {
157 const glm::vec2 ndc((sp.x - viewport.x) / viewport.width * 2.f - 1.f,
158 (sp.y - viewport.y) / viewport.height * 2.f - 1.f);
159 glm::mat4 tform(1.f);
160 if (hasScene() && scenes.back().camera) {
161 tform = scenes.back().camera->getProjectionMatrix(viewport) *
162 scenes.back().camera->getViewMatrix();
163 }
164 tform = glm::inverse(tform);
165 const glm::vec4 result = tform * glm::vec4(ndc, 0.f, 1.f);
166 return {result.x, result.y};
167}

220void RenderTarget::renderOverlay(VkCommandBuffer commandBuffer) {
221 if (hasScene()) {
222 if (scenes.back().overlay) {
223 vkCmdSetViewport(commandBuffer, 0, 1, &viewport);
224 vkCmdSetScissor(commandBuffer, 0, 1, &scissor);
225
226 VkClearAttachment attachment{};
227 attachment.aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT;
228 attachment.clearValue = clearColors[1];
229
230 VkClearRect rect{};
231 rect.rect = scissor;
232 rect.baseArrayLayer = 0;
233 rect.layerCount = 1;
234
235 vkCmdClearAttachments(commandBuffer, 1, &attachment, 1, &rect);
236
237 scene::SceneRenderContext ctx(commandBuffer,
238 scenes.back().overlayIndex,
239 viewport,
240 RenderPhase::Overlay,
241 isRenderTexture ?
242 Config::RenderPassIds::StandardAttachmentDefault :
243 Config::RenderPassIds::SwapchainDefault,
244 isRenderTexture);
