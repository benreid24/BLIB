19 const float ot = settings.outlineThickness.value_or(1.f);
20 const glm::vec2 bsize{owner.getAcquisition().width - ot * 2.f,
21 owner.getAcquisition().height - ot * 2.f};
22
23 // size bar based on progress
24 const bool hor = owner.getFillDirection() == ProgressBar::LeftToRight ||
25 owner.getFillDirection() == ProgressBar::RightToLeft;
26 const glm::vec2 size{hor ? owner.getProgress() : 1.f, hor ? 1.f : owner.getProgress()};
27 bar.scaleToSize({bsize.x * size.x, bsize.y * size.y});
28
29 // position based on fill dir

33 break;

35 case ProgressBar::BottomToTop:
36 bar.getTransform().setPosition({0.f, bsize.y - size.y});
37 break;
38
39 case ProgressBar::TopToBottom:
40 case ProgressBar::LeftToRight:
41 default:
42 bar.getTransform().setPosition({0.f, 0.f});
43 break;
44 }
