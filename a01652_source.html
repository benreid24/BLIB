57bool Configuration::load(const std::string& file) {
58 BL_LOG_INFO << "Loading configuration from " << file;
59
60 std::ifstream input(file.c_str(), std::ios::in);
61 if (!input.good()) return false;
62
63 unsigned int lineNo = 1;
64 while (input.good()) {
65 std::string lineStr;
66 std::getline(input, lineStr);
67 if (lineStr.empty()) break;
68
69 std::stringstream line(lineStr);
70 std::string key, value;
71 const char t = line.get();
72 if (line.eof()) return false;
73 std::getline(line, key, '=');
74 if (line.eof()) return false;
75 std::getline(line, value);
76
77 switch (t) {
78 case 'b':
79 if (value == "1") { Configuration::set<bool>(key, true); }
80 else if (value == "0") { Configuration::set<bool>(key, false, true); }
81 else { BL_LOG_WARN << "Bad bool value for key '" << key << "' value: " << value; }
82 break;
83
84 case 'i':
85 parseValue<int>("integer", value, key, lineNo);
86 break;
87
88 case 'u':
89 parseValue<unsigned int>("unsigned integer", value, key, lineNo);
90 break;
91
92 case 'f':
93 parseValue<float>("float", value, key, lineNo);
94 break;
95
96 case 's':
97 Configuration::set<std::string>(key, value, true);
98 break;
99
100 default:
101 BL_LOG_WARN << "Invalid type '" << t << "'"
102 << "on line " << lineNo << " (" << lineStr << ")";
103 break;
104 }
105 ++lineNo;
106 }
