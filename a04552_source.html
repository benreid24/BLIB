125 void Renderer::renderFrame() {
126  // kick off transfers
127  for (vk::RenderTexture* rt : renderTextures) { rt->handleDescriptorSync(); }
128  if (commonObserver.hasScene()) { commonObserver.handleDescriptorSync(); }
129  else {
130  for (auto& o : observers) { o->handleDescriptorSync(); }
131  }
132  state.transferEngine.executeTransfers();
133 
134  // begin frame
135  vk::StandardAttachmentSet* currentFrame = nullptr;
136  VkCommandBuffer commandBuffer = nullptr;
137  state.beginFrame(currentFrame, commandBuffer);
138  framebuffers.current().recreateIfChanged(*currentFrame);
139 
140  // begin render texture rendering
141  for (vk::RenderTexture* rt : renderTextures) { rt->renderScene(commandBuffer); }
142 
143  // record commands to render scenes
144  if (commonObserver.hasScene()) { commonObserver.renderScene(commandBuffer); }
145  else {
146  // record all before blocking to apply postfx
147  for (auto& o : observers) { o->renderScene(commandBuffer); }
148  }
149 
150  // begin render pass to composite content into swapchain image
151  framebuffers.current().beginRender(commandBuffer,
152  {{0, 0}, currentFrame->renderExtent()},
153  clearColors,

155  false);
156 
157  // apply rendered scenes to swap image with postfx
158  if (commonObserver.hasScene()) { commonObserver.compositeSceneWithEffects(commandBuffer); }
159  else {
160  for (auto& o : observers) { o->compositeSceneWithEffects(commandBuffer); }
161  }
162 
163  // render overlays
164  for (auto& o : observers) { o->renderOverlay(commandBuffer); }
165  commonObserver.renderOverlay(commandBuffer);
166 
167  // complete frame

169  state.completeFrame();
