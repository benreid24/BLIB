105bool WriteContext::setupWriteStream(std::string_view filename, stream::OutputStream& output) {
106 switch (getMode()) {

125void WriteContext::flushBundleWrite() {
126 if (!priorFile.empty() && bundle) {
127 const std::size_t end = bundle->data.size();
128 const std::size_t size = end > priorFileOffset ? end - priorFileOffset : 0;
129 auto& fileMap = bundle->assetFileManifest[asset.getUUID()];
130 std::string path(priorFile);
131 auto result = fileMap.try_emplace(path, path, priorFileOffset, size);
132 if (!result.second) {
133 BL_LOG_ERROR << "Failed to save duplicate file '" << path << "' for asset "
134 << asset.getUUID() << " to bundle";
135 }
136
137 // clear so we do not double write in destructor
138 priorFile = std::string_view();
139 }
