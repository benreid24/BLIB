48 const Element::Ptr& content, const PageChangedCb& onOpen,

50 if (pageMap.find(name) == pageMap.end()) {
51 Button::Ptr label = Button::create(title);
52 label->setOutlineThickness(1.f);
53 const auto it = pages.emplace(pages.end(), name, label, content, onOpen, onClose);
54 Page& p = *it;

63 if (pages.size() == 1) makePageActiveDirect(&*pages.begin());
64
65 updateTabBoxes();

139sf::Vector2f Notebook::minimumRequisition() const {
140 sf::Vector2f tabReq = tabArea->getRequisition();
141 tabReq.x = maxWidth > 0.f ? std::min(maxWidth, tabReq.x) : tabReq.x;
142 tabReq.y += TabTopPadding;
143 sf::Vector2f contentReq(0, 0);
144 for (const Page& p : pages) {
145 const sf::Vector2f creq = p.content->getRequisition();
146 contentReq.x = std::max(contentReq.x, creq.x);
147 contentReq.y = std::max(contentReq.y, creq.y);
148 }
149 contentReq += 2.f * sf::Vector2f(ContentPadding, ContentPadding);
150 return {std::max(tabReq.x, contentReq.x), tabReq.y + contentReq.y};
151}

153void Notebook::onAcquisition() {
154 const float contentOutline =
155 activePage ?
156 activePage->content->renderSettings().outlineThickness.value_or(TabTopPadding) :
157 TabTopPadding;
158 tabAcquisition = {getAcquisition().left + contentOutline,
159 getAcquisition().top + TabTopPadding,
160 std::max(tabArea->getRequisition().x, getAcquisition().width) -
161 contentOutline * 2.f,
162 tabArea->getRequisition().y - TabTopPadding};
163 Packer::manuallyPackElement(tabArea, tabAcquisition, true);
164 if (maxWidth > 0.f) {
165 tabAcquisition.width =
166 std::min(getAcquisition().width, tabAcquisition.width) - contentOutline * 2.f;
167 }
168 if (activePage) Packer::manuallyPackElement(activePage->content, contentArea());
169
170 constrainScroll();
171}

177void Notebook::makePageActiveDirect(Page* page) {
178 if (activePage) {
179 activePage->content->setVisible(false, false);
180 if (page != activePage) activePage->onClose();
181 }
182 page->content->setVisible(true, false);
183 page->content->moveToTop();
184 Packer::manuallyPackElement(page->content, contentArea(), true);
185 if (page != activePage) { page->onOpen(); }
186 activePage = page;
187
188 if (maxWidth > 0.f) {
189 const float labelX =
190 (page->label->getPosition().x - tabArea->getPosition().x) + tabAcquisition.left;
191 const float x = labelX - scroll;
192 if (x < getPosition().x) { scroll = labelX - getPosition().x; }
193 else {
194 const float right = getAcquisition().left + getAcquisition().width;
195 const float xr = x + page->label->getAcquisition().width;
196 if (xr > right) {
197 scroll = (labelX - getPosition().x) - getAcquisition().width +
198 page->label->getAcquisition().width;
199 }
200 }
201 constrainScroll();
202 }
203 updateTabBoxes();
204}

240bool Notebook::propagateEvent(const Event& e) {
241 const bool inTabArea = tabAcquisition.contains(e.mousePosition());
242 if (e.type() == Event::Scrolled && inTabArea) {
243 scroll += e.scrollDelta() * -4.f;
244 constrainScroll();
245 const Event fakeMove(Event::MouseMoved,
246 sf::Vector2f(e.mousePosition().x, e.mousePosition().y));
247 tabArea->processEvent(fakeMove);
248 return true;
249 }
250
251 if (inTabArea && tabArea->processEvent(e)) return true;
252 if (activePage) return activePage->content->processEvent(e);
253 return false;
254}

256void Notebook::constrainScroll() {
257 if (scroll < 0.f) { scroll = 0.f; }
258 if (maxWidth > 0.f && tabArea->getAcquisition().width > getAcquisition().width) {
259 const float ms = tabArea->getAcquisition().width - tabAcquisition.width;
260 if (scroll > ms) { scroll = ms; }
261 }
262
263 if (maxWidth < 0.f) { scroll = 0.f; }
264
265 tabArea->setPosition({tabAcquisition.left - scroll, tabAcquisition.top});
266}
