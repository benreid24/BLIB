116bool Element::processEvent(const Event& event) {
117 if (!visible()) return false;
118
119 if (propagateEvent(event)) return true;
120
121 const bool eventOnMe = cachedArea.contains(event.mousePosition());
122
123 if (event.type() == Event::Scrolled) {
124 if (eventOnMe) { return handleScroll(event); }
125 return false;
126 }
127
128 switch (event.type()) {
129 case Event::TextEntered:
130 case Event::KeyPressed:
131 case Event::KeyReleased:
132 return processAction(event);
133
134 case Event::LeftMousePressed:
135 case Event::RightMousePressed:
136 if (eventOnMe) {
137 if (!active()) return true;
138
139 if (takeFocus()) {
140 if (event.type() == Event::LeftMousePressed) {
141 dragStart = event.mousePosition();
142 isLeftPressed = true;
143 updateUiState();
144 return processAction(event);
145 }
146 else {
147 isRightPressed = true;
148 updateUiState();
149 return true;
150 }
151 }
152 return true; // click blocked
153 }
154 return false;
155
156 case Event::LeftMouseReleased:
157 case Event::RightMouseReleased:
158 if (!active()) {
159 isLeftPressed = isRightPressed = false;
160 return eventOnMe;
161 }
162 processAction(event);
163 if (event.type() == Event::LeftMouseReleased) {
164 if (isLeftPressed) {
165 isLeftPressed = false;
166 updateUiState();
167 if (eventOnMe) {
168 processAction(Event(Event::LeftClicked, event.mousePosition()));
169 return true;
170 }
171 }
172 }
173 else if (event.type() == Event::RightMouseReleased) {
174 if (isRightPressed) {
175 isRightPressed = false;
176 updateUiState();
177 if (eventOnMe) {
178 processAction(Event(Event::RightClicked, event.mousePosition()));
179 return true;
180 }
181 }
182 }
183 return false;
184
185 case Event::MouseMoved:
186 if (!active()) { return eventOnMe; }
187 if (showingTooltip && component) {
188 showingTooltip = false;
189 component->dismissTooltip();
190 }
191 if (eventOnMe) { fireSignal(event); }
192 if (isLeftPressed) {
193 isMouseOver = eventOnMe;
194 const bool r = processAction(Event(Event::Dragged, dragStart, event.mousePosition()));
195 dragStart = event.mousePosition();
196 updateUiState();
197 return r;
198 }
199 else if (eventOnMe) {
200 if (!isMouseOver) {
201 isMouseOver = true;
202 updateUiState();
203 processAction(Event(Event::MouseEntered, event.mousePosition()));
204 }
205 hoverTime = 0.f;
206 return true;
207 }
208 else if (isMouseOver) {
209 isMouseOver = false;
210 updateUiState();
211 processAction(Event(Event::MouseLeft, event.mousePosition()));
212 }
213 return false;
214
215 case Event::MouseOutsideWindow:
216 isMouseOver = false;
217 isLeftPressed = false;
218 isRightPressed = false;
219 updateUiState();
220 return false;
221
222 default:
223 return false;
224 }
225}

254bool Element::shouldMarkSelfDirty() {
255 constexpr float ShrinkThresh = 0.75f;
256 const sf::Vector2f newReq = getRequisition();
257 const sf::FloatRect& acq = getAcquisition();
258 if (newReq.x > acq.width || newReq.y > acq.height ||
259 (!fillX && newReq.x < acq.width * ShrinkThresh) ||
260 (!fillY && newReq.y < acq.height * ShrinkThresh)) {
261 return true;
262 }
263 return false;
