86bool MusicDriver::doWrite(as::WriteContext& ctx, const MusicPayload&) {
87 // only write in Export mode otherwise we clobber the source file
88 if (ctx.getMode() != as::Mode::BundleCreation) { return true; }
89
90 stream::InputStream readStream;
91 if (!readStream.open(ctx.getFilePath("music.ogg"))) {
92 BL_LOG_ERROR << "Failed to open source music file";
93 return false;
94 }
95 stream::OutputStream writeStream;
96 if (!ctx.setupWriteStream("music.ogg", writeStream)) {
97 BL_LOG_ERROR << "Failed to setup music write stream";
98 return false;
99 }
100 constexpr std::size_t ChunkSize = 8192;
101 const std::size_t fullChunks = readStream.getSize() / ChunkSize;
102 const std::size_t lastChunkSize = readStream.getSize() % ChunkSize;
103 const std::size_t totalChunks = fullChunks + (lastChunkSize > 0 ? 1 : 0);
104
105 std::vector<char> buffer;
106 buffer.resize(ChunkSize);
107 for (std::size_t i = 0; i < totalChunks; ++i) {
108 const std::size_t toRead = i < fullChunks ? ChunkSize : lastChunkSize;
109 if (!readStream.read(buffer, toRead)) {
110 BL_LOG_ERROR << "Failed to read music chunk";
111 return false;
112 }
113 if (!writeStream.write(buffer.data(), toRead)) {
114 BL_LOG_ERROR << "Failed to write music chunk";
115 return false;
116 }
117 }
118 return true;
119}
