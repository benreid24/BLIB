33MaterialRef MaterialPool::getOrCreateFromTexture(const res::TextureRef& texture) {
34 std::unique_lock lock(mutex);
35
36 checkLazyInit();
37
38 const bool canCache = texture->getType() != vk::Texture::Type::Cubemap;
39
40 if (texture.id() < textureIdToMaterialId.size() &&
41 textureIdToMaterialId[texture.id()] != InvalidId && canCache) {
42 return MaterialRef(this, textureIdToMaterialId[texture.id()]);
43 }
44
45 const auto newId = freeIds.allocate();
46 if (canCache) {
47 if (textureIdToMaterialId.size() <= texture.id()) {
48 textureIdToMaterialId.resize(texture.id() + 1, InvalidId);
49 }
50 textureIdToMaterialId[texture.id()] = newId;
51 }
52 materials[newId] = mat::Material(texture, texture, defaultNormalMap, defaultParallaxMap);
53 markForUpdate(newId);
54 return MaterialRef(this, newId);
55}
